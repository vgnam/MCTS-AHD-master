def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) == 1:
            future_min = distance_matrix[node][destination_node]
            future_variance = 0.0
        else:
            future_distances = [distance_matrix[node][other] for other in unvisited_nodes if other != node]
            future_min = min(future_distances)
            future_mean = sum(future_distances) / len(future_distances)
            future_variance = sum((d - future_mean) ** 2 for d in future_distances) / len(future_distances)
        total_score = current_distance + (0.5 * future_min) + (0.3 * future_variance)
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

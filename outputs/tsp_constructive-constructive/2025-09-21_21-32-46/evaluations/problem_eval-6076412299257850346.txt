def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            nearest_neighbor = min(remaining_nodes, key=lambda x: distance_matrix[node][x])
            lookahead_distance = distance_matrix[node][nearest_neighbor]
            total_score = current_distance + lookahead_distance
        else:
            total_score = current_distance
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if remaining_nodes == 1:
            future_distance = distance_matrix[node][destination_node]
            weight = 1.0
        else:
            future_distances = [distance_matrix[node][other] for other in unvisited_nodes if other != node]
            future_distance = sum(future_distances) / (remaining_nodes - 1)
            weight = (remaining_nodes / (remaining_nodes + 1)) ** 2
        combined_score = (current_distance + future_distance) * weight
        if combined_score < min_score:
            min_score = combined_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    avg_unvisited_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic_factor = (distance_matrix[node][destination_node] if node != destination_node else 0) / (avg_unvisited_distance + 1e-6)
        weighted_score = 0.7 * distance + 0.3 * heuristic_factor

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

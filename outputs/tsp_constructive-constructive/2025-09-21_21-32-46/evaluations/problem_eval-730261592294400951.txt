def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][node]
        neighbor_sum = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        score = direct_distance + 0.5 * neighbor_sum
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

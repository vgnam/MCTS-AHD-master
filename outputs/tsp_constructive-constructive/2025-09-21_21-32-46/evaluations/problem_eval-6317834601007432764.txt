def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    alpha = 0.7  # Weight for immediate distance
    beta = 0.3   # Weight for estimated remaining cost
    next_node = None
    best_score = float('inf')
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_remaining = distance_matrix[node][destination_node]  # Simplified heuristic
        score = alpha * immediate_distance + beta * estimated_remaining
        if score < best_score:
            best_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    min_score = float('inf')
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        future_cost = distance_matrix[node][destination_node] if len(unvisited_nodes) == 1 else 0
        score = distance + 0.3 * future_cost
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

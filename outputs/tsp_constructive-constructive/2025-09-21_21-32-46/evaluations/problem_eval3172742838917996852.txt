def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic_factor = distance_matrix[node][destination_node] if node != destination_node else 0
        score = 0.7 * distance + 0.3 * heuristic_factor
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) == 1:
            future_average = distance_matrix[node][destination_node]
        else:
            future_sum = 0.0
            for other in unvisited_nodes:
                if other != node:
                    distance = distance_matrix[node][other]
                    future_sum += 1.0 / distance
            future_average = 1.0 / future_sum
        total_score = current_distance * future_average
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, balance_factor=0.5):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        score = (1 - balance_factor) * immediate_distance + balance_factor * future_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

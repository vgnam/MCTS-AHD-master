def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    avg_distances = {node: sum(distance_matrix[node]) / len(distance_matrix[node]) for node in unvisited_nodes}
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        score = distance + 0.3 * avg_distances[node]
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, lambda_param=0.5):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        heuristic_score = lambda_param * distance_to_current + (1 - lambda_param) * distance_to_destination
        if heuristic_score < min_score:
            min_score = heuristic_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    heuristic_factor = 0.3  # Weight for heuristic estimate

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic_estimate = distance_matrix[node][destination_node]  # Simplified heuristic
        score = distance + heuristic_factor * heuristic_estimate

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

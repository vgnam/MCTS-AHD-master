def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    avg_unvisited_distances = [sum(distance_matrix[node]) / len(distance_matrix[node]) for node in unvisited_nodes]
    for i, node in enumerate(unvisited_nodes):
        current_distance = distance_matrix[current_node][node]
        avg_distance = avg_unvisited_distances[i]
        score = 0.7 * current_distance + 0.3 * avg_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

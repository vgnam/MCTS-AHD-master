def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            average_distance = 0
        else:
            average_distance = sum(distance_matrix[node][other] for other in unvisited_nodes - {node}) / len(unvisited_nodes - {node})
        score = current_distance + 0.5 * average_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

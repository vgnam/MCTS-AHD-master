def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        if remaining_nodes == 1:
            future_distance = distance_matrix[node][destination_node]
            future_average = future_distance
        else:
            future_sum = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
            future_average = future_sum / (remaining_nodes - 1)
        weight = 1.0 / (1.0 + 0.1 * remaining_nodes)
        total_score = (1 - weight) * current_distance + weight * future_average
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

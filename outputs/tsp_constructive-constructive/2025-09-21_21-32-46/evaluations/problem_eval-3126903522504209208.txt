def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    avg_unvisited_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        heuristic_term = (current_distance - avg_unvisited_distance) ** 2
        total_score = 0.6 * current_distance + 0.3 * future_distance + 0.1 * heuristic_term
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

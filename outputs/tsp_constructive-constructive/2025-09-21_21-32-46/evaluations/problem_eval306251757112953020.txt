def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    min_score = float('inf')
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        future_estimate = distance_matrix[node][destination_node] if destination_node in unvisited_nodes else 0
        score = 0.7 * distance + 0.3 * future_estimate
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

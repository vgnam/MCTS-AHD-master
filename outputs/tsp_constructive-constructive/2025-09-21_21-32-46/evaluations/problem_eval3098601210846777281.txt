def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        estimated_future_distance = distance_matrix[node][destination_node]
        weighted_score = alpha * current_distance + (1 - alpha) * estimated_future_distance
        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node
    return next_node

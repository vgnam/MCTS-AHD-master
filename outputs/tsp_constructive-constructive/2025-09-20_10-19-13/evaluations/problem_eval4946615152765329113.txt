def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    avg_distances = [sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0 for node in unvisited_nodes]
    next_node = max(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] - avg_distances[unvisited_nodes.index(node)]))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic(node):
        proximity = distance_matrix[current_node][node]
        remaining_avg = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes) if unvisited_nodes else 0
        return 0.7 * proximity + 0.3 * remaining_avg

    next_node = min(unvisited_nodes, key=heuristic)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    novelty_factor = 0.3
    next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] * (1 - novelty_factor) + novelty_factor * (1 / (1 + len(unvisited_nodes)))))
    return next_node

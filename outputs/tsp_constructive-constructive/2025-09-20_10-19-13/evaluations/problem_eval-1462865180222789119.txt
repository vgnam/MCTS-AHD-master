def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    avg_distance = total_distance / len(unvisited_nodes) if unvisited_nodes else 0
    weight_factor = 1 + (avg_distance / (total_distance + 1e-10)) ** 0.5
    next_node = min(unvisited_nodes, key=lambda node: weight_factor * distance_matrix[current_node][node])
    return next_node

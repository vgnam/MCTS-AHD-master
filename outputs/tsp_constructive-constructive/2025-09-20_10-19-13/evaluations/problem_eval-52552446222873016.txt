def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    alpha = 0.6  # Weighting factor for immediate distance vs. destination proximity
    beta = 0.4    # Weighting factor for destination proximity

    def combined_score(node):
        immediate_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        return alpha * immediate_dist + beta * dest_dist

    next_node = min(unvisited_nodes, key=combined_score)
    return next_node

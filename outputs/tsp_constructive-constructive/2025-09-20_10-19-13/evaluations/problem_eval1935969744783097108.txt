def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def weighted_cost(node):
        current_distance = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            return current_distance
        avg_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes - {node}) / len(unvisited_nodes - {node})
        return current_distance + 0.3 * avg_future_distance

    next_node = min(unvisited_nodes, key=weighted_cost)
    return next_node

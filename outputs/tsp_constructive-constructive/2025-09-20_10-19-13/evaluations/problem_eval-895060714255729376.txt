def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def weighted_score(node):
        distance = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        return distance + 0.5 * avg_distance_to_unvisited

    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, exploration_factor=0.1):
    if not unvisited_nodes:
        return destination_node
    perturbed_distances = {
        node: distance_matrix[current_node][node] + exploration_factor * np.random.rand()
        for node in unvisited_nodes
    }
    next_node = min(unvisited_nodes, key=lambda node: perturbed_distances[node])
    return next_node

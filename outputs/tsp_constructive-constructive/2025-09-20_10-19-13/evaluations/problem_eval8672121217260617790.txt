import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    total_distance = sum(distances)
    probabilities = [1 / (distance + 1e-10) for distance in distances]  # Avoid division by zero
    total_prob = sum(probabilities)
    normalized_probs = [prob / total_prob for prob in probabilities]

    next_node = random.choices(unvisited_nodes, weights=normalized_probs, k=1)[0]
    return next_node

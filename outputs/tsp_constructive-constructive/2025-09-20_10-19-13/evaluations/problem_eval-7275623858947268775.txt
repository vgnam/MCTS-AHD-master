def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = max(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node], -len([n for n in unvisited_nodes if distance_matrix[node][n] > 0])))
    return next_node

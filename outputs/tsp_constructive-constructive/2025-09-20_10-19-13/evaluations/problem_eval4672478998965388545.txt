def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    heuristic_factor = 0.3
    weighted_scores = {}

    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        future_cost = distance_matrix[node][destination_node]
        weighted_scores[node] = (1 - heuristic_factor) * immediate_cost + heuristic_factor * future_cost

    next_node = min(weighted_scores, key=weighted_scores.get)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0, iteration=0):
    if not unvisited_nodes:
        return destination_node

    nearest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    # Calculate probability to choose random node (decreases with iteration)
    random_prob = temperature * (1.0 / (1.0 + iteration))

    if random.random() < random_prob:
        next_node = random.choice(list(unvisited_nodes))
    else:
        next_node = nearest_node

    return next_node

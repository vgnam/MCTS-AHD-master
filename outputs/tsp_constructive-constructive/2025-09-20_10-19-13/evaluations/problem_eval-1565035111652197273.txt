def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    adjustment_factor = 1.0 - (remaining_nodes / total_nodes) ** 0.5

    next_node = min(unvisited_nodes, key=lambda node: (
        distance_matrix[current_node][node] * (1 - adjustment_factor) +
        adjustment_factor * distance_matrix[node][destination_node]
    ))
    return next_node

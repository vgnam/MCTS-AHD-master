def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def penalty(node):
        distance = distance_matrix[current_node][node]
        if distance < 0.1:  # Threshold to avoid revisiting nearby nodes
            return float('inf')
        return distance

    next_node = min(unvisited_nodes, key=lambda node: penalty(node))
    return next_node

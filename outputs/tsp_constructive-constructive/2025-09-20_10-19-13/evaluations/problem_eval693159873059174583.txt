def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def calculate_score(node):
        distance = distance_matrix[current_node][node]
        potential = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        return distance + 0.3 * potential

    next_node = min(unvisited_nodes, key=lambda node: calculate_score(node))
    return next_node

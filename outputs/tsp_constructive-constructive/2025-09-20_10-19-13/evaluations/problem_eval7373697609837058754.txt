import numpy as np

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    distances = np.array([distance_matrix[current_node][node] for node in unvisited_nodes])
    probabilities = np.exp(-distances / temperature) / np.sum(np.exp(-distances / temperature))
    next_node = np.random.choice(unvisited_nodes, p=probabilities)

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    penalty_factor = 0.1  # Adjustable parameter to control exploration
    node_scores = {
        node: distance_matrix[current_node][node] * (1 + penalty_factor * (len(unvisited_nodes) - unvisited_nodes.index(node)))
        for node in unvisited_nodes
    }

    next_node = min(unvisited_nodes, key=lambda node: node_scores[node])
    return next_node

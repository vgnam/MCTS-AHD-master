def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def adjusted_distance(node):
        immediate_dist = distance_matrix[current_node][node]
        heuristic_bias = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
        return immediate_dist * (1 - heuristic_bias) + heuristic_bias * distance_matrix[node][destination_node]

    next_node = min(unvisited_nodes, key=adjusted_distance)
    return next_node

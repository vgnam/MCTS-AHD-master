def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, node_values, alpha=0.5):
    if not unvisited_nodes:
        return destination_node
    next_node = max(unvisited_nodes, key=lambda node: alpha * (1 / distance_matrix[current_node][node]) + (1 - alpha) * node_values[node])
    return next_node

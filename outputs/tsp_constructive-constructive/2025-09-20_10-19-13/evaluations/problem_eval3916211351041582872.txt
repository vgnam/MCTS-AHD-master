def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    alpha = 0.7  # Weighting factor for balancing distance and destination proximity
    next_node = min(unvisited_nodes, key=lambda node: alpha * distance_matrix[current_node][node] + (1 - alpha) * distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=0.1):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    min_distance = min(distances)
    penalties = [distance - min_distance for distance in distances]

    probabilities = [np.exp(-penalty / temperature) for penalty in penalties]
    probabilities = [p / sum(probabilities) for p in probabilities]

    next_node = np.random.choice(unvisited_nodes, p=probabilities)
    return next_node

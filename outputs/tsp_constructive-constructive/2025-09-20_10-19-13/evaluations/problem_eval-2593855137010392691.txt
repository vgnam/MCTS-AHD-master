def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_value(node):
        distance = distance_matrix[current_node][node]
        proximity_to_dest = distance_matrix[node][destination_node]
        weight = 0.7  # Adjustable weight for balancing distance and heuristic
        return weight * distance + (1 - weight) * proximity_to_dest

    next_node = min(unvisited_nodes, key=heuristic_value)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_weight(node):
        distance = distance_matrix[current_node][node]
        degree = sum(1 for neighbor in unvisited_nodes if neighbor != node and distance_matrix[node][neighbor] > 0)
        return distance * (1 + 1 / (degree + 1))

    next_node = min(unvisited_nodes, key=heuristic_weight)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Calculate a weighted score combining distance and remaining connections
    def score(node):
        distance = distance_matrix[current_node][node]
        remaining_connections = sum(1 for n in unvisited_nodes if n != node)
        weight_factor = 0.7 if remaining_connections > len(unvisited_nodes) / 2 else 0.3
        return distance + weight_factor * remaining_connections

    next_node = min(unvisited_nodes, key=score)
    return next_node

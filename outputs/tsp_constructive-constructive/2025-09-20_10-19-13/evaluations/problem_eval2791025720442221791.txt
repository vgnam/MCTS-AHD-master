def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    avg_distances = {node: sum(distance_matrix[node]) / len(distance_matrix[node]) for node in unvisited_nodes}
    next_node = max(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] - avg_distances[node]))
    return next_node

defaulting to the destination node if no unvisited nodes remain.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    min_distance = min(distances)
    normalized_distances = [min_distance / d for d in distances]

    weights = [math.exp(-d / temperature) for d in normalized_distances]
    total_weight = sum(weights)
    probabilities = [w / total_weight for w in weights]

    next_node = random.choices(unvisited_nodes, weights=probabilities, k=1)[0]
    return next_node

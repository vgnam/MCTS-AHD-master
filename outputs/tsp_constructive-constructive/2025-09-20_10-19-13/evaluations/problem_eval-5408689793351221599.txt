import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, exploration_weight=0.3):
    if not unvisited_nodes:
        return destination_node

    def heuristic_score(node):
        distance = distance_matrix[current_node][node]
        exploration_factor = 1.0 / (1 + len(unvisited_nodes)) if len(unvisited_nodes) > 0 else 1.0
        return (1 - exploration_weight) * distance + exploration_weight * exploration_factor

    next_node = min(unvisited_nodes, key=lambda node: heuristic_score(node))
    return next_node

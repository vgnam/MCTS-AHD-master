def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = 1.0 / len(unvisited_nodes) if unvisited_nodes else 1.0
    next_node = min(
        unvisited_nodes,
        key=lambda node: (distance_matrix[current_node][node] * weight_factor) +
                         (sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1 if len(unvisited_nodes) > 1 else 1))
    )
    return next_node

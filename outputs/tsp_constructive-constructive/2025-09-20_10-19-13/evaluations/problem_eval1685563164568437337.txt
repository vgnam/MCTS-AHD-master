def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    gains = [(node, distance_matrix[current_node][destination_node] - distance_matrix[current_node][node] - distance_matrix[node][destination_node])
             for node in unvisited_nodes]
    next_node = max(gains, key=lambda x: x[1])[0]
    return next_node

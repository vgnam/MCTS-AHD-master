def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_estimate(node):
        distance_to_node = distance_matrix[current_node][node]
        estimated_remaining_path = distance_matrix[node][destination_node] if destination_node in unvisited_nodes else 0
        return distance_to_node + 0.3 * estimated_remaining_path

    next_node = min(unvisited_nodes, key=heuristic_estimate)
    return next_node

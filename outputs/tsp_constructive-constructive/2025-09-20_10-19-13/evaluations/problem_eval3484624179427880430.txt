def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    num_unvisited = len(unvisited_nodes)
    distance_weight = 0.7 if num_unvisited > 1 else 0.5
    centrality_weight = 1.0 - distance_weight

    def node_score(node):
        distance = distance_matrix[current_node][node]
        centrality = sum(distance_matrix[node]) / (len(distance_matrix[node]) - 1)
        return distance_weight * distance + centrality_weight * centrality

    next_node = min(unvisited_nodes, key=node_score)
    return next_node

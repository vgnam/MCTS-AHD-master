def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    penalty_weight = 1.0 / len(unvisited_nodes) if unvisited_nodes else 0
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] + penalty_weight * (1 if node in unvisited_nodes else 0))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    revisit_penalty = 0.1  # Adjusts the influence of revisit penalty
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] + revisit_penalty * (1 if node in unvisited_nodes else 0))
    return next_node

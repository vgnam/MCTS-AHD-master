def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    heuristic_weights = {node: len(unvisited_nodes) - distance_matrix[current_node][node] for node in unvisited_nodes}
    next_node = max(unvisited_nodes, key=lambda node: heuristic_weights[node])
    return next_node

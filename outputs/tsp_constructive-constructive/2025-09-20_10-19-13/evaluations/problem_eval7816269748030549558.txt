def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Calculate a score combining immediate distance and potential future connections
    def score(node):
        immediate_distance = distance_matrix[current_node][node]
        future_potential = min(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        return immediate_distance * 0.7 + future_potential * 0.3

    next_node = min(unvisited_nodes, key=score)
    return next_node

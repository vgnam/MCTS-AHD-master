def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, exploration_factor=0.5):
    if not unvisited_nodes:
        return destination_node

    def heuristic_score(node):
        immediate_distance = distance_matrix[current_node][node]
        remaining_distance_estimate = distance_matrix[node][destination_node] if node != destination_node else 0
        return (1 - exploration_factor) * immediate_distance + exploration_factor * remaining_distance_estimate

    next_node = min(unvisited_nodes, key=heuristic_score)
    return next_node

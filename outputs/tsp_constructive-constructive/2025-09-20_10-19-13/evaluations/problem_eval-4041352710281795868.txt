def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Calculate weighted scores: distance + small random factor to avoid deterministic ties
    def weighted_score(node):
        distance = distance_matrix[current_node][node]
        random_factor = 0.1 * (hash(node) % 10) / 10  # Simple deterministic pseudo-randomness
        return distance + random_factor

    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

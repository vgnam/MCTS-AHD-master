def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    heuristic_factor = 0.3  # Weight for balancing distance and heuristic
    next_node = max(unvisited_nodes, key=lambda node: (1 / distance_matrix[current_node][node]) + heuristic_factor * (1 / distance_matrix[node][destination_node]))
    return next_node

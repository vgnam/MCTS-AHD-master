def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=0.5):
    if not unvisited_nodes:
        return destination_node

    def weighted_score(node):
        distance = distance_matrix[current_node][node]
        exploration_bias = 1.0 / (1.0 + len(unvisited_nodes)) ** temperature
        return distance * (1.0 - exploration_bias)

    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

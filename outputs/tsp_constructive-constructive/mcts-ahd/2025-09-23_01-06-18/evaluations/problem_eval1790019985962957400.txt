def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, threshold=1.5):
    if not unvisited_nodes:
        return destination_node
    closest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    if destination_node in unvisited_nodes and distance_matrix[current_node][destination_node] <= threshold * distance_matrix[current_node][closest_node]:
        return destination_node
    return next_node

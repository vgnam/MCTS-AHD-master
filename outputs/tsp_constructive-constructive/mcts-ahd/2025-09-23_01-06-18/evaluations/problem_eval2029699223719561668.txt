def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if destination_node in unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress_ratio = (total_nodes - remaining_nodes) / total_nodes if total_nodes > 0 else 0.0

    exploration_factor = 0.2 + 0.6 * (1 - progress_ratio) ** 2
    connectivity_factor = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

    def heuristic(node):
        immediate_cost = distance_matrix[current_node][node]
        detour_cost = distance_matrix[node][destination_node] - immediate_cost
        detour_scaling = (remaining_nodes / total_nodes) ** 2 * (1 - progress_ratio)
        connectivity_adjustment = detour_cost * (1 - connectivity_factor / max(distance_matrix[current_node]))
        adjusted_detour = (detour_cost + connectivity_adjustment) * detour_scaling
        return immediate_cost + (1 - exploration_factor) * adjusted_detour

    next_node = min(unvisited_nodes, key=heuristic)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, max_distance=100, threshold_distance=50):
    if not unvisited_nodes:
        return destination_node
    if destination_node in unvisited_nodes:
        if distance_matrix[current_node][destination_node] <= threshold_distance:
            return destination_node
    closest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    if distance_matrix[current_node][closest_node] <= max_distance:
        return closest_node

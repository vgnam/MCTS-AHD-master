def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node
    if destination_node in unvisited_nodes:
        distance_to_dest = distance_matrix[current_node][destination_node]
        min_distance = min(distance_matrix[current_node][node] for node in unvisited_nodes)
        distance_ratio = min_distance / (distance_to_dest + 1e-10)
        prob = 1 / (1 + math.exp(-temperature * (distance_ratio - 0.5) * (len(unvisited_nodes) / 10)))

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if destination_node in unvisited_nodes:
        return destination_node

    def cluster_nodes(nodes):
        clusters = []
        nodes = list(nodes)
        while nodes:
            cluster = [nodes.pop(0)]
            for node in list(nodes):
                max_distance = max(distance_matrix[cluster[0]][c] for c in cluster)
                if distance_matrix[cluster[0]][node] <= max_distance:
                    cluster.append(node)
                    nodes.remove(node)
            clusters.append(cluster)
        return clusters

    clusters = cluster_nodes(unvisited_nodes)
    cluster_scores = []

    for cluster in clusters:
        cluster_score = 0
        for node in cluster:
            immediate_cost = distance_matrix[current_node][node]
            cluster_score += immediate_cost
        cluster_score /= len(cluster)
        cluster_scores.append(cluster_score)

    selected_cluster = clusters[cluster_scores.index(min(cluster_scores))]
    next_node = min(selected_cluster, key=lambda node: distance_matrix[current_node][node] + distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_value = float('inf')

    # Dynamic weight factor based on remaining nodes
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 0.7 if remaining_nodes > 5 else 0.9  # Prioritize immediate distance when few nodes left

    # Penalty for nodes that were previously considered but not chosen
    penalty = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        estimated_cost = distance_matrix[node][destination_node]
        total_cost = (weight_factor * distance_to_node) + ((1 - weight_factor) * estimated_cost) + penalty[node]

        if total_cost < best_value:
            best_value = total_cost
            next_node = node

    # Apply penalty to the best node to discourage revisiting in the next step
    if next_node is not None:
        penalty[next_node] += 0.1

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    temperature = 0.5  # Controls exploration vs. exploitation

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            normalization_factor = distance_matrix[current_node][destination_node] / distance_to_destination
            adaptive_penalty = 0.2 * (normalization_factor - 1) * distance_to_destination
            exploration_term = temperature * (distance_to_node * (1 / (distance_to_destination + 1e-5)))
            score = distance_to_node + adaptive_penalty + exploration_term

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

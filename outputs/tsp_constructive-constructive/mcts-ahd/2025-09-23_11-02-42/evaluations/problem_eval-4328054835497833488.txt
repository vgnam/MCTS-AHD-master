def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes == 1:
            score = -distance_to_node - distance_to_destination
        else:
            direct_distance = distance_matrix[current_node][destination_node]
            detour_ratio = direct_distance / (distance_to_node + distance_to_destination)
            detour_threshold = 1.0 + 0.5 * (1.0 - (total_nodes - remaining_nodes) / total_nodes)
            detour_penalty = max(0, (1.0 - detour_ratio) * direct_distance - (detour_threshold * direct_distance))

            node_centrality = 0.0
            for other_node in unvisited_nodes:
                if other_node != node:
                    node_centrality += distance_matrix[node][other_node]
            if remaining_nodes > 1:
                node_centrality /= (remaining_nodes - 1)

            exploration_bonus = node_centrality * (1.0 - (total_nodes - remaining_nodes) / total_nodes)
            progress_factor = (total_nodes - remaining_nodes) / total_nodes
            exploration_weight = 0.4 + 0.3 * (1 - progress_factor)

            score = -distance_to_node - 0.6 * detour_penalty + exploration_weight * exploration_bonus - 0.3 * progress_factor

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

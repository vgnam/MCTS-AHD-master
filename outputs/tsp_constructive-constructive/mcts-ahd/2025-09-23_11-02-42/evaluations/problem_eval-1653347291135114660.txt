def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes == 1:
            score = distance_to_destination - distance_to_node
        else:
            direct_distance = distance_matrix[current_node][destination_node]
            detour_ratio = direct_distance / (distance_to_node + distance_to_destination)
            detour_penalty = (1.0 - detour_ratio) * direct_distance

            angular_deviation = 0.0
            for other_node in unvisited_nodes:
                if other_node != node:
                    angular_deviation += abs(distance_matrix[current_node][other_node] - distance_to_node)
            angular_deviation /= (remaining_nodes - 1) if remaining_nodes > 1 else 1

            node_centrality = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
            node_centrality /= (remaining_nodes - 1) if remaining_nodes > 1 else 1

            progress_factor = (total_nodes - remaining_nodes) / total_nodes
            dynamic_weight = 0.6 - 0.2 * progress_factor

            score = (distance_to_node + 0.5 * detour_penalty) - (0.4 * angular_deviation + 0.3 * node_centrality) + dynamic_weight * progress_factor

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

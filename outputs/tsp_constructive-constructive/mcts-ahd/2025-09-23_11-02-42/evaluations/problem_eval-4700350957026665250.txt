def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_value = float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 0.7 if remaining_nodes > 3 else 0.9  # Adjust weight based on remaining nodes

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        estimated_cost = distance_matrix[node][destination_node]
        total_cost = (weight_factor * distance_to_node) + ((1 - weight_factor) * estimated_cost)

        if total_cost < best_value:
            best_value = total_cost
            next_node = node

    return next_node

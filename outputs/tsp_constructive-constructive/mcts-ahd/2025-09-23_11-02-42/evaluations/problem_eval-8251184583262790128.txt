def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    exploration_factor = 0.7 * (1 - (remaining_nodes / 10)) if remaining_nodes > 0 else 0.7

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        direct_distance = distance_matrix[current_node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            detour_ratio = direct_distance / distance_to_destination
            penalty = 1 + (0.1 * (1 / (remaining_nodes + 1)))
            score = distance_to_node + (detour_ratio * 0.3 + exploration_factor * 0.7) * direct_distance * penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

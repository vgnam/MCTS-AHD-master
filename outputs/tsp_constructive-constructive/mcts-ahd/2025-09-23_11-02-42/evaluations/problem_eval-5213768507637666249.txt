def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    remaining_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            detour_factor = (distance_to_node + distance_to_destination) / remaining_distance
            exploration_weight = 1.0 - (0.5 * detour_factor)
            exploitation_weight = 0.5 * (1.0 - exploration_weight)
            score = (exploration_weight * distance_to_node) + (exploitation_weight * distance_to_destination)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_value = float('inf')
    weight_current = 0.6  # Weight for distance from current node
    weight_destination = 0.4  # Weight for distance to destination

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_sum = weight_current * distance_to_node + weight_destination * distance_to_destination

        if weighted_sum < best_value:
            best_value = weighted_sum
            next_node = node

    return next_node

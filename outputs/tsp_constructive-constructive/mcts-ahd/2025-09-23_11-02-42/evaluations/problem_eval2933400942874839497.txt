def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes == 1:
            score = distance_matrix[current_node][destination_node] / (distance_to_node + distance_to_destination)
        else:
            direct_distance = distance_matrix[current_node][destination_node]
            direct_ratio = direct_distance / (distance_to_node + distance_to_destination)

            connectivity_penalty = 0.0
            for other_node in unvisited_nodes:
                if other_node != node:
                    connectivity_penalty += distance_matrix[node][other_node]
            if remaining_nodes > 1:
                connectivity_penalty /= (remaining_nodes - 1)

            score = 0.6 * direct_ratio - 0.4 * connectivity_penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

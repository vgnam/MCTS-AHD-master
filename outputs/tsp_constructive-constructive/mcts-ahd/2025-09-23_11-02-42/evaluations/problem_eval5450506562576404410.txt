def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        estimated_total_distance = distance_to_node + distance_matrix[node][destination_node]

        if remaining_nodes <= 3:
            weight_local = 0.9
            weight_global = 0.1
        else:
            weight_local = 0.5
            weight_global = 0.5

        score = weight_local * distance_to_node + weight_global * estimated_total_distance
        score *= (1 + 0.1 * (remaining_nodes / len(distance_matrix)))

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

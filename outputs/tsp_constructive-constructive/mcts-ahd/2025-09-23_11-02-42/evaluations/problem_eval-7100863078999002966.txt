def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_dest == 0:
            score = dist_current
        else:
            ratio = dist_current / (dist_dest + 1e-6)
            angle = (distance_matrix[current_node][destination_node] - (dist_current + dist_dest)) ** 2
            detour_penalty = (dist_current + dist_dest) / distance_matrix[current_node][destination_node]
            score = ratio * (1 + angle) + detour_penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

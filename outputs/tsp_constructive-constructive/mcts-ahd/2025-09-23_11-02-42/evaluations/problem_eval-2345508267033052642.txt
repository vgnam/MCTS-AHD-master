def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    direct_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            detour_penalty = (distance_to_node + distance_to_destination - direct_distance) ** 2
            global_centrality = sum(distance_matrix[node][n] for n in range(len(distance_matrix))) / len(distance_matrix)
            local_centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / remaining_nodes
            exploration_factor = (remaining_nodes / (remaining_nodes + 1)) ** 2
            detour_weight = 1 - exploration_factor
            centrality_weight = exploration_factor * (1 + global_centrality)

            score = (distance_to_node + detour_penalty * detour_weight) * (1 + centrality_weight * local_centrality)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

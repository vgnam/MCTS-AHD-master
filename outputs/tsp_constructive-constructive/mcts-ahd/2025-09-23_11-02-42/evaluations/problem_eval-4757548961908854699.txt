def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_path_length = sum(distance_matrix[current_node][destination_node] for node in unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            direct_distance = distance_matrix[current_node][destination_node]
            detour_penalty = (distance_to_node + distance_to_destination - direct_distance) / direct_distance
            dynamic_weight = (remaining_nodes / total_path_length) if total_path_length > 0 else 1.0
            exploration_bonus = (1.0 / (1.0 + distance_to_node)) * dynamic_weight
            exploitation_factor = (1.0 - detour_penalty) * (1.0 - dynamic_weight)
            score = distance_to_node + (1.0 - exploitation_factor) * direct_distance - exploration_bonus

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

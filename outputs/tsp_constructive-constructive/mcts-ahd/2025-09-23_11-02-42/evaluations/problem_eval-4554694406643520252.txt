def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        estimated_total_distance = distance_to_node + distance_matrix[node][destination_node]

        # Weight the distance to current node more heavily as more nodes remain unvisited
        weight = 1.0 + (1.0 / (remaining_nodes + 1))
        score = (weight * distance_to_node) + estimated_total_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

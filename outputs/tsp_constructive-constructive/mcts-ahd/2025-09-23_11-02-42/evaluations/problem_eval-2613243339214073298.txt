def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            score = distance_to_node
        else:
            direct_distance = distance_matrix[current_node][destination_node]
            exploration_weight = (remaining_nodes / total_nodes) ** 2
            exploration_bonus = exploration_weight * (1.0 / (1.0 + distance_to_node))

            centrality_penalty = sum(distance_matrix[node]) / total_nodes
            exploitation_factor = (direct_distance / distance_to_destination) * (1.0 + 0.5 * (centrality_penalty / distance_to_node))

            score = distance_to_node + 0.3 * (1.0 - exploitation_factor) * direct_distance - exploration_bonus

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

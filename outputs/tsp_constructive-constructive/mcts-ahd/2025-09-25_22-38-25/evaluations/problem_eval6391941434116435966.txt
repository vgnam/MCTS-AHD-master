def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        distance_to_dest = distance_matrix[current_node][destination_node]
        prob_dest = 1 / (1 + distance_to_dest)
        if np.random.rand() < prob_dest:
            return destination_node
    weighted_nodes = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        random_factor = np.random.uniform(0.1, 1.0)
        weight = distance * random_factor
        weighted_nodes.append((node, weight))
    weighted_nodes.sort(key=lambda x: x[1])
    return next_node

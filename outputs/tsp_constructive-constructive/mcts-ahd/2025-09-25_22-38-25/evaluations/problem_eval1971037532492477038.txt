def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, iteration, max_iterations):
    cooling_rate = 0.95
    temperature = max(0.1, 0.7 * (cooling_rate ** iteration))
    if destination_node in unvisited_nodes:
        if random.random() < temperature:
            return destination_node
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

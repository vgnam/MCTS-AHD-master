import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        if random.random() < 0.35:
            return destination_node
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
    if next_node is not None and random.random() < 0.6:
        return next_node
    if next_node is not None and random.random() < 0.05:
        faraway_nodes = [node for node in unvisited_nodes if distance_matrix[current_node][node] >= min_distance * 1.5]
        if faraway_nodes:
            return random.choice(faraway_nodes)
    return next_node

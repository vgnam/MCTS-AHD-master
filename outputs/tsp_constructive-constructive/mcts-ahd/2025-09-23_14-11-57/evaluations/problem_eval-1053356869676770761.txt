def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    immediate_weight = 0.5 if remaining_nodes > 3 else 0.7
    future_weight = 0.3 if remaining_nodes > 3 else 0.5

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        best_lookahead_distance = float('inf')
        for next_node in unvisited_nodes:
            if next_node != node:
                lookahead_distance = distance_matrix[node][next_node]
                if lookahead_distance < best_lookahead_distance:
                    best_lookahead_distance = lookahead_distance

        score = immediate_weight * immediate_distance + future_weight * best_lookahead_distance

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    min_value = float('inf')

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        sum_dist = current_to_node + node_to_dest
        product_dist = current_to_node * node_to_dest
        ratio_dist = node_to_dest / (current_to_node + 1e-10)  # Avoid division by zero
        combined_value = sum_dist + 0.5 * product_dist + 0.3 * ratio_dist

        if combined_value < min_value:
            min_value = combined_value
            best_node = node

    return next_node

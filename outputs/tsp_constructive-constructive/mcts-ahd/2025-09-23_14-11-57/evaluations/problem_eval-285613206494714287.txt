def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    min_weighted_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    progress = 1 - len(unvisited_nodes) / total_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_potential = distance_matrix[node][destination_node]

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            reconsideration_penalty = sum(distance_matrix[current_node][n] for n in remaining_nodes) / len(remaining_nodes) if progress > 0.5 else 0
        else:
            avg_remaining_dist = 0
            reconsideration_penalty = 0

        dynamic_weight = 0.5 * (1 - 0.6 * progress)
        weighted_score = immediate_distance + dynamic_weight * (future_potential + avg_remaining_dist) + reconsideration_penalty

        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        total_increase = current_to_node + node_to_dest

        # Stronger penalty for nodes within a closer distance
        penalty = 0.2 * (1.0 / (current_to_node + 1e-6)) if current_to_node < 0.3 else 0

        # Weighted score: prioritize long-term path efficiency
        score = 0.4 * current_to_node + 0.6 * total_increase - penalty

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

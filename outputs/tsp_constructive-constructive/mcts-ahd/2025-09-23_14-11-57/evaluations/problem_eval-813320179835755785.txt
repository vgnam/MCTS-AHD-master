def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = []
    regrets = []
    proximity_potentials = []
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_potential = distance_matrix[node][destination_node]
        distances.append(immediate_distance)

        # Calculate regret: difference between best and second-best options
        other_nodes = [n for n in unvisited_nodes if n != node]
        if other_nodes:
            second_best = min(distance_matrix[current_node][n] for n in other_nodes)
            regret = (second_best - distance_matrix[current_node][node]) if distance_matrix[current_node][node] < second_best else 0
        else:
            regret = 0
        regrets.append(regret)

        # Calculate proximity potential: estimate of how well the node bridges current and future paths
        proximity_potential = (future_potential + immediate_distance) / (1 + sum(distance_matrix[n][node] for n in unvisited_nodes if n != node))
        proximity_potentials.append(proximity_potential)

    # Adaptive weights: prioritize distance and potential early, balance regret and proximity later
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    distance_weight = max(0.1, 0.8 * (remaining_nodes / total_nodes))
    regret_weight = 1.0 - distance_weight
    proximity_weight = max(0.1, 0.5 * (remaining_nodes / total_nodes))

    # Combine weighted scores
    scores = [
        distance_weight * distance + regret_weight * regret + proximity_weight * potential
        for distance, regret, potential in zip(distances, regrets, proximity_potentials)
    ]
    min_score = min(scores)
    selected_index = scores.index(min_score)
    next_node = list(unvisited_nodes)[selected_index]

    return next_node

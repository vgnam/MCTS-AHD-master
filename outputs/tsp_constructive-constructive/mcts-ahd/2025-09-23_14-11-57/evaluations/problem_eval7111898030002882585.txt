def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        product = current_to_node * node_to_dest

        # Normalize by average distance to destination
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        score = product / (avg_distance + 1e-6)  # Avoid division by zero

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')
    base_penalty = 0.3

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            remaining_distances = [distance_matrix[node][n] for n in remaining_nodes]
            variance = sum((d - avg_remaining_dist) ** 2 for d in remaining_distances) / len(remaining_distances)
            dynamic_penalty = base_penalty * (1 + variance / (avg_remaining_dist + 1e-6))
            revisit_penalty = dynamic_penalty * distance_matrix[current_node][node]
        else:
            avg_remaining_dist = 0
            revisit_penalty = 0

        score = current_to_node + avg_remaining_dist + revisit_penalty

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

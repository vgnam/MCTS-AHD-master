def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')
    penalty_factor = 0.5 * (len(unvisited_nodes) / (len(unvisited_nodes) + 1))

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            min_remaining_dist = min(distance_matrix[node][n] for n in remaining_nodes)
            revisit_penalty = penalty_factor * distance_matrix[current_node][node]
            score = current_to_node + 0.5 * (avg_remaining_dist + min_remaining_dist) + revisit_penalty
        else:
            score = current_to_node

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        # Look-ahead: evaluate the worst possible next node after 'node'
        worst_lookahead_distance = 0
        for next_node in unvisited_nodes:
            if next_node != node:
                lookahead_distance = distance_matrix[node][next_node]
                if lookahead_distance > worst_lookahead_distance:
                    worst_lookahead_distance = lookahead_distance

        # Weighted score: prioritize immediate distance with conservative lookahead
        score = 0.7 * immediate_distance + 0.3 * worst_lookahead_distance

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    # Dynamic weighting based on remaining nodes
    weight_proximity = 0.6 if remaining_nodes > 3 else 0.4
    weight_efficiency = 0.4 if remaining_nodes > 3 else 0.6
    weight_heuristic = 0.2

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        total_increase = current_to_node + node_to_dest

        # Penalize nodes too close to avoid cycles
        penalty = 0.3 * (1.0 / (current_to_node + 1e-6)) if current_to_node < 0.4 else 0

        # Heuristic for path efficiency (estimates impact on remaining tour)
        heuristic = sum(min(distance_matrix[node][n], distance_matrix[n][node]) for n in unvisited_nodes if n != node)

        # Combined score with dynamic weights and heuristic
        score = (weight_proximity * current_to_node +
                 weight_efficiency * total_increase +
                 weight_heuristic * heuristic) - penalty

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

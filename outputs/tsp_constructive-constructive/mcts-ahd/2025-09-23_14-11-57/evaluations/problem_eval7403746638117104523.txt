def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = []
    regrets = []
    proximity_potentials = []
    future_diversities = []
    smoothness_penalties = []
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_potential = distance_matrix[node][destination_node]
        distances.append(immediate_distance)

        other_nodes = [n for n in unvisited_nodes if n != node]
        if other_nodes:
            second_best = min(distance_matrix[current_node][n] for n in other_nodes)
            regret = (distance_matrix[current_node][node] - second_best) if distance_matrix[current_node][node] > second_best else 0
        else:
            regret = 0
        regrets.append(regret)

        proximity_potential = (immediate_distance + future_potential) / (1 + sum(distance_matrix[node][n] for n in unvisited_nodes if n != node))
        proximity_potentials.append(proximity_potential)

        future_diversity = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        future_diversities.append(future_diversity)

        smoothness_penalty = abs(immediate_distance - future_potential) / (1 + immediate_distance + future_potential)
        smoothness_penalties.append(smoothness_penalty)

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    regret_weight = max(0.1, 0.8 * (remaining_nodes / total_nodes))
    distance_weight = 1.0 - regret_weight
    proximity_weight = max(0.1, 0.5 * (remaining_nodes / total_nodes))
    diversity_weight = max(0.1, 0.3 * (remaining_nodes / total_nodes))
    smoothness_weight = max(0.1, 0.2 * (remaining_nodes / total_nodes))

    scores = [
        regret_weight * regret + distance_weight * distance + proximity_weight * potential + diversity_weight * diversity + smoothness_weight * penalty
        for regret, distance, potential, diversity, penalty in zip(regrets, distances, proximity_potentials, future_diversities, smoothness_penalties)
    ]
    min_score = min(scores)
    selected_index = scores.index(min_score)
    next_node = list(unvisited_nodes)[selected_index]

    return next_node

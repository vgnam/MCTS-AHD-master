def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    min_weighted_score = float('inf')
    progress = 1 - len(unvisited_nodes) / (len(unvisited_nodes) + 1)  # Normalized progress (0 to 1)
    revisit_penalties = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_potential = distance_matrix[node][destination_node]

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            penalty_decay = 0.9 ** (progress * len(unvisited_nodes))  # Decay penalty based on progress
            revisit_penalties[node] += penalty_decay * (1 - progress)  # Increase penalty for revisitation
            penalty = revisit_penalties[node] if progress > 0.5 else 0
        else:
            avg_remaining_dist = 0
            penalty = 0

        dynamic_weight = 1 / (1 + math.exp(-5 * (progress - 0.5)))  # Sigmoid-based weight adjustment
        weighted_score = immediate_distance + dynamic_weight * future_potential + penalty

        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            best_node = node

    return next_node

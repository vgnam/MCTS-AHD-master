def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            penalty = 0.1 * current_to_node if node in remaining_nodes else 0
        else:
            avg_remaining_dist = 0
            penalty = 0

        score = current_to_node + 0.7 * avg_remaining_dist + penalty

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

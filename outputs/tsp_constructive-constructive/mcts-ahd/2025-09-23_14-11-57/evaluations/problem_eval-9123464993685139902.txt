import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    scores = []
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        product = current_to_node * node_to_dest
        scores.append(1.0 / product if product != 0 else float('inf'))

    total_score = sum(scores)
    normalized_scores = [score / total_score for score in scores] if total_score != 0 else [1.0 / len(unvisited_nodes)] * len(unvisited_nodes)

    selected_index = random.choices(range(len(unvisited_nodes)), weights=normalized_scores, k=1)[0]
    next_node = unvisited_nodes[selected_index]

    return next_node

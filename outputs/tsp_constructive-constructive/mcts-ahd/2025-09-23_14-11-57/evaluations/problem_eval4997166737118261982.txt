def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = []
    future_potentials = []
    penalties = []
    progress = 1 - len(unvisited_nodes) / (len(unvisited_nodes) + 1)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_potential = distance_matrix[node][destination_node]
        distances.append(immediate_distance)
        future_potentials.append(future_potential)

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distance_matrix[current_node][n] for n in remaining_nodes) / len(remaining_nodes)
            penalty = (avg_remaining_dist ** 2) if progress > 0.7 else 0
        else:
            penalty = 0
        penalties.append(penalty)

    total_nodes = len(distance_matrix)
    distance_weight = 0.5 * (len(unvisited_nodes) / total_nodes)
    future_weight = 0.3 + 0.2 * (1 - len(unvisited_nodes) / total_nodes)
    penalty_weight = 0.2 * (1 - len(unvisited_nodes) / total_nodes)

    scores = [
        distance_weight * distance + future_weight * potential + penalty_weight * penalty
        for distance, potential, penalty in zip(distances, future_potentials, penalties)
    ]
    min_score = min(scores)
    selected_index = scores.index(min_score)
    next_node = list(unvisited_nodes)[selected_index]

    return next_node

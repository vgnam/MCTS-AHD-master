def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        total_increase = immediate_distance + lookahead_distance

        penalty = 0.1 * (1.0 / (immediate_distance + 1e-6)) if immediate_distance < 0.5 else 0
        score = 1.0 * immediate_distance + 0.5 * lookahead_distance + 0.3 * total_increase - penalty

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

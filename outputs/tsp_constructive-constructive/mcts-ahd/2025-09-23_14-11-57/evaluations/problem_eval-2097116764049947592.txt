def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        # Look-ahead: evaluate the best possible next node after 'node'
        best_lookahead_distance = float('inf')
        for next_node in unvisited_nodes:
            if next_node != node:
                lookahead_distance = distance_matrix[node][next_node]
                if lookahead_distance < best_lookahead_distance:
                    best_lookahead_distance = lookahead_distance

        # Dynamic weighting: prioritize lookahead when few nodes remain
        weight_immediate = 0.5 if remaining_nodes > 2 else 0.3
        weight_lookahead = 0.5 if remaining_nodes > 2 else 0.7

        # Additional heuristic: favor nodes closer to destination when few nodes left
        destination_distance = distance_matrix[node][destination_node] if remaining_nodes <= 2 else 0
        score = (weight_immediate * immediate_distance +
                 weight_lookahead * best_lookahead_distance +
                 0.2 * destination_distance)

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

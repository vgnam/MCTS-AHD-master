def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    weight_immediate = max(0.5, 0.9 - 0.4 * (total_unvisited / len(distance_matrix)))
    weight_lookahead = 1 - weight_immediate

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        # Look-ahead with penalty for revisiting
        best_lookahead_distance = float('inf')
        for next_node in unvisited_nodes:
            if next_node != node:
                lookahead_distance = distance_matrix[node][next_node]
                if lookahead_distance < best_lookahead_distance:
                    best_lookahead_distance = lookahead_distance

        # Dynamic penalty for revisiting nodes
        penalty = 0.1 if node in unvisited_nodes and current_node == node else 0

        score = (weight_immediate * immediate_distance +
                 weight_lookahead * best_lookahead_distance +
                 penalty * max(distance_matrix))

        if score < best_score:
            best_score = score
            best_node = node

    return next_node

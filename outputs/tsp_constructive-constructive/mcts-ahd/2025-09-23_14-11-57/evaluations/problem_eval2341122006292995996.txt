def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    base_immediate_weight = 0.5
    base_future_weight = 0.3
    path_memory_weight = 0.2

    node_degrees = [sum(1 for d in distance_matrix[node] if d > 0) for node in unvisited_nodes]
    max_degree = max(node_degrees) if node_degrees else 1
    degree_weights = [1 - (d / max_degree) for d in node_degrees]

    current_path_length = sum(distance_matrix[current_node][node] for node in unvisited_nodes if node != current_node)
    avg_path_length = current_path_length / len(unvisited_nodes) if unvisited_nodes else 0

    adaptive_immediate_weight = base_immediate_weight * (1 + 0.1 * (avg_path_length / (sum(sum(row) for row in distance_matrix) / (len(distance_matrix)**2))))
    adaptive_future_weight = base_future_weight * (1 - 0.1 * (avg_path_length / (sum(sum(row) for row in distance_matrix) / (len(distance_matrix)**2))))

    scores = []
    for i, node in enumerate(unvisited_nodes):
        immediate_distance = distance_matrix[current_node][node]
        future_potential = 0

        if len(unvisited_nodes) > 1:
            next_nodes = [n for n in unvisited_nodes if n != node]
            future_potential = sum(distance_matrix[node][n] + distance_matrix[n][destination_node] for n in next_nodes) / len(next_nodes)

        penalty = 0.5 * (1.0 / (immediate_distance + 1e-6)) if immediate_distance < 0.3 else 0
        memory_factor = path_memory_weight * (1 - degree_weights[i]) * (0.5 + 0.5 * (immediate_distance / (avg_path_length + 1e-6)))

        combined_score = adaptive_immediate_weight * immediate_distance + adaptive_future_weight * future_potential - penalty + memory_factor
        scores.append(combined_score)

    selected_index = scores.index(min(scores))
    next_node = list(unvisited_nodes)[selected_index]

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    nodes = list(unvisited_nodes)
    scores = []

    for node in nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        combined_score = 0.7 * (1 / immediate_distance) + 0.3 * (1 / remaining_distance)
        scores.append(combined_score)

    total_score = sum(scores)
    normalized_scores = [score / total_score for score in scores]

    selected_index = random.choices(range(len(nodes)), weights=normalized_scores, k=1)[0]
    next_node = nodes[selected_index]

    return next_node

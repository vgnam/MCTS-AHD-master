def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = -float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            score = -distance_to_node
        else:
            total_remaining_distance = sum(distance_matrix[node][n] for n in remaining_nodes)
            avg_remaining_distance = total_remaining_distance / len(remaining_nodes)
            weight = len(remaining_nodes) / (len(unvisited_nodes) + 1)
            score = (weight * avg_remaining_distance) - ((1 - weight) * distance_to_node)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

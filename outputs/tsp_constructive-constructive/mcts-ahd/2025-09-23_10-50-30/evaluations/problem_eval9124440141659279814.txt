def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    closest_neighbor = None
    min_neighbor_distance = float('inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_neighbor_distance:
            min_neighbor_distance = distance
            closest_neighbor = node

    if destination_node in unvisited_nodes:
        distance_to_destination = distance_matrix[current_node][destination_node]
        if distance_to_destination < min_neighbor_distance:
            return destination_node
        else:
            if distance_matrix[closest_neighbor][destination_node] < distance_matrix[current_node][destination_node]:
                return destination_node
            else:
                return closest_neighbor

        return next_node

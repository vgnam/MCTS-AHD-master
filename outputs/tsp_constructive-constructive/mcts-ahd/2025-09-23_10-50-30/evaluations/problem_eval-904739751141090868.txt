def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight = 0.5 ** (1 / (remaining_nodes + 1))  # Exponentially decreasing weight

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = (1 - weight) * distance_to_destination - weight * distance_to_node

        # Bonus for nodes that are both close to current and far from destination
        if distance_to_node < distance_to_destination:
            score += 0.1 * distance_to_destination

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

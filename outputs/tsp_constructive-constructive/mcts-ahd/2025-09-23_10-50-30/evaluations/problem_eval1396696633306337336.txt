def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_distance = float('inf')
    remaining_nodes = len(unvisited_nodes)
    dynamic_bias = 0.3 * (1 - (remaining_nodes / (remaining_nodes + 10)))  # Bias decreases as more nodes remain

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        distance_to_destination = distance_matrix[current_node][destination_node]
        if distance_to_destination < min_distance * (1 - dynamic_bias):
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_distance = float('inf')
    remaining_nodes = len(unvisited_nodes)
    dynamic_bias = 0.1 + 0.9 * (1 - remaining_nodes / len(distance_matrix))  # Bias increases as nodes are visited
    distance_to_destination = distance_matrix[current_node][destination_node] if destination_node in unvisited_nodes else float('inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        if distance_to_destination < min_distance * (1 - dynamic_bias):
            next_node = destination_node

    return next_node

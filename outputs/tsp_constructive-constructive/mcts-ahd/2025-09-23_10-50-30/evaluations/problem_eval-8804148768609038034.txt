def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = -float('inf')

    remaining_nodes = unvisited_nodes - {current_node}
    if remaining_nodes:
        total_avg_distance = sum(distance_matrix[n1][n2] for n1 in remaining_nodes for n2 in remaining_nodes if n1 != n2)
        avg_distance = total_avg_distance / (len(remaining_nodes) * (len(remaining_nodes) - 1)) if len(remaining_nodes) > 1 else 0
    else:
        avg_distance = 0

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            max_remaining_distance = 0
            avg_remaining_distance = 0
        else:
            max_remaining_distance = max(distance_matrix[node][n] for n in remaining_nodes)
            total_remaining_distance = sum(distance_matrix[node][n] for n in remaining_nodes)
            avg_remaining_distance = total_remaining_distance / len(remaining_nodes)

        reduction = max_remaining_distance - distance_to_node
        savings = avg_remaining_distance - distance_to_node
        weight = len(remaining_nodes) / (len(unvisited_nodes) + 1) if unvisited_nodes else 0

        immediate_weight = 0.3 * (len(unvisited_nodes) / (len(unvisited_nodes) + 1))
        distance_penalty = max(0, (distance_to_node - avg_distance) / (avg_distance + 1e-6)) if avg_distance > 0 else 0

        score = (1 - weight) * reduction + weight * savings - immediate_weight * distance_to_node - distance_penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

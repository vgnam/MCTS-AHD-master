def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    progress = (total_nodes - len(unvisited_nodes)) / total_nodes

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Calculate node density: average distance to other unvisited nodes
        node_density = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0

        # Dynamic weight adjustment
        immediate_weight = 0.8 * (1 - progress) + 0.1 * progress
        future_weight = 0.2 * (1 - progress) + 0.3 * progress
        proximity_penalty = 0.5 * (1 - progress) * (distance_to_destination / (1 + distance_to_current))

        # Novel factor: balance between immediate and future potential with node density
        score = immediate_weight * distance_to_current + future_weight * node_density + proximity_penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('-inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    visited_ratio = 1 - (remaining_count / total_nodes)
    exploration_factor = 1.0 - (remaining_count / total_nodes) ** 2

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            node_centrality = avg_distance
            connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.2 * avg_distance)
            connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * 0.5
            lookahead_depth = min(3, remaining_count // 3) if remaining_count > 3 else 1
            nearest_neighbors = sorted(remaining_nodes, key=lambda n: distance_matrix[node][n])[:lookahead_depth]
            avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0
        else:
            avg_distance = 0
            node_centrality = 0
            connectivity_factor = 1.0
            avg_neighbor_distance = 0

        immediate_weight = 0.5 - 0.2 * exploration_factor
        lookahead_weight = 0.4 * (1.0 - exploration_factor)
        centrality_weight = 0.7 * (1.0 - lookahead_weight)
        penalty_weight = 0.5 * (1 - visited_ratio)

        score = (immediate_weight * (1 / (immediate_distance + 1e-6)) +
                lookahead_weight * (avg_distance / (immediate_distance + 1e-6)) -
                centrality_weight * (node_centrality / (immediate_distance + 1e-6)))
        connectivity_penalty = 1 - (0.3 * avg_neighbor_distance / (immediate_distance + 1e-6)) * (1 - visited_ratio)
        score *= connectivity_factor * connectivity_penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

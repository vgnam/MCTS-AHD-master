def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_distance = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        total_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)

        combined_score = distance_to_current + total_distance_to_remaining

        if combined_score < min_total_distance:
            min_total_distance = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes == 1:
            score = distance_to_current + distance_to_destination
        else:
            avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1) if remaining_nodes > 1 else 0
            connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.2 * avg_distance_to_unvisited)
            connectivity_weight = 1.0 + (1.0 - (remaining_nodes / total_nodes)) * (connectivity / (total_nodes - 1)) * 0.3

            centrality = 1.0 / (avg_distance_to_unvisited) if avg_distance_to_unvisited > 0 else 0
            weight_factor = 0.4 + 0.6 * (remaining_nodes / (remaining_nodes + 3))
            immediate_weight = 0.5 * weight_factor
            centrality_weight = 0.3 * (1 - weight_factor)
            penalty_weight = 0.2 * (1.0 - centrality_weight)
            penalty = penalty_weight * (distance_to_destination / (1.0 + distance_to_current)) if distance_to_current > 0 else 0

            score = (immediate_weight * distance_to_current + centrality_weight * centrality) * connectivity_weight - penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

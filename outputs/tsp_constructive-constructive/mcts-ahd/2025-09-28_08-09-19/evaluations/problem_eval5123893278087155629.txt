def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes == 1:
            score = distance_to_current + distance_to_destination
        else:
            # Calculate centrality (inverse of average distance to unvisited nodes)
            centrality = 1.0 / (sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1)) if remaining_nodes > 1 else 0

            # Calculate exploration bonus (inverse of average distance to all nodes)
            exploration_bonus = 1.0 / (sum(distance_matrix[node][n] for n in range(len(distance_matrix))) / len(distance_matrix))

            # Dynamic weights based on remaining nodes and exploration progress
            progress = 1.0 - (remaining_nodes / len(unvisited_nodes))
            weight_factor = 0.3 + 0.7 * (1.0 - progress**2)  # Non-linear decrease
            immediate_weight = 0.4 * weight_factor
            centrality_weight = 0.3 * (1 - weight_factor)
            exploration_weight = 0.3 * (progress**2)  # Increase exploration as progress advances
            penalty_weight = 0.1 * (1.0 - centrality_weight)

            # Penalty for nodes too close to destination
            penalty = penalty_weight * (distance_to_destination / (1.0 + distance_to_current)) if distance_to_current > 0 else 0

            # Combine components into final score
            score = (immediate_weight * distance_to_current +
                     centrality_weight * centrality +
                     exploration_weight * exploration_bonus) - penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

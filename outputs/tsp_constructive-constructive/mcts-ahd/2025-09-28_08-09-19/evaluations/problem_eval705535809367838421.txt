def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    temperature = 1.0 / (1.0 + remaining_count)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        centrality = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_count - 1) if remaining_count > 1 else 0
        distance_to_destination = distance_matrix[node][destination_node]

        immediate_weight = 0.6 + 0.2 * (1.0 - temperature)
        centrality_weight = 0.3 * temperature
        penalty_weight = 0.5 * (1.0 - centrality_weight)

        penalty = penalty_weight * (distance_to_destination + 0.5 * centrality)
        score = immediate_weight * immediate_distance + centrality_weight * centrality - penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

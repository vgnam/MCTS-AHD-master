def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    visited_ratio = 1 - (remaining_count / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.2 * avg_distance)
        else:
            avg_distance = 0
            connectivity = 0

        connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * 0.3
        weight = 1.0 + (remaining_count / total_nodes) * connectivity_factor * (1.0 - visited_ratio)

        score = (immediate_distance + avg_distance) * weight

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

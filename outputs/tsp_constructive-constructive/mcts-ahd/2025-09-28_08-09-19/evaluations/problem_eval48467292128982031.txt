def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = -float('inf')
    remaining_count = len(unvisited_nodes)
    lookahead_depth = min(3, max(1, remaining_count // 2))

    avg_total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_count if remaining_count > 0 else 0

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        avg_remaining_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes) if remaining_nodes else 0

        connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.2 * avg_remaining_distance)
        connectivity_factor = 1.0 + (connectivity / remaining_count) * 0.4

        deviation_penalty = abs(immediate_distance - avg_total_distance) * (1.0 + (remaining_count / len(unvisited_nodes)) * 0.2)

        future_impact = 0
        if remaining_nodes:
            future_impact = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)

        weight = 0.4 * (immediate_distance / (future_impact + 1)) + 0.3 * connectivity_factor - 0.2 * deviation_penalty

        probability = 1 / (1 + math.exp(-weight))
        score = probability * (1 + connectivity_factor) - deviation_penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

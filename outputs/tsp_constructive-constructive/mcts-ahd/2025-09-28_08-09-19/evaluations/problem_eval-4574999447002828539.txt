def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        local_neighbors = [n for n in unvisited_nodes if n != node]
        avg_neighbor_distance = sum(distance_matrix[node][n] for n in local_neighbors) / len(local_neighbors) if local_neighbors else 0
        score = distance_to_current + avg_neighbor_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

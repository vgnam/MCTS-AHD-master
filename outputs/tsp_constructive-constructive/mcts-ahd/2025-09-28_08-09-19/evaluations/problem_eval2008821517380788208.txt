def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    exploration_factor = 1.0 - (remaining_count / (remaining_count + 1)) * 0.7

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = 0
        lookahead_depth = min(2, remaining_count // 3) if remaining_count > 1 else 1
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            current = node
            for _ in range(lookahead_depth):
                if not remaining_nodes:
                    break
                next_candidate = min(remaining_nodes, key=lambda x: distance_matrix[current][x])
                lookahead_distance += distance_matrix[current][next_candidate]
                current = next_candidate
                remaining_nodes -= {next_candidate}

        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

        immediate_weight = 0.5 + 0.3 * exploration_factor
        lookahead_weight = 0.3 * (1.0 - exploration_factor)
        continuity_weight = 0.4 * exploration_factor
        penalty_weight = 0.4 * (1.0 - lookahead_weight)

        penalty = penalty_weight * (distance_to_destination + 0.3 * avg_distance_to_unvisited)
        score = immediate_weight * immediate_distance + lookahead_weight * (lookahead_distance / lookahead_depth if lookahead_depth > 0 else 0) + continuity_weight * (distance_to_destination + avg_distance_to_unvisited) - penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

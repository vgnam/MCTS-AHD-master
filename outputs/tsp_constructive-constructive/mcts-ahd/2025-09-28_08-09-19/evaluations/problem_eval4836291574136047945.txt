def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_count = len(unvisited_nodes)
    exploration_factor = 1.0 / (1.0 + math.exp(-0.5 * (remaining_count - 5)))

    node_scores = []
    node_frequencies = defaultdict(int)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        lookahead_depth = min(3, max(1, remaining_count // 3))
        lookahead_distance = 0
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            current = node
            for _ in range(lookahead_depth):
                if not remaining_nodes:
                    break
                next_candidate = min(remaining_nodes, key=lambda x: distance_matrix[current][x])
                lookahead_distance += distance_matrix[current][next_candidate]
                current = next_candidate
                remaining_nodes -= {next_candidate}

        node_centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
        popularity = 1.0 / (1.0 + node_centrality)
        distance_to_destination = distance_matrix[node][destination_node]

        immediate_weight = 0.4 + 0.3 * exploration_factor
        lookahead_weight = 0.3 * (1.0 - exploration_factor)
        popularity_weight = 0.2 * exploration_factor
        penalty_weight = 0.5 * (1.0 - exploration_factor)

        frequency_penalty = 0.1 * node_frequencies[node]
        normalized_penalty = penalty_weight * (distance_to_destination + 0.3 * node_centrality + frequency_penalty)

        score = (immediate_weight * immediate_distance +
                lookahead_weight * (lookahead_distance / lookahead_depth if lookahead_depth > 0 else 0) +
                popularity_weight * popularity - normalized_penalty)

        node_scores.append((node, score))
        node_frequencies[node] += 1

    if not node_scores:
        return destination_node

    scores = [score for _, score in node_scores]
    min_score, max_score = min(scores), max(scores)
    normalized_scores = [(node, (score - min_score) / (max_score - min_score + 1e-6)) for node, score in node_scores]

    probabilities = [1.0 - score for node, score in normalized_scores]
    total_prob = sum(probabilities)
    normalized_probabilities = [p / total_prob for p in probabilities]

    next_node = random.choices([node for node, _ in normalized_scores], weights=normalized_probabilities, k=1)[0]

    return next_node

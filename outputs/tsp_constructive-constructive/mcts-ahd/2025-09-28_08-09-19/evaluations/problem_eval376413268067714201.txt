def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    progress_factor = 1.0 - (remaining_count / total_nodes) ** 2

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            node_centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / total_nodes
            path_diversity = sum(distance_matrix[node][n] for n in unvisited_nodes if distance_matrix[node][n] > avg_distance) / len(unvisited_nodes)
        else:
            avg_distance = 0
            node_centrality = 0
            path_diversity = 0

        connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.2 * avg_distance)
        connectivity_weight = 0.5 + 0.5 * progress_factor

        future_potential = 0
        if remaining_nodes:
            future_potential = sum(sorted(distance_matrix[node][n] for n in remaining_nodes)[:max(1, remaining_count // 3)]) / len(remaining_nodes)

        immediate_weight = 0.5 - 0.3 * progress_factor
        diversity_weight = 0.3 * progress_factor
        potential_weight = 0.4 * progress_factor
        centrality_weight = 0.2 * (1.0 - progress_factor)

        score = immediate_weight * immediate_distance + diversity_weight * path_diversity - potential_weight * future_potential - centrality_weight * node_centrality
        score *= (1.0 + connectivity_weight * connectivity / (total_nodes - 1))

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    exploration_progress = 1.0 - (len(unvisited_nodes) / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        lookahead_depth = max(1, min(4, int(2 + exploration_progress * 2)))
        lookahead_distance = 0
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            current = node
            for _ in range(lookahead_depth):
                if not remaining_nodes:
                    break
                next_candidate = min(remaining_nodes, key=lambda x: distance_matrix[current][x])
                lookahead_distance += distance_matrix[current][next_candidate]
                current = next_candidate
                remaining_nodes -= {next_candidate}

        local_centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
        global_centrality = sum(distance_matrix[node][n] for n in range(len(distance_matrix))) / len(distance_matrix)
        node_centrality = 0.7 * local_centrality + 0.3 * global_centrality

        distance_to_destination = distance_matrix[node][destination_node]
        time_decay = 1.0 - exploration_progress

        immediate_weight = 0.6
        lookahead_weight = 0.3 * (1.0 - exploration_progress)
        centrality_weight = 0.1 * exploration_progress
        penalty_weight = 0.4 * time_decay

        penalty = penalty_weight * (distance_to_destination + 0.5 * node_centrality)
        score = immediate_weight * immediate_distance + lookahead_weight * (lookahead_distance / lookahead_depth) + centrality_weight * node_centrality - penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

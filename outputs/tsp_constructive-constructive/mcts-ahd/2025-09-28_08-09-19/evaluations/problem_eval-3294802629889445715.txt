def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if len(unvisited_nodes) == 1:
            avg_distance = 0
        else:
            avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1)

        dynamic_weight = 1 - (1 / (len(unvisited_nodes) ** 0.5))
        score = (1 - dynamic_weight) * avg_distance + dynamic_weight * distance_to_current

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

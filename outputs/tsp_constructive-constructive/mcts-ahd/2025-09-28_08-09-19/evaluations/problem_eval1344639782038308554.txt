def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    visited_ratio = 1 - (len(unvisited_nodes) / total_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if len(unvisited_nodes) == 1:
            avg_distance = 0
        else:
            avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1)

        base_score = distance_to_current - avg_distance

        if len(unvisited_nodes) > 1:
            nearest_neighbors = sorted(unvisited_nodes, key=lambda n: distance_matrix[node][n])[:min(4, len(unvisited_nodes))]
            avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0

            attraction_field = sum(1 / (1 + distance_matrix[node][n]) for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

            weight_factor = 0.4 + 0.6 * (1 - visited_ratio)
            score = weight_factor * base_score + (1 - weight_factor) * (0.7 * avg_neighbor_distance + 0.3 * attraction_field)

            if distance_to_current > 1.2 * avg_neighbor_distance:
                score *= 1.2
            if attraction_field < 0.5:
                score *= 1.1

        else:
            score = base_score

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

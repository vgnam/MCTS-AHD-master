def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    progress_ratio = 1 - (len(unvisited_nodes) / total_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if len(unvisited_nodes) == 1:
            score = distance_to_current + distance_to_destination
        else:
            # Dynamic neighbor exploration
            neighbor_count = min(4, len(unvisited_nodes))
            nearest_neighbors = sorted(unvisited_nodes, key=lambda n: distance_matrix[node][n])[:neighbor_count]
            avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / neighbor_count if neighbor_count else 0

            # Node centrality calculation
            centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)

            # Adaptive weight components
            distance_weight = 0.6 - 0.3 * progress_ratio
            centrality_weight = 0.4 + 0.3 * progress_ratio

            # Non-linear transformations
            transformed_current = distance_to_current ** 0.6
            transformed_centrality = centrality ** 0.9

            # Dynamic exploration penalty
            exploration_penalty = 1 + 0.5 * (transformed_centrality / (transformed_current + 1e-6)) * (1 - progress_ratio)

            # Hybrid scoring
            score = (distance_weight * transformed_current +
                    centrality_weight * transformed_centrality) * exploration_penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

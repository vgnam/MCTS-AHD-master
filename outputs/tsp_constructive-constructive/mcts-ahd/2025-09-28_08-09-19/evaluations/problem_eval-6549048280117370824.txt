def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1
    weight_factor = 1.0 - (1.0 / (1.0 + remaining_nodes))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        degree = sum(1 for n in unvisited_nodes if distance_matrix[node][n] > 0)
        degree_factor = 1.0 + (degree / (total_nodes - 1)) * 0.8

        immediate_weight = 0.5
        centrality_weight = 0.5 + weight_factor * degree_factor
        proximity_weight = 0.3 * (1.0 - weight_factor)

        penalty = proximity_weight * (1.0 / (1.0 + distance_to_destination)) if degree > 0 else 0

        score = -distance_to_current * immediate_weight + degree_factor * centrality_weight - penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

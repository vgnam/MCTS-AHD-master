def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if total_unvisited == 1:
            avg_distance = 0
        else:
            avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (total_unvisited - 1)

        # Normalize avg_distance by the maximum possible distance in the matrix
        max_possible_distance = max(max(row) for row in distance_matrix)
        normalized_avg = avg_distance / max_possible_distance if max_possible_distance != 0 else 0

        # Adjust weights based on remaining nodes density
        density_weight = 1 / (1 + total_unvisited)  # Higher density reduces weight of avg_distance
        score = (1 - density_weight) * distance_to_current + density_weight * (1 - normalized_avg)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

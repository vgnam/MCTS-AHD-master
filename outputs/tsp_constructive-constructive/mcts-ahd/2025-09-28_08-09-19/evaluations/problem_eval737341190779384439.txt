def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if not unvisited_nodes:
            avg_distance_to_remaining = 0
        else:
            avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes) / remaining_nodes

        centrality_factor = 1.0 + (remaining_nodes / total_nodes) * 0.5

        score = distance_to_current + (avg_distance_to_remaining * centrality_factor)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

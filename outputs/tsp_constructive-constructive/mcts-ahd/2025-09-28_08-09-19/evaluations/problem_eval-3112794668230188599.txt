def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        lookahead_depth = min(3, remaining_nodes // 2) if remaining_nodes > 2 else 1
        remaining_unvisited = unvisited_nodes - {node}

        if remaining_unvisited:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_unvisited) / len(remaining_unvisited)
        else:
            avg_distance = 0

        detour_penalty = max(0, (immediate_distance + distance_to_destination) - (distance_matrix[current_node][destination_node] * 1.1))
        continuity_factor = (immediate_distance + distance_to_destination) / (1 + avg_distance + detour_penalty)

        weight_factor = (remaining_nodes / total_nodes) ** 2
        lookahead_factor = sum(min(distance_matrix[node][n] for n in unvisited_nodes if n != node) for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

        score = (1 - weight_factor) * immediate_distance + weight_factor * (continuity_factor - 0.3 * distance_to_destination + 0.3 * lookahead_factor)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

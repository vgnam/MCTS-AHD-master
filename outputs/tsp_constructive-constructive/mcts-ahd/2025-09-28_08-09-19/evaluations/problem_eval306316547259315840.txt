def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    visited_ratio = 1 - (remaining_count / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.2 * avg_distance)
            node_degree = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.5 * avg_distance)
        else:
            avg_distance = 0
            connectivity = 0
            node_degree = 0

        centrality_factor = (node_degree / remaining_count) if remaining_count > 0 else 0
        lookahead_depth = max(1, int((remaining_count ** 0.5) + 0.5))
        nearest_neighbors = sorted(remaining_nodes, key=lambda n: distance_matrix[node][n])[:lookahead_depth] if remaining_nodes else []
        avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0

        immediate_weight = 0.5
        centrality_weight = 0.3 * (1.0 - visited_ratio)
        penalty_weight = 0.2 * (remaining_count / total_nodes)

        score = immediate_weight * immediate_distance + centrality_weight * (immediate_distance * centrality_factor)
        penalty = 1 + penalty_weight * (avg_neighbor_distance / (immediate_distance + 1e-6))
        score *= penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

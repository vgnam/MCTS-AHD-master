def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if len(unvisited_nodes) == 1:
            global_score = 0
        else:
            distances = [distance_matrix[node][n] for n in unvisited_nodes if n != node]
            avg_distance = sum(distances) / (len(unvisited_nodes) - 1)
            variance = sum((d - avg_distance) ** 2 for d in distances) / len(distances) if len(distances) > 1 else 0
            potential_score = avg_distance * (1 / (1 + variance))
            penalty = max(distances) - min(distances) if len(distances) > 1 else 0
            global_score = potential_score * (1 - 0.5 * (penalty / (avg_distance + 1e-6)))

        score = distance_to_current + global_score

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 0.5 * distance_to_destination / (1 + avg_distance_to_unvisited) if avg_distance_to_unvisited > 0 else 0
        total_score = distance_to_current + 0.7 * avg_distance_to_unvisited - penalty

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    lookahead_factor = 1.0 + (remaining_nodes / (remaining_nodes + 1)) * 0.5

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        lookahead_sum = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        score = distance_to_current + lookahead_factor * lookahead_sum

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

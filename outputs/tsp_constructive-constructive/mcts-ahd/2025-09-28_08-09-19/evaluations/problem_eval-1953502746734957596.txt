def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1
    connectivity_weight = 0.3 + 0.6 * (remaining_nodes / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        avg_distance_to_all = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

        connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.2 * avg_distance_to_all)
        connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * connectivity_weight

        penalty = 1.0 + (0.4 * connectivity / (total_nodes - 1)) * (1 - remaining_nodes / total_nodes)

        score = (immediate_distance + avg_distance_to_all) * connectivity_factor * penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    weight_factor = 0.5
    penalty_threshold = 1.5

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        dynamic_weight = 1.0 - (len(unvisited_nodes) / (len(distance_matrix) - 1))
        weighted_destination = distance_to_destination * (1 + dynamic_weight * weight_factor)

        penalty = max(0, distance_to_current - penalty_threshold * distance_matrix[current_node][destination_node])

        score = distance_to_current + weighted_destination + penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

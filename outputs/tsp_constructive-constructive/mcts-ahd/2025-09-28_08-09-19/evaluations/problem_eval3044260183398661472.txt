def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    visited_ratio = 1 - (remaining_count / total_nodes)
    exploration_factor = 1.0 - (visited_ratio ** 2) * 0.8

    lookahead_depth = min(4, remaining_count // 3) if remaining_count > 3 else 1
    avg_total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_count

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        avg_remaining_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes) if remaining_nodes else 0
        node_centrality = avg_remaining_distance

        connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.1 * avg_remaining_distance)
        risk_factor = (connectivity / remaining_count) ** 1.5
        hub_bias = node_centrality / (immediate_distance + 1e-6)

        nearest_neighbors = sorted(remaining_nodes, key=lambda n: distance_matrix[node][n])[:lookahead_depth] if remaining_nodes else []
        avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0

        immediate_weight = 0.5 - 0.2 * exploration_factor
        lookahead_weight = 0.4 * (1.0 - exploration_factor)
        centrality_weight = 0.7 * (1.0 - lookahead_weight)
        risk_weight = 0.3 * (visited_ratio ** 1.5)

        score = immediate_weight * immediate_distance + lookahead_weight * (immediate_distance - avg_remaining_distance)
        score -= centrality_weight * node_centrality * (0.8 + 0.2 * visited_ratio)
        score += risk_weight * risk_factor * (1.5 * immediate_distance)

        hub_penalty = 1 - (0.4 * hub_bias) if hub_bias > 1.2 else 1.0
        score *= hub_penalty * (1 + 0.5 * risk_factor)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

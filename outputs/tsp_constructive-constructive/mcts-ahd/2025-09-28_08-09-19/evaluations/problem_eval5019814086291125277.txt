def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    exploration_progress = 1.0 - (remaining_count / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            node_centrality = avg_distance
            centrality_diversity = sum(abs(distance_matrix[node][n] - avg_distance) for n in remaining_nodes) / len(remaining_nodes)
        else:
            avg_distance = 0
            node_centrality = 0
            centrality_diversity = 0

        connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.1 * avg_distance)
        connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * 0.4

        lookahead_depth = min(4, remaining_count // 3) if remaining_count > 3 else 1
        nearest_neighbors = sorted(remaining_nodes, key=lambda n: distance_matrix[node][n])[:lookahead_depth] if remaining_nodes else []
        avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0

        immediate_weight = 0.6 - 0.3 * exploration_progress
        lookahead_weight = 0.4 * (1.0 - exploration_progress)
        centrality_weight = 0.3 * (1.0 - lookahead_weight)
        significance_weight = 0.5 * exploration_progress

        node_significance = (node_centrality + centrality_diversity) / (1 + connectivity)
        score = immediate_weight * immediate_distance + lookahead_weight * (immediate_distance - avg_neighbor_distance) - centrality_weight * node_centrality
        score += significance_weight * node_significance
        penalty = 1 + (0.6 * avg_neighbor_distance / (immediate_distance + 1e-6)) * (1 - exploration_progress)
        score *= penalty * connectivity_factor

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')
    remaining_count = len(unvisited_nodes)
    lookahead_depth = min(5, max(1, remaining_count // 2))

    avg_total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_count

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        avg_remaining_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes) if remaining_nodes else 0

        connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.2 * avg_remaining_distance)
        connectivity_factor = 1.0 + (connectivity / remaining_count) * 0.7

        novelty_factor = (immediate_distance / (1 + sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes))) if remaining_nodes else 0

        normalized_distance = (immediate_distance - avg_total_distance) / (avg_total_distance + 1e-6)
        normalized_connectivity = (connectivity_factor - 1) * 2
        normalized_novelty = novelty_factor * 0.5

        weight = 0.5 * normalized_distance + 0.3 * normalized_connectivity + 0.2 * normalized_novelty

        score = weight

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

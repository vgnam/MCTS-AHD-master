def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_all = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

        connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.1 * avg_distance_to_all)
        connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * 0.5

        weight = 1.0 + (remaining_nodes / total_nodes) * connectivity_factor
        dynamic_weight = 0.6 * (1.0 + (remaining_nodes / total_nodes) * 0.5)

        score = (distance_to_current * dynamic_weight + avg_distance_to_all * 0.4) * weight

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

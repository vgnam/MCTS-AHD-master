def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_ratio = float('inf')
    base_decay_factor = 0.7
    current_to_destination = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            ratio = float('inf')
        else:
            dynamic_decay = base_decay_factor * (1 - (distance_to_destination / (current_to_destination + 1)))
            weighted_distance = distance_to_current * (distance_to_destination ** dynamic_decay)
            sum_distances = distance_to_current + distance_to_destination
            ratio = weighted_distance / sum_distances

        if ratio < min_ratio:
            min_ratio = ratio
            next_node = node

    return next_node

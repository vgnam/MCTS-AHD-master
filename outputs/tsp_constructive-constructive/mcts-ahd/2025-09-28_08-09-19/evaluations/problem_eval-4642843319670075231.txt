def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

        connectivity = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 1.2 * avg_distance_to_remaining)
        connectivity_factor = 1.0 + (connectivity / (total_nodes - 1)) * 0.7

        weight = 1.0 + (remaining_nodes / total_nodes) * connectivity_factor * 1.5

        score = (distance_to_current + avg_distance_to_remaining) * weight

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

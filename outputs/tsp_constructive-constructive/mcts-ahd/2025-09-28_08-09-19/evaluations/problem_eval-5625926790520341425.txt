def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_count = len(unvisited_nodes)
    total_nodes = remaining_count + 1
    lookahead_depth = min(4, max(2, remaining_count // 3))
    exploration_factor = 1.0 - (remaining_count / total_nodes) * 0.5

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            node_centrality = avg_distance
        else:
            avg_distance = 0
            node_centrality = 0

        connectivity = sum(1 for n in remaining_nodes if distance_matrix[node][n] < 1.2 * avg_distance)
        connectivity_factor = 1.0 + (connectivity / (remaining_count - 1)) * 0.4

        nearest_neighbors = sorted(remaining_nodes, key=lambda n: distance_matrix[node][n])[:lookahead_depth] if remaining_nodes else []
        avg_neighbor_distance = sum(distance_matrix[node][n] for n in nearest_neighbors) / len(nearest_neighbors) if nearest_neighbors else 0

        deviation_penalty = abs(immediate_distance - avg_distance) * (1.0 + (remaining_count / total_nodes) * 0.2)

        immediate_weight = 0.4 + 0.3 * exploration_factor
        centrality_weight = 0.3 * (1.0 - exploration_factor)
        connectivity_weight = 0.2 * (1.0 - exploration_factor)
        penalty_weight = 0.1 * (1.0 - exploration_factor)

        score = immediate_weight * immediate_distance + centrality_weight * node_centrality + connectivity_weight * (immediate_distance * connectivity_factor) + penalty_weight * deviation_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

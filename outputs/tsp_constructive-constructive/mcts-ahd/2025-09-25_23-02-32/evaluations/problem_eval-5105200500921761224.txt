def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_destination = 1.5
    min_distance = float('inf')
    next_node = None

    if destination_node in unvisited_nodes:
        candidate_distance = distance_matrix[current_node][destination_node]
        weighted_distance = candidate_distance * weight_destination
        min_distance = weighted_distance
        next_node = destination_node

    threshold = min_distance * 0.9 if next_node else float('inf')

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        distance = distance_matrix[current_node][node]
        if distance < threshold:
            min_distance = distance
            next_node = node
            threshold = min_distance * 0.9

    return next_node

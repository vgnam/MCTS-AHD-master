def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]

        if len(unvisited_nodes) > 1:
            avg_distance_to_others = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1)
        else:
            avg_distance_to_others = 0

        score = distance_to_node + 0.7 * avg_distance_to_others

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        remaining_nodes = len(unvisited_nodes)
        total_nodes = remaining_nodes + 1  # +1 for the destination
        destination_weight = (total_nodes - remaining_nodes) / total_nodes
        next_node = min(unvisited_nodes, key=lambda node: (
            (1 - destination_weight) * distance_matrix[current_node][node] +
            destination_weight * distance_matrix[node][destination_node]
        ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.5):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] + weight * distance_matrix[node][destination_node]))
    return next_node

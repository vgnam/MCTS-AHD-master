def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
        weighting_factor = distance_matrix[current_node][destination_node] / avg_distance if avg_distance > 0 else 1.0
        next_node = min(unvisited_nodes, key=lambda node: (1 - weighting_factor) * distance_matrix[current_node][node] + weighting_factor * distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)
        if distance_matrix[current_node][destination_node] <= avg_distance_to_dest * 1.2:
            next_node = destination_node
        else:
            next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    else:
        next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] * 0.7 + distance_matrix[node][destination_node] * 0.3))
    return next_node

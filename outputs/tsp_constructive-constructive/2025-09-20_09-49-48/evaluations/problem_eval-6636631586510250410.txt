def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, balance=0.5):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        next_node = min(unvisited_nodes, key=lambda node: balance * distance_matrix[current_node][node] + (1 - balance) * distance_matrix[node][destination_node])
    return next_node

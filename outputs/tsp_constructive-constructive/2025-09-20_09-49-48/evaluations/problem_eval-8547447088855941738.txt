def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        next_node = min(unvisited_nodes, key=lambda node: 0.6 * distance_matrix[current_node][node] + 0.4 * (distance_matrix[current_node][node] / distance_matrix[node][destination_node] if distance_matrix[node][destination_node] != 0 else float('inf')))
    return next_node

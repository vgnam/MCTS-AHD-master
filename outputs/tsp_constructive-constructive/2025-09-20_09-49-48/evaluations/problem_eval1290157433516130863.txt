def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        remaining_nodes = len(unvisited_nodes)
        weight = 1.0 / (remaining_nodes + 1) if remaining_nodes > 0 else 1.0
        next_node = min(unvisited_nodes, key=lambda node: (1 - weight) * distance_matrix[current_node][node] + weight * distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] + distance_matrix[node][destination_node]) + 2 * max(distance_matrix[current_node][node], distance_matrix[node][destination_node]))
    return next_node

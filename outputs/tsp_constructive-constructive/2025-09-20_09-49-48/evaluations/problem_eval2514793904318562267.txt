def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        remaining_nodes = len(unvisited_nodes)
        next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] * (remaining_nodes / (remaining_nodes + 1)) + distance_matrix[node][destination_node] * (1 / (remaining_nodes + 1))))
    return next_node

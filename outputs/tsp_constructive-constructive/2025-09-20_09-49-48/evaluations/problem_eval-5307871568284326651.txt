def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        next_node = destination_node
    else:
        alpha = 0.6
        next_node = min(unvisited_nodes, key=lambda node: alpha * distance_matrix[current_node][node] + (1 - alpha) * max(distance_matrix[current_node][node], distance_matrix[node][destination_node]))
    return next_node

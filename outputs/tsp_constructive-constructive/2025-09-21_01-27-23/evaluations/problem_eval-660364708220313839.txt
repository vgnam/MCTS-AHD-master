def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, penalty=1.3):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        score = distance
        if distance_to_destination < distance_matrix[current_node][destination_node]:
            score *= penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

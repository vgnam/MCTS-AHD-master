def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    # Dynamic weight factor based on progress (higher as more nodes are visited)
    progress = (len(unvisited_nodes) / (len(unvisited_nodes) + 1)) if unvisited_nodes else 0
    weight_factor = 0.3 + 0.7 * (1 - progress)  # Weight shifts from 0.3 to 1.0 as more nodes are visited

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        # Weighted score combining distance to current and distance to destination
        score = (1 - weight_factor) * distance_to_current + weight_factor * distance_to_dest

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + remaining_nodes)  # Dynamic weighting based on remaining nodes

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        combined_score = (1 - weight_factor) * current_distance + weight_factor * dest_distance

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

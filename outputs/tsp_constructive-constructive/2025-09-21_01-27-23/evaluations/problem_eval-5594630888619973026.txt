def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    if total_unvisited == 1:
        return unvisited_nodes[0]

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = (total_unvisited / (total_unvisited + 1)) ** 0.5
        combined_score = (1 - weight) * distance_to_current + weight * distance_to_destination

        if combined_score < min_distance:
            min_distance = combined_score
            next_node = node
        elif combined_score == min_distance:
            if distance_to_destination < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

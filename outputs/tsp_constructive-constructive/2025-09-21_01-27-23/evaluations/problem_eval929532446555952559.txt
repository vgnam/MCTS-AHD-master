def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        # Dynamic weight adjustment: prioritize distance to current node when few nodes remain
        weight_current = 0.8 if remaining_nodes > 2 else 0.6
        weight_dest = 1.0 - weight_current

        # Combined score: weighted sum of distances
        score = (weight_current * distance_to_current) + (weight_dest * distance_to_dest)

        if score < min_distance:
            min_distance = score
            next_node = node
        elif score == min_distance:
            # Tie-breaker: prefer node with lower distance to destination
            if distance_to_dest < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    avg_dest_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # If distances are equal, choose the node closer to the destination
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node
            # If still equal, choose the node farther from the average destination distance
            elif abs(distance_matrix[node][destination_node] - avg_dest_distance) > abs(distance_matrix[next_node][destination_node] - avg_dest_distance):
                next_node = node

    return next_node

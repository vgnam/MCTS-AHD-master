def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weights = {
        'current': len(unvisited_nodes) / (len(unvisited_nodes) + 1),
        'destination': 1 - len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    }

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        score = (weights['current'] * distance_matrix[current_node][node] +
                 weights['destination'] * distance_matrix[node][destination_node])

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

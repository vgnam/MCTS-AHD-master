def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # If distances are equal, choose the node with the lower total path cost (current + to destination)
            total_cost_new = distance + distance_matrix[node][destination_node]
            total_cost_current = distance + distance_matrix[next_node][destination_node]
            if total_cost_new < total_cost_current:
                next_node = node

    return next_node

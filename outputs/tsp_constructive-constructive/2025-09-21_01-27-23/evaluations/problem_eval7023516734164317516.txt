def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight_current=0.7, weight_dest=0.3):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_dest = distance_matrix[node][destination_node]
        score = weight_current * distance_current + weight_dest * distance_dest

        if score < best_score:
            best_score = score
            next_node = node
        elif score == best_score:
            if distance_dest < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # If distances are equal, choose the node closer to the destination with higher weight
            if 0.3 * distance_matrix[node][destination_node] + 0.7 * distance_matrix[current_node][node] < 0.3 * distance_matrix[next_node][destination_node] + 0.7 * distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

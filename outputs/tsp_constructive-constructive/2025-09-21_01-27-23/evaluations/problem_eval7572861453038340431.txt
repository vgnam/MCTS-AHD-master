def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        avg_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)
        total = current_to_node + avg_to_unvisited

        if total < min_total:
            min_total = total
            next_node = node
        elif total == min_total:
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance_to_dest = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_dest = distance_matrix[node][destination_node]
        if distance_to_dest < min_distance_to_dest:
            min_distance_to_dest = distance_to_dest
            next_node = node
        elif distance_to_dest == min_distance_to_dest:
            # If distances to destination are equal, choose the node closer to the current node
            if distance_matrix[current_node][node] < distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

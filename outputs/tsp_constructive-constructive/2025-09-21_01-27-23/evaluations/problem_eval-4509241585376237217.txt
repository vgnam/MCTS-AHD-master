def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.7):
    if not unvisited_nodes:
        return destination_node

    min_weighted_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        weighted_distance = weight * distance_to_current + (1 - weight) * distance_to_dest

        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node
        elif weighted_distance == min_weighted_distance:
            if distance_to_dest > distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # Among nodes with equal minimal distance, choose the one with the smallest ratio
            if distance_matrix[node][destination_node] == 0:
                continue  # Avoid division by zero if node is the destination
            current_ratio = distance / distance_matrix[node][destination_node]
            next_ratio = distance_matrix[current_node][next_node] / distance_matrix[next_node][destination_node]
            if current_ratio < next_ratio:
                next_node = node

    return next_node

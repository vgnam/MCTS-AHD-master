def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        # Weighted score: balance between current distance and distance to destination
        weight_current = 0.7 if remaining_nodes > 2 else 0.3
        weight_dest = 1 - weight_current

        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        score = (weight_current * current_dist) + (weight_dest * dest_dist)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

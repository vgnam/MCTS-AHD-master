def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_remaining = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)
        total_score = distance_to_current + avg_distance_to_remaining

        if total_score < min_score:
            min_score = total_score
            next_node = node
        elif total_score == min_score:
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

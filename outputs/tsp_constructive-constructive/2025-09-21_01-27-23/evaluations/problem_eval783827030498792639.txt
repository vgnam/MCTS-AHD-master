def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_avg_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)
        total_distance = distance_matrix[current_node][node] + avg_distance
        if total_distance < min_avg_distance:
            min_avg_distance = total_distance
            next_node = node

    return next_node

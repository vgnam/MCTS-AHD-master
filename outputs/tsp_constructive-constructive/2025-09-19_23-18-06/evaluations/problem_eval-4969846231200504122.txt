def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_count = len(unvisited_nodes)
    immediate_weight = max(0.3, 0.6 - 0.05 * (10 - remaining_count))
    future_weight = 1 - immediate_weight

    candidates = []
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            future_distances = [distance_matrix[node][n] for n in remaining_nodes if n != node]
            future_distances_sorted = sorted(future_distances)
            top_k = min(3, len(future_distances_sorted))
            avg_future_distance = sum(future_distances_sorted[:top_k]) / top_k
            total_score = immediate_weight * immediate_distance + future_weight * avg_future_distance
        else:
            total_score = immediate_distance
        candidates.append((node, total_score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

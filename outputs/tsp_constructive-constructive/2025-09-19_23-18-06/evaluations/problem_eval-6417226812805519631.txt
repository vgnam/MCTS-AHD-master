def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress_ratio = 1.0 - (remaining_nodes / (remaining_nodes + total_nodes))
    temperature = max(0.1, 1.0 - progress_ratio)

    candidates = []
    for node in unvisited_nodes:
        immediate_dist = distance_matrix[current_node][node]

        if remaining_nodes > 1:
            remaining_unvisited = unvisited_nodes - {node}

            # Calculate average future distance
            avg_future_dist = sum(distance_matrix[node][n] for n in remaining_unvisited) / len(remaining_unvisited)

            # Calculate locality (similar to No.2)
            locality = 1.0 / (avg_future_dist + 1e-6)

            # Calculate node centrality (similar to No.2 and No.3)
            node_centrality = 1.0 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-6)

            # Dynamic exploration term combining multiple factors
            exploration_term = temperature * locality * (1.0 + 0.5 * node_centrality)

            # Detour penalty based on progress (similar to No.2)
            detour_penalty = max(distance_matrix[node][n] for n in remaining_unvisited) * progress_ratio
        else:
            exploration_term = 0.0
            detour_penalty = 0.0

        # Adaptive weighting based on remaining path length (similar to No.3)
        remaining_path_factor = (remaining_nodes / total_nodes) ** 2
        weighted_score = (1.0 - temperature) * immediate_dist + exploration_term * remaining_path_factor + 0.3 * detour_penalty
        candidates.append((node, weighted_score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    candidates = []
    for node in unvisited_nodes:
        next_distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)

        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            locality = 1.0 / (avg_distance + 1e-6)

            # Calculate path potential: balance between immediate distance and future connectivity
            path_potential = (next_distance + 0.5 * (1.0 / (degree + 1e-6))) * locality

            # Dynamic weight adjustment based on remaining nodes
            weight_distance = 1.0
            weight_degree = 0.3 * (len(unvisited_nodes) / len(distance_matrix))
            weight_locality = 0.4 * (1.0 / (len(unvisited_nodes) + 1e-6))

            total_score = (weight_distance * next_distance +
                          weight_degree * (1.0 / (degree + 1e-6)) +
                          weight_locality * locality +
                          0.2 * path_potential)
        else:
            total_score = next_distance

        candidates.append((node, total_score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

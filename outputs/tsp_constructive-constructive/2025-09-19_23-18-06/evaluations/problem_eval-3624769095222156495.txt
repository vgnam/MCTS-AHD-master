def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    candidates = []
    remaining_count = len(unvisited_nodes)
    weight_factor = 1 / (1 + math.exp(-(remaining_count - 5)))  # Sigmoid function to adjust weights dynamically
    immediate_weight = 0.8 - 0.6 * weight_factor
    future_weight = 0.2 + 0.6 * weight_factor

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            future_distances = [distance_matrix[node][n] for n in remaining_nodes if n != node]
            future_distances_sorted = sorted(future_distances)
            avg_future_distance = sum(future_distances_sorted[:3]) / 3 if len(future_distances_sorted) >= 3 else sum(future_distances_sorted) / len(future_distances_sorted)
            penalty = 0.1 * max(0, avg_future_distance - 2 * immediate_distance)  # Penalize nodes with poor future connectivity
            total_score = immediate_weight * immediate_distance + future_weight * avg_future_distance + penalty
        else:
            total_score = immediate_distance
        candidates.append((node, total_score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    temperature = max(0.1, 1.0 - (remaining_nodes / total_nodes))

    candidates = []
    for node in unvisited_nodes:
        immediate_dist = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)

        if remaining_nodes > 1:
            remaining_unvisited = unvisited_nodes - {node}
            avg_future_dist = sum(distance_matrix[node][n] for n in remaining_unvisited) / len(remaining_unvisited)
            locality = 1.0 / (avg_future_dist + 1e-6)
            centrality = 1.0 / (sum(distance_matrix[node]) / (total_nodes - 1) + 1e-6)
            detour_penalty = max(distance_matrix[node][n] for n in remaining_unvisited) / (remaining_nodes ** 0.5)

            dynamic_weight = 1.0 / (1.0 + 1.5 * (remaining_nodes / total_nodes))
            weighted_score = (1.0 - temperature) * immediate_dist + temperature * (0.4 * degree * locality * centrality - 0.3 * detour_penalty)
        else:
            weighted_score = immediate_dist

        candidates.append((node, weighted_score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

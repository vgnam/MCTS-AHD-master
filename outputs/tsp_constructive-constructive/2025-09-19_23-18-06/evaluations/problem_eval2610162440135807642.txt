importance scores, where distance is weighted by a time-decaying factor, node importance is based on a combination of degree centrality and locality, and a stochastic element is added to explore diverse paths.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def score(node):
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            locality = 1.0 / (sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes) + 1e-6)
            time_decay = 1.0 / (1.0 + 0.1 * len(remaining_nodes))
            importance = 0.7 * degree + 0.3 * locality
            return (1.5 * time_decay * distance) + (0.5 * importance)
        else:
            return distance

    next_node = min(unvisited_nodes, key=score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    candidates = []
    remaining_count = len(unvisited_nodes)
    total_nodes = len(distance_matrix)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)

        if remaining_count > 1:
            remaining_nodes = unvisited_nodes - {node}
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            locality = 1.0 / (avg_distance + 1e-6)

            centrality = 1.0 / (sum(distance_matrix[node]) / (total_nodes - 1) + 1e-6)
            detour_penalty = max(distance_matrix[node][n] for n in remaining_nodes) / (remaining_count ** 0.5)

            connectivity = degree / (total_nodes - 1)
            dynamic_locality = locality * (remaining_count / total_nodes) ** 0.5

            score = (distance * 0.4 + 0.3 * dynamic_locality * connectivity * centrality +
                     0.2 * detour_penalty + 0.1 * avg_distance)
            score *= (1.0 + 0.1 * (1.0 - (remaining_count / total_nodes)))
        else:
            score = distance

        candidates.append((node, score))

    if remaining_count > 1:
        next_node, _ = min(candidates, key=lambda x: x[1])
    else:
        next_node = min(candidates, key=lambda x: x[1])[0]

    return next_node

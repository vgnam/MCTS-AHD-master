def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    progress_factor = (total_nodes - remaining_nodes) / total_nodes if total_nodes > 0 else 0.0
    adjusted_progress = progress_factor ** 2  # Non-linear adjustment for stronger later emphasis

    candidates = []
    for node in unvisited_nodes:
        next_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_unvisited = unvisited_nodes - {node}
            avg_distance = sum(distance_matrix[node][n] for n in remaining_unvisited) / len(remaining_unvisited)
            centrality = sum(distance_matrix[node][n] for n in range(total_nodes)) / (total_nodes - 1)
            total_distance = (1 - adjusted_progress) * next_distance + adjusted_progress * (next_distance + 0.4 * avg_distance + 0.2 * centrality)
        else:
            total_distance = next_distance
        candidates.append((node, total_distance))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

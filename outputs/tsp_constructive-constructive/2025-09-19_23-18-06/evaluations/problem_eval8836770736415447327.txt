def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    progress_ratio = 1.0 - (total_unvisited / (total_unvisited + len(distance_matrix)))

    candidates = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)

        if total_unvisited > 1:
            remaining_nodes = unvisited_nodes - {node}
            avg_distance = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
            locality = 1.0 / (avg_distance + 1e-6)
        else:
            avg_distance = 0.0
            locality = 1.0

        proximity_pressure = (1.0 - progress_ratio) ** 2
        dynamic_weight = 0.7 + 0.3 * progress_ratio

        score = (distance * (1.0 + 0.5 * (degree + 1e-6) * locality) +
                0.3 * avg_distance * dynamic_weight +
                0.2 * proximity_pressure * distance)

        candidates.append((node, score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

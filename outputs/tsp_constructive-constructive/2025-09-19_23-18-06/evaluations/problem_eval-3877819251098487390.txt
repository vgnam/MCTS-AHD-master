def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def node_degree(node):
        return sum(1 for d in distance_matrix[node] if d > 0)

    def node_locality(node, unvisited):
        if len(unvisited) <= 1:
            return 0.0
        remaining = unvisited - {node}
        avg_dist = sum(distance_matrix[node][n] for n in remaining) / len(remaining)
        return 1.0 / (avg_dist + 1e-6)

    def locality_scaling_factor(unvisited):
        if len(unvisited) <= 1:
            return 0.0
        avg_distances = []
        for node in unvisited:
            remaining = unvisited - {node}
            avg_dist = sum(distance_matrix[node][n] for n in remaining) / len(remaining)
            avg_distances.append(avg_dist)
        overall_avg = sum(avg_distances) / len(avg_distances)
        return 1.0 / (1.0 + overall_avg)

    exploration_factor = 0.5 ** (10 - len(unvisited_nodes))
    candidates = []

    for node in unvisited_nodes:
        immediate_dist = distance_matrix[current_node][node]
        degree = node_degree(node)
        locality = node_locality(node, unvisited_nodes)
        locality_scale = locality_scaling_factor(unvisited_nodes)
        weight_dist = 0.4 + 0.4 * exploration_factor
        weight_degree = 0.3 * (1 - exploration_factor)
        weight_locality = 0.3 * exploration_factor * locality_scale
        penalty_degree = 0.1 * (1 - exploration_factor)
        score = (weight_dist * immediate_dist +
                 weight_degree * degree * (1 - penalty_degree) +
                 weight_locality * locality)
        candidates.append((node, score))

    next_node, _ = min(candidates, key=lambda x: x[1])
    return next_node

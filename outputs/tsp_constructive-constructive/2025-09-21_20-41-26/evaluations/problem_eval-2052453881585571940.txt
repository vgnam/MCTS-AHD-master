def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes and len(unvisited_nodes) == 1:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        if node == destination_node:
            continue

        distance = distance_matrix[current_node][node]
        estimated_path_length = distance + distance_matrix[node][destination_node]

        # Weighted score combining immediate distance and estimated path length
        score = 0.6 * distance + 0.4 * estimated_path_length

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

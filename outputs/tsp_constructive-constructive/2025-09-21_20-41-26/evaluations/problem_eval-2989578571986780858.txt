import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes and len(unvisited_nodes) > 1:
        candidates = [destination_node]
        weights = [1.0]
    else:
        candidates = unvisited_nodes
        weights = [1.0 / (distance_matrix[current_node][node] + 1e-10) for node in candidates]

    total_weight = sum(weights)
    normalized_weights = [w / total_weight for w in weights]
    next_node = random.choices(candidates, weights=normalized_weights, k=1)[0]
    return next_node

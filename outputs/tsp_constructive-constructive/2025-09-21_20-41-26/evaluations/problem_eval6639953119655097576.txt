def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes and len(unvisited_nodes) == 1:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        neighbor_distance = min(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        total_score = distance_matrix[current_node][node] + 0.5 * neighbor_distance
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes and len(unvisited_nodes) > 1:
        return destination_node
    weights = {}
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 2:
            potential_future_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1)
            weights[node] = 0.7 * immediate_distance + 0.3 * potential_future_distance
        else:
            weights[node] = immediate_distance
    return next_node

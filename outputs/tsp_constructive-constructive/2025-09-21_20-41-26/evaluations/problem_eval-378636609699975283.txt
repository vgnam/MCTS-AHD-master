def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        remaining_nodes = len(unvisited_nodes)
        if remaining_nodes == 1:
            return destination_node
        elif remaining_nodes <= 3:
            probability = 0.7  # Higher probability to prioritize destination
        else:
            probability = 0.3   # Lower probability to prioritize nearest node
        if random.random() < probability:
            return destination_node

    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

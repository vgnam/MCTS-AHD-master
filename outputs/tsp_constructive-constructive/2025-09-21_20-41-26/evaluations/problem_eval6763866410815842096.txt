def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None
    if destination_node in unvisited_nodes:
        remaining_unvisited = len(unvisited_nodes)
        probability = 0.5 * (remaining_unvisited / (remaining_unvisited + 1))  # Decaying probability
        if random.random() < probability:
            return destination_node
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

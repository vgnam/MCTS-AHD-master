def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if len(unvisited_nodes) == 1:
        return unvisited_nodes[0]

    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic = distance_matrix[node][destination_node] if node != destination_node else 0
        score = 0.7 * distance + 0.3 * heuristic

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

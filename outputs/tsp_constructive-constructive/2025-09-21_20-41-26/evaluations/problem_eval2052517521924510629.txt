def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        remaining_nodes = len(unvisited_nodes)
        if remaining_nodes <= 2:
            return destination_node
        elif remaining_nodes <= 4:
            closest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
            if distance_matrix[current_node][destination_node] <= 1.2 * distance_matrix[current_node][closest_node]:
                return destination_node
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

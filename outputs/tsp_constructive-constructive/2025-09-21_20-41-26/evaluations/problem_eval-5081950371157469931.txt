def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes and len(unvisited_nodes) > 1:
        if len(unvisited_nodes) == 2:
            return destination_node
        else:
            prob = 0.7 if (len(unvisited_nodes) < 5) else 0.4
            if random.random() < prob:
                return destination_node

    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        candidate_distance = distance_matrix[current_node][node] + 0.3 * distance_matrix[node][destination_node]
        if candidate_distance < min_distance:
            min_distance = candidate_distance
            next_node = node

    return next_node

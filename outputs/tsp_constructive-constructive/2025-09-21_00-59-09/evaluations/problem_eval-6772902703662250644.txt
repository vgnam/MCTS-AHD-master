def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    # Compute mean distance to destination for unvisited nodes
    if unvisited_nodes:
        mean_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        mean_distance = 0

    # Compute centroid of unvisited nodes
    if unvisited_nodes:
        centroid_x = sum(distance_matrix[destination_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        centroid_x = 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        mean_penalty = abs(distance_to_dest - mean_distance)
        centroid_penalty = abs(distance_matrix[node][destination_node] - centroid_x)

        # Weighted score: prioritize proximity to current node, penalize deviation from mean distance, moderate centroid alignment
        score = distance_to_current + 0.2 * mean_penalty + 0.5 * centroid_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

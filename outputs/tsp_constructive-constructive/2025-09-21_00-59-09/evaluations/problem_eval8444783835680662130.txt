def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        max_future_dist = max(distance_matrix[node][n] for n in unvisited_nodes if n != node) if unvisited_nodes else 0

        score = distance_to_current + 0.7 * distance_to_dest + 0.2 * max_future_dist

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

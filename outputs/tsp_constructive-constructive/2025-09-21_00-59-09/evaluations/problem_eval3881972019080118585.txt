def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    # Compute centroid of unvisited nodes based on distances to current node
    if unvisited_nodes:
        centroid_x = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        centroid_x = 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        centroid_penalty = abs(distance_matrix[node][current_node] - centroid_x)

        # Weighted score: prioritize proximity to current node, then destination, then penalize deviation from centroid
        score = distance_to_current + 0.4 * distance_to_dest + 0.5 * centroid_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

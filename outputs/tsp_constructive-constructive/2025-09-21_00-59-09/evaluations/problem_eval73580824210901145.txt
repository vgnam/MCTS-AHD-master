def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    # Compute centroid of unvisited nodes
    if unvisited_nodes:
        centroid_x = sum(distance_matrix[destination_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        centroid_x = 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        centroid_penalty = (distance_matrix[node][destination_node] - centroid_x) ** 2

        # Dynamic weight adjustment: prioritize destination proximity when far from centroid
        weight_dest = 0.7 if distance_to_dest > centroid_x else 0.3

        # Score combines distance to current, dynamic destination weight, and squared centroid penalty
        score = distance_to_current + weight_dest * distance_to_dest + 0.2 * centroid_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    # Compute centroid of unvisited nodes
    if unvisited_nodes:
        centroid_x = sum(distance_matrix[destination_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        centroid_x = 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_unvisited_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)
        centroid_penalty = abs(distance_matrix[node][destination_node] - centroid_x)

        # Weighted score: prioritize proximity to current node and average unvisited distance, penalize deviation from centroid
        score = distance_to_current + 0.7 * avg_unvisited_distance + 0.2 * centroid_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

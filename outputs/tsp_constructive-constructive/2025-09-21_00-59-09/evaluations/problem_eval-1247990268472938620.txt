def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    # Compute centroid of unvisited nodes
    if unvisited_nodes:
        centroid_x = sum(distance_matrix[destination_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    else:
        centroid_x = 0

    # Dynamic weight adjustment based on remaining nodes
    remaining_nodes = len(unvisited_nodes)
    weight_current = 1.0 if remaining_nodes > 2 else 0.7
    weight_dest = 0.5 if remaining_nodes > 2 else 0.8
    weight_centroid = 0.3 if remaining_nodes > 2 else 0.5

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        centroid_penalty = abs(distance_matrix[node][destination_node] - centroid_x)

        # Normalize centroid penalty to avoid dominance
        max_centroid_penalty = max(abs(distance_matrix[n][destination_node] - centroid_x) for n in unvisited_nodes) if unvisited_nodes else 1
        normalized_penalty = centroid_penalty / max_centroid_penalty if max_centroid_penalty != 0 else 0

        # Weighted score with dynamic weights
        score = (weight_current * distance_to_current +
                 weight_dest * distance_to_dest +
                 weight_centroid * normalized_penalty)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

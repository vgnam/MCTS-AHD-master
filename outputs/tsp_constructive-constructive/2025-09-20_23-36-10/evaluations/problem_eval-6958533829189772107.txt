def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, visit_counts):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        ratio = destination_distance / distance
        revisit_penalty = visit_counts.get(node, 0) * 0.1
        score = ratio - revisit_penalty
        if score > max_score:
            max_score = score
            next_node = node
    return next_node

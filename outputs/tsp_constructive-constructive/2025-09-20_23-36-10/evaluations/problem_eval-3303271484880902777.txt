def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    max_distance_from_current = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_dest = distance_matrix[node][destination_node]
        distance_from_current = distance_matrix[current_node][node]
        if distance_from_current == 0:
            continue
        if distance_to_dest < min_distance or (distance_to_dest == min_distance and distance_from_current > max_distance_from_current):
            min_distance = distance_to_dest
            max_distance_from_current = distance_from_current
            next_node = node
    return next_node

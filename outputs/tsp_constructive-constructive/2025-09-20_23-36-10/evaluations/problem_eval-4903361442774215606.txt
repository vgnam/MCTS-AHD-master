def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    current_destination_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        progress = (current_destination_distance - destination_distance) / current_destination_distance
        efficiency = destination_distance / distance
        score = 0.7 * progress + 0.3 * efficiency

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

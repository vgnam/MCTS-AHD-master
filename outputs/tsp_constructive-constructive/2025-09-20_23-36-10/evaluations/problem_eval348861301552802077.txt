def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    total_distance = sum(sum(row) for row in distance_matrix)
    avg_distance = total_distance / (len(distance_matrix) * len(distance_matrix))

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        proximity_weight = 1 - (destination_distance / avg_distance)
        score = (destination_distance / distance) * (1 + proximity_weight)
        if score > max_score:
            max_score = score
            next_node = node

    return next_node

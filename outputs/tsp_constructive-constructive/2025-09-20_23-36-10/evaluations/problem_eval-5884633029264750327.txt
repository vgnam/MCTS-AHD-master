def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    current_dest_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        ratio = dest_distance / distance
        proximity_bonus = (current_dest_distance - dest_distance) / (distance + 1e-10)
        score = ratio + 0.5 * proximity_bonus

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

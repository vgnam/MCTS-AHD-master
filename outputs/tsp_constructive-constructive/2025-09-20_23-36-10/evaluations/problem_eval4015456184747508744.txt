def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    alpha = 0.7  # Weight for the destination-distance ratio
    beta = 0.3   # Weight for the inverse of current distance

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        ratio = destination_distance / distance
        inverse_distance = 1 / distance
        score = alpha * ratio + beta * inverse_distance
        if score > max_score:
            max_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    visited_penalty = 0.1  # Penalty for revisiting nodes to encourage exploration

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]

        if distance == 0:
            continue

        # Dynamic weighting: balance ratio with a penalty for revisiting nodes
        score = (destination_distance / distance) * (1 - visited_penalty)

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

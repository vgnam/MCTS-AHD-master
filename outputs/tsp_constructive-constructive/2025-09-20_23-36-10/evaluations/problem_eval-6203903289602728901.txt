def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    max_weighted_ratio = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        weight = 1.0 / (1.0 + remaining_nodes)  # Dynamic weight based on remaining nodes
        weighted_ratio = (destination_distance ** 2) / (distance * weight)
        if weighted_ratio > max_weighted_ratio:
            max_weighted_ratio = weighted_ratio
            next_node = node
    return next_node

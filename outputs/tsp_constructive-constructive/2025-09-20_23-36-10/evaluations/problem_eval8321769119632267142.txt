def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    current_to_dest = distance_matrix[current_node][destination_node]
    weight = 1.0 / (1.0 + current_to_dest)  # Higher weight when closer to destination

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        score = (dest_distance / distance) * (1 + weight * (1 - dest_distance / current_to_dest))
        if score > max_score:
            max_score = score
            next_node = node
    return next_node

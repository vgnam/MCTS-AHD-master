def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Original ratio term
        ratio = destination_distance / distance

        # Novel detour penalty term: penalize nodes that are far from the destination relative to their distance from current
        detour_penalty = (distance_matrix[node][destination_node] / distance) ** 2

        # Combined score: prioritize nodes that progress toward destination while minimizing detours
        score = ratio - 0.5 * detour_penalty

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

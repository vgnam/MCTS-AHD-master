def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Original ratio term
        ratio = destination_distance / distance

        # Dynamic penalty term based on visit frequency (simulated here)
        # In practice, this would require tracking visit counts
        penalty = 0.1 * (len(unvisited_nodes) - len(distance_matrix))  # Simplified approximation

        # Combined score with penalty
        score = ratio - penalty

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Calculate centrality score (average distance to other unvisited nodes)
        centrality = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1)

        # Combine ratio and centrality with weighted sum
        ratio = destination_distance / distance
        score = 0.6 * ratio + 0.4 * (1 / (centrality + 1e-6))  # Avoid division by zero

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

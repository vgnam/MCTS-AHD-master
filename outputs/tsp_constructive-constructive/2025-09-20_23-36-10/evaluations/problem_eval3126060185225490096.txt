def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    adjustment_factor = 1.0 / (1.0 + remaining_nodes)  # Decreases as more nodes remain

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue
        # Weighted combination of proximity to destination and distance from current node
        score = (destination_distance ** 2) / (distance * (1.0 + adjustment_factor))
        if score > max_score:
            max_score = score
            next_node = node
    return next_node

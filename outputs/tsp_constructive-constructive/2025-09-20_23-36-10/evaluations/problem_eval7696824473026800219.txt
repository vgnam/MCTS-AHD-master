def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Original ratio
        ratio = destination_distance / distance

        # Lookahead heuristic: average distance to remaining unvisited nodes
        lookahead = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

        # Weighted score: balance between progress and lookahead
        score = 0.7 * ratio + 0.3 * (1 / (1 + lookahead))

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

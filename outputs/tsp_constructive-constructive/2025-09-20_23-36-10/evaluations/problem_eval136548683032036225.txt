def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Calculate centrality: average distance to all other nodes
        centrality = sum(distance_matrix[node]) / (len(distance_matrix[node]) - 1)

        # Combine ratio and centrality (weighted sum)
        ratio = destination_distance / distance
        score = 0.7 * ratio + 0.3 * (1 / centrality)  # Higher centrality gets lower penalty

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        if distance == 0:
            continue

        # Calculate hub potential: average distance to all other unvisited nodes
        hub_potential = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

        # Combine ratio and hub potential with adaptive weights
        ratio_weight = 0.7 if len(unvisited_nodes) > 3 else 0.9
        hub_weight = 1 - ratio_weight
        score = (ratio_weight * (destination_distance / distance)) + (hub_weight * hub_potential)

        if score > max_score:
            max_score = score
            next_node = node
    return next_node

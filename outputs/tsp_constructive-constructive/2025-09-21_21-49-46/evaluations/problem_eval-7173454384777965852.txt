def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_current == 0:
            angular_progress = 1.0
        else:
            angular_progress = dist_dest / dist_current

        penalty_factor = 0.1 if len(unvisited_nodes) > 3 else 0.05
        bonus_factor = 0.3 if len(unvisited_nodes) > 3 else 0.2

        score = angular_progress - dist_current * (1 + penalty_factor) + bonus_factor * dist_dest

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_distance_to_dest = distance_matrix[current_node][destination_node]
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        if remaining_distance_to_dest > 0:
            weight = 1 / (1 + math.exp(-(distance_to_dest - remaining_distance_to_dest) / remaining_distance_to_dest))
        else:
            weight = 0.5

        score = (1 - weight) * distance_to_current + weight * distance_to_dest

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

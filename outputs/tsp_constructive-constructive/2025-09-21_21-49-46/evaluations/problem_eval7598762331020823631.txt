def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_current == 0:
            angular_progress = 1.0
        else:
            angular_progress = dist_dest / dist_current

        if len(unvisited_nodes) > 1:
            avg_dist = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1)
        else:
            avg_dist = 0

        score = dist_current - angular_progress + 0.5 * avg_dist

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        dest_dist_current = distance_matrix[current_node][destination_node]
        dest_dist_dest = distance_matrix[destination_node][destination_node]

        if dest_dist_current == 0:
            dest_angular_progress = 1.0
        else:
            dest_angular_progress = dest_dist_dest / dest_dist_current

        if len(unvisited_nodes) > 1:
            dest_avg_dist = sum(distance_matrix[destination_node][other] for other in unvisited_nodes if other != destination_node) / (len(unvisited_nodes) - 1)
        else:
            dest_avg_dist = 0

        dest_score = dest_dist_current - dest_angular_progress + 0.5 * dest_avg_dist

        if dest_score < min_score:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance_to_current = float('inf')
    min_distance_to_dest = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        combined_score = 0.5 * distance_to_current + 1.5 * distance_to_dest

        if combined_score < min_distance_to_current + 1.5 * min_distance_to_dest:
            min_distance_to_current = distance_to_current
            min_distance_to_dest = distance_to_dest
            next_node = node

    return next_node

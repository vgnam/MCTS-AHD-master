def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    dest_weight = 1.0 + (1.0 / (remaining_nodes + 1))
    ratio_weight = (remaining_nodes + 1) / (remaining_nodes + 2)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        if distance_to_dest == 0:
            combined_score = distance_to_current
        else:
            combined_score = (distance_to_current + dest_weight * distance_to_dest) * (1 + ratio_weight)

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

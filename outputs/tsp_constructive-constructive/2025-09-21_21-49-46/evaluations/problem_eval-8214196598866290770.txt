def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_distance_to_dest = -float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        if distance_to_dest > max_distance_to_dest or (distance_to_dest == max_distance_to_dest and distance_to_node < distance_matrix[current_node][next_node]):
            max_distance_to_dest = distance_to_dest
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance < distance_matrix[current_node][next_node]:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    dest_weight = 1.0 + (1.0 / (remaining_nodes + 1))  # Dynamic weight based on remaining nodes

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_current == 0:
            angular_progress = 1.0
        else:
            angular_progress = dist_dest / dist_current

        weighted_angular = angular_progress * 0.5
        combined_score = dist_current + dest_weight * (dist_dest - weighted_angular)

        # Penalize revisits
        if node in [current_node, destination_node]:
            combined_score += 10.0

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

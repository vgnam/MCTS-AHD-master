def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_destination_distance = float('inf')
    candidates = []

    for node in unvisited_nodes:
        distance_to_destination = distance_matrix[node][destination_node]
        if distance_to_destination < min_destination_distance:
            min_destination_distance = distance_to_destination
            candidates = [node]
        elif distance_to_destination == min_destination_distance:
            candidates.append(node)

    if len(candidates) == 1:
        return candidates[0]
    else:
        min_current_distance = float('inf')
        next_node = None
        for node in candidates:
            distance = distance_matrix[current_node][node]
            if distance < min_current_distance:
                min_current_distance = distance
                next_node = node
        return next_node

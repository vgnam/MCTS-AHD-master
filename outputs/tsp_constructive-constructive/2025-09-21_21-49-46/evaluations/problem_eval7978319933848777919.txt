def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')
    total_nodes = len(unvisited_nodes) + 1
    visited_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes

    # Track selection frequency for dynamic penalty
    selection_frequency = {node: 0 for node in unvisited_nodes}
    for node in unvisited_nodes:
        if node in selection_frequency:
            selection_frequency[node] += 1

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_current == 0:
            angular_progress = 1.0
        else:
            angular_progress = dist_dest / dist_current

        # Non-linear weight adjustment
        weight = (1 + visited_ratio) ** 2
        weighted_angular = angular_progress * weight

        # Dynamic penalty based on selection frequency
        penalty = 1 - (0.1 * selection_frequency.get(node, 0))
        score = (weighted_angular * penalty) - dist_current

        if node in [current_node, destination_node]:
            score *= 0.5

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

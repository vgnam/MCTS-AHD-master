def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]
        max_dist = max(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        score = 0.5 * dist_current + 0.3 * dist_dest - 0.2 * max_dist

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_score = 0.5 * distance_matrix[current_node][destination_node] + 0.3 * 0 - 0.2 * 0
        if destination_score < min_score:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weights = {'distance': 0.7, 'detour_penalty': 0.3}
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        detour_penalty = distance_matrix[node][destination_node] if node != destination_node else 0
        score = weights['distance'] * distance + weights['detour_penalty'] * detour_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

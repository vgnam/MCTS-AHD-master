def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        weight_current = 0.7 if remaining_nodes > 1 else 0.5
        weight_dest = 0.3 if remaining_nodes > 1 else 0.5

        combined_score = (weight_current * distance_to_current) + (weight_dest * distance_to_dest)

        if node in unvisited_nodes and distance_matrix[current_node][node] > 0:
            combined_score += 0.1 * distance_to_current

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    weight_current = 0.7 - 0.3 * (remaining_nodes / len(distance_matrix))
    weight_dest = 1.0 - weight_current

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]
        score = weight_current * dist_current + weight_dest * dist_dest

        if node == current_node:
            score += 1000

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_score = weight_current * distance_matrix[current_node][destination_node] + weight_dest * 0
        if destination_score < min_score:
            next_node = destination_node

    return next_node

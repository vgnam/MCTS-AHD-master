import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')

    for node in unvisited_nodes:
        dist_current = distance_matrix[current_node][node]
        dist_dest = distance_matrix[node][destination_node]

        if dist_current == 0:
            angular_progress = 1.0
        else:
            angular_progress = dist_dest / dist_current

        visibility = angular_progress / (1 + dist_current)
        weight = 1 / (1 + math.exp(-(angular_progress - 1)))
        power_weight = math.pow(visibility, 2)
        score = (angular_progress * power_weight * weight) - (dist_current * (1 - weight))

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

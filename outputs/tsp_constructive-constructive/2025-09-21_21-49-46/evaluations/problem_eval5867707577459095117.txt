def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_distance = distance_matrix[current_node][destination_node]
    weight_proximity = max(0.5, 0.9 - 0.01 * remaining_distance)
    weight_destination = 1 - weight_proximity

    min_combined_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        penalty = 0.1 * (1 if node == current_node else 0)
        combined_score = weight_proximity * distance_to_current + weight_destination * distance_to_dest + penalty

        if combined_score < min_combined_score:
            min_combined_score = combined_combined_score
            next_node = node

    return next_node

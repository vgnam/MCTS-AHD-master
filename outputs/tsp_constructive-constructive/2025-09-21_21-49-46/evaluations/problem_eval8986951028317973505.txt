def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    dest_weight = 1.0 + (1.0 / (remaining_nodes + 1))
    exploration_factor = 1.0 / (remaining_nodes + 1)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        normalized_distance = (distance_to_current + distance_to_dest) / 2
        combined_score = distance_to_current + (dest_weight * distance_to_dest) + (exploration_factor * normalized_distance)

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes)
    dynamic_factor = (1 - (remaining_nodes / (remaining_nodes + 1)))
    next_node = max(unvisited_nodes, key=lambda node: (
        (distance_matrix[node][destination_node] / distance_matrix[current_node][node]) * dynamic_factor +
        (1 / distance_matrix[current_node][node]) * (1 + dynamic_factor)
    ))
    return next_node

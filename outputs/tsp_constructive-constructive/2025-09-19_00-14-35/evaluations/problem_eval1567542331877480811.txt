def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes

    def evaluate_node(node):
        immediate_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            remaining_nodes_list = [n for n in unvisited_nodes if n != node]
            sum_inv_dists = sum(1.0 / (1.0 + distance_matrix[node][n]) for n in remaining_nodes_list)
            harmonic_connectivity = len(remaining_nodes_list) / sum_inv_dists if sum_inv_dists != 0 else 0
            potential_gain = (distance_matrix[current_node][destination_node] - immediate_dist - dest_dist) / (1 + immediate_dist)
            normalized_gain = potential_gain / (1 + immediate_dist + dest_dist)
        else:
            harmonic_connectivity = 0
            normalized_gain = 0

        weight_local = 0.4 * (1 - exploration_factor**2)
        weight_connectivity = 0.3 + 0.2 * exploration_factor
        weight_dest = -0.2 * exploration_factor**1.5
        weight_gain = 0.1 * (1 - exploration_factor)

        score = (weight_local * immediate_dist) + (weight_connectivity * harmonic_connectivity) + \
                (weight_dest * dest_dist) + (weight_gain * normalized_gain)
        return score

    next_node = min(unvisited_nodes, key=lambda node: evaluate_node(node))
    return next_node

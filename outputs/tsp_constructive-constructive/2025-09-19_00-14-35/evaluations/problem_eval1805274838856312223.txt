def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes

    scores = []
    for node in unvisited_nodes:
        local_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            distances_to_unvisited = [distance_matrix[node][n] for n in unvisited_nodes if n != node]
            avg_remaining_dist = sum(distances_to_unvisited) / (remaining_nodes - 1)
            variance_remaining_dist = sum((d - avg_remaining_dist) ** 2 for d in distances_to_unvisited) / (remaining_nodes - 1)
            connectivity_score = 1.0 / (1.0 + avg_remaining_dist) * (1.0 + (variance_remaining_dist ** 0.5))

            node_degree = sum(1 for n in range(total_nodes) if distance_matrix[node][n] > 0 and n != node)
            centrality_score = node_degree / (total_nodes - 1)

            dynamic_threshold = 0.7 * exploration_factor + 0.3 * (1 - (sum(distances_to_unvisited) / (remaining_nodes - 1)) / total_nodes)
        else:
            connectivity_score = 0
            centrality_score = 0
            dynamic_threshold = 0

        weight_local = 0.4 + 0.2 * exploration_factor - 0.1 * (sum(distance_matrix[current_node][n] for n in unvisited_nodes) / (remaining_nodes + 1e-6) / total_nodes)
        weight_connectivity = 0.3 + 0.1 * exploration_factor + 0.1 * (sum(distance_matrix[current_node][n] for n in unvisited_nodes) / (remaining_nodes + 1e-6) / total_nodes)
        weight_dest = -0.3 + 0.2 * (1 - exploration_factor)
        weight_centrality = 0.2 * dynamic_threshold

        score = (weight_local * local_distance) + (weight_connectivity * connectivity_score) + (weight_dest * dest_distance) + (weight_centrality * centrality_score)
        scores.append((node, score))

    next_node = min(scores, key=lambda x: x[1])[0]
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] * distance_matrix[node][destination_node])
    if destination_node in unvisited_nodes and distance_matrix[current_node][destination_node] < 0.8 * distance_matrix[current_node][next_node]:
        return destination_node
    return next_node

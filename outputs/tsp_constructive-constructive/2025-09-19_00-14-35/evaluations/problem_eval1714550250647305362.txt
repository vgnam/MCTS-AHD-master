def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        local_distance = distance_matrix[current_node][node]
        global_distance = distance_matrix[node][destination_node]
        remaining_nodes = len(unvisited_nodes)
        weight_factor = 0.5 ** remaining_nodes if remaining_nodes else 0
        return local_distance + (global_distance * weight_factor)
    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

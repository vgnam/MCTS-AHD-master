def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=0.5):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    min_distance = min(distances)
    max_distance = max(distances)

    if max_distance == min_distance:
        return random.choice(unvisited_nodes)

    weights = [(max_distance - d) / (max_distance - min_distance) for d in distances]
    weights = [w ** (1 / temperature) for w in weights]
    weights = [w / sum(weights) for w in weights]

    next_node = random.choices(unvisited_nodes, weights=weights, k=1)[0]
    return next_node

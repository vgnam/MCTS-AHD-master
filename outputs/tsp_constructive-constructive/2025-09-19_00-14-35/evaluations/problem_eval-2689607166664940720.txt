def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes

    scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            connectivity_score = sum(1.0 / distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1)
            flow_disruption = max(distance_matrix[node][n] for n in unvisited_nodes if n != node) - current_dist
            path_quality = (current_dist + dest_dist) / connectivity_score if connectivity_score > 0 else float('inf')
        else:
            connectivity_score = 0
            flow_disruption = 0
            path_quality = current_dist + dest_dist

        weight_local = 0.4 + 0.3 * (1 - exploration_factor)
        weight_connectivity = 0.2 + 0.3 * exploration_factor
        weight_flow = -0.2 * flow_disruption if flow_disruption > 0 else 0
        weight_quality = 0.1 * (1 / path_quality)

        score = (weight_local * current_dist) + (weight_connectivity * connectivity_score) + (weight_flow * flow_disruption) + (weight_quality * path_quality)

        scores.append((node, score))

    next_node = min(scores, key=lambda x: x[1])[0]
    return next_node

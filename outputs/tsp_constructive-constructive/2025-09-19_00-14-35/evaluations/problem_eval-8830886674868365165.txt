def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes

    # Stage 1: Dynamic neighborhood search
    neighborhood_size = max(1, int(3 + 2 * (1 - exploration_factor)))
    candidates = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])[:neighborhood_size]

    # Stage 2: Weighted ratio evaluation with adaptive weights
    def evaluate_node(node):
        ratio = distance_matrix[node][destination_node] / distance_matrix[current_node][node]
        penalty = distance_matrix[node][destination_node]

        # Adaptive weights
        weight_ratio = 0.4 + 0.2 * (1 - exploration_factor)
        weight_penalty = -0.3 - 0.1 * (1 - exploration_factor)

        return weight_ratio * ratio + weight_penalty * penalty

    next_node = max(candidates, key=lambda node: evaluate_node(node))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def evaluate_node(node):
        progress_ratio = distance_matrix[node][destination_node] / distance_matrix[current_node][node]
        nearest_neighbor_dist = min(distance_matrix[node][n] for n in unvisited_nodes if n != node) if len(unvisited_nodes) > 1 else 0
        return -progress_ratio + 0.5 * nearest_neighbor_dist
    next_node = min(unvisited_nodes, key=lambda node: evaluate_node(node))
    return next_node

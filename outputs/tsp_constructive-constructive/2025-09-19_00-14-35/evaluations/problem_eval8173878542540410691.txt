def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_unvisited = len(unvisited_nodes)
    next_node = max(unvisited_nodes, key=lambda node: (
        (0.5 * (distance_matrix[node][destination_node] / distance_matrix[current_node][node])) -
        (0.2 * distance_matrix[node][destination_node] / max(distance_matrix[node][destination_node] for node in unvisited_nodes)) -
        (0.3 * (total_unvisited / (total_unvisited + 1)))
    ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    candidate_scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        avg_dist = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)
        ratio = current_dist / distance_matrix[node][destination_node] if distance_matrix[node][destination_node] != 0 else 0
        score = 0.6 * current_dist + 0.3 * avg_dist + 0.1 * ratio + 0.01 * (hash(node) % 10)
        candidate_scores.append((node, score))
    next_node = min(candidate_scores, key=lambda x: x[1])[0]
    return next_node

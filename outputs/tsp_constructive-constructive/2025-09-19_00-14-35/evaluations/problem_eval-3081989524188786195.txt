def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes

    # Stage 1: Filter candidates more aggressively
    candidates = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    candidate_size = max(1, min(2, len(candidates)))
    top_candidates = candidates[:candidate_size]

    # Stage 2: Evaluate with modified weights
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        # Connectivity score (max distance to remaining nodes)
        if remaining_nodes > 1:
            max_remaining_dist = max(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        else:
            max_remaining_dist = 0

        # Dynamic weights
        weight_dist = 0.3 + 0.1 * (1 - exploration_factor)
        weight_connectivity = 0.5 - 0.1 * exploration_factor
        weight_penalty = 0.1 * (1 - exploration_factor)
        weight = 1.0 / (remaining_nodes + 1)

        return (weight_dist * current_dist +
                weight_connectivity * weight * max_remaining_dist -
                weight_penalty * dest_dist)

    next_node = min(top_candidates, key=lambda node: evaluate_node(node))
    return next_node

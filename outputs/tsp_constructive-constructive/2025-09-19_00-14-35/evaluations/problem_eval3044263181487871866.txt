def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    current_to_dest = distance_matrix[current_node][destination_node]
    avg_dist = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    next_node = min(unvisited_nodes, key=lambda node: (
        (distance_matrix[current_node][node] * distance_matrix[node][destination_node]) /
        (1 + current_to_dest * (1 + (distance_matrix[current_node][node] / (avg_dist + 1e-9))))
    ))
    return next_node

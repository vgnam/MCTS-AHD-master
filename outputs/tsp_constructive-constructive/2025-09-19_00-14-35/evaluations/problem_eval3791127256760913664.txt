def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_distance = distance_matrix[current_node][destination_node]
    weight = min(1.0, 0.5 + (remaining_distance / max(distance_matrix[node][destination_node] for node in unvisited_nodes)))
    next_node = max(unvisited_nodes, key=lambda node: (weight * distance_matrix[node][destination_node] / distance_matrix[current_node][node]) + ((1 - weight) * distance_matrix[node][destination_node] / distance_matrix[current_node][node]))
    return next_node

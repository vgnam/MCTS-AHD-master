def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, visited_history=None, penalty_factor=10.0):
    if not unvisited_nodes:
        return destination_node
    if visited_history is None:
        visited_history = set()
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] * (1 + distance_matrix[node][destination_node]) + (node in visited_history) * penalty_factor)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = max(unvisited_nodes, key=lambda node: (0.8 * (1 / (distance_matrix[current_node][node] + 1e-6))) + (0.2 * (1 / (sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) + 1e-6))))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_distance = distance_matrix[current_node][destination_node]
    total_nodes = len(unvisited_nodes) + 1
    weight = 0.5 + (0.5 * (remaining_distance / sum(distance_matrix[current_node][node] for node in unvisited_nodes)))

    next_node = max(unvisited_nodes, key=lambda node: (
        weight * (distance_matrix[node][destination_node] / distance_matrix[current_node][node]) +
        (1 - weight) * (distance_matrix[current_node][node] / distance_matrix[node][destination_node])
    ))
    return next_node

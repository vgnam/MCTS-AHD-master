def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = 1.0 - (remaining_nodes / total_nodes)

    scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            connectivity_score = sum(1.0 for n in unvisited_nodes if distance_matrix[node][n] < current_dist) / (remaining_nodes - 1)
        else:
            connectivity_score = 0

        weight_local = 0.4
        weight_connectivity = 0.6 * exploration_factor
        penalty_factor = 1.0 - (dest_dist / (sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) + 1e-6))
        score = (weight_local * current_dist) + (weight_connectivity * connectivity_score * penalty_factor)

        scores.append((node, score))

    next_node = min(scores, key=lambda x: x[1])[0]
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    temperature = 1.0 - (1.0 / (1.0 + len(unvisited_nodes)))  # Increases as more nodes are visited
    candidate_scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        score = (dest_dist * 0.7) + (current_dist * 0.3) + (temperature * (hash(node) % 5))
        candidate_scores.append((node, score))
    next_node = min(candidate_scores, key=lambda x: x[1])[0]
    return next_node

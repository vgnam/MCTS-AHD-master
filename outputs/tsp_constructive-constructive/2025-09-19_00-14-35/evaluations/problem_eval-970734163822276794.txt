def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))

    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    closest_distance = distance_matrix[current_node][next_node]
    threshold_factor = 0.6 + (0.2 * (len(unvisited_nodes) / len(distance_matrix)))
    threshold = threshold_factor * closest_distance

    if destination_node in unvisited_nodes and distance_matrix[current_node][destination_node] <= threshold:
        return destination_node
    return next_node

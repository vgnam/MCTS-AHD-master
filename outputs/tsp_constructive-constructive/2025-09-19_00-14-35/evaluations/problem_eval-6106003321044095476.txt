def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = (remaining_nodes / total_nodes) ** 2

    scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            min_remaining_dist = min(distance_matrix[node][n] for n in unvisited_nodes if n != node)
            max_remaining_dist = max(distance_matrix[node][n] for n in unvisited_nodes if n != node)
            connectivity_score = (min_remaining_dist + max_remaining_dist) / (2 * max_remaining_dist)
            flow_disruption = (max_remaining_dist - current_dist) / max_remaining_dist if max_remaining_dist > 0 else 0
        else:
            connectivity_score = 0
            flow_disruption = 0

        weight_local = 0.5 * (1 - exploration_factor)
        weight_connectivity = 0.5 * exploration_factor
        weight_dest = -0.3 * (1 - exploration_factor)
        weight_flow = -0.2 * flow_disruption

        score = (weight_local * current_dist) + (weight_connectivity * connectivity_score) + (weight_dest * dest_dist) + (weight_flow * flow_disruption)

        scores.append((node, score))

    if exploration_factor > 0.7:
        next_node = max(scores, key=lambda x: x[1])[0]
    else:
        next_node = min(scores, key=lambda x: x[1])[0]

    return next_node

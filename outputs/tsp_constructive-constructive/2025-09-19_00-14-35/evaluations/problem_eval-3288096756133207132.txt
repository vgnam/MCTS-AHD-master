def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    weight = 1.0 / (len(unvisited_nodes) + 1)
    next_node = min(unvisited_nodes, key=lambda node: weight * (distance_matrix[current_node][node] / distance_matrix[node][destination_node]) + (1 - weight) * distance_matrix[current_node][node])
    return next_node

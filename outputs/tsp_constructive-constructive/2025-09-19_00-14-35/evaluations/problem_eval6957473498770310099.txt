def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    proximity_weight = 0.7 if total_nodes <= 3 else 0.5
    destination_weight = 1.0 - proximity_weight
    next_node = min(unvisited_nodes, key=lambda node: proximity_weight * distance_matrix[current_node][node] + destination_weight * distance_matrix[node][destination_node])
    return next_node

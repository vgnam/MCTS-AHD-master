import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5, beta=0.5, epsilon=0.1):
    if not unvisited_nodes:
        return destination_node
    next_node = min(unvisited_nodes, key=lambda node: (
        alpha * distance_matrix[current_node][node] +
        beta * distance_matrix[node][destination_node] +
        epsilon * random.random()
    ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_weight = remaining_nodes / total_nodes

    next_node = min(unvisited_nodes, key=lambda node: (
        exploration_weight * (distance_matrix[current_node][node] / distance_matrix[node][destination_node]) +
        (1 - exploration_weight) * (distance_matrix[current_node][node] / sum(distance_matrix[node]))
    ))

    return next_node

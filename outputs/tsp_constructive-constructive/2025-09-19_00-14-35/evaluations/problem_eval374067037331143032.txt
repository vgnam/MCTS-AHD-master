def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        local_distance = distance_matrix[current_node][node]
        global_distance = distance_matrix[node][destination_node]
        remaining_nodes = len(unvisited_nodes)
        weight_factor = 1 - (1 / (remaining_nodes + 1)) if remaining_nodes else 1
        return (local_distance * (1 - weight_factor)) + (global_distance * weight_factor)
    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

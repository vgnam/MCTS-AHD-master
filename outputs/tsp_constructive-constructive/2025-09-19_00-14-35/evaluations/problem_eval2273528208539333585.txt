def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node] * 0.9
        other_distances = [distance_matrix[current_node][node] for node in unvisited_nodes if node != destination_node]
        if not other_distances or destination_distance <= min(other_distances):
            return destination_node
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    return next_node

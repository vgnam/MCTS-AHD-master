def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress_factor = (total_nodes - remaining_nodes) / total_nodes
    local_weight = 0.7 - 0.4 * progress_factor
    ratio_weight = 0.3 + 0.4 * progress_factor
    next_node = min(unvisited_nodes, key=lambda node: local_weight * distance_matrix[current_node][node] + ratio_weight * (distance_matrix[current_node][node] / (distance_matrix[node][destination_node] + 1e-10)))
    return next_node

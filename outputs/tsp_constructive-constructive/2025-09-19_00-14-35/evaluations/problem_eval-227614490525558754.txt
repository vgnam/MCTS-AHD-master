def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    remaining_distance = distance_matrix[current_node][destination_node]

    next_node = max(unvisited_nodes, key=lambda node: (
        (0.6 * (1 / (distance_matrix[current_node][node] + 1e-6))) +
        (0.35 * (1 / (distance_matrix[node][destination_node] + 1e-6))) -
        (0.05 * (distance_matrix[current_node][node] / remaining_distance if remaining_distance > 0 else 0))
    ))

    return next_node

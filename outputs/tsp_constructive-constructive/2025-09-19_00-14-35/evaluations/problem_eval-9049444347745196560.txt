def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    avg_dest_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)
    next_node = min(unvisited_nodes, key=lambda node: 0.75 * distance_matrix[current_node][node] + 0.25 * distance_matrix[node][destination_node] + 0.05 * (distance_matrix[node][destination_node] - avg_dest_distance))
    return next_node

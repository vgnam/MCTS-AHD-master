def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def score(node):
        dist_to_dest = distance_matrix[node][destination_node]
        dist_from_current = distance_matrix[current_node][node]
        return (dist_to_dest / dist_from_current) * (1 - (dist_from_current / max(distance_matrix[current_node].values())))

    next_node = max(unvisited_nodes, key=score)
    return next_node

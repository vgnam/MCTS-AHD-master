def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))

    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    threshold_factor = 0.7 - (0.2 * (len(unvisited_nodes) / len(distance_matrix))) + (0.1 * (distance_matrix[current_node][destination_node] / avg_distance))
    threshold = threshold_factor * distance_matrix[current_node][next_node]

    if destination_node in unvisited_nodes and distance_matrix[current_node][destination_node] < threshold:
        return destination_node
    return next_node

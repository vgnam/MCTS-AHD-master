def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if len(unvisited_nodes) > 1:
            remaining_dists = [distance_matrix[node][n] for n in unvisited_nodes if n != node]
            avg_remaining = sum(remaining_dists) / len(remaining_dists) if remaining_dists else 0
            connection_strength = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
            decay_factor = 1 / (1 + 0.1 * len(unvisited_nodes))
            return (current_dist + 0.4 * dest_dist + 0.3 * avg_remaining) * (1 - decay_factor) + 0.3 * connection_strength * decay_factor
        else:
            return current_dist + dest_dist

    next_node = min(unvisited_nodes, key=lambda node: evaluate_node(node))
    return next_node

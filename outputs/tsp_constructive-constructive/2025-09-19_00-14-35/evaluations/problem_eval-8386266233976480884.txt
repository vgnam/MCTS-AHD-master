def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    exploration_factor = remaining_nodes / total_nodes
    temperature = 1.0 / (1.0 + exploration_factor)

    scores = []
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            connectivity_score = sum(1.0 / (1.0 + distance_matrix[node][n]) for n in unvisited_nodes if n != node) / (remaining_nodes - 1)
            centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / (remaining_nodes - 1)
            flow_disruption = max(distance_matrix[node][n] for n in unvisited_nodes if n != node) - current_dist
            path_coherence = (current_dist + dest_dist) / (centrality * (1 + exploration_factor))
        else:
            connectivity_score = 0
            centrality = 0
            flow_disruption = 0
            path_coherence = 0

        weight_local = 0.6 * (1 - exploration_factor) + 0.2 * (1 - centrality / sum(distance_matrix[node]))
        weight_connectivity = 0.3 * exploration_factor + 0.1 * (1 - connectivity_score)
        weight_flow = -0.2 * flow_disruption if flow_disruption > 0 else 0
        weight_coherence = 0.1 * path_coherence + 0.05 * (1 - exploration_factor)

        score = (weight_local * current_dist) + (weight_connectivity * connectivity_score) + (weight_flow * flow_disruption) + (weight_coherence * path_coherence)

        scores.append((node, score))

    if temperature > 0:
        exp_scores = [math.exp(-s[1]/temperature) for s in scores]
        probs = [e / sum(exp_scores) for e in exp_scores]
        next_node = random.choices([s[0] for s in scores], weights=probs)[0]
    else:
        next_node = min(scores, key=lambda x: x[1])[0]

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    weight_factor = 1.0 / (len(unvisited_nodes) + 1)
    next_node = max(unvisited_nodes, key=lambda node: (distance_matrix[node][destination_node] ** 2) / (distance_matrix[current_node][node] * (1 + weight_factor * len(unvisited_nodes))))
    return next_node

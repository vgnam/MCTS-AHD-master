def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    weight_current = 0.6
    weight_destination = 0.4
    next_node = max(unvisited_nodes, key=lambda node: (weight_current * distance_matrix[current_node][node]) + (weight_destination * distance_matrix[node][destination_node]))
    return next_node

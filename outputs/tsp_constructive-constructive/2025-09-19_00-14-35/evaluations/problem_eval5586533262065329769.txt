def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    base_distance = distance_matrix[current_node][destination_node]
    next_node = max(unvisited_nodes, key=lambda node: (
        (base_distance / distance_matrix[current_node][node]) *
        (1 / (distance_matrix[node][destination_node] + 1e-6)) *
        (1 + 0.1 * (distance_matrix[current_node][node] / (distance_matrix[current_node][destination_node] + 1e-6)))
    ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    base_distance = distance_matrix[current_node][destination_node]
    dynamic_weight = 0.5 + 0.5 * (1 / (base_distance + 1e-6))
    next_node = max(unvisited_nodes, key=lambda node: (dynamic_weight * (base_distance / distance_matrix[current_node][node])) + ((1 - dynamic_weight) * (1 / (distance_matrix[node][destination_node] + 1e-6))))
    return next_node

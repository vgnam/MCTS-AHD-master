def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    destination_in_unvisited = destination_node in unvisited_nodes

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_in_unvisited:
        destination_distance = distance_matrix[current_node][destination_node]
        if min_distance == 0:
            bias_factor = 1.0
        else:
            bias_factor = (min_distance - destination_distance) / min_distance
        remaining_nodes = len(unvisited_nodes)
        dynamic_bias = 0.3 * (1 - (1 - remaining_nodes / len(unvisited_nodes + [current_node])) ** 2)
        probability = 1 / (1 + math.exp(-5 * (bias_factor - dynamic_bias)))
        if random.random() < probability:
            next_node = destination_node

    return next_node

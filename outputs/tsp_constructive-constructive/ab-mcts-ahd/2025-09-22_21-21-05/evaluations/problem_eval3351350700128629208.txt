def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, threshold=1.5):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            ratio = 0
        else:
            ratio = distance_to_current / distance_to_destination

        combined_score = distance_to_current + ratio

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance <= distance_matrix[current_node][next_node] * threshold:
            next_node = destination_node

    return next_node

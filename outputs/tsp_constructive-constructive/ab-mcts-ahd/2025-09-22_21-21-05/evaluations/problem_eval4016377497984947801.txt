def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    if total_unvisited == 0:
        return destination_node

    exploration_factor = 1.0 / (1 + math.exp(-0.5 * (total_unvisited - 5)))
    penalty_factor = 0.1 * (10 - total_unvisited) if total_unvisited < 10 else 0

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        score = (1 - exploration_factor) * distance_to_current + exploration_factor * distance_to_destination
        score += penalty_factor * distance_to_current

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

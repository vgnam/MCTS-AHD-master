def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_ratio = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    destination_bias = 1 / (remaining_ratio + 1e-6)  # Prevent division by zero

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            weighted_score = distance * destination_bias
        else:
            weighted_score = distance * (1 - remaining_ratio)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

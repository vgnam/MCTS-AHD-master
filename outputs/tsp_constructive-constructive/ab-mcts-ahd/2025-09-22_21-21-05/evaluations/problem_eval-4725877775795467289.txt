def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    total_nodes = len(distance_matrix)
    remaining_unvisited = len(unvisited_nodes)
    destination_bias = (1 - (remaining_unvisited / total_nodes)) ** 2

    if not unvisited_nodes:
        return destination_node

    candidates = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            selection_prob = destination_bias
        else:
            selection_prob = (1 - destination_bias) * (1 / (1 + distance))
        candidates.append((node, selection_prob))

    candidates.sort(key=lambda x: x[1], reverse=True)
    return next_node

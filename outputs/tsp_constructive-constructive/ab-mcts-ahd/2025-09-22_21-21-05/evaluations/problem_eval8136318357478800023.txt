def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    centrality_weight = 0.4
    exploration_bonus = 0.3

    avg_distances = []
    for node in unvisited_nodes:
        avg_dist = sum(distance_matrix[node][other] for other in unvisited_nodes) / remaining_nodes
        avg_distances.append(avg_dist)

    avg_avg_distance = sum(avg_distances) / remaining_nodes if remaining_nodes > 0 else 0

    for i, node in enumerate(unvisited_nodes):
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        centrality_factor = 1.0 - (avg_distances[i] / (avg_avg_distance + 1e-6))

        weight_current = 1.0 / (remaining_nodes ** 2)
        weight_destination = 1.0 - weight_current

        penalty = max(0, distance_to_current - avg_distances[i])

        weighted_score = (weight_current * distance_to_current) + (weight_destination * distance_to_destination) + centrality_weight * centrality_factor - exploration_bonus * penalty

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

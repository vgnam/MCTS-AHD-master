def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    if total_unvisited == 0:
        return destination_node

    avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / total_unvisited
    remaining_nodes_factor = 1.0 / (total_unvisited ** 2)

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        weight = remaining_nodes_factor * (1.0 / (1.0 + (distance_to_destination / avg_distance_to_dest) * (total_unvisited / len(distance_matrix))))
        max_distance = max(distance_to_current, distance_to_destination)

        score = (1 - weight) * distance_to_current + weight * distance_to_destination + 0.1 * max_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            scaling_factor = (remaining_nodes / total_nodes) ** 3
            weighted_score = distance * scaling_factor
        else:
            distance_to_dest = distance_matrix[node][destination_node]
            penalty_factor = 1 + (distance_to_dest / sum(distance_matrix[node]))
            weighted_score = distance * penalty_factor

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

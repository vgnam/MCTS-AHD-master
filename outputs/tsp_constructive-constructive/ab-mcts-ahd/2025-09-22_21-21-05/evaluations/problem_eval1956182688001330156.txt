def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1  # +1 to include current node
    progress_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes

    # Dynamic weight based on progress (sigmoid function)
    weight_factor = 1 / (1 + math.exp(-10 * (progress_ratio - 0.5)))
    distance_weight = weight_factor
    destination_weight = 1 - weight_factor

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_score = distance_weight * distance_to_current + destination_weight * distance_to_destination

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

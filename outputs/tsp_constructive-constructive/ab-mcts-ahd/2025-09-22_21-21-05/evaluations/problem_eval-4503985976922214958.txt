def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)
    progress = (total_nodes - remaining_nodes) / total_nodes
    destination_weight = 1 + (1 - progress) * (remaining_nodes ** 0.5)

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            weighted_score = distance * destination_weight
        else:
            weighted_score = distance * (1 + progress * 0.5)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

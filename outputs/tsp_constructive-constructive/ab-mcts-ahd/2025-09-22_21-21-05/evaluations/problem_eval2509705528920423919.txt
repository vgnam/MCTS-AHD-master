def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = -float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        sum_distances = distance_to_current + distance_to_destination

        distances_to_remaining = [distance_matrix[node][n] for n in unvisited_nodes if n != node]
        avg_distance_to_remaining = sum(distances_to_remaining) / len(unvisited_nodes) if unvisited_nodes else 0
        variance = sum((d - avg_distance_to_remaining) ** 2 for d in distances_to_remaining) / len(unvisited_nodes) if unvisited_nodes else 0
        max_distance_to_remaining = max(distances_to_remaining) if distances_to_remaining else 0

        # Dynamic weight using sigmoid function
        dynamic_weight = 1 / (1 + math.exp(-0.5 * (num_unvisited - len(distance_matrix)/2)))
        centrality = avg_distance_to_remaining * (1 - dynamic_weight)

        combined_score = (sum_distances * dynamic_weight) + centrality - (variance * 0.2) - (max_distance_to_remaining * 0.3)

        if combined_score > best_score:
            best_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    if total_unvisited == 0:
        return destination_node

    avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / total_unvisited
    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        centrality = sum(distance_matrix[node][other] for other in unvisited_nodes) / total_unvisited
        weight = 1.0 / (1.0 + (distance_to_destination / avg_distance_to_dest) * (total_unvisited / len(distance_matrix)))

        penalty = 0
        if distance_to_destination > 1.5 * avg_distance_to_dest:
            penalty = distance_to_destination - avg_distance_to_dest

        score = (1 - weight) * distance_to_current + weight * distance_to_destination + 0.3 * centrality + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

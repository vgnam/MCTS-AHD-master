def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        max_distance = max(distance_to_current, distance_to_destination)

        # Non-linear weight based on remaining nodes
        weight = 1.0 - (total_unvisited / (total_unvisited + 2)) ** 2
        score = (distance_to_current + weight * distance_to_destination) / (1 + weight)

        if max_distance < best_score or (max_distance == best_score and score < best_score):
            best_score = score
            next_node = node

    return next_node

import math
import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    temperature = max(0.1, 1.0 - (remaining_nodes / len(distance_matrix)))

    candidates = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node] if node != destination_node else 0
        score = (distance + destination_distance) * (1.0 - temperature) + distance * temperature
        candidates.append((node, score))

    candidates.sort(key=lambda x: x[1])
    selected = random.choices([x[0] for x in candidates], weights=[math.exp(-x[1]) for x in candidates], k=1)[0]
    return next_node

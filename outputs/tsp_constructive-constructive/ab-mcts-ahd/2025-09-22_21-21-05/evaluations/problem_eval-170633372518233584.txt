def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    remaining_nodes = len(unvisited_nodes)
    weight_current = remaining_nodes / (remaining_nodes + 1)  # Decreases as remaining_nodes decreases
    weight_remaining = 1 - weight_current

    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        remaining_dist = distance_matrix[node][destination_node]
        score = (weight_current * current_dist) + (weight_remaining * remaining_dist)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

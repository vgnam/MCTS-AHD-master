def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('-inf')
    next_node = None
    remaining_count = len(unvisited_nodes)
    total_nodes = len(distance_matrix)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Calculate local density penalty
        local_density = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < 0.3 * (sum(distance_matrix[node]) / total_nodes))

        # Dynamic exploration factor based on remaining path progress
        exploration_factor = (remaining_count / total_nodes) ** 2

        # Global distance contribution
        global_contribution = sum(distance_matrix[node][n] for n in unvisited_nodes) / (remaining_count ** 0.5)

        # Novel scoring equation combining local and global factors
        score = (distance_to_destination - exploration_factor * global_contribution) / (1 + local_density)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

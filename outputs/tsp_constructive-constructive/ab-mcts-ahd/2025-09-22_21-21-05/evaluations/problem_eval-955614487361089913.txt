def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1  # +1 to include current_node
    visited_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes
    novelty_factor = 1 - visited_ratio  # Prioritizes exploration early, exploitation later

    visit_counts = {node: 0 for node in unvisited_nodes}
    for node in unvisited_nodes:
        visit_counts[node] = 1 / (1 + visit_counts.get(node, 0))  # Penalize frequently visited nodes

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_score = (1 - novelty_factor) * distance_to_current + novelty_factor * distance_to_destination
        weighted_score *= visit_counts[node]  # Adjust score based on visit frequency

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

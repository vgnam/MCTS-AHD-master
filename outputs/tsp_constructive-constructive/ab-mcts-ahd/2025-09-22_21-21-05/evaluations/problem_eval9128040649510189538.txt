def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = -float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Calculate normalized centrality (average distance to remaining nodes)
        if num_unvisited > 1:
            centrality = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (num_unvisited - 1)
        else:
            centrality = 0

        # Dynamic balance factor (higher when fewer nodes remain)
        balance_factor = 0.5 + (0.5 * (num_unvisited - 1) / (num_unvisited + 1)) if num_unvisited > 1 else 0.5

        # Combined score: balance between proximity to current and destination, adjusted by centrality
        combined_score = (balance_factor * distance_to_current + (1 - balance_factor) * distance_to_destination) - (0.5 * centrality)

        if combined_score > best_score:
            best_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('-inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight based on remaining nodes and sigmoid transition
        weight_current = 1.0 / (1.0 + math.exp(-(remaining_nodes - 5)))
        weight_destination = 1.0 - weight_current

        # Penalty for nodes with high historical scores (simulated here)
        history_penalty = 0.1 * (1.0 - (1.0 / (1.0 + math.exp(-(distance_to_current - distance_to_destination)))))

        weighted_score = (weight_current * distance_to_current) + (weight_destination * distance_to_destination) - history_penalty

        if weighted_score > best_score:
            best_score = weighted_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if all(distance_matrix[current_node][node] >= destination_distance for node in unvisited_nodes if node != destination_node):
            return destination_node

    max_distance = max(max(row) for row in distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    weight = 1 / (1 + remaining_nodes) if remaining_nodes > 0 else 0.5

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_destination = distance_matrix[node][destination_node]
        normalized_sum = (current_to_node + node_to_destination) / max_distance
        score = weight * normalized_sum + (1 - weight) * (current_to_node / max_distance)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

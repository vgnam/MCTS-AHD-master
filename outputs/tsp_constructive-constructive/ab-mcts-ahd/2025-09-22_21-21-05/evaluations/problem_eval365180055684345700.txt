def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_sum = float('inf')
    next_node = None
    min_distance = float('inf')

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_destination = distance_matrix[node][destination_node]
        total_distance = current_to_node + node_to_destination

        if total_distance < min_sum:
            min_sum = total_distance
            next_node = node

        if current_to_node < min_distance:
            min_distance = current_to_node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance < min_distance:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    time_decay_factor = 0.6
    exploration_bonus = 0.4
    penalty_scaling = 1.0 + (len(unvisited_nodes) / len(distance_matrix))

    avg_distances = []
    for node in unvisited_nodes:
        avg_dist = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)
        avg_distances.append(avg_dist)

    avg_avg_distance = sum(avg_distances) / len(unvisited_nodes) if unvisited_nodes else 0

    for i, node in enumerate(unvisited_nodes):
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        centrality_factor = 1.0 - (avg_distances[i] / (avg_avg_distance + 1e-6))
        penalty = max(0, (distance_to_current + distance_to_destination) - 2 * avg_distances[i]) * penalty_scaling

        combined_score = (distance_to_current + distance_to_destination) + time_decay_factor * centrality_factor - exploration_bonus * penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

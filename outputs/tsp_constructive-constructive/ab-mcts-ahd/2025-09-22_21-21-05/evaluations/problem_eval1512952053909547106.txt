def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    unvisited_count = len(unvisited_nodes)
    total_nodes = unvisited_count + 1
    destination_bias = (total_nodes - unvisited_count) / total_nodes

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if node == destination_node:
            adjusted_distance = distance_to_current * (1 - destination_bias)
        else:
            adjusted_distance = distance_to_current

        if distance_to_destination == 0:
            ratio = 0
        else:
            ratio = adjusted_distance / distance_to_destination

        combined_score = adjusted_distance + ratio

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_progress = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    exploration_factor = 1 - remaining_progress  # Decays as progress increases

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        dynamic_weight = 3 * exploration_factor  # Higher weight as exploration decays
        penalty = 0.5 * distance_matrix[current_node][node]  # Penalty for revisiting
        combined_score = distance_to_current + dynamic_weight * distance_to_destination + penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

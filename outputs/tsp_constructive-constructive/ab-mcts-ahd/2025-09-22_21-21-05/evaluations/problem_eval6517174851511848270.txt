def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    # Calculate centrality of current node
    centrality = 1.0 / (sum(distance_matrix[current_node]) / len(distance_matrix[current_node]))

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        dynamic_weight = 0.3 * centrality + 0.2 * (1 / (1 + len(unvisited_nodes)))
        threshold = 1.5 * (1 - dynamic_weight)
        if destination_distance <= threshold * min_distance:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    memory = {}
    exploration_bonus = 0.7
    heuristic_factor = 0.4

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0

        if node == destination_node:
            weighted_score = distance_to_current * (1 - heuristic_factor)
        else:
            weighted_score = distance_to_current * (1 + heuristic_factor)

        memory_factor = 1 + 0.15 * memory.get(node, 0)
        exploration_factor = 1 - (distance_to_destination / max(distance_matrix[node])) * exploration_bonus
        combined_score = weighted_score + 0.3 * avg_distance_to_remaining
        score = combined_score * memory_factor * exploration_factor

        if score < best_score:
            best_score = score
            next_node = node

    memory[next_node] = memory.get(next_node, 0) + 1
    return next_node

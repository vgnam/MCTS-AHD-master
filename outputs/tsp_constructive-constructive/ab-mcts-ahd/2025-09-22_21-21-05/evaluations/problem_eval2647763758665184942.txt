def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    temperature = remaining_nodes / (remaining_nodes + 1)
    probabilities = []

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        weighted_score = (1 - temperature) * distance + temperature * destination_distance
        probabilities.append((node, weighted_score))

    probabilities.sort(key=lambda x: x[1])
    selected_node = probabilities[0][0] if remaining_nodes > 1 else destination_node

    return next_node

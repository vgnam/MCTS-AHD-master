def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_sum = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_destination = distance_matrix[node][destination_node]
        total_distance = current_to_node + node_to_destination

        if destination_node == node:
            if total_distance < min_sum:
                min_sum = total_distance
                next_node = node
        else:
            if total_distance < min_sum and (next_node is None or destination_node not in unvisited_nodes or distance_matrix[current_node][destination_node] >= current_to_node):
                min_sum = total_distance
                next_node = node

    return next_node

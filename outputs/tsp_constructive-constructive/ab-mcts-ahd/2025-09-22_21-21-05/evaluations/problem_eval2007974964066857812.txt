def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    nearest_node = None
    min_distance = float('inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            nearest_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        total_unvisited_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
        avg_unvisited_distance = total_unvisited_distance / len(unvisited_nodes) if unvisited_nodes else 0

        dynamic_threshold = 0.8 * avg_unvisited_distance
        if destination_distance < dynamic_threshold:
            return destination_node

    return next_node

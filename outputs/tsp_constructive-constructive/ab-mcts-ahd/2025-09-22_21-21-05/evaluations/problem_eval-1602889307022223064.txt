def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        ratio = current_distance / (remaining_distance + 1e-10)
        score = current_distance + ratio

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        destination_ratio = destination_distance / (distance_matrix[destination_node][destination_node] + 1e-10)
        destination_score = destination_distance + destination_ratio
        if destination_score < min_score:
            next_node = destination_node

    return next_node

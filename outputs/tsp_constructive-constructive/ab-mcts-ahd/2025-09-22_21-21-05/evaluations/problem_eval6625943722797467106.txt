def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)
    weight_avg = 0.3 * (num_unvisited / (num_unvisited + 1))  # Dynamic weight adjustment
    penalty_threshold = 0.1 * sum(distance_matrix[0]) / len(distance_matrix)  # Dynamic penalty threshold
    adaptive_penalty_factor = 1.0 + (0.1 * (len(unvisited_nodes) / len(distance_matrix)))  # Adaptive penalty
    weight_proximity = 0.5 if num_unvisited > 1 else 0.8  # Prioritize proximity as fewer nodes remain

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / num_unvisited if num_unvisited else 0

        # Normalized score combining immediate and long-term distances
        normalized_score = (distance_to_current / max(1, sum(distance_matrix[current_node]))) * weight_proximity + \
                          (distance_to_destination / max(1, sum(distance_matrix[node]))) * (1 - weight_proximity)

        # Penalty for nodes too close to current node
        penalty = 1.0 if distance_to_current < penalty_threshold else 1.0
        penalty *= adaptive_penalty_factor  # Apply adaptive penalty

        combined_score = (normalized_score + weight_avg * avg_distance_to_remaining) * penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

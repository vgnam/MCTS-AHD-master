def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    global_weight = 0.5 if remaining_nodes > 2 else 0.2
    avg_remaining_distance = sum(distance_matrix[current_node][n] for n in unvisited_nodes) / remaining_nodes if remaining_nodes else 0

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / remaining_nodes if remaining_nodes else 0

        proximity_penalty = 1.0 if distance_to_current < 0.1 * avg_remaining_distance else 0.0
        normalized_avg_distance = avg_distance_to_remaining / avg_remaining_distance if avg_remaining_distance else 0

        combined_score = (distance_to_current + distance_to_destination) + global_weight * normalized_avg_distance + proximity_penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

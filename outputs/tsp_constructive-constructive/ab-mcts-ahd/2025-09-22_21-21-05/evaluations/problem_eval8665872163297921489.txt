def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    unvisited_count = len(unvisited_nodes)
    total_nodes = unvisited_count + 1
    destination_bias = (total_nodes - unvisited_count) / total_nodes
    remaining_fraction = unvisited_count / total_nodes

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            adjusted_distance = distance * (1 - (destination_bias * (1 - remaining_fraction)))
        else:
            adjusted_distance = distance * (1 + remaining_fraction)
        if adjusted_distance < min_distance:
            min_distance = adjusted_distance
            next_node = node

    return next_node

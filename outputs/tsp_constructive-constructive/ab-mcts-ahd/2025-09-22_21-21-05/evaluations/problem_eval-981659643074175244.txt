def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Fixed weight for immediate distance
        weight = 0.7

        # Linear penalty for detours
        detour_penalty = distance_to_current + distance_to_destination - distance_matrix[current_node][destination_node]

        # Combine scores with fixed weight and linear penalty
        combined_score = distance_to_current + weight * distance_to_destination + detour_penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight for sum of distances based on remaining nodes
        sum_distance_weight = 0.3 if remaining_nodes_count > 5 else 0.7
        sum_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)

        # Penalty for revisiting nodes
        penalty = 1.0 if node in unvisited_nodes else 1.5

        combined_score = (distance_to_current + distance_to_destination) * penalty + sum_distance_weight * sum_distance_to_remaining

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

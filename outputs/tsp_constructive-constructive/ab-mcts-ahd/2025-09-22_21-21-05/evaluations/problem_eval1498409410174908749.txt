def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        weight_current = remaining_nodes / (remaining_nodes + 2)
        weight_destination = 1.0 - weight_current
        weighted_score = (weight_current * min_distance) + (weight_destination * destination_distance)

        if destination_node in unvisited_nodes:
            destination_score = (weight_current * destination_distance) + (weight_destination * distance_matrix[destination_node][destination_node])
            if destination_score < weighted_score:
                next_node = destination_node

    return next_node

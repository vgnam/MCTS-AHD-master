def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        weighted_distance = distance + 0.7 * distance_matrix[node][destination_node]
        if weighted_distance < min_distance:
            min_distance = weighted_distance
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node] + 0.7 * distance_matrix[destination_node][destination_node]
        if destination_distance < min_distance:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    ratio = remaining_nodes / total_nodes if total_nodes > 0 else 0
    weight_to_destination = 0.7 * (1 - ratio)

    best_score = float('-inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        combined_score = -distance_to_current + weight_to_destination * distance_to_destination

        if distance_matrix[node][current_node] > 0:
            combined_score -= 0.3 * distance_matrix[node][current_node]

        if remaining_nodes <= 2:
            combined_score += 0.2 * sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)

        if combined_score > best_score:
            best_score = combined_score
            next_node = node

    return next_node

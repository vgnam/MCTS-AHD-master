def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    log_weight = min(1.0, math.log(remaining_nodes + 1) / math.log(remaining_nodes + 2))  # Logarithmic decay
    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Novel detour penalty based on remaining nodes
        detour_penalty = (distance_to_current + distance_to_destination - distance_matrix[current_node][destination_node]) ** 1.5
        detour_penalty *= (remaining_nodes / (remaining_nodes + 1))  # Scale with remaining nodes

        combined_score = distance_to_current + log_weight * distance_to_destination + detour_penalty

        if combined_score < best_score:
            best_score = combined_score
            next_node = node

    return next_node

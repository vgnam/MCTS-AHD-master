def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_cost = float('inf')

    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]

        remaining_nodes = unvisited_nodes - {node}
        if remaining_nodes:
            next_best_cost = min(distance_matrix[node][n] for n in remaining_nodes)
            total_cost = immediate_cost + next_best_cost
        else:
            total_cost = immediate_cost

        if total_cost < min_total_cost:
            min_total_cost = total_cost
            next_node = node

    if destination_node in unvisited_nodes:
        destination_cost = distance_matrix[current_node][destination_node]
        if destination_cost <= min_total_cost:
            next_node = destination_node

    return next_node

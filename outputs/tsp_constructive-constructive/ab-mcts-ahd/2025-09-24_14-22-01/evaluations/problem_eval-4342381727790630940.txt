def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    total_nodes = len(unvisited_nodes) + 1  # +1 to account for the current node
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 - (remaining_nodes / total_nodes)
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_return_distance = distance_matrix[node][destination_node]
        score = immediate_distance + weight_factor * estimated_return_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight = 0.5 if remaining_nodes > 1 else 1.0
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        centrality_score = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1)
        combined_score = (1 - weight) * distance_to_node + weight * (0.7 * distance_to_dest + 0.3 * centrality_score)
        if combined_score < min_score:
            min_score = combined_score
            next_node = node
    return next_node

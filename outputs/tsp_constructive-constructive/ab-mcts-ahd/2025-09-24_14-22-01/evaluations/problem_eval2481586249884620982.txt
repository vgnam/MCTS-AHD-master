def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1
    progress_ratio = remaining_nodes / total_nodes if total_nodes > 0 else 0.0

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_distance = distance_matrix[node][destination_node]

        if estimated_distance > 0:
            distance_ratio = immediate_distance / estimated_distance
        else:
            distance_ratio = 1.0

        weight_immediate = 0.5 + 0.3 * progress_ratio
        weight_estimated = 0.5 - 0.3 * progress_ratio
        penalty = 0.1 * distance_ratio * (1.0 - progress_ratio)

        score = (weight_immediate * immediate_distance) + (weight_estimated * estimated_distance) + penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

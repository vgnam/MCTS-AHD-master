def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_immediate = 0.5 if remaining_nodes > 3 else 0.2 * (1 - 0.1 * (3 - remaining_nodes))
    weight_potential = 0.3 if remaining_nodes > 3 else 0.6 * (1 + 0.1 * (3 - remaining_nodes))
    weight_centrality = 0.2 if remaining_nodes > 3 else 0.2 * (1 + 0.2 * (3 - remaining_nodes))

    avg_distance = sum(sum(row) for row in distance_matrix) / (len(distance_matrix) * (len(distance_matrix) - 1))
    node_degree = {node: sum(1 for d in distance_matrix[node] if d > 0) for node in range(len(distance_matrix))}
    centrality_penalty = {node: node_degree[node] / max(node_degree.values()) if max(node_degree.values()) > 0 else 0 for node in range(len(distance_matrix))}

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        centrality = sum(distance_matrix[node]) / (len(distance_matrix) - 1)
        normalized_immediate = immediate_distance / avg_distance
        normalized_potential = potential_distance / avg_distance
        dynamic_centrality = centrality * centrality_penalty[node] / avg_distance
        total_cost = (weight_immediate * normalized_immediate +
                      weight_potential * normalized_potential +
                      weight_centrality * dynamic_centrality)
        if total_cost < min_cost:
            min_cost = total_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, gamma=0.6, delta=0.4):
    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        connectivity = len(unvisited_nodes) - 1  # Approximate connectivity (all unvisited nodes are neighbors)
        weighted_score = gamma * (1 / immediate_distance) + delta * connectivity
        if weighted_score > max_score:
            max_score = weighted_score
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

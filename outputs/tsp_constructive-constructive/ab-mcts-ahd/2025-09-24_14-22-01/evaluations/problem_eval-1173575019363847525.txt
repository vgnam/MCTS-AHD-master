def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_remaining_distance = distance_matrix[node][destination_node]
        weight = 0.7 if len(unvisited_nodes) > 2 else 0.3  # Higher weight for immediate distance when many nodes remain
        score = (weight * immediate_distance) + ((1 - weight) * estimated_remaining_distance)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

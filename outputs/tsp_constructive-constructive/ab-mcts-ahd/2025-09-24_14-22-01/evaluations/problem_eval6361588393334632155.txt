def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_distance = distance_matrix[node][destination_node]
        penalty = (1.0 / (remaining_nodes + 1)) ** 2 if remaining_nodes > 0 else 0.0
        weight_immediate = 0.8 - 0.3 * (remaining_nodes / len(unvisited_nodes))
        weight_estimated = 0.2 + 0.3 * (remaining_nodes / len(unvisited_nodes))
        score = (weight_immediate * immediate_distance) + (weight_estimated * estimated_distance) + (penalty * immediate_distance * remaining_nodes)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

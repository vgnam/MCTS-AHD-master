def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    max_distance_threshold = 1.5 * distance_matrix[current_node][destination_node]
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        if immediate_distance > max_distance_threshold:
            continue
        potential_distance = distance_matrix[node][destination_node]
        diversity_factor = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1 if len(unvisited_nodes) > 1 else 1)
        path_momentum = sum(distance_matrix[current_node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes)
        weighted_cost = (0.5 * immediate_distance) + (0.2 * potential_distance) + (0.2 * diversity_factor) + (0.1 * path_momentum)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, gamma=0.7, delta=0.3):
    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        connectivity = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
        weighted_score = gamma * (1 / immediate_distance) + delta * (1 / (1 + connectivity))
        if weighted_score > max_score:
            max_score = weighted_score
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

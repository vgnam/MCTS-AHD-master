def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes) - 1
    weight_current = 0.5 + 0.1 * remaining_nodes
    weight_destination = 1.5 - 0.1 * remaining_nodes
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 0.2 * (1 if node in unvisited_nodes else 0)
        score = weight_current * distance_to_current + weight_destination * distance_to_destination + penalty
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

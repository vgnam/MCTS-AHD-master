def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.4, beta=0.3, gamma=0.3):
    min_cost = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        connectivity = len(unvisited_nodes) - 1
        combined_cost = alpha * immediate_distance + beta * potential_distance + gamma * (1 / (1 + connectivity))
        if combined_cost < min_cost:
            min_cost = combined_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    initial_distance = distance_matrix[current_node][destination_node]
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        local_distance = distance_matrix[current_node][node]
        global_distance = distance_matrix[node][destination_node]
        weight = 0.5 * (global_distance / initial_distance) if initial_distance != 0 else 0.5
        score = local_distance + weight * global_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

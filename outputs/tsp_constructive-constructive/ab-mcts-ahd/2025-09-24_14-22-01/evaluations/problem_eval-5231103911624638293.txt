def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    alpha = 0.5
    beta = 0.3
    gamma = 0.2
    num_unvisited = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    centrality = [sum(distance_matrix[node]) / (total_nodes - 1) for node in range(total_nodes)]

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        if immediate_distance == 0 or potential_distance == 0:
            geometric_mean = float('inf')
        else:
            geometric_mean = (immediate_distance * potential_distance) ** 0.5
        position_factor = (distance_matrix[current_node][destination_node] - immediate_distance) / distance_matrix[current_node][destination_node] if distance_matrix[current_node][destination_node] > 0 else 0
        centrality_factor = centrality[node] / max(centrality) if max(centrality) > 0 else 0
        weighted_cost = geometric_mean * (1 - alpha - beta) + position_factor * alpha + centrality_factor * beta
        if num_unvisited > total_nodes / 2:
            weighted_cost *= (1 - gamma)
        else:
            weighted_cost *= (1 + gamma)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

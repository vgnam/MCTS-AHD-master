def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    exploration_weight = 0.7

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        distance_ratio = immediate_distance / (potential_distance + 1e-10)  # Avoid division by zero
        exploration_factor = 1 / (1 + distance_ratio)  # Dynamic adjustment based on distance ratio
        penalty = 1 + (immediate_distance / max(distance_matrix[current_node]))  # Penalize distant nodes
        weighted_cost = (1 - exploration_weight) * immediate_distance + exploration_weight * exploration_factor * penalty
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node

    if next_node is None:
        next_node = destination_node
    return next_node

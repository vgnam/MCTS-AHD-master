def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        avg_remaining_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes) if unvisited_nodes else 0
        weighted_cost = 0.2 * immediate_distance + 0.5 * potential_distance + 0.3 * avg_remaining_distance
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    return next_node

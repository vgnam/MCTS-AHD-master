def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1  # +1 for the current node
    remaining_ratio = remaining_nodes / total_nodes if total_nodes > 0 else 0
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_return_distance = distance_matrix[node][destination_node]
        score = immediate_distance + (1 - remaining_ratio) * estimated_return_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    weight = 0.2 + 0.2 * (len(unvisited_nodes) / 10)  # Dynamic weight between 0.2 and 0.4
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_return_distance = distance_matrix[node][destination_node]
        score = immediate_distance + weight * estimated_return_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

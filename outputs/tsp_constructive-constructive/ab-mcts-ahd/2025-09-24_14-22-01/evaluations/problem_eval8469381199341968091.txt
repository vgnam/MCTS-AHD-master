def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = 1.0 if remaining_nodes <= 2 else 0.5 + (remaining_nodes / len(unvisited_nodes))
        score = distance_to_current + weight * distance_to_destination
        if distance_to_current < 0.1 * distance_matrix[current_node][destination_node]:
            score += 10.0
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_cost = float('inf')
    next_node = None
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    gamma = len(unvisited_nodes) / len(distance_matrix)
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes)

    for node in unvisited_nodes:
        if node == destination_node:
            continue

        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]

        if immediate_distance == 0 or potential_distance == 0:
            combined_weight = float('inf')
        else:
            weighted_immediate = immediate_distance * (1 + centrality)
            weighted_potential = potential_distance * (1 - centrality)
            combined_weight = (weighted_immediate + weighted_potential) / 2

        deviation = abs(immediate_distance - avg_distance)
        sigmoid_gamma = 1 / (1 + math.exp(-gamma))
        weighted_cost = (0.5 * combined_weight) + (0.3 * deviation * sigmoid_gamma) + (0.2 * sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1 if len(unvisited_nodes) > 1 else 1))

        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node

    if next_node is None:
        next_node = destination_node

    return next_node

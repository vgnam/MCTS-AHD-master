def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    if remaining_nodes == 0:
        return None
    min_total_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_weight = 0.7 * (remaining_nodes / (remaining_nodes + 1))
        estimated_weight = 0.3 * (1 - (remaining_nodes / (remaining_nodes + 1)))
        immediate_distance = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        total_distance = immediate_weight * immediate_distance + estimated_weight * estimated_distance_to_destination
        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node
    return next_node

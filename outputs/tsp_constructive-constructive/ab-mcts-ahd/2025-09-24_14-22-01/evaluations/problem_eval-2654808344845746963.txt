def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_value = float('inf')
    next_node = None
    alpha = 0.7  # Weight for distance_to_current (higher alpha favors local proximity)
    beta = 0.3    # Weight for distance_to_destination (higher beta favors global progress)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_value = alpha * distance_to_current + beta * distance_to_destination
        if combined_value < min_value:
            min_value = combined_value
            next_node = node
    return next_node

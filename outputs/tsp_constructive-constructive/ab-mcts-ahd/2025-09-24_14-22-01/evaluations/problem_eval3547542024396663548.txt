def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    best_score = float('inf')
    next_node = None
    avg_unvisited_dist = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        weighted_score = 0.6 * current_to_node + 0.4 * (node_to_dest - avg_unvisited_dist)
        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node
    return next_node

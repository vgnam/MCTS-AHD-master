def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_cost = float('inf')
    next_node = None
    base_decay = 0.7  # Higher decay for future distance
    threshold_decay = 0.5  # More aggressive threshold adjustment
    dynamic_threshold = 1.2  # Lower base threshold

    # Calculate max remaining distance to destination
    max_remaining = max(distance_matrix[node][destination_node] for node in unvisited_nodes)

    for node in unvisited_nodes:
        if node == destination_node:
            continue

        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]

        # Dynamic threshold adjustment based on max remaining distance
        adjusted_threshold = dynamic_threshold * (1 + threshold_decay * (potential_distance / max_remaining))

        # Linear path potential metric with higher decay
        path_potential = immediate_distance + base_decay * potential_distance

        # Higher penalty for exceeding threshold
        if immediate_distance > adjusted_threshold:
            penalty = 3.0 + (immediate_distance - adjusted_threshold) * 0.5
            path_potential *= penalty

        if path_potential < min_cost:
            min_cost = path_potential
            next_node = node

    return next_node

import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes) - 1
    total_nodes = len(distance_matrix)
    weight_immediate = 0.6 + 0.4 * (remaining_nodes / (remaining_nodes + 1))
    penalty_factor = 0.5 / (remaining_nodes + 1)
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes)
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        penalty = penalty_factor * potential_distance
        deviation = abs(immediate_distance - avg_distance)
        weighted_cost = (weight_immediate * immediate_distance * (1 + centrality) + (1 - weight_immediate) * potential_distance * (1 - centrality)) - penalty + deviation
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

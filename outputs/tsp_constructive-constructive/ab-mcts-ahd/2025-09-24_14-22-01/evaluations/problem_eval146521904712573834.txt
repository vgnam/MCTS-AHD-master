def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        score = distance_matrix[current_node][node] + 0.5 * distance_matrix[node][destination_node]
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

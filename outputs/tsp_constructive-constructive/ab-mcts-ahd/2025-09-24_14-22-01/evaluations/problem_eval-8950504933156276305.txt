def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    beta = 0.7
    gamma = 1.0 / len(unvisited_nodes) if unvisited_nodes else 1.0
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        if immediate_distance == 0 or potential_distance == 0:
            geometric_mean = float('inf')
        else:
            geometric_mean = (immediate_distance * potential_distance) ** 0.5
        distance_ratio = immediate_distance / potential_distance if potential_distance > 0 else float('inf')
        weighted_cost = geometric_mean * (1 - beta) + distance_ratio * beta * gamma
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

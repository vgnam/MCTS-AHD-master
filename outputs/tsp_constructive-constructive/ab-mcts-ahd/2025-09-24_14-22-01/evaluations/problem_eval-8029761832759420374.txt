def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    exploration_weight = min(0.7 + (1 / (1 + remaining_nodes)), 0.9)  # Dynamic exploration weight

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        exploration_factor = 1 / (1 + potential_distance)
        penalty_factor = 1 + (potential_distance / (sum(distance_matrix[current_node]) + 1e-6))  # Penalty for distant nodes
        weighted_cost = (1 - exploration_weight) * immediate_distance + exploration_weight * exploration_factor * penalty_factor
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node

    if next_node is None:
        next_node = destination_node
    return next_node

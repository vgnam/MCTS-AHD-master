import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        if current_to_node == 0 or node_to_dest == 0:
            cost = float('inf')
        else:
            cost = (current_to_node * node_to_dest) * (1 + centrality)
        if cost < min_cost:
            min_cost = cost
            next_node = node
    return next_node

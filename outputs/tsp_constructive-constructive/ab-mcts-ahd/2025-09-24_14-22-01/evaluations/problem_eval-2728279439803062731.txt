def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    best_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        fixed_weight = 0.3  # Fixed weight for destination distance
        penalty = 1.5 if immediate_distance > 1.5 * distance_matrix[current_node][destination_node] else 1.0
        combined_score = immediate_distance + fixed_weight * destination_distance * penalty
        if combined_score < best_score:
            best_score = combined_score
            next_node = node
    return next_node

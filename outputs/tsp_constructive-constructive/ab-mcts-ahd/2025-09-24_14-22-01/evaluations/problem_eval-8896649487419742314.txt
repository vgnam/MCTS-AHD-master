def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_value = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress_weight = remaining_nodes / total_nodes

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            future_impact = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1)
        else:
            future_impact = 0

        combined_value = (1 - progress_weight) * distance_to_current + \
                         progress_weight * (0.7 * distance_to_destination + 0.3 * future_impact)

        if combined_value < min_value:
            min_value = combined_value
            next_node = node

    return next_node

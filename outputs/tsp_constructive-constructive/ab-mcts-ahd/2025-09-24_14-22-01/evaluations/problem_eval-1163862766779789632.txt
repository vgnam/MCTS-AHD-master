import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        deviation = abs(distance_to_current - avg_distance)
        score = distance_to_current + 0.5 * distance_to_destination + 0.1 * deviation

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

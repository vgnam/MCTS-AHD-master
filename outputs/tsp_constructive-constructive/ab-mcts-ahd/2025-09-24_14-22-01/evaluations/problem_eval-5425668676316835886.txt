def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = 0.6 * distance_to_current + 0.4 * distance_to_destination + 0.1 * (distance_to_destination < 0.1 * sum(distance_matrix[node]) / len(distance_matrix[node]))
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

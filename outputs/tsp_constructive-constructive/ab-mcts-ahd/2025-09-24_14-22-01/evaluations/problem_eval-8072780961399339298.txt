import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    current_to_dest = distance_matrix[current_node][destination_node]
    max_distance = max(max(row) for row in distance_matrix)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        weight_factor = math.exp(-current_to_dest / max_distance)
        weighted_score = (1 - weight_factor) * immediate_distance + weight_factor * future_distance

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node
    return next_node

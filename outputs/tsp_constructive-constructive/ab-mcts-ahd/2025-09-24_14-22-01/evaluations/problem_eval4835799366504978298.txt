def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    num_unvisited = len(unvisited_nodes)
    min_cost = float('inf')
    next_node = None
    current_to_dest = distance_matrix[current_node][destination_node]
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        exploration_weight = (num_unvisited ** 2) / (num_unvisited + 10)
        exploitation_weight = (current_to_dest / (potential_distance + 1)) ** 0.5
        weighted_cost = immediate_distance + exploration_weight * potential_distance * exploitation_weight
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

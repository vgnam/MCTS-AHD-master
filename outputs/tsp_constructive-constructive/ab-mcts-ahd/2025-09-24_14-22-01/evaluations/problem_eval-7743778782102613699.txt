def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    centroid_node = sum(unvisited_nodes) / len(unvisited_nodes)
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    min_total_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        centroid_weight = abs(node - centroid_node) / len(unvisited_nodes)
        total_distance = (current_to_node * 0.6) + (node_to_dest * 0.3) + (centroid_weight * 0.1) + (abs(current_to_node - avg_distance) * 0.2)
        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    best_score = float('inf')
    next_node = None
    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    weight = 0.5 + (0.5 * (remaining_nodes / total_nodes))
    penalty_factor = 1.0 + (0.2 * (total_nodes - remaining_nodes) / total_nodes)
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        combined_score = weight * immediate_distance + (1 - weight) * potential_distance
        combined_score *= penalty_factor
        if combined_score < best_score:
            best_score = combined_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5):
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        direct_distance = distance_matrix[node][destination_node]
        immediate_distance = distance_matrix[current_node][node]
        score = alpha * direct_distance + (1 - alpha) * immediate_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

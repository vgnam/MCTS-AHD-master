def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_current = distance_to_current * (1 + centrality)
        weighted_destination = distance_to_destination * (1 - centrality)
        score = 0.5 * weighted_current + weighted_destination
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

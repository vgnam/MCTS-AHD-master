def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, gamma=0.5, delta=0.3, alpha=0.2):
    max_score = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if len(unvisited_nodes) == 1:
            connectivity = 1.0
        else:
            avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1)
            connectivity = 1 / avg_distance_to_unvisited
        weighted_score = gamma * (1 / immediate_distance) + delta * connectivity + alpha * (1 / distance_matrix[node][destination_node])
        if weighted_score > max_score:
            max_score = weighted_score
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

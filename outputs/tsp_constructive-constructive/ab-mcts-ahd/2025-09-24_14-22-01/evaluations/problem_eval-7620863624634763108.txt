def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    beta = 0.5  # Static weighting factor
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        if immediate_distance == 0 or potential_distance == 0:
            arithmetic_mean = float('inf')
        else:
            arithmetic_mean = (immediate_distance + potential_distance) / 2
        position_factor = (distance_matrix[current_node][destination_node] - immediate_distance) / distance_matrix[current_node][destination_node] if distance_matrix[current_node][destination_node] > 0 else 0
        revisit_penalty = 1 - (remaining_nodes / (remaining_nodes + 1))  # Linear penalty
        weighted_cost = arithmetic_mean * (1 - beta) + (position_factor * beta) * (1 + revisit_penalty)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

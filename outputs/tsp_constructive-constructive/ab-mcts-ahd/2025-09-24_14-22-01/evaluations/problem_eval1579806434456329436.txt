def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = unvisited_nodes - {destination_node}
    if not remaining_nodes:
        return destination_node
    avg_distance = sum(distance_matrix[current_node][node] for node in remaining_nodes) / len(remaining_nodes)
    node_centrality = {}
    for node in remaining_nodes:
        neighbors = [n for n in remaining_nodes if n != node]
        if neighbors:
            node_centrality[node] = len(neighbors) / len(remaining_nodes)
        else:
            node_centrality[node] = 0
    avg_centrality = sum(node_centrality.values()) / len(node_centrality) if node_centrality else 0
    for node in remaining_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        heuristic_factor = (immediate_distance + potential_distance) / avg_distance
        clustering_factor = (node_centrality[node] / avg_centrality) if avg_centrality else 1
        total_cost = 0.5 * immediate_distance + 0.2 * potential_distance + 0.2 * clustering_factor + 0.1 * heuristic_factor
        if total_cost < min_cost:
            min_cost = total_cost
            next_node = node
    return next_node

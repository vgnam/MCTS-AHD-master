def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_unvisited = len(unvisited_nodes)
    destination_weight = 0.5 / total_unvisited if total_unvisited > 1 else 1.0
    min_cost = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        if node == destination_node:
            weighted_cost = immediate_distance * (1 - destination_weight) + potential_distance * destination_weight
        else:
            weighted_cost = immediate_distance * (1 + destination_weight) + potential_distance * (1 - destination_weight)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

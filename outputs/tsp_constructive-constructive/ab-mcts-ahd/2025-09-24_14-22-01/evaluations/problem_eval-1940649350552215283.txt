def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        if immediate_distance == 0:
            immediate_distance = 1e-6
        if future_distance == 0:
            future_distance = 1e-6
        dynamic_weight = 0.5 + 0.4 * (1 - (num_unvisited / len(distance_matrix)))
        score = dynamic_weight * (1 / immediate_distance) + (1 - dynamic_weight) * (1 / future_distance)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

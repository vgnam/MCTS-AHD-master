def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        distance_ratio = immediate_distance / (potential_distance + 1e-10)
        weight_factor = (remaining_nodes / (remaining_nodes + 1)) ** 2
        penalty = 1 + (potential_distance / (distance_matrix[current_node][destination_node] + 1e-10)) ** 0.5
        weighted_cost = immediate_distance * (potential_distance ** weight_factor) * penalty
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

importance factor (20%) based on both distance and connectivity, while prioritizing immediate distance (40%), potential return distance (20%), heuristic factor (10%), and node importance (10%).}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = unvisited_nodes - {destination_node}
    if not remaining_nodes:
        return destination_node
    avg_distance = sum(distance_matrix[current_node][node] for node in remaining_nodes) / len(remaining_nodes)
    node_importance = {}
    for node in remaining_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        neighbors = [n for n in remaining_nodes if n != node]
        if neighbors:
            avg_neighbor_distance = sum(distance_matrix[node][n] for n in neighbors) / len(neighbors)
            node_importance[node] = (immediate_distance + potential_distance) / (avg_neighbor_distance + 1e-10)
        else:
            node_importance[node] = 0
    avg_importance = sum(node_importance.values()) / len(node_importance) if node_importance else 0
    for node in remaining_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        heuristic_factor = (immediate_distance + potential_distance) / avg_distance
        importance_factor = (node_importance[node] / avg_importance) if avg_importance else 1
        total_cost = 0.4 * immediate_distance + 0.2 * potential_distance + 0.1 * heuristic_factor + 0.1 * (immediate_distance * importance_factor) + 0.2 * (potential_distance * importance_factor)
        if total_cost < min_cost:
            min_cost = total_cost
            next_node = node
    return next_node

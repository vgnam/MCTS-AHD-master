def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    if remaining_nodes == 0:
        return destination_node

    current_weight = 0.6 * (0.9 ** (remaining_nodes - 1))
    destination_weight = 0.4 - 0.05 * (0.9 ** (remaining_nodes - 1))
    proximity_weight = 0.2

    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = (current_weight * distance_to_node +
                 destination_weight * distance_to_destination +
                 proximity_weight * distance_to_node)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

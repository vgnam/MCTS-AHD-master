def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    weight_immediate = 0.7  # Higher weight for immediate distance
    weight_long_term = 0.3  # Lower weight for long-term distance
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        score = (weight_immediate * current_to_node) + (weight_long_term * node_to_dest)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

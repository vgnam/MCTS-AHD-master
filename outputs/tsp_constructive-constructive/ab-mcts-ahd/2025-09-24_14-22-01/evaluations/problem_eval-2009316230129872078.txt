def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_total_score = float('inf')
    next_node = None
    avg_distance_to_unvisited = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        score = current_to_node + (avg_distance_to_unvisited - node_to_dest)
        if score < min_total_score:
            min_total_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + remaining_nodes) if remaining_nodes > 0 else 1.0
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes)
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        normalized_immediate = immediate_distance / (avg_distance + 1e-6)
        normalized_potential = potential_distance / (max(distance_matrix[node]) + 1e-6)
        weighted_immediate = normalized_immediate * (1 + centrality)
        weighted_potential = normalized_potential * (1 - centrality)
        combined_weight = (weighted_immediate + weighted_potential) / 2
        deviation = abs(normalized_immediate - 1)
        penalty = 0.1 * (normalized_potential ** 2)
        dynamic_cost = (combined_weight ** weight_factor) * (deviation ** (1 - weight_factor)) - penalty

        if dynamic_cost < min_cost:
            min_cost = dynamic_cost
            next_node = node

    if next_node is None:
        next_node = destination_node
    return next_node

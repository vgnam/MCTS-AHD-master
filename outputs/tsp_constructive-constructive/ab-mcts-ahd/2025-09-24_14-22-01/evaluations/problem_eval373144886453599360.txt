def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        estimated_remaining_distance = distance_matrix[node][destination_node]
        weight = 0.7 if remaining_nodes > 3 else 0.4 if remaining_nodes > 1 else 0.3
        score = (weight * immediate_distance) + ((1 - weight) * estimated_remaining_distance)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

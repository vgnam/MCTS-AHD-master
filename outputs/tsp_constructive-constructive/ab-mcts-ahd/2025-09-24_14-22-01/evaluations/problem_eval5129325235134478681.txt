defaulting to the destination if no other options remain.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    proximity_weight = 0.2 if remaining_nodes > 2 else 0.0  # Reduce proximity influence as tour nears completion
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        diversity_factor = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1 if remaining_nodes > 1 else 1)
        proximity_factor = distance_matrix[node][destination_node] if proximity_weight > 0 else 0
        weighted_cost = (0.5 * immediate_distance) + (0.3 * diversity_factor) + (proximity_weight * proximity_factor)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    if not unvisited_nodes:
        return destination_node
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes if node != destination_node) / len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        distance_ratio = immediate_distance / avg_distance if avg_distance > 0 else 1.0
        spatial_dispersion = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 1.0
        weighted_cost = immediate_distance + (0.4 * distance_ratio) * potential_distance + (0.3 * spatial_dispersion)
        penalty = 1.0 + 0.1 * (potential_distance / (sum(distance_matrix[node][other] for other in range(total_nodes)) / total_nodes)) if total_nodes > 0 else 1.0
        weighted_cost *= penalty
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    return next_node

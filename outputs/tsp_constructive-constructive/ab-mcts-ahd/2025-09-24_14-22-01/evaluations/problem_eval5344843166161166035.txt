def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_value = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    alpha = 0.7 if remaining_nodes > 2 else 0.3  # Adjust weights based on remaining nodes
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_value = alpha * distance_to_current + (1 - alpha) * distance_to_destination
        if weighted_value < min_value:
            min_value = weighted_value
            next_node = node
    return next_node

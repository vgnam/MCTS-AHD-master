import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    avg_distance = total_distance / remaining_nodes if remaining_nodes > 0 else 0
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

    for node in unvisited_nodes:
        if node == destination_node:
            continue

        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]

        if immediate_distance == 0 or potential_distance == 0:
            weighted_cost = float('inf')
        else:
            exploration_factor = (1 - centrality) * (1 - (remaining_nodes / len(distance_matrix))) if remaining_nodes > 0 else 0
            weighted_immediate = immediate_distance * (1 + centrality)
            weighted_potential = potential_distance * exploration_factor
            weighted_cost = weighted_immediate + weighted_potential

        deviation = abs(immediate_distance - avg_distance)
        cost = weighted_cost + deviation * (0.5 + exploration_factor)

        if cost < min_cost:
            min_cost = cost
            next_node = node

    if next_node is None:
        next_node = destination_node

    return next_node

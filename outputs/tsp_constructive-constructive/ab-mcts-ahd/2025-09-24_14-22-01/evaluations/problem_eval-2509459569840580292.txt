def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    if remaining_nodes == 0:
        return destination_node
    weight_local = 1.0 if remaining_nodes > 2 else 0.3
    weight_global = 0.5 if remaining_nodes > 2 else 0.7
    avg_unvisited_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes
    for node in unvisited_nodes:
        local_distance = distance_matrix[current_node][node]
        global_distance = distance_matrix[node][destination_node]
        score = (weight_local * local_distance) + (weight_global * global_distance) + (0.2 * avg_unvisited_distance)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    if remaining_nodes == 1:
        return unvisited_nodes[0]
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        distance_ratio = immediate_distance / (destination_distance + 1e-10)
        penalty = max(0, immediate_distance - 1.5 * avg_distance)
        weight = 1.0 / (remaining_nodes ** 0.5)
        score = (1 - weight) * immediate_distance + weight * destination_distance + penalty
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    min_cost = float('inf')
    next_node = None
    centrality_weight = 1.0 + (1.0 / (remaining_nodes + 1))  # Higher weight as tour progresses

    for node in unvisited_nodes:
        if node == destination_node:
            continue

        immediate_distance = distance_matrix[current_node][node]
        node_centrality = sum(1 / (distance_matrix[node][other] + 1e-6) for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1)
        cost = immediate_distance * (1 + centrality_weight * node_centrality)

        if cost < min_cost:
            min_cost = cost
            next_node = node

    if next_node is None:
        next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = [node for node in unvisited_nodes if node != destination_node]
    for node in remaining_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        if remaining_nodes:
            avg_distance = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        else:
            avg_distance = 0
        weighted_cost = 0.4 * immediate_distance + 0.3 * potential_distance + 0.3 * avg_distance
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

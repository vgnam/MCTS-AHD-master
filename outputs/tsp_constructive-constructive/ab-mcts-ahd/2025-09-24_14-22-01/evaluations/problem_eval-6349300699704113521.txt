def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_total_distance = float('inf')
    next_node = None
    alpha = 0.6  # Weight for distance to current node
    beta = 0.4   # Weight for distance to destination node
    gamma = 0.1  # Penalty weight for visited nodes
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        total_distance = alpha * distance_to_current + beta * distance_to_destination + gamma * (1 if node in unvisited_nodes else 0)
        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        connectivity_heuristic = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        total_score = 0.5 * distance_to_current + 0.2 * distance_to_destination + 0.3 * connectivity_heuristic
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_score = float('inf')
    next_node = None
    remaining_nodes = unvisited_nodes - {destination_node}
    if not remaining_nodes:
        return destination_node

    remaining_count = len(remaining_nodes)
    weight_immediate = 0.7 - (0.3 * (remaining_count / (remaining_count + 1)))
    weight_potential = 0.3 + (0.3 * (remaining_count / (remaining_count + 1)))

    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in remaining_nodes) / remaining_count

    for node in remaining_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        combined_score = (weight_immediate * immediate_distance * (1 + centrality)) + (weight_potential * potential_distance * (1 - centrality))
        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    if next_node is None:
        next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    alpha = 0.5  # Tunable parameter
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1  # +1 for the destination
    min_weighted_product = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        total_product = current_to_node * node_to_dest
        weight = 1 + alpha * (remaining_nodes / total_nodes)
        weighted_product = total_product * weight
        if weighted_product < min_weighted_product:
            min_weighted_product = weighted_product
            next_node = node
    return next_node

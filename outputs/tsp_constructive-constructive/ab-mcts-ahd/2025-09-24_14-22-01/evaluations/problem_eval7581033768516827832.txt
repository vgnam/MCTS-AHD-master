def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_immediate = 0.8
    weight_factor = (remaining_nodes + 1) / (remaining_nodes + 2)
    penalty_factor = 1.0 / (remaining_nodes + 1)

    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        penalty = penalty_factor * potential_distance
        weighted_cost = weight_immediate * immediate_distance + (potential_distance ** weight_factor) - penalty
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

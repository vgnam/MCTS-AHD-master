def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    destination_proximity = 1 / (1 + distance_matrix[current_node][destination_node]) if remaining_nodes > 1 else 0
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        diversity_factor = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1 if len(unvisited_nodes) > 1 else 1)
        novelty_factor = 1 / (1 + sum(1 for other in unvisited_nodes if distance_matrix[node][other] < distance_matrix[current_node][other]))
        weighted_cost = (0.3 * immediate_distance) + (0.4 * potential_distance) + (0.2 * diversity_factor) + (0.1 * novelty_factor)
        weighted_cost *= (1 + destination_proximity * 0.5)
        if weighted_cost < min_cost:
            min_cost = weighted_cost
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

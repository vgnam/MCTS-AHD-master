def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_combined_distance = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = 1.0 / remaining_nodes if remaining_nodes > 0 else 1.0
        combined_distance = (1 - weight) * distance_to_current + weight * distance_to_destination
        if combined_distance < min_combined_distance:
            min_combined_distance = combined_distance
            next_node = node
    return next_node

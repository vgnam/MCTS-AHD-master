def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    min_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight = 0.3 + 0.4 * (remaining_nodes / len(distance_matrix))
    centrality = sum(1 / (distance_matrix[current_node][node] + 1e-6) for node in unvisited_nodes) / len(unvisited_nodes)
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        weighted_immediate = immediate_distance * (2 + centrality)
        weighted_potential = potential_distance * (0.5 - centrality)
        combined_weight = weighted_immediate + (weighted_potential * weight)
        if combined_weight < min_cost:
            min_cost = combined_weight
            next_node = node
    if next_node is None:
        next_node = destination_node
    return next_node

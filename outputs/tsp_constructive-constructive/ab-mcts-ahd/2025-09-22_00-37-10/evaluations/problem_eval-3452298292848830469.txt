import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    temperature = len(unvisited_nodes) / len(distance_matrix)  # Normalized temperature
    candidate_nodes = []
    scores = []

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        penalty = 1.0 / (distance_matrix[node][destination_node] + 1e-6)  # Penalty for nodes closer to destination
        score = (1 - temperature) * (1.0 / (distance + 1e-6)) + temperature * random.random() - penalty
        scores.append(score)
        candidate_nodes.append(node)

    if candidate_nodes:
        next_node = candidate_nodes[scores.index(max(scores))]

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5):
    candidate_nodes = []
    scores = []

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_score = (1 - alpha) * distance_to_current + alpha * distance_to_destination
        candidate_nodes.append(node)
        scores.append(combined_score)

    if not candidate_nodes:
        return None

    weights = [1.0 / (score + 1e-6) for score in scores]
    weights = [w / sum(weights) for w in weights]
    next_node = random.choices(candidate_nodes, weights=weights, k=1)[0]

    return next_node

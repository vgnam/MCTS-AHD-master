def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    visited_recently = set()

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        bias_factor = 1 / (1 + math.log(len(unvisited_nodes) + 1))
        penalty = sum(distance_matrix[node][v] for v in visited_recently) if visited_recently else 0
        score = -current_distance + bias_factor * dest_distance - penalty

        if score > best_score:
            best_score = score
            next_node = node

    if next_node is not None:
        visited_recently.add(next_node)

    return next_node

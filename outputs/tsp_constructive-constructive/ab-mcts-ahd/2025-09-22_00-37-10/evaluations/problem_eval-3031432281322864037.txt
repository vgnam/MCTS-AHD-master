def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    weight = 0.5  # Static weight from No.1
    bias_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)  # Dynamic bias from No.2

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        score = -current_distance + bias_factor * (weight * dest_distance)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

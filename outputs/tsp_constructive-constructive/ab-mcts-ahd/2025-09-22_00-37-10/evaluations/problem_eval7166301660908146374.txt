def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')

    total_unvisited = len(unvisited_nodes)
    if total_unvisited == 0:
        return destination_node

    avg_distance_to_unvisited = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / total_unvisited
    bias_factor = 1.0 - (total_unvisited / (total_unvisited + 1)) ** 2

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        normalized_current = current_distance / avg_distance_to_unvisited
        score = -normalized_current + bias_factor * (1 / (1 + dest_distance))

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

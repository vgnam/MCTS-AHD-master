def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        distance_ratio = current_distance / (dest_distance + 1e-10)  # Avoid division by zero
        bias_factor = 1 / (1 + math.exp(-0.1 * (remaining_nodes - 5)))  # Sigmoid transition
        score = -current_distance + bias_factor * (dest_distance / (1 + distance_ratio))

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_unvisited = len(unvisited_nodes)
    progress_ratio = (total_unvisited - len(unvisited_nodes)) / total_unvisited if total_unvisited > 0 else 0.0

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        dynamic_weight = 0.4 + 0.2 * (1 - progress_ratio)
        score = (1 - dynamic_weight) * immediate_distance + dynamic_weight * lookahead_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

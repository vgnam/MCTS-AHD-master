def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_distance = float('inf')
    candidate_nodes = []
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if immediate_distance < min_distance:
            min_distance = immediate_distance
            candidate_nodes = [node]
        elif immediate_distance == min_distance:
            candidate_nodes.append(node)

    if len(candidate_nodes) > 1:
        best_score = float('inf')
        for node in candidate_nodes:
            lookahead_distance = distance_matrix[node][destination_node]
            weight = 1.0 / (1.0 + remaining_nodes) + 0.5
            score = immediate_distance - weight * lookahead_distance
            if score < best_score:
                best_score = score
                next_node = node
    else:
        next_node = candidate_nodes[0]

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes) if len(unvisited_nodes) > 1 else 0
        score = 0.7 * distance_to_current + 0.1 * distance_to_destination + 0.2 * avg_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

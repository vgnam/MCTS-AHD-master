import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_distance = float('inf')
    candidate_nodes = []

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_distance = distance_to_current + distance_to_destination

        if combined_distance < min_distance:
            min_distance = combined_distance
            candidate_nodes = [node]
        elif combined_distance == min_distance:
            candidate_nodes.append(node)

    if len(candidate_nodes) > 1:
        weights = [1.0 / (distance_matrix[current_node][node] + distance_matrix[node][destination_node] + 1e-6) for node in candidate_nodes]
        weights = [w / sum(weights) for w in weights]
        next_node = random.choices(candidate_nodes, weights=weights, k=1)[0]
    else:
        next_node = candidate_nodes[0]

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    base_weight = 0.5 * (1.0 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))) / (1.0 + remaining_nodes)
    random_weight = 0.1 * random.random()
    weight = base_weight + random_weight

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        score = immediate_distance - weight * lookahead_distance + random_weight * random.random()

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    distance_to_destination = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        detour_penalty = max(0, distance_to_current + distance_to_dest - distance_to_destination)

        if distance_to_destination > 0:
            weight = 0.5 * (1 - (distance_to_dest / distance_to_destination))
        else:
            weight = 0.5

        score = distance_to_current + weight * distance_to_dest + 0.2 * detour_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_weight = -0.5 * (remaining_nodes / len(distance_matrix))  # Dynamic lookahead weight
        score = immediate_distance + lookahead_weight * distance_matrix[node][destination_node]

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

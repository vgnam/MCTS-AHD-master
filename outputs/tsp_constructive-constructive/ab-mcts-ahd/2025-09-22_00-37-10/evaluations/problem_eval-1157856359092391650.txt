def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    avg_distance_to_unvisited = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes
    weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1))) * (0.9 ** (1 / remaining_nodes))

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        avg_adjusted_score = (immediate_distance + lookahead_distance) / avg_distance_to_unvisited
        score = immediate_distance + weight * lookahead_distance * (1 + avg_adjusted_score)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

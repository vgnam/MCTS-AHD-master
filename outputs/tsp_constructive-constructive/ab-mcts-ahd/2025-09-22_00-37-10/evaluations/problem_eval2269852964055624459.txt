import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    max_score = -float('inf')
    candidate_nodes = []

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        utility_score = (1 / (distance_to_node + 1e-6)) + (0.5 * (1 / (distance_to_dest + 1e-6)))

        if utility_score > max_score:
            max_score = utility_score
            candidate_nodes = [node]
        elif utility_score == max_score:
            candidate_nodes.append(node)

    if len(candidate_nodes) > 1:
        weights = [(1 / (distance_matrix[current_node][node] + 1e-6)) for node in candidate_nodes]
        weights = [w / sum(weights) for w in weights]
        next_node = random.choices(candidate_nodes, weights=weights, k=1)[0]
    else:
        next_node = candidate_nodes[0]

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    dynamic_weight = 0.3 + (1.0 - 0.3) * (1.0 - remaining_nodes / (remaining_nodes + 1))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        detour_penalty = distance_to_current + distance_to_destination - distance_matrix[current_node][destination_node]

        score = distance_to_current + dynamic_weight * distance_to_destination + 0.5 * detour_penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

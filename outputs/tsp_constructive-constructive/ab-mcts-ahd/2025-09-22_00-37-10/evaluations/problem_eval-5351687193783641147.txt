def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    total_unvisited = len(unvisited_nodes)
    remaining_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        progress_factor = (total_unvisited - len(unvisited_nodes)) / total_unvisited
        bias_factor = 1 - progress_factor**2
        time_decay = (total_unvisited / (total_unvisited + 1))**2
        score = (-current_distance * (1 + time_decay)) + (bias_factor * dest_distance)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

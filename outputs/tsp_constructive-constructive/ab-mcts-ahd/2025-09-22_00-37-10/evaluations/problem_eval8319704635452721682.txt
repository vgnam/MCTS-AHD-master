def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        proximity_factor = 1 - (immediate_distance / (avg_distance + 1e-6))
        weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))
        score = immediate_distance + weight * lookahead_distance + (1 - weight) * proximity_factor

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    # Calculate mean and variance of distances from current_node to unvisited nodes
    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    mean_distance = sum(distances) / remaining_nodes
    variance = sum((d - mean_distance) ** 2 for d in distances) / remaining_nodes

    # Calculate centrality of each node as average distance to other unvisited nodes
    centrality = {}
    for node in unvisited_nodes:
        centrality[node] = sum(distance_matrix[node][other] for other in unvisited_nodes) / remaining_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        centrality_score = centrality[node]
        dynamic_weight = (1.0 / (remaining_nodes + 1)) * (1 + variance) * (1 + (1 / (centrality_score + 1e-6)))

        score = immediate_distance + dynamic_weight * lookahead_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    progress_ratio = 1 - (len(unvisited_nodes) / total_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        bias_factor = 0.8 - 0.6 * progress_ratio

        immediate_weight = 0.8 - 0.6 * progress_ratio
        lookahead_weight = 0.2 + 0.6 * progress_ratio

        score = bias_factor * (immediate_weight * immediate_distance + lookahead_weight * lookahead_distance)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

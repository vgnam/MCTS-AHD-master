import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    candidate_nodes = []
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_score = distance_to_current + 0.5 * distance_to_destination  # Adjust weight for destination bias

        if combined_score < min_score:
            min_score = combined_score
            candidate_nodes = [node]
        elif combined_score == min_score:
            candidate_nodes.append(node)

    if len(candidate_nodes) > 1:
        weights = [1.0 / (distance_matrix[current_node][node] + 1e-6) for node in candidate_nodes]
        weights = [w / sum(weights) for w in weights]
        next_node = random.choices(candidate_nodes, weights=weights, k=1)[0]
    else:
        next_node = candidate_nodes[0]

    return next_node

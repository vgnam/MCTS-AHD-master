def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    visit_frequency = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        bias_factor = (len(unvisited_nodes) / (len(unvisited_nodes) + 1)) * (1 - visit_frequency[node] / (sum(visit_frequency.values()) + 1))
        score = -current_distance + bias_factor * dest_distance

        if score > best_score:
            best_score = score
            next_node = node

    visit_frequency[next_node] += 1
    return next_node

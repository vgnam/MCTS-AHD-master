def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        priority_factor = 1 + (len(unvisited_nodes) / (len(unvisited_nodes) + 1)) ** 2
        harmonic_mean = (2 * current_distance * dest_distance) / (current_distance + dest_distance)
        score = -harmonic_mean * priority_factor

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

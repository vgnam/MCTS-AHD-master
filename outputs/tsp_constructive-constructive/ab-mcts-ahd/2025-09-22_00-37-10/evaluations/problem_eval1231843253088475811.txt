import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    temperature = max(0.1, 1.0 - 0.5 * (1.0 - remaining_nodes / len(distance_matrix)))

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        lookahead_penalty = (len(unvisited_nodes) - 1) * min(immediate_distance, lookahead_distance)
        score = immediate_distance + lookahead_distance + lookahead_penalty

        if random.random() < temperature:
            next_node = node
            break
        elif score < best_score:
            best_score = score
            next_node = node

    return next_node

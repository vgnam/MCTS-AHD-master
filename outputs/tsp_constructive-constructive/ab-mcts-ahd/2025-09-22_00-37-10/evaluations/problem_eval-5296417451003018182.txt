def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_unvisited = len(unvisited_nodes)
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / total_unvisited if total_unvisited > 0 else 0

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        weight_immediate = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))
        weight_lookahead = 1 - weight_immediate
        penalty = (total_unvisited - remaining_nodes) / total_unvisited if total_unvisited > 0 else 0
        score = immediate_distance + weight_lookahead * lookahead_distance - penalty * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

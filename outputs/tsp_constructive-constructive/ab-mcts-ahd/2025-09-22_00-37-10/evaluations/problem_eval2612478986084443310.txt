def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if not unvisited_nodes:
        return destination_node

    centroid = sum(unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        centroid_distance = abs(node - centroid)
        weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))
        penalty = 0.1 * (len(unvisited_nodes) - remaining_nodes)
        score = immediate_distance + weight * lookahead_distance + penalty * centroid_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    candidate_nodes = []
    min_distance = float('inf')

    # Find nearest neighbors (same as original)
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            candidate_nodes = [node]
        elif distance == min_distance:
            candidate_nodes.append(node)

    # Calculate heuristic scores (new addition)
    heuristic_scores = []
    for node in candidate_nodes:
        # Inverse distance component
        inv_distance = 1.0 / (distance_matrix[current_node][node] + 1e-6)

        # Heuristic estimate: sum of distances to remaining nodes
        heuristic = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)

        # Combined score (weighted sum)
        score = 0.7 * inv_distance + 0.3 * (1.0 / (heuristic + 1e-6))
        heuristic_scores.append(score)

    # Normalize scores to probabilities
    total_score = sum(heuristic_scores)
    probabilities = [s / total_score for s in heuristic_scores]

    # Select node with weighted random choice
    next_node = random.choices(candidate_nodes, weights=probabilities, k=1)[0]

    return next_node

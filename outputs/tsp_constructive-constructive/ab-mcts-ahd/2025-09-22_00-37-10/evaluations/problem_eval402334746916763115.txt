    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]

        # Calculate average distance to other unvisited nodes (hub centrality)
        avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (total_unvisited - 1) if total_unvisited > 1 else 0

        score = -current_distance + bias_factor * (dest_distance + hub_factor * avg_distance)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

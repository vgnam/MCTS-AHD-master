def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1))) + (0.2 * (lookahead_distance / max(distance_matrix[node])))
        penalty = 0.1 * (1 - (remaining_nodes / total_nodes)) if node in unvisited_nodes else 0
        score = immediate_distance + weight * lookahead_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

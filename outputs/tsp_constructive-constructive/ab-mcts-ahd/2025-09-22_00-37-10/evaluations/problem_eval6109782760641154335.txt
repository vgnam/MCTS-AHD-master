def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.3, decay_factor=0.9):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = distance_to_current + alpha * distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node

    alpha *= decay_factor
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    average_unvisited_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes
    weight = 1.0 / (1.0 + remaining_nodes) ** 2  # Non-linear weight decay

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        score = immediate_distance - weight * (lookahead_distance + 0.3 * average_unvisited_distance)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    base_weight = 1.0 / (1.0 + remaining_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        penalty = 0.0

        if immediate_distance < 0.1 * distance_matrix[current_node][destination_node]:
            penalty = 0.5 * (distance_matrix[current_node][destination_node] - immediate_distance)

        score = immediate_distance - base_weight * lookahead_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]

        if total_unvisited > 1:
            remaining_ratio = (total_unvisited - 1) / total_unvisited
            bias_factor = remaining_ratio ** 2
        else:
            bias_factor = 0

        score = -current_distance + bias_factor * (1 - dest_distance / (max(distance_matrix[node]) + 1e-6))

        if score > best_score or (score == best_score and dest_distance < distance_matrix[next_node][destination_node]):
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]

        # Find the next closest node after the candidate
        next_closest_distance = float('inf')
        for next_node_candidate in unvisited_nodes:
            if next_node_candidate != node:
                next_distance = distance_matrix[node][next_node_candidate]
                if next_distance < next_closest_distance:
                    next_closest_distance = next_distance

        score = immediate_distance + 0.3 * next_closest_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

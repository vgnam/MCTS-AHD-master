def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes)
    avg_distance_to_dest = total_distance_to_dest / remaining_nodes if remaining_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = 0.3 * (1 - (remaining_nodes / (remaining_nodes + 1))) + 0.7 * (distance_to_destination / (avg_distance_to_dest + 1e-6))
        score = distance_to_current + weight * distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

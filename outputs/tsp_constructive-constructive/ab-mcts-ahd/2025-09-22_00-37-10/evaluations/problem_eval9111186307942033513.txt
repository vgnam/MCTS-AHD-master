def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    # Calculate mean and variance of immediate distances
    immediate_distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    mean_distance = sum(immediate_distances) / remaining_nodes
    variance = sum((d - mean_distance) ** 2 for d in immediate_distances) / remaining_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]

        # Dynamic weight adjustment based on remaining nodes and variance
        weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1))) + 0.2 * (variance / (mean_distance + 1e-6))

        score = immediate_distance + weight * lookahead_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

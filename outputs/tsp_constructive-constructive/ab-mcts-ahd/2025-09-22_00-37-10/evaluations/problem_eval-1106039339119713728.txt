import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    max_ratio = -float('inf')
    candidate_nodes = []

    for node in unvisited_nodes:
        dist_to_current = distance_matrix[current_node][node]
        dist_to_dest = distance_matrix[node][destination_node]
        ratio = dist_to_dest / (dist_to_current + 1e-6)
        if ratio > max_ratio:
            max_ratio = ratio
            candidate_nodes = [node]
        elif ratio == max_ratio:
            candidate_nodes.append(node)

    if len(candidate_nodes) > 1:
        weights = []
        for node in candidate_nodes:
            dist_to_current = distance_matrix[current_node][node]
            weights.append(1.0 / (dist_to_current + 1e-6))
        weights = [w / sum(weights) for w in weights]
        next_node = random.choices(candidate_nodes, weights=weights, k=1)[0]
    else:
        next_node = candidate_nodes[0]

    return next_node

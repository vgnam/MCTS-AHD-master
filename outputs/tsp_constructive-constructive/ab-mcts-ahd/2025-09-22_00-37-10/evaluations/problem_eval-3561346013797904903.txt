def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    # Precompute average lookahead distance for unvisited nodes
    avg_lookahead = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        lookahead_distance = distance_matrix[node][destination_node]
        dynamic_weight = (1.0 / (remaining_nodes + 1)) * (1 + (lookahead_distance / (avg_lookahead + 1e-6)))
        score = immediate_distance + dynamic_weight * lookahead_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

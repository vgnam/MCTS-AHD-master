def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    min_weighted_sum = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_destination = distance_matrix[node][destination_node]
        weight = 1.0 / (1.0 + remaining_nodes)  # Adjust weight based on remaining nodes
        weighted_sum = (current_to_node * (1 - weight)) + (node_to_destination * weight)
        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
    return next_node

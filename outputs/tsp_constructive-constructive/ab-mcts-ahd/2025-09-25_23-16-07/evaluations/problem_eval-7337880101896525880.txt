def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return unvisited_nodes.pop()
    avg_distances = {node: sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes) for node in unvisited_nodes}
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] / avg_distances[node])
    return next_node

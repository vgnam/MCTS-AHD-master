def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    if len(unvisited_nodes) == 1:
        return unvisited_nodes.pop()

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    avg_distance = sum(distances) / len(distances)

    min_cost = float('inf')
    next_node = None
    node_scores = {}

    total_nodes = len(distance_matrix)
    phase_factor = 1 - (0.5 ** ((total_nodes - len(unvisited_nodes)) / total_nodes))

    for node in unvisited_nodes:
        cost = distance_matrix[current_node][node]
        centrality = sum(distance_matrix[node]) / (total_nodes - 1)
        weighted_score = (1 - phase_factor) * cost + phase_factor * (centrality / avg_distance)

        node_scores[node] = weighted_score

        if cost < min_cost:
            min_cost = cost
            next_node = node

    if len(unvisited_nodes) > 2:
        sorted_nodes = sorted(node_scores.items(), key=lambda x: x[1])
        candidate_nodes = [node for node, score in sorted_nodes if score <= 1.2 * min_cost]

        if candidate_nodes:
            best_node = candidate_nodes[0]
            for node in candidate_nodes:
                if node == best_node:
                    continue
                potential_cost = distance_matrix[current_node][node] + distance_matrix[node][best_node]
                if potential_cost < distance_matrix[current_node][best_node] * 1.2:
                    best_node = node
                    break
            next_node = best_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    heuristic_weight = 0.3 * (remaining_nodes / total_nodes)
    penalty = 0.1 if current_node in unvisited_nodes else 0

    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic_value = distance_matrix[node][destination_node] if destination_node != node else 0
        weighted_score = distance + heuristic_value * heuristic_weight - penalty

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        destination_urgency = 1.0 / (distance_matrix[current_node][destination_node] + 1e-6)
        min_score = -float('inf')
        next_node = None

        for node in unvisited_nodes:
            if node == destination_node:
                score = float('inf')
            else:
                distance = distance_matrix[current_node][node]
                score = (1.0 / (distance + 1e-6)) * (1 + destination_urgency)

            if score > min_score:
                min_score = score
                next_node = node

        return next_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node

    return next_node

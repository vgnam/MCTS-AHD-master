def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    total_unvisited = len(unvisited_nodes)
    remaining_distance_to_dest = distance_matrix[current_node][destination_node]
    exploration_factor = 1.0 / (total_unvisited + 1)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        proximity_to_dest = 1 / (1 + distance_matrix[node][destination_node])
        dynamic_weight = (1 - exploration_factor) * 0.5 + exploration_factor * 0.9
        score = distance_to_current * (1 - dynamic_weight * proximity_to_dest)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

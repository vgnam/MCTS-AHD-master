def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        min_ratio = float('inf')
        next_node = None

        for node in unvisited_nodes:
            if node == destination_node:
                continue
            distance_to_node = distance_matrix[current_node][node]
            distance_to_dest = distance_matrix[node][destination_node]
            ratio = distance_to_node / (distance_to_dest + 1e-10)  # Avoid division by zero

            if ratio < min_ratio:
                min_ratio = ratio
                next_node = node

        return next_node if next_node is not None else destination_node
    else:
        min_distance = float('inf')
        next_node = None

        for node in unvisited_nodes:
            if distance_matrix[current_node][node] < min_distance:
                min_distance = distance_matrix[current_node][node]
                next_node = node

        return next_node

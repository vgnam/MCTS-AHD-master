def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    if len(unvisited_nodes) == 1:
        return unvisited_nodes.pop()

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    min_distance = min(distances)
    max_distance = max(distances)

    weighted_threshold = min_distance + (max_distance - min_distance) * (0.5 / len(unvisited_nodes))

    candidates = []
    for node in unvisited_nodes:
        cost = distance_matrix[current_node][node]
        proximity_to_dest = 1 / (1 + distance_matrix[node][destination_node])
        dynamic_weight = 0.5 + 0.5 * (1 - (len(unvisited_nodes) / (len(unvisited_nodes) + 1)))
        score = cost + (dynamic_weight * proximity_to_dest)
        if cost <= weighted_threshold:
            candidates.append((node, score))

    if candidates:
        candidates.sort(key=lambda x: x[1])
        best_node = candidates[0][0]
        if len(unvisited_nodes) > 2:
            threshold = min_distance * (1.0 + 0.1 * (len(unvisited_nodes) / len(distance_matrix)))
            for node, score in candidates[1:]:
                potential_cost = score + (dynamic_weight * proximity_to_dest)
                if score <= threshold and potential_cost < min_distance * 1.2:
                    best_node = node
                    break
        return best_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    avg_distance = sum(distances) / len(distances)
    max_distance = max(distances)

    if len(unvisited_nodes) > 2:
        threshold = max_distance * 0.8
        candidates = [node for node in unvisited_nodes if distance_matrix[current_node][node] >= threshold]
        if candidates:
            return candidates[0]

    next_node = unvisited_nodes[0]
    min_distance = distance_matrix[current_node][next_node]
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node

    return next_node

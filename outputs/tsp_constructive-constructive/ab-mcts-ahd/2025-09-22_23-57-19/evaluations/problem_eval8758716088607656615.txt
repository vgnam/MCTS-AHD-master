def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    if len(unvisited_nodes) == 1:
        return unvisited_nodes.pop()

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    avg_distance = sum(distances) / len(distances)

    min_cost = min(distances)
    min_cost_node = unvisited_nodes[distances.index(min_cost)]

    escape_prob = 1 / len(unvisited_nodes)
    if random.random() < escape_prob:
        threshold = avg_distance / min_cost if min_cost != 0 else 1.2
        candidates = [node for node in unvisited_nodes if distance_matrix[current_node][node] < avg_distance * threshold]
        if candidates:
            return random.choice(candidates)

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        distance_to_dest = distance_matrix[current_node][destination_node]
        if distance_to_dest <= 2 * sum(distance_matrix[current_node]) / len(distance_matrix[current_node]):
            return destination_node

    max_distance = -float('inf')
    next_node = None

    for node in unvisited_nodes:
        if distance_matrix[current_node][node] > max_distance:
            max_distance = distance_matrix[current_node][node]
            next_node = node

    return next_node

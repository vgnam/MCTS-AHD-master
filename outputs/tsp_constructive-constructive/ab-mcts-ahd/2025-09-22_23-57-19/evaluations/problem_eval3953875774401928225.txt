def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    if len(unvisited_nodes) == 1:
        return unvisited_nodes.pop()

    max_cost = -float('inf')
    min_cost = float('inf')
    farthest_node = None
    nearest_node = None

    for node in unvisited_nodes:
        cost = distance_matrix[current_node][node]
        if cost > max_cost:
            max_cost = cost
            farthest_node = node
        if cost < min_cost:
            min_cost = cost
            nearest_node = node

    if farthest_node is None or nearest_node is None:
        return unvisited_nodes.pop()

    threshold = 0.5 + 0.5 * (len(unvisited_nodes) / len(distance_matrix))

    if len(unvisited_nodes) > 2:
        for node in unvisited_nodes:
            if node == farthest_node or node == nearest_node:
                continue
            potential_cost = distance_matrix[current_node][node] + distance_matrix[node][farthest_node]
            if potential_cost > max_cost * threshold:
                farthest_node = node
                break

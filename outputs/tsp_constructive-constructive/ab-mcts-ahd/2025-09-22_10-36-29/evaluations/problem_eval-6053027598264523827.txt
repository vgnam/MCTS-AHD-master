def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    candidates = []
    for node in unvisited_nodes:
        if node == destination_node:
            return node

        current_distance = distance_matrix[current_node][node]
        total_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        avg_distance = total_distance / (len(unvisited_nodes) - 1) if (len(unvisited_nodes) - 1) > 0 else 0
        score = current_distance + 0.3 * avg_distance

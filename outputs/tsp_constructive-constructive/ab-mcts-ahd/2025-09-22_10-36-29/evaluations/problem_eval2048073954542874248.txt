import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    temperature = len(unvisited_nodes) / len(distance_matrix)
    if temperature < 0.1:
        temperature = 0.1

    candidates = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            candidates.append((node, distance, 1.0))
        else:
            candidates.append((node, distance, math.exp(-distance / temperature)))

    candidates.sort(key=lambda x: x[1])
    probabilities = [x[2] for x in candidates]
    total_prob = sum(probabilities)
    normalized_probs = [p / total_prob for p in probabilities]

    selected_node = random.choices([x[0] for x in candidates], weights=normalized_probs, k=1)[0]

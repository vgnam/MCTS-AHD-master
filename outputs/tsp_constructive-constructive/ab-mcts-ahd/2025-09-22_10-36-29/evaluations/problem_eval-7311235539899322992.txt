def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        if node == destination_node:
            return node

        current_distance = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            total_score = current_distance
        else:
            future_distances = [distance_matrix[node][next_node] for next_node in unvisited_nodes - {node}]
            avg_future_distance = sum(future_distances) / len(future_distances)
            total_score = current_distance + avg_future_distance

        if total_score < best_score:
            best_score = total_score
            best_node = node

    return next_node

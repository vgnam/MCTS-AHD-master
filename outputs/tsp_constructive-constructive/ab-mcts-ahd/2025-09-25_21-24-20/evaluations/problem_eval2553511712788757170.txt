def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        min_distance_to_unvisited = min(distance_matrix[node][n] for n in unvisited_nodes if n != node) if len(unvisited_nodes) > 1 else 0
        dest_distance = distance_matrix[node][destination_node]
        score = distance_to_node + min_distance_to_unvisited + 0.5 * dest_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

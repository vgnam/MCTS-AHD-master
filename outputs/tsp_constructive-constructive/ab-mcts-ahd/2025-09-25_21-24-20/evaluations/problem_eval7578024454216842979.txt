def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    alpha = 0.7  # Weighting factor for balancing local and global criteria
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        local_distance = distance_matrix[current_node][node]
        global_estimate = distance_matrix[node][destination_node] if node != destination_node else 0
        score = alpha * local_distance + (1 - alpha) * global_estimate

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

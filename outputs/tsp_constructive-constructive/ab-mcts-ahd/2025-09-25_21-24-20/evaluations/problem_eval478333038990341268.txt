def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        potential_to_destination = distance_matrix[node][destination_node]
        min_distance_to_unvisited = min(distance_matrix[node][n] for n in unvisited_nodes if n != node) if len(unvisited_nodes) > 1 else 0
        balance_factor = 1 - (remaining_nodes / (remaining_nodes + 1))
        dynamic_weight = 1 - (0.5 * (remaining_nodes / (remaining_nodes + 1)))
        score = (balance_factor * distance_to_node) + (dynamic_weight * min_distance_to_unvisited) + ((1 - balance_factor - dynamic_weight) * potential_to_destination)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        distance_to_destination = distance_matrix[node][destination_node]

        weight = 0.5 + 0.5 * (num_unvisited / (num_unvisited + 1))
        score = (1 - weight) * distance_to_node + weight * avg_future_distance
        score += 0.2 * distance_to_destination

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        distance_to_dest = distance_matrix[node][destination_node]
        weight_factor = remaining_nodes / (remaining_nodes + 1)
        score = (-1.5 * distance_to_node * weight_factor) + (0.5 * avg_distance * (1 - weight_factor)) + (0.3 * distance_to_dest)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    if not unvisited_nodes:
        return destination_node

    num_unvisited = len(unvisited_nodes)
    weight_immediate = 0.6 if num_unvisited > 1 else 1.0
    weight_average = 0.4 if num_unvisited > 1 else 0.0

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (num_unvisited - 1) if num_unvisited > 1 else 0
        score = weight_immediate * distance_to_node + weight_average * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, visited_count=None):
    next_node = None
    best_score = -float('inf')

    if visited_count is None:
        visited_count = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = distance_matrix[node][destination_node]
        revisit_penalty = visited_count.get(node, 0)

        score = -1.5 * distance_to_node + 1.0 * avg_distance_to_dest - 0.5 * revisit_penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

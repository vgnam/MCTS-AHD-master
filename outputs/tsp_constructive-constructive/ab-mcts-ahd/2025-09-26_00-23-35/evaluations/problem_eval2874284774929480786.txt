def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        penalty = 0.05 * distance_matrix[node][destination_node] if len(unvisited_nodes) > 1 else 0
        score = 0.7 * distance_to_node + 0.3 * avg_future_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

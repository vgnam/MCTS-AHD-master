def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_others = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        combined_score = distance_to_current + (0.5 * distance_to_destination) + (0.3 * avg_distance_to_others)

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        weight_factor = 1.0 + (0.7 * (len(distance_matrix) - unvisited_count) / (len(distance_matrix) - 2)) if unvisited_count > 1 else 1.5
        destination_factor = (0.5 - weight_factor) * distance_matrix[node][destination_node]

        score = weight_factor * distance_to_node + 0.2 * distance_to_node + destination_factor

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    weight_proximity = 0.7
    weight_centrality = 0.3

    if len(unvisited_nodes) > 2:
        dynamic_weight = 0.5 + 0.5 * (len(unvisited_nodes) / len(unvisited_nodes + [current_node]))
        weight_proximity = dynamic_weight
        weight_centrality = 1 - dynamic_weight

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        centrality_score = avg_distance_to_dest

        penalty = 0
        if len(unvisited_nodes) > 3 and distance_matrix[node][destination_node] < 0.5 * sum(distance_matrix[destination_node][n] for n in unvisited_nodes if n != destination_node) / max(1, len(unvisited_nodes) - 1):
            penalty = 0.2

        score = -weight_proximity * distance_to_node + weight_centrality * centrality_score - penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    weight_proximity = 0.6
    weight_centrality = 0.4
    dynamic_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        centrality_score = avg_distance_to_dest
        distance_to_dest = distance_matrix[node][destination_node]

        adjusted_weight_centrality = weight_centrality + dynamic_factor * (1 - weight_centrality)
        penalty = 0.1 * distance_to_dest if distance_to_dest < 2 * distance_to_node else 0

        score = -weight_proximity * distance_to_node + adjusted_weight_centrality * centrality_score - penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        bias_factor = sum(1 / (distance_matrix[node][u] + 1e-10) for u in unvisited_nodes if u != node)
        adjusted_distance = distance_to_node + (bias_factor * distance_to_destination)

        if adjusted_distance < min_score:
            min_score = adjusted_distance
            next_node = node

    return next_node

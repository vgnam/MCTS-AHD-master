def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1  # +1 to include current node
    immediate_weight = max(0.5, 1.2 * (remaining_nodes / total_nodes))
    long_term_weight = 1.0 - immediate_weight

    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = immediate_weight * distance_to_node + long_term_weight * distance_to_destination

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    weight_proximity = 0.8 * (num_unvisited / (num_unvisited + 1)) + 0.2
    weight_centrality = 0.2 * (num_unvisited / (num_unvisited + 1)) + 0.8

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        dest_distance = distance_matrix[node][destination_node]
        penalty = 0.1 * dest_distance if dest_distance > 2 * avg_distance else 0
        score = weight_proximity * distance_to_node + weight_centrality * avg_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

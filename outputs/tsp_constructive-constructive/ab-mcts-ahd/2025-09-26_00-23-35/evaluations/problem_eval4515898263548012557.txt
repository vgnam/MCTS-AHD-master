def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_other = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        distance_to_destination = distance_matrix[node][destination_node]
        score = 1.0 * (1 / distance_to_node) + 0.5 * (1 / avg_distance_to_other) - 0.5 * distance_to_destination

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

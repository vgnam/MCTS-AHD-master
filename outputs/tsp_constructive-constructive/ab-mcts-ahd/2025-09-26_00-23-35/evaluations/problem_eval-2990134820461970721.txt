def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    next_node = None
    min_distance = float('inf')
    dest_distance = distance_matrix[current_node][destination_node]
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if next_node == destination_node:
        return next_node

    if destination_node in unvisited_nodes:
        ratio = min_distance / (dest_distance + 1e-6)
        sigmoid_bias = 1 / (1 + math.exp(-(ratio - 0.5) * 10))
        prob_threshold = sigmoid_bias * (1.0 / (1 + remaining_nodes))
        if random.random() < prob_threshold:
            next_node = destination_node
        elif random.random() < 0.1 * (1.0 / (1 + remaining_nodes)):
            next_node = random.choice(list(unvisited_nodes))

    return next_node

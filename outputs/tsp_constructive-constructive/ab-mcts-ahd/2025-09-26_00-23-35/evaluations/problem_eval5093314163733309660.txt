def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1

    # Calculate average distance from unvisited nodes to destination
    avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / total_nodes

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight adjustment with centrality consideration
        centrality_factor = sum(distance_matrix[node][other] for other in unvisited_nodes) / total_nodes
        weight_immediate = max(0.3, 1.0 * (remaining_nodes / total_nodes))
        weight_long_term = 1.0 - weight_immediate
        score = (weight_immediate * distance_to_node + weight_long_term * distance_to_destination) * (1 + centrality_factor / avg_distance_to_dest)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

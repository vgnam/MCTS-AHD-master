def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        weight = 0.5 + 0.5 * (math.sqrt(remaining_nodes) / math.sqrt(total_nodes))
        ratio = distance_to_node / (distance_to_destination + 1e-10)
        penalty = 1.0 + (ratio ** 2) if ratio > 1.0 else 1.0
        score = distance_to_node + weight * distance_to_destination * penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

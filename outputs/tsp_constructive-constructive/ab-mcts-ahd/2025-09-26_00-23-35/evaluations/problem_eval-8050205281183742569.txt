def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    visited_nodes = len(distance_matrix) - len(unvisited_nodes)
    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    avg_distance = total_distance / len(unvisited_nodes) if unvisited_nodes else 0

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight_current = 1.0 / (1 + visited_nodes)
        weight_dest = 1.0 - weight_current
        score = weight_current * distance_to_node + weight_dest * distance_to_destination + avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

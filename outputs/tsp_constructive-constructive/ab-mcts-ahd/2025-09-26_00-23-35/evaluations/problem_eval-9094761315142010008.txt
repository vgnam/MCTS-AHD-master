def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic adaptive weighting using sigmoid function
        weight = 1 / (1 + math.exp(-0.1 * remaining_distance))
        score = (1 - weight) * distance_to_node + weight * distance_to_destination

        # Penalize revisiting nearby nodes to avoid cycles
        if distance_to_node < 0.2 * remaining_distance:
            score += 0.3 * distance_to_node

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

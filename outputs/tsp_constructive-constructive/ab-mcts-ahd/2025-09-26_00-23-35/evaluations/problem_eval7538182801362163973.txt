def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)

    # Inverse adaptive weights
    progress_factor = (total_nodes - remaining_nodes) / total_nodes
    weight_proximity = 1.0 - 0.5 * progress_factor
    weight_centrality = 0.5 + 0.5 * progress_factor
    weight_future = 0.3 * progress_factor
    weight_random = 0.2 * progress_factor

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Centrality measure: average distance to all other unvisited nodes
        centrality = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, remaining_nodes - 1)

        # Future potential: weighted by inverse of remaining nodes
        future_potential = distance_to_destination * (1.0 / (remaining_nodes + 1))

        # Probabilistic component scaled by progress
        random_factor = weight_random * (distance_matrix[current_node][node] / max(distance_matrix[current_node]))

        # Combined score with inverse adaptive weights
        score = (weight_proximity * distance_to_node -
                weight_centrality * centrality +
                weight_future * future_potential +
                random_factor)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

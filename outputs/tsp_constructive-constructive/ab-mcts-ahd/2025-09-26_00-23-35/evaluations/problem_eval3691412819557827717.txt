def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    density_factor = 1 - (len(unvisited_nodes) / total_nodes) if total_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        opportunity_cost = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes) if unvisited_nodes else 0
        score = distance_to_node + (0.6 * distance_to_destination * density_factor) + (0.4 * opportunity_cost)

        if score < best_score:
            best_score = best_score
            next_node = node

    return next_node

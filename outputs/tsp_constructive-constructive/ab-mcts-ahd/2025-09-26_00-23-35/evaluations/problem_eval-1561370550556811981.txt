def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)

        proximity_weight = 1.2 * (remaining_nodes / (remaining_nodes + 2))  # More gradual decrease
        avg_weight = 0.8 * (1 - (remaining_nodes / (remaining_nodes + 2)))    # More gradual increase

        # Penalty for high average distances
        penalty = 0.3 * (avg_distance_to_dest / max(1, sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)))

        score = -proximity_weight * distance_to_node + avg_weight * avg_distance_to_dest - penalty

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    # Dynamic weights (70% proximity early, 30% novelty later)
    weight_proximity = 0.7 * (0.6 + 0.4 * (remaining_nodes / (remaining_nodes + 1)))
    weight_novelty = 0.3 * (0.4 * (1 - (remaining_nodes / (remaining_nodes + 1))))

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        future_distances = [distance_matrix[node][n] for n in unvisited_nodes if n != node]

        # Novelty factor: variance of future distances (higher variance = more novel)
        novelty = np.var(future_distances) if len(future_distances) > 1 else 0

        # Penalty for distant nodes when few nodes remain
        penalty = 0.3 * distance_matrix[node][destination_node] if remaining_nodes <= 4 else 0

        # Centrality reward: inverse of average distance to unvisited nodes
        avg_future_distance = sum(future_distances) / max(1, remaining_nodes - 1)
        centrality_reward = -0.2 * avg_future_distance

        # Score with dynamic weights, novelty, penalty, and centrality reward
        score = weight_proximity * distance_to_node + weight_novelty * novelty + penalty + centrality_reward

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

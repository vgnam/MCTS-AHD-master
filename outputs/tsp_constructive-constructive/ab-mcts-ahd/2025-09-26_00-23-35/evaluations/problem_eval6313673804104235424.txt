def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, unvisited_count - 1)
        weight_factor = 1.0 + (0.5 * (unvisited_count - 1) / (len(distance_matrix) - 2)) if unvisited_count > 1 else 1.5
        destination_factor = (1.0 - weight_factor) * distance_matrix[node][destination_node]

        score = weight_factor * distance_to_node + 0.3 * avg_distance + destination_factor

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Heuristic score prioritizes long-term progress with a lower weight on immediate distance
        heuristic_factor = 0.3  # Weight for immediate distance (closer to 0 prioritizes long-term steps)
        score = (heuristic_factor * (1 / distance_to_node)) + ((1 - heuristic_factor) * (1 / distance_to_destination))

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        centrality = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        future_potential = sum(distance_matrix[n][destination_node] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        destination_penalty = distance_matrix[node][destination_node] * 0.15

        if remaining_nodes <= 3:
            weight_immediate = 0.7
            weight_centrality = 0.25
            weight_future = 0.05
        else:
            weight_immediate = 0.5
            weight_centrality = 0.3
            weight_future = 0.2

        score = (weight_immediate * distance_to_node +
                weight_centrality * centrality +
                weight_future * future_potential +
                destination_penalty)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

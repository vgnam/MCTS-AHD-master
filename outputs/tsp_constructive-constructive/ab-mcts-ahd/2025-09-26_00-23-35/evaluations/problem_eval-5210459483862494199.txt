def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_distance = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight adjustment: prioritize long-term optimization as fewer nodes remain
        weight_current = 0.7 * (remaining_nodes / (remaining_nodes + 1))
        weight_destination = 0.3 * (1 - remaining_nodes / (remaining_nodes + 1))
        weighted_distance = weight_current * distance_to_current + weight_destination * distance_to_destination

        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    remaining_nodes = len(unvisited_nodes)
    weight_dest = 0.4 + (0.6 * (1 - remaining_nodes / len(distance_matrix))) if remaining_nodes > 0 else 0.4
    weight_central = 0.2 + (0.3 * (remaining_nodes / len(distance_matrix))) if remaining_nodes > 0 else 0.2

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        score = distance_to_node + weight_dest * distance_to_destination + weight_central * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

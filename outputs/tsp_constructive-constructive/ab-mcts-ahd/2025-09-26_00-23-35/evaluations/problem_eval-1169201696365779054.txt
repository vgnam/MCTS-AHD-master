def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, unvisited_count - 1)
        destination_weight = 3.0 / (1 + 0.1 * unvisited_count)
        score = destination_weight * distance_to_destination + 0.5 * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

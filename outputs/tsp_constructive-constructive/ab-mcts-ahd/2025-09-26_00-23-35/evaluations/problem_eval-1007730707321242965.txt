importance (weight 0.2) as the tour progresses. It penalizes nodes that are outliers in connectivity or distance.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        connectivity_score = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < distance_matrix[node][destination_node]) / max(1, len(unvisited_nodes) - 1)
        destination_alignment = 1 / (1 + distance_matrix[node][destination_node])

        # Dynamic weights
        weight_proximity = 0.6 if remaining_nodes > 5 else 0.7
        weight_connectivity = 0.2 * (1 - remaining_nodes / total_nodes)
        weight_alignment = 0.3 + 0.1 * (1 - remaining_nodes / total_nodes)

        # Penalty for outliers
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        penalty = max(0, distance_to_node - 2 * avg_distance)

        score = (weight_proximity * distance_to_node +
                 weight_connectivity * (1 - connectivity_score) +
                 weight_alignment * destination_alignment +
                 penalty)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        base_weight = 0.5 * (1 - (num_unvisited / (num_unvisited + 2)))
        weight_proximity = base_weight + 0.3 * (num_unvisited / (num_unvisited + 1))
        weight_future = 0.7 * (1 - base_weight) * (1 - (num_unvisited / (num_unvisited + 1)))
        penalty = 1.5 if distance_to_node > 1.5 * distance_to_destination else 1.0
        score = weight_proximity * distance_to_node + weight_future * distance_to_destination * penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

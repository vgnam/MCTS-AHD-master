def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_distance = float('inf')
    dest_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if node == destination_node:
            next_node = node
            break
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if next_node != destination_node and len(unvisited_nodes) > 1:
        distance_ratio = min_distance / (dest_distance + 1e-6)
        prob_destination = 1.0 / (1.0 + distance_ratio)
        threshold = 0.5 + 0.5 * (len(unvisited_nodes) / len(unvisited_nodes + [current_node]))
        if prob_destination > threshold and destination_node in unvisited_nodes:
            next_node = destination_node

    return next_node

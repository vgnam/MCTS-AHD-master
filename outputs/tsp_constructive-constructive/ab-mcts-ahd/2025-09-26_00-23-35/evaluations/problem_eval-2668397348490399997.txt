def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    base_weight_centrality = 0.6
    base_weight_proximity = 0.4
    dynamic_factor = len(unvisited_nodes) / max(1, len(distance_matrix))
    weight_centrality = base_weight_centrality * dynamic_factor
    weight_proximity = base_weight_proximity * (1 - dynamic_factor)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        centrality_score = avg_distance_to_dest
        distance_to_dest = distance_matrix[node][destination_node]

        penalty = 0.4 if distance_to_dest < 0.25 * max(distance_matrix[node][n] for n in unvisited_nodes) else 0
        score = weight_centrality * centrality_score - weight_proximity * distance_to_node - penalty

        if score > best_score:
            best_score = score
            next_node = next_node if next_node is not None else node

    return next_node

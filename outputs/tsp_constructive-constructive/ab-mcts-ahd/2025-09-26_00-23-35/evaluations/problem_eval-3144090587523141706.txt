def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1
    density_factor = 1 - (remaining_nodes / total_nodes) if total_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 0.1 * distance_to_destination if remaining_nodes > 3 else 0

        if remaining_nodes <= 3:
            weight_proximity = 0.8
            weight_future = 0.2
        else:
            weight_proximity = 0.5
            weight_future = 0.5

        score = (weight_proximity * distance_to_node) + (weight_future * distance_to_destination * density_factor) + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        sum_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node)
        score = 0.5 * distance_to_node + 0.5 * sum_future_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

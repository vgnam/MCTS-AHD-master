def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        centrality_weight = 0.3 * (1 + (1 - num_unvisited / len(distance_matrix)) * 0.5)
        score = distance_to_node + centrality_weight * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

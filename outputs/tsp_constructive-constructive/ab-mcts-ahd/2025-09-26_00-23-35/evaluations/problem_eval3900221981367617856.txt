def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        distance_current_to_destination = distance_matrix[current_node][destination_node]

        if distance_current_to_destination == 0:
            weight = 1.0
        else:
            weight = (distance_to_current + distance_to_destination) / distance_current_to_destination

        score = (1 - weight) * distance_to_current + weight * distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

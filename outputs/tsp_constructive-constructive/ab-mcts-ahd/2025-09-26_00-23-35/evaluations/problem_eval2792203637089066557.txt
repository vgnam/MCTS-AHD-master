def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = -float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_dest = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)

        weight_proximity = 1.2 * (num_unvisited / (num_unvisited + 1))
        weight_future = 0.8 * (1 - (num_unvisited / (num_unvisited + 1)))
        score = -weight_proximity * distance_to_node + weight_future * avg_distance_to_dest

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

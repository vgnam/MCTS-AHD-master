import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    
    temperature = 100
    cooling_rate = 0.95
    max_score = float('-inf')
    next_node = None
    
    while temperature > 1:
        candidate_nodes = list(unvisited_nodes)
        for node in candidate_nodes:
            distance_to_current = distance_matrix[current_node][node]
            distance_to_destination = distance_matrix[node][destination_node]
            score = (1 / distance_to_current) * distance_to_destination
            
            if score > max_score or (random.random() < math.exp((score - max_score) / temperature)):
                max_score = score
                next_node = node
        
        if next_node is not None:
            return next_node
        
        temperature *= cooling_rate
    
    if next_node is None:
        next_node = max(unvisited_nodes, key=lambda node: (1 / distance_matrix[current_node][node]) * distance_matrix[node][destination_node])
    
    return next_node

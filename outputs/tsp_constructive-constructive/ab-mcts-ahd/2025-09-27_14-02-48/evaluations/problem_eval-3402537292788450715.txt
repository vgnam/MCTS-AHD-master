def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    
    min_score = float('inf')
    next_node = None
    
    for node in unvisited_nodes:
        local_cost = distance_matrix[current_node][node]
        
        # More sophisticated global cost estimation
        global_cost = distance_matrix[node][destination_node]
        if len(unvisited_nodes) > 1:
            remaining_nodes = unvisited_nodes - {node}
            if destination_node in remaining_nodes:
                remaining_nodes.remove(destination_node)
            if remaining_nodes:
                nearest_remaining = min(remaining_nodes, key=lambda x: distance_matrix[node][x])
                global_cost = min(global_cost, distance_matrix[node][nearest_remaining] + distance_matrix[nearest_remaining][destination_node])
        
        # Weighted scoring with a balance between local and global costs
        score = 0.5 * local_cost + 0.5 * global_cost
        
        if score < min_score:
            min_score = score
            next_node = node
    
    # Explicit check for destination node
    if destination_node in unvisited_nodes:
        local_cost = distance_matrix[current_node][destination_node]
        global_cost = 0  # Since we're directly going to the destination, global cost is 0
        score = 0.5 * local_cost + 0.5 * global_cost
        if score < min_score:
            next_node = destination_node
    
    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    
    distances = {node: distance_matrix[current_node][node] for node in unvisited_nodes}
    sorted_nodes = sorted(distances, key=distances.get)
    
    N = max(1, len(unvisited_nodes) // 3)
    top_N_nodes = sorted_nodes[:N]
    
    if random.random() < 0.7:  # 70% chance to choose the nearest neighbor
        next_node = top_N_nodes[0]
    else:
        next_node = random.choice(top_N_nodes)
    
    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance < distances[next_node]:
            next_node = destination_node
    
    return next_node

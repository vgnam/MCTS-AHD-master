import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    initial_temperature = 100
    cooling_steps = 50
    step = 0

    min_distance = float('inf')
    nearest_neighbor = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            nearest_neighbor = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
    else:
        destination_distance = float('inf')

    while step < cooling_steps:
        current_temp = initial_temperature * (1 - step / cooling_steps)

        if random.random() < current_temp / 100:
            next_node = random.choice(list(unvisited_nodes))
        else:
            if destination_distance < min_distance:
                next_node = destination_node
            else:
                next_node = nearest_neighbor

        if next_node not in unvisited_nodes:
            step += 1
            continue

        if next_node == destination_node or random.random() < math.exp((min_distance - distance_matrix[current_node][next_node]) / (current_temp * 0.2)):
            return next_node

        step += 1

    return next_node

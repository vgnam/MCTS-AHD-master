def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1 if len(unvisited_nodes) > 1 else 1)
        score = current_to_node + avg_distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

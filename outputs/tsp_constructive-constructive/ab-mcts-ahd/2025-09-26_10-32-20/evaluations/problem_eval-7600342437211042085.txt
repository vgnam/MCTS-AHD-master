def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress = 1 - (remaining_nodes / total_nodes)

    weight_local = 0.4 + 0.6 * progress
    weight_centrality = 0.5 * (0.5 ** progress)
    weight_coherence = 0.2 + 0.8 * progress
    weight_factor = 1 - math.log(remaining_nodes + 1) / (math.log(remaining_nodes + 1) + 1)

    def calculate_score(node):
        local_distance = distance_matrix[current_node][node]
        centrality = 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)
        coherence = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes)
        global_distance = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        return (weight_local * local_distance) + (weight_centrality * centrality) - (weight_coherence * coherence) + (weight_factor * global_distance)

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    nearest_neighbor = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    farthest_insertion = max(unvisited_nodes, key=lambda node: distance_matrix[node][destination_node])
    dynamic_weight = 0.5 + 0.5 * (len(unvisited_nodes) / len(distance_matrix))
    hybrid_score = lambda node: dynamic_weight * distance_matrix[current_node][node] + (1 - dynamic_weight) * distance_matrix[node][destination_node]
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

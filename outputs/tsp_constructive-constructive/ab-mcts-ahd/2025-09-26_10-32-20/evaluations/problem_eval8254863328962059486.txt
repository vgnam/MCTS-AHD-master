def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def weighted_score(node):
        direct_distance = distance_matrix[current_node][node]
        avg_remaining_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        exploration_factor = 0.5 + 0.4 * (1 / (1 + math.exp(-(len(unvisited_nodes) - 5))))
        return direct_distance * (1 - exploration_factor) + avg_remaining_distance * exploration_factor

    next_node = max(unvisited_nodes, key=weighted_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        direct_distance = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][n] for n in range(len(distance_matrix))) / len(distance_matrix)
        return direct_distance * 0.5 + avg_distance * 0.5
    next_node = max(unvisited_nodes, key=weighted_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    progress_factor = 1 - (len(unvisited_nodes) / len(distance_matrix))
    weight_immediate = 0.7 - 0.4 * progress_factor
    weight_destination = 0.3 + 0.4 * progress_factor
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] * weight_immediate + distance_matrix[node][destination_node] * weight_destination)
    return next_node

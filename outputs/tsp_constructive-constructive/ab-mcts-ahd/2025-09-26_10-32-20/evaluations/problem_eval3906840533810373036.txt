def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_ratio = len(unvisited_nodes) / total_nodes
    local_weight = 0.6 + 0.4 * (1 - remaining_ratio)
    global_weight = 1 - local_weight

    nearest_neighbor = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    def improved_centrality(node):
        return 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)

    def improved_unvisited_avg_distance(node):
        return 1 / (sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes) + 1e-10)

    hybrid_score = lambda node: (
        local_weight * (0.6 * distance_matrix[current_node][node] + 0.4 * distance_matrix[node][destination_node]) +
        global_weight * (0.5 * improved_centrality(node) + 0.5 * improved_unvisited_avg_distance(node))
    )

    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    exploration_factor = 1.0 / (1.0 + len(unvisited_nodes))
    adaptive_weight = 0.6 - 0.2 * exploration_factor
    hybrid_score = lambda node: adaptive_weight * distance_matrix[current_node][node] + (1.0 - adaptive_weight) * distance_matrix[node][destination_node] + exploration_factor * (distance_matrix[current_node][node] ** 2)
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

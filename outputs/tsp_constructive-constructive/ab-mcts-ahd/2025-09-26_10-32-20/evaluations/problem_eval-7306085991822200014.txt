def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    visited_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes
    exploration_weight = 0.6 * (1 - visited_ratio)
    exploitation_weight = 0.4 * visited_ratio

    def calculate_score(node):
        if not unvisited_nodes:
            return distance_matrix[current_node][node]

        distances_to_unvisited = [distance_matrix[node][n] for n in unvisited_nodes]
        avg_distance = sum(distances_to_unvisited) / len(unvisited_nodes)
        variance = sum((d - avg_distance) ** 2 for d in distances_to_unvisited) / len(unvisited_nodes)
        entropy = -sum((d / sum(distances_to_unvisited) * np.log(d / sum(distances_to_unvisited) + 1e-10)) for d in distances_to_unvisited)

        exploration_term = exploration_weight * distance_matrix[current_node][node]
        exploitation_term = exploitation_weight * avg_distance
        variance_penalty = 0.3 * variance
        entropy_bonus = 0.2 * entropy

        return exploration_term + exploitation_term + variance_penalty - entropy_bonus

    if np.random.rand() < 0.7 * visited_ratio:
        next_node = min(unvisited_nodes, key=calculate_score)
    else:
        next_node = np.random.choice(unvisited_nodes)

    return next_node

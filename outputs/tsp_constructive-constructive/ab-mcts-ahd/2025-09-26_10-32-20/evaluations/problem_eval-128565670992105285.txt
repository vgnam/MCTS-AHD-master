def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress = 1 - (remaining_nodes / total_nodes)

    def sigmoid(x, k=10):
        return 1 / (1 + np.exp(-k * (x - 0.5)))

    alpha = sigmoid(progress)
    beta = 1 - alpha

    exploration_bias = (remaining_nodes / total_nodes) ** 2

    weight_local = 0.5 * (1 - alpha) + 0.2 * exploration_bias
    weight_centrality = 0.3 * (1 - beta) + 0.2 * exploration_bias
    weight_coherence = 0.2 + 0.5 * beta

    def calculate_score(node):
        local_distance = distance_matrix[current_node][node]
        centrality = 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)
        coherence = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes)
        return (weight_local * local_distance) + (weight_centrality * centrality) - (weight_coherence * coherence)

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Calculate centroid of unvisited nodes
    centroid = sum(unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

    # Weighted cost function: immediate distance + destination distance + penalty for distance from centroid
    next_node = min(unvisited_nodes, key=lambda node: (
        0.4 * distance_matrix[current_node][node] +
        0.4 * distance_matrix[node][destination_node] +
        0.2 * abs(node - centroid)
    ))
    return next_node

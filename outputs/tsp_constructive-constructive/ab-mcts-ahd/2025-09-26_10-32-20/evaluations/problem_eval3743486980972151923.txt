def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes)
    local_weight = max(0.1, 0.5 - (remaining_nodes / 20))
    global_weight = 1 - local_weight
    dynamic_score = lambda node: local_weight * distance_matrix[current_node][node] + global_weight * distance_matrix[node][destination_node]
    next_node = min(unvisited_nodes, key=dynamic_score)
    return next_node

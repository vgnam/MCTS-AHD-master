def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_ratio = len(unvisited_nodes) / total_nodes
    progress_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes

    if progress_ratio < 0.3:
        local_weight = 0.8
        global_weight = 0.2
    elif progress_ratio < 0.7:
        local_weight = 0.5 * (1 - remaining_ratio) + 0.5 * progress_ratio
        global_weight = 1 - local_weight
    else:
        local_weight = 0.2
        global_weight = 0.8

    def centrality(node):
        return 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)

    def connectivity_potential(node):
        return sum(1 / (distance_matrix[node][n] + 1e-10) for n in unvisited_nodes if n != node)

    hybrid_score = lambda node: (
        local_weight * distance_matrix[current_node][node] +
        global_weight * (0.6 * centrality(node) + 0.4 * connectivity_potential(node))
    )

    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

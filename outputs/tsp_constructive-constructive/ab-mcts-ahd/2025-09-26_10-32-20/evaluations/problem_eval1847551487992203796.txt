def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        immediate_cost = distance_matrix[current_node][node]
        future_cost = distance_matrix[node][destination_node] if unvisited_nodes == {node} else min(distance_matrix[node][n] for n in unvisited_nodes - {node})
        return immediate_cost + 0.3 * future_cost
    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    weight = 0.3  # Prioritize global optimization (distance to destination)
    next_node = min(unvisited_nodes, key=lambda node: (1 - weight) * distance_matrix[node][destination_node] + weight * distance_matrix[current_node][node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    proximity_factor = 1.0 / (1.0 + distance_matrix[current_node][destination_node])
    weight_current = 0.5 * (1 - proximity_factor)
    weight_destination = 0.5 * (1 + proximity_factor)
    hybrid_score = lambda node: weight_current * distance_matrix[current_node][node] + weight_destination * distance_matrix[node][destination_node]
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    global_centrality = lambda node: 0.8 * (1 / (sum(distance_matrix[node][n] for n in range(len(distance_matrix))) + 1e-10)) + 0.2 * (distance_matrix[current_node][node])
    next_node = min(unvisited_nodes, key=global_centrality)
    return next_node

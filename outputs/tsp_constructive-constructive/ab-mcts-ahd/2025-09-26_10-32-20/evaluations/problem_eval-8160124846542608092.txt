def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    local_weight = 0.7 - 0.4 * (remaining_nodes / len(distance_matrix))
    global_weight = 0.3 + 0.2 * (remaining_nodes / len(distance_matrix))
    coherence_weight = 0.4 - 0.2 * (remaining_nodes / len(distance_matrix))

    next_node = min(unvisited_nodes, key=lambda node:
        local_weight * distance_matrix[current_node][node] +
        global_weight * distance_matrix[node][destination_node] +
        coherence_weight * (sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, remaining_nodes))
    )
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_ratio = len(unvisited_nodes) / total_nodes
    local_weight = 0.6 + 0.4 * (1 - remaining_ratio)
    global_weight = 0.3 + 0.2 * remaining_ratio
    degree_weight = 0.1 * (1 - remaining_ratio)

    def centrality(node):
        return 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)

    def node_degree(node):
        return sum(1 for n in range(total_nodes) if distance_matrix[node][n] > 0)

    hybrid_score = lambda node: (
        local_weight * distance_matrix[current_node][node] +
        global_weight * centrality(node) +
        degree_weight * node_degree(node)
    )

    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

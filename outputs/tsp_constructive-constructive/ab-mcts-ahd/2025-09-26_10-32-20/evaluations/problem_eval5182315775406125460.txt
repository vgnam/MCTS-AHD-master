def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    remaining_ratio = remaining_nodes / total_nodes
    local_weight = remaining_nodes / (remaining_nodes + 1)
    global_weight = 1 - local_weight
    centrality_weight = 0.3 + 0.5 * (1 - remaining_ratio)
    coherence_weight = 0.2 + 0.5 * (1 - remaining_ratio)

    def calculate_score(node):
        local_distance = distance_matrix[current_node][node]
        global_distance = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        centrality = 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)
        coherence = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes)
        return (local_weight * local_distance +
                global_weight * global_distance +
                centrality_weight * centrality -
                coherence_weight * coherence)

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

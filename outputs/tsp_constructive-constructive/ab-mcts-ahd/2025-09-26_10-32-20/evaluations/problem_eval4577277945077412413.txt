def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    progress = 1 - (remaining_nodes / total_nodes)

    weight_local = 0.5 * (1 - progress) + 0.2 * progress
    weight_centrality = 0.3 * (1 - progress) + 0.1 * progress
    weight_coherence = 0.2 * (1 - progress) + 0.5 * progress
    weight_coverage = 0.1 * (1 - progress) + 0.2 * progress

    def calculate_score(node):
        local_distance = distance_matrix[current_node][node]
        centrality = 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)
        coherence = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        coverage = sum(1 for n in unvisited_nodes if distance_matrix[node][n] < sum(distance_matrix[node][k] for k in range(total_nodes)) / total_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
        return (weight_local * local_distance) + (weight_centrality * centrality) - (weight_coherence * coherence) + (weight_coverage * coverage)

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes)
    exploration_weight = 0.7 * (remaining_nodes / (remaining_nodes + 1))
    exploitation_weight = 0.3 * (1 - (remaining_nodes / (remaining_nodes + 1)))
    hybrid_score = lambda node: exploitation_weight * distance_matrix[current_node][node] + exploration_weight * distance_matrix[node][destination_node]
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

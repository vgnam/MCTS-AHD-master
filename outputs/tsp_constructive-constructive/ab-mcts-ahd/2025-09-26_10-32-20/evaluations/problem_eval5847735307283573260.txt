def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes)
    k = 0.1  # steepness parameter for sigmoid
    dynamic_weight = 1.0 / (1.0 + math.exp(-k * (remaining_nodes - 5)))  # sigmoid transition
    hybrid_score = lambda node: (1 - dynamic_weight) * distance_matrix[current_node][node] + dynamic_weight * distance_matrix[node][destination_node]
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

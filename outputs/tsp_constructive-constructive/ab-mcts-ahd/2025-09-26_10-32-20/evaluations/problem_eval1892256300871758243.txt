def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    temperature = max(0.1, 1.0 / (1.0 + remaining_nodes))  # Decreasing temperature

    def hybrid_score(node):
        local_distance = distance_matrix[current_node][node]
        global_distance = distance_matrix[node][destination_node]
        penalty = 1.0 if node in unvisited_nodes else 1.5  # Penalize revisited nodes
        return (local_distance + global_distance) * penalty

    scores = [hybrid_score(node) for node in unvisited_nodes]
    min_score = min(scores)
    max_score = max(scores)

    normalized_scores = [(max_score - score) / (max_score - min_score + 1e-10) for score in scores]
    probabilities = [math.exp(normalized_score / temperature) for normalized_score in normalized_scores]
    total_prob = sum(probabilities)
    probabilities = [p / total_prob for p in probabilities]

    next_node = random.choices(unvisited_nodes, weights=probabilities, k=1)[0]
    return next_node

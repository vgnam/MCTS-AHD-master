def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_ratio = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] * (1 - remaining_ratio) + distance_matrix[node][destination_node] * remaining_ratio)
    return next_node

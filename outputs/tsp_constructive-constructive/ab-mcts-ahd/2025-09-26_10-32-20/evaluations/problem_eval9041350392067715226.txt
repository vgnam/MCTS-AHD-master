def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.7):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] if node != destination_node else 0
        return weight * distance + (1 - weight) * future_savings
    next_node = min(unvisited_nodes, key=weighted_score)
    return next_node

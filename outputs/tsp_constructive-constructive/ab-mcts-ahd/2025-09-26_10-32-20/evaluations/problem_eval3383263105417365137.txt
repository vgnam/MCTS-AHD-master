def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_score(node):
        direct_distance = distance_matrix[current_node][node]
        return_to_dest = distance_matrix[node][destination_node]
        return direct_distance + 0.3 * return_to_dest

    next_node = min(unvisited_nodes, key=lambda node: heuristic_score(node))
    return next_node

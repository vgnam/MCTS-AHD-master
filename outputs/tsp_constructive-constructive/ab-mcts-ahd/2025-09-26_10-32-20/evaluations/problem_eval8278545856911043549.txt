def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(distance_matrix)
    remaining_ratio = len(unvisited_nodes) / total_nodes
    local_weight = 0.8 * (1 - remaining_ratio)
    global_weight = 1 - local_weight

    def centrality(node):
        return 1 / (sum(distance_matrix[node][n] for n in range(total_nodes)) + 1e-10)

    def penalty(node):
        return 1.0 if distance_matrix[current_node][node] < 0.1 * sum(distance_matrix[current_node]) / total_nodes else 1.0

    hybrid_score = lambda node: (
        local_weight * distance_matrix[current_node][node] +
        global_weight * centrality(node) +
        penalty(node)
    )

    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    hybrid_score = lambda node: 0.5 * distance_matrix[current_node][node] + 0.5 * distance_matrix[node][destination_node] + 0.1 * (distance_matrix[node][destination_node] > 1.5 * distance_matrix[current_node][destination_node])
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

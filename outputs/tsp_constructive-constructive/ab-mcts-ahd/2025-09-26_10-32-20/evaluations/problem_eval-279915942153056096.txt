def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    exploration_factor = 1.0 / (1.0 + len(unvisited_nodes))
    local_weight = 0.5 + 0.5 * exploration_factor
    global_weight = 1 - local_weight
    hybrid_score = lambda node: local_weight * distance_matrix[current_node][node] + global_weight * (distance_matrix[node][destination_node] + exploration_factor * distance_matrix[current_node][node] ** 2)
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

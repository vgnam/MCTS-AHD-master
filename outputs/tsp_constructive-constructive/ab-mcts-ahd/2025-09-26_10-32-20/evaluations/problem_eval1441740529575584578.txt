def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    exploitation_weight = max(0.1, 0.5 - (remaining_nodes / len(distance_matrix)))
    exploration_weight = 1 - exploitation_weight

    hybrid_score = lambda node: (exploitation_weight * distance_matrix[current_node][node] +
                                exploration_weight * distance_matrix[node][destination_node])

    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

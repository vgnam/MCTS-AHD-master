def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    exploration_factor = 1.0 / (1.0 + len(unvisited_nodes))
    hybrid_score = lambda node: (0.2 + exploration_factor * 0.3) * distance_matrix[current_node][node] + (0.8 - exploration_factor * 0.5) * distance_matrix[node][destination_node] + exploration_factor * (distance_matrix[node][destination_node] ** 2)
    next_node = min(unvisited_nodes, key=hybrid_score)
    return next_node

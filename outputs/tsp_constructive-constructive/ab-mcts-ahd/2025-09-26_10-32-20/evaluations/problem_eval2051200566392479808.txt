def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    penalties = {node: 1.0 / (1 + sum(distance_matrix[current_node][node] for node in unvisited_nodes)) for node in unvisited_nodes}
    next_node = max(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] * penalties[node])
    return next_node

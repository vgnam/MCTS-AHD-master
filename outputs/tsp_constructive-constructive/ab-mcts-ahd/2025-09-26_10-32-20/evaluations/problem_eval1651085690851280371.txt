def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = max(unvisited_nodes, key=lambda node: (0.7 * (distance_matrix[node][destination_node] / (distance_matrix[current_node][node] + 1e-6)) + 0.3 * (distance_matrix[current_node][node] / (distance_matrix[node][destination_node] + 1e-6))))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def cost_function(node):
        immediate_cost = distance_matrix[current_node][node]
        future_cost = distance_matrix[node][destination_node] if destination_node in distance_matrix[node] else float('inf')
        return immediate_cost + 0.5 * future_cost
    next_node = min(unvisited_nodes, key=cost_function)
    return next_node

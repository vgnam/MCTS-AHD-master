def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(unvisited_nodes) + 1
    dynamic_weight = 0.7 * (remaining_nodes / total_nodes) + 0.3 * (1 - remaining_nodes / total_nodes)

    base_threshold = 1.2 * distance_matrix[current_node][destination_node]
    adaptive_threshold = base_threshold * (1 + 0.1 * (1 - remaining_nodes / total_nodes))

    def novelty_factor(node):
        local_cost = distance_matrix[current_node][node]
        global_cost = distance_matrix[node][destination_node]
        penalty = max(0, (local_cost + global_cost) - adaptive_threshold) ** 2
        return dynamic_weight * local_cost + (1 - dynamic_weight) * global_cost + penalty

    next_node = min(unvisited_nodes, key=novelty_factor)
    return next_node

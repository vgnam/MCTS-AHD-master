def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def centrality(node):
        return sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)

    next_node = max(
        unvisited_nodes,
        key=lambda node: (0.7 * distance_matrix[current_node][node]) - (0.3 * centrality(node))
    )
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_score(node):
        distance = distance_matrix[current_node][node]
        penalty = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        return distance - 0.3 * penalty
    next_node = max(unvisited_nodes, key=weighted_score)
    return next_node

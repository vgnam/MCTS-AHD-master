def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    # Balance between farthest and nearest by selecting nodes with moderate distances
    next_node = min(unvisited_nodes, key=lambda node: abs(distance_matrix[current_node][node] - distance_matrix[current_node][destination_node]))
    return next_node

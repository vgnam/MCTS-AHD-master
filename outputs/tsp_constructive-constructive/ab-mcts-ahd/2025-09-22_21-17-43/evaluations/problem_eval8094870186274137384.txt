def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def weighted_penalty_score(node):
        local_weight = 0.6
        penalty_weight = 0.5
        local_distance = distance_matrix[current_node][node]
        penalty = penalty_weight * distance_matrix[node][destination_node]
        return local_weight * local_distance + penalty
    next_node = min(unvisited_nodes, key=lambda node: weighted_penalty_score(node))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    progress = 1 - (len(unvisited_nodes) / total_nodes)
    weight_local = 1 / (1 + math.exp(5 * (progress - 0.5)))
    weight_global = 1 - weight_local

    def combined_score(node):
        local_distance = distance_matrix[current_node][node]
        global_factor = 0
        if len(unvisited_nodes) > 1:
            for other_node in unvisited_nodes:
                if other_node != node:
                    global_factor += distance_matrix[node][other_node]
            global_factor /= (len(unvisited_nodes) - 1)
        return weight_local * local_distance + weight_global * global_factor

    next_node = min(unvisited_nodes, key=lambda node: combined_score(node))
    return next_node

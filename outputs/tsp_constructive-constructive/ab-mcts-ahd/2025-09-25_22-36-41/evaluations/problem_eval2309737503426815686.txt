def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_value = float('inf')
    next_node = None
    for node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][node]
        total_increment = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        value = direct_distance + 0.5 * total_increment
        if value < min_value:
            min_value = value
            next_node = node
    if next_node is None:
        return destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    best_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        direct_distance = distance_matrix[current_node][node]
        total_increment = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if total_increment == 0:
            ratio = float('inf')
        else:
            ratio = direct_distance / total_increment
        score = product * (1 / ratio) if ratio != 0 else float('inf')
        if score < best_score:
            best_score = score
            next_node = node
    if next_node is None:
        return destination_node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    max_gain = -float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))  # Dynamic weight based on remaining nodes
    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        savings = distance_matrix[node][destination_node] - (distance_matrix[current_node][destination_node] - immediate_cost)
        gain = (weight * (immediate_cost - savings)) + ((1 - weight) * distance_matrix[node][destination_node])
        if gain > max_gain:
            max_gain = gain
            next_node = node
    if next_node is None:
        return destination_node
    return next_node

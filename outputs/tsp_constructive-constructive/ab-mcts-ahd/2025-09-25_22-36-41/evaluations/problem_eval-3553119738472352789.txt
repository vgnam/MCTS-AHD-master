def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        future_cost = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / remaining_nodes if remaining_nodes > 1 else 0
        score = 0.4 * immediate_cost + 0.3 * distance_matrix[node][destination_node] + 0.3 * future_cost
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

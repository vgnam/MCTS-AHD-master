def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_ratio = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        sum_distance = current_distance + destination_distance
        product_distance = (current_distance ** 2) * (destination_distance ** 2)
        ratio = sum_distance / product_distance if product_distance != 0 else float('inf')

        if ratio < min_ratio:
            min_ratio = ratio
            next_node = node
        elif ratio == min_ratio:
            if destination_distance < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

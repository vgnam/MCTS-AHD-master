def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_metric = float('inf')
    next_node = None
    weight_factor = 0.7

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        combined_metric = weight_factor * current_distance + (1 - weight_factor) * destination_distance
        total_distance = current_distance + destination_distance

        if combined_metric < min_metric:
            min_metric = combined_metric
            next_node = node
            min_total = total_distance
        elif combined_metric == min_metric:
            if total_distance < min_total:
                next_node = node
                min_total = total_distance
            elif total_distance == min_total:
                if destination_distance < distance_matrix[next_node][destination_node]:
                    next_node = node

    return next_node

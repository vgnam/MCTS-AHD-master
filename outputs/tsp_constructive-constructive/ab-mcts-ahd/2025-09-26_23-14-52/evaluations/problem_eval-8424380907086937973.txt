def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_destination_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        destination_distance = distance_matrix[node][destination_node]
        current_distance = distance_matrix[current_node][node]

        if destination_distance < min_destination_distance:
            min_destination_distance = destination_distance
            next_node = node
        elif destination_distance == min_destination_distance:
            if current_distance < distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

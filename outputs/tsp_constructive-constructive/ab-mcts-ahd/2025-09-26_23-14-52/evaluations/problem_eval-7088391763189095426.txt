import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    exploration_temp = max(0.1, math.exp(-0.2 * total_unvisited))
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        exploration_penalty = exploration_temp * random.gauss(0, 1)
        score = current_distance + 0.3 * future_distance + exploration_penalty

        if score < best_score:
            best_score = score
            next_node = node
        elif score == best_score and random.random() < 0.5:
            next_node = node

    return next_node

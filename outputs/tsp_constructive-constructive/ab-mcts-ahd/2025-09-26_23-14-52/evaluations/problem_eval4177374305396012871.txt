def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('-inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        avg_distance = sum(distance_matrix[node]) / len(distance_matrix[node])

        if num_unvisited > len(distance_matrix) // 2:
            weighted_score = 2.0 * current_distance + 0.5 * future_distance + 0.5 * avg_distance
        else:
            weighted_score = 1.0 * current_distance + 1.0 * future_distance + 1.5 * avg_distance

        penalty = 0.1 * avg_distance if num_unvisited > 2 else 0
        weighted_score -= penalty

        if weighted_score > best_score:
            best_score = weighted_score
            next_node = node
        elif weighted_score == best_score:
            if avg_distance < sum(distance_matrix[next_node]) / len(distance_matrix[next_node]):
                next_node = node

    return next_node

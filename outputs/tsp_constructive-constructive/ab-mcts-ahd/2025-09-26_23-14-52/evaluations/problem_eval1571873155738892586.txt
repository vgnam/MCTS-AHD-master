def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_weighted_ratio = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        ratio = current_distance / destination_distance if destination_distance != 0 else float('inf')
        weighted_ratio = 0.7 * ratio + 0.3 * (current_distance + destination_distance)

        if weighted_ratio < min_weighted_ratio:
            min_weighted_ratio = weighted_ratio
            next_node = node
        elif weighted_ratio == min_weighted_ratio:
            if destination_distance < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

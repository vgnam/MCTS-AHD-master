def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('-inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    exploration_factor = 0.3 + (0.6 * (1 - (total_unvisited / (total_unvisited + 1))))

    # Track frequency of nodes selected in past iterations (simulated here)
    node_frequencies = {node: random.uniform(0, 0.5) for node in unvisited_nodes}

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        frequency_penalty = node_frequencies[node]

        weighted_score = (0.3 * current_distance + 0.6 * future_distance) * (1 + frequency_penalty) + exploration_factor * random.uniform(0, 1)

        if weighted_score > best_score:
            best_score = weighted_score
            next_node = node
        elif weighted_score == best_score:
            if current_distance < distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

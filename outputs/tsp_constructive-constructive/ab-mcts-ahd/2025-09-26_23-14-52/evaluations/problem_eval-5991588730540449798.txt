def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_combined_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        combined_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        current_distance = distance_matrix[current_node][node]

        if combined_distance < min_combined_distance:
            min_combined_distance = combined_distance
            next_node = node
        elif combined_distance == min_combined_distance:
            if current_distance < distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

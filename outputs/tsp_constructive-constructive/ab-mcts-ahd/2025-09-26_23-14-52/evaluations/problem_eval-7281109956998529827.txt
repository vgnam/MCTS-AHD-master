def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    weight = 0.5 if total_unvisited > len(unvisited_nodes) * 0.5 else 0.6

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        centrality = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        sum_distance = current_distance + future_distance
        product_distance = current_distance * future_distance
        ratio = sum_distance / product_distance if product_distance != 0 else float('inf')
        weighted_score = ratio + weight * (0.5 * centrality + 0.5 * future_distance)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node
        elif weighted_score == min_score:
            if future_distance < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

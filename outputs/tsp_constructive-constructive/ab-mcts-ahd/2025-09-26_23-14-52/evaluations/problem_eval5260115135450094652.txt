def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    penalty_factor = 1.0 - (total_unvisited / len(distance_matrix)) ** 2

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        penalty = penalty_factor * future_distance
        score = current_distance + (1 + penalty) * future_distance
        tie_breaker = current_distance + future_distance

        if score < min_score:
            min_score = score
            next_node = node
            min_tie = tie_breaker
        elif score == min_score:
            if tie_breaker < min_tie:
                next_node = node
                min_tie = tie_breaker
            elif tie_breaker == min_tie:
                if future_distance < distance_matrix[next_node][destination_node]:
                    next_node = node

    return next_node

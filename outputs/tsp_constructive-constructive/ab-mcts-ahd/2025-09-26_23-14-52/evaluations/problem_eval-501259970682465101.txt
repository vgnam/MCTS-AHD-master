def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_product = float('inf')
    min_ratio = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        product = current_distance * destination_distance
        ratio = current_distance / destination_distance if destination_distance != 0 else float('inf')

        if product < min_product:
            min_product = product
            min_ratio = ratio
            next_node = node
        elif product == min_product:
            if ratio < min_ratio:
                min_ratio = ratio
                next_node = node
            elif ratio == min_ratio:
                if destination_distance < distance_matrix[next_node][destination_node]:
                    next_node = node

    return next_node

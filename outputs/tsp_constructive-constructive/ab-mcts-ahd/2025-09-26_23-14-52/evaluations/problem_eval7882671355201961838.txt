def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        sum_distance = current_distance + destination_distance
        product_distance = current_distance * destination_distance
        ratio = sum_distance / product_distance if product_distance != 0 else float('inf')
        score = ratio * (1 + 1 / (destination_distance + 1e-10))

        if score < min_score:
            min_score = score
            next_node = node
        elif score == min_score:
            if destination_distance < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

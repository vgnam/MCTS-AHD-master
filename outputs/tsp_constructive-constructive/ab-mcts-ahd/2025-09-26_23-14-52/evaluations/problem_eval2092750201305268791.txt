def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    normalization_factor = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / total_unvisited if total_unvisited != 0 else 1

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        normalized_distance = current_distance / normalization_factor if normalization_factor != 0 else current_distance
        score = 0.7 * normalized_distance + 0.3 * destination_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

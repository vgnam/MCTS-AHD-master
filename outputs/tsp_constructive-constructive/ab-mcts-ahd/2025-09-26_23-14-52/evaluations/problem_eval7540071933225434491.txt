def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]

        # Dynamic weight adjustment
        weight_current = 1.0 / (remaining_nodes ** 1.5 + 1)
        weight_destination = 1.0 - weight_current

        # Combined score with additional factors
        score = (weight_current * current_distance) + (weight_destination * destination_distance) + (current_distance * destination_distance / 1000)

        if score < min_score:
            min_score = score
            next_node = node
        elif score == min_score:
            if destination_distance < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    weight_factor = min(0.9, (total_unvisited / len(distance_matrix)))

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        weighted_score = current_distance + weight_factor * future_distance

        if current_distance < min_distance:
            min_distance = current_distance
            next_node = node
        elif current_distance == min_distance:
            if weighted_score < distance_matrix[next_node][current_node] + weight_factor * distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

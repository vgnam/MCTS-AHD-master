import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    max_unvisited = len(distance_matrix) - 1
    annealing_factor = 0.5 * (1 + math.cos(math.pi * total_unvisited / max_unvisited))
    exploitation_weight = annealing_factor
    exploration_weight = 1.0 - annealing_factor

    best_score = float('-inf')
    next_node = None
    node_popularity = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]

        base_score = (exploitation_weight * current_distance) - (exploration_weight * future_distance)

        popularity_score = node_popularity[node] * 0.3
        weighted_score = base_score + popularity_score + random.gauss(0, 0.1)

        if weighted_score > best_score:
            best_score = weighted_score
            next_node = node
            node_popularity[node] += 1

    return next_node

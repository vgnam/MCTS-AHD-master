def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    current_to_dest = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]

        # Dynamic weighting: prioritize immediate distance when far from destination, future distance when close
        weight = 1.0 if current_to_dest > 10 else 0.3  # Adjusted threshold and weight
        weighted_score = current_distance + weight * future_distance

        # Centrality tiebreaker: nodes with higher centrality (lower average distance to all other nodes) are preferred
        centrality = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0

        # Penalize nodes that are closer to the destination to avoid revisiting
        penalty = 0.2 * (1 / (1 + future_distance))  # Higher penalty for nodes closer to destination
        adjusted_score = weighted_score + penalty

        if adjusted_score < best_score:
            best_score = adjusted_score
            next_node = node
            best_centrality = centrality
        elif adjusted_score == best_score:
            if centrality < best_centrality:  # Prefer nodes with higher centrality (lower centrality value)
                next_node = node
                best_centrality = centrality

    return next_node

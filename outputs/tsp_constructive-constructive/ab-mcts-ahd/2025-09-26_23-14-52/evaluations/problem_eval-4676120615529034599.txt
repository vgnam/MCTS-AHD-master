import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    temperature = max(0.1, 1.0 - 0.2 * (10 - total_unvisited))

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]

        proximity_score = current_distance + 0.3 * future_distance
        randomness_factor = temperature * random.uniform(0.5, 1.5)
        weighted_score = proximity_score + randomness_factor * (1.0 - math.exp(-0.1 * future_distance))

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node
        elif weighted_score == best_score and future_distance < distance_matrix[next_node][destination_node]:
            next_node = node

    return next_node

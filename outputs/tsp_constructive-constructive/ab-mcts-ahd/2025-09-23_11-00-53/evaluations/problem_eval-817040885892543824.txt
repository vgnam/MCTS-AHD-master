def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining = len(unvisited_nodes)
    weight = max(0.1, 1 - 0.5 * (remaining / (remaining + 1)))
    next_node = min(unvisited_nodes, key=lambda node: weight * distance_matrix[current_node][node] + (1 - weight) * sum(distance_matrix[node][other] for other in unvisited_nodes))
    return next_node

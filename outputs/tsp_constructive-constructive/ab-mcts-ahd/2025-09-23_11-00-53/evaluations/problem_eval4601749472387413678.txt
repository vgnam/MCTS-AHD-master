def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_count = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + (sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_count))
    next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] * weight_factor) + (sum(distance_matrix[node][other] for other in unvisited_nodes) / remaining_count))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_cost(node):
        current_dist = distance_matrix[current_node][node]
        avg_remaining_dist = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        return 0.7 * current_dist + 0.3 * avg_remaining_dist

    next_node = min(unvisited_nodes, key=lambda node: heuristic_cost(node))
    return next_node

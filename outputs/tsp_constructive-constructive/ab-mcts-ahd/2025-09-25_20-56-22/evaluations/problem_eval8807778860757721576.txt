def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = max(unvisited_nodes, key=lambda node: (len(unvisited_nodes) - 1 if node == destination_node else 0) - distance_matrix[current_node][node])
    return next_node

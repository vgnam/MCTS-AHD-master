def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def node_score(node):
        current_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_distance

        avg_future_distance = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        min_future_distance = min(distance_matrix[node][other] for other in remaining_nodes)

        weight = len(remaining_nodes) / (len(unvisited_nodes) + 1)
        return (1 - weight) * current_distance + weight * (0.7 * avg_future_distance + 0.3 * min_future_distance)

    next_node = min(unvisited_nodes, key=node_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def node_score(node):
        current_distance = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_distance

        avg_future_distance = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        local_density = len([other for other in remaining_nodes if distance_matrix[node][other] < 2 * current_distance])

        weight_factor = 0.7 if len(remaining_nodes) > 3 else 0.3
        distance_penalty = (current_distance ** 2) / (1 + local_density)

        return (weight_factor * current_distance) + ((1 - weight_factor) * avg_future_distance) + distance_penalty

    next_node = min(unvisited_nodes, key=node_score)
    return next_node

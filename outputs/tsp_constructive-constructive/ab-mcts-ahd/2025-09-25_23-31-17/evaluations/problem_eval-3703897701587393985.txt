defaults to the closest unvisited node if no nodes are left.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.7):
    if not unvisited_nodes:
        return destination_node
    next_node = min(unvisited_nodes, key=lambda node: weight * distance_matrix[current_node][node] + (1 - weight) * distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_distance_to_destination = sum(distance_matrix[n][destination_node] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        score = distance_to_node + avg_distance_to_destination
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

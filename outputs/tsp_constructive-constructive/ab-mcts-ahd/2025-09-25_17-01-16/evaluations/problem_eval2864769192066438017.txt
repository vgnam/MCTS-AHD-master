def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_product_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)
        product_score = distance_to_current * avg_distance_to_unvisited

        if product_score < min_product_score:
            min_product_score = product_score
            next_node = node

    return next_node

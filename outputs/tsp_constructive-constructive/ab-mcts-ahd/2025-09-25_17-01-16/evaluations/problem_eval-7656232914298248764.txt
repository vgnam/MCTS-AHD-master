def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    weight_current = 1.5 * (1 + 0.1 * num_unvisited)
    weight_dest = 0.5 + 0.05 * (num_unvisited ** 2)
    weight_hub = 0.1 + 0.2 * (num_unvisited ** 0.5)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        progress_factor = (num_unvisited / (num_unvisited + 1)) ** 2
        total_score = weight_current * distance_to_current + progress_factor * weight_dest * distance_to_destination + weight_hub * hub_score

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if estimated_distance_to_destination == 0:
            ratio = float('inf')
        else:
            ratio = distance_to_current / estimated_distance_to_destination

        score = -ratio + 0.5 * distance_to_current  # Balance between progress toward destination and distance to current node

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

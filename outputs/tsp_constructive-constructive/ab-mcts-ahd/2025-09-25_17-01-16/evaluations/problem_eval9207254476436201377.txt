def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_distance = float('inf')
    alpha = 0.3  # Weight for distance_to_current
    beta = 0.7   # Weight for estimated_distance_to_destination

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        total_distance = alpha * distance_to_current + beta * estimated_distance_to_destination

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_distance = float('inf')
    total_nodes = len(unvisited_nodes) + 1  # Including the current node
    visited_ratio = 1 - (len(unvisited_nodes) / total_nodes)
    weight = 0.5 * (1 + visited_ratio)  # Higher weight on local early, global later

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        weighted_distance = weight * distance_to_current + (1 - weight) * estimated_distance_to_destination

        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node

    return next_node

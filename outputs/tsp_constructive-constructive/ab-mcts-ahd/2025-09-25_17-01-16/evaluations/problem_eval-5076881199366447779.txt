def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)

        dynamic_weight = 0.5 + 0.1 * (1 - remaining_nodes / len(unvisited_nodes))
        progress_weight = 0.3 + 0.1 * (remaining_nodes / len(unvisited_nodes))
        hub_weight = 0.2

        cycle_penalty = 1.0 + 0.5 * (1 - math.log(remaining_nodes + 1)) if node != destination_node else 0.0
        novelty_score = len([n for n in unvisited_nodes if distance_matrix[node][n] > 0 and n != node])

        total_score = (dynamic_weight * distance_to_current +
                      progress_weight * distance_to_destination +
                      hub_weight * hub_score +
                      cycle_penalty -
                      0.1 * novelty_score)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    weight_current = 1.0 + 0.3 * num_unvisited
    weight_dest = 1.0 - 0.2 * num_unvisited
    weight_centrality = 0.5 + 0.1 * num_unvisited

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        centrality_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (num_unvisited - 1)
        historical_penalty = 0.0 if num_unvisited == len(distance_matrix) - 1 else sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(distance_matrix) - num_unvisited)

        total_score = (weight_current * distance_to_current +
                       weight_dest * distance_to_destination +
                       weight_centrality * centrality_score +
                       historical_penalty)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

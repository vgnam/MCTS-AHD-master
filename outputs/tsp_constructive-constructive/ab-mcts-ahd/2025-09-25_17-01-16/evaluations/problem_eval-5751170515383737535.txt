def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_distance = float('inf')
    total_nodes = len(unvisited_nodes) + 1  # +1 to include the current node
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight adjustment based on remaining nodes ratio
        if remaining_nodes > 0:
            weight = (remaining_nodes / total_nodes) ** 2  # Squared to emphasize long-term progress
        else:
            weight = 1.0

        # Penalty for revisiting nodes to encourage exploration
        penalty = 1.0 - (remaining_nodes / total_nodes)

        total_distance = distance_to_current + weight * estimated_distance_to_destination + penalty * distance_to_current

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

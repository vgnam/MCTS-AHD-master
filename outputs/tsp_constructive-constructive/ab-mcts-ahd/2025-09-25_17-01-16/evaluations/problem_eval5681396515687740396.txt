def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    total_unvisited = len(unvisited_nodes)
    num_nodes = len(distance_matrix)

    # Calculate centrality of the current node (average distance to all other nodes)
    centrality = sum(distance_matrix[current_node]) / (num_nodes - 1)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight based on remaining unvisited nodes and centrality
        weight = 0.5 * (1 - (total_unvisited / (total_unvisited + 1))) + 0.5 * (centrality / max(distance_matrix[current_node]))

        # Weighted score: closer nodes, destination proximity, and centrality all contribute
        total_score = (1 - weight) * distance_to_current + weight * distance_to_destination + 0.3 * centrality

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

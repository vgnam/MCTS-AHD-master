def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + 0.1 * remaining_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)

        total_score = distance_to_current + (weight_factor * distance_to_destination) + (0.3 * hub_score)

        if remaining_nodes > 2:
            total_score += 0.1 * distance_to_current

        if total_score < best_score:
            best_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if estimated_distance_to_destination == 0:
            ratio = float('inf')
        else:
            ratio = distance_to_current / estimated_distance_to_destination

        avg_distance_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes) / total_unvisited
        connectivity_penalty = sum(1 for other in unvisited_nodes if distance_matrix[node][other] < avg_distance_to_unvisited) / total_unvisited

        dynamic_weight = 1.0 / (1.0 + connectivity_penalty)
        weighted_score = (dynamic_weight * ratio) + distance_to_current + (0.5 * avg_distance_to_unvisited)

        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            next_node = node

    return next_node

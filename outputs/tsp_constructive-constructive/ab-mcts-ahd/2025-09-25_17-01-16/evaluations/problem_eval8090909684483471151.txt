def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        distance_sum = distance_to_current + estimated_distance_to_destination
        distance_product = distance_to_current * estimated_distance_to_destination
        weight = 1.0 / (1.0 + num_unvisited)
        score = weight * distance_sum + (1 - weight) * distance_product

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

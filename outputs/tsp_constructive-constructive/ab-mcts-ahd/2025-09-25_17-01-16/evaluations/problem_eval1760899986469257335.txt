def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_sum = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        total_sum = distance_to_current + (estimated_distance_to_destination ** 2)

        if total_sum < min_total_sum:
            min_total_sum = total_sum
            next_node = node

    return next_node

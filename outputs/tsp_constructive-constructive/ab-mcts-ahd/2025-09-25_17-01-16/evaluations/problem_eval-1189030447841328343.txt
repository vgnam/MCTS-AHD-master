def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_value = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        total_product = distance_to_current * estimated_distance_to_destination
        total_sum = distance_to_current + estimated_distance_to_destination
        total_value = 0.5 * total_product + 0.5 * total_sum

        if total_value < min_total_value:
            min_total_value = total_value
            next_node = node

    return next_node

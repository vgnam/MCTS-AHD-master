def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.6):
    next_node = None
    min_total_score = float('inf')

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        distance_score = current_distance + destination_distance
        product_score = current_distance * destination_distance
        total_score = 0.6 * distance_score + 0.4 * product_score
        total_score = alpha * current_distance + (1 - alpha) * total_score

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

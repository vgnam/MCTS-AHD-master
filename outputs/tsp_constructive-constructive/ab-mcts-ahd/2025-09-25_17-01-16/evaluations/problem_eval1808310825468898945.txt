def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    current_to_dest = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        dynamic_weight = 1.0 - (current_to_dest / (sum(distance_matrix[current_node]) + 1e-6))
        penalty = dynamic_weight * future_savings
        score = immediate_distance + penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    max_remaining = max(1, remaining_nodes)  # Avoid division by zero

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        dynamic_weight = 0.7 - 0.4 * (1 - remaining_nodes / max_remaining)  # Weight decreases as nodes decrease
        penalty = 1.0 + 0.3 * (1 if node == current_node else 0)  # Larger penalty for revisiting current node
        total_score = immediate_distance + dynamic_weight * future_distance + penalty

        if total_score < min_score:
            min_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if estimated_distance_to_destination == 0:
            ratio = float('inf')
        else:
            ratio = distance_to_current / estimated_distance_to_destination

        distances_to_unvisited = [distance_matrix[node][other] for other in unvisited_nodes]
        variance = sum((d - (sum(distances_to_unvisited) / len(distances_to_unvisited))) ** 2 for d in distances_to_unvisited) / len(distances_to_unvisited)
        weighted_score = (2 * ratio) + distance_to_current + (3 * variance)

        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            next_node = node

    return next_node

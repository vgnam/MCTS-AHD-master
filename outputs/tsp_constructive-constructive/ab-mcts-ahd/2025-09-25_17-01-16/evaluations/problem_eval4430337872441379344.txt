def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)
        distance_to_destination = distance_matrix[node][destination_node]

        proximity_weight = 0.5 * (1.0 / (1.0 + remaining_nodes)) if remaining_nodes > 0 else 0.0
        detour_weight = 0.3 * (avg_distance_to_unvisited / (distance_to_current + 1e-6))
        alignment_weight = 0.2 * (1.0 - (distance_to_destination / (distance_matrix[current_node][destination_node] + 1e-6)))

        total_score = (distance_to_current * proximity_weight) + (avg_distance_to_unvisited * detour_weight) + (distance_to_destination * alignment_weight)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

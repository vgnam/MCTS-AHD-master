def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1  # +1 for the current node
    current_to_dest = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        exploration_factor = (remaining_nodes / total_nodes) ** 1.5  # Steeper curve for exploration
        proximity_penalty = 1 - (destination_distance / current_to_dest) if current_to_dest > 0 else 0
        score = -current_distance + exploration_factor * destination_distance - proximity_penalty * 0.5

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

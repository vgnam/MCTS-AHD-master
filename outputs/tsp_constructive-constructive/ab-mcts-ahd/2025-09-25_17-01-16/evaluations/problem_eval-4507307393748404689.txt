def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes) / len(unvisited_nodes)
        score = distance_to_node + 0.3 * distance_to_destination + 0.2 * avg_distance_to_unvisited

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        remaining_distance_factor = distance_to_destination / sum(distance_matrix[current_node][n] for n in unvisited_nodes) if unvisited_nodes else 0
        dynamic_weight = 0.7 - 0.4 * remaining_distance_factor
        average_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        weighted_score = dynamic_weight * distance_to_current + (1 - dynamic_weight) * average_future_distance

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

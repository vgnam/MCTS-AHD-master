def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_product_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        weight = 0.5 + (1.0 - 0.5) * (1.0 - remaining_nodes / len(unvisited_nodes))
        penalty = 1.0 if current_distance < 0.1 * distance_matrix[current_node][destination_node] else 0.0
        product_score = current_distance * (1.0 - penalty) * (1.0 - weight) + destination_distance * weight

        if product_score < min_product_score:
            min_product_score = product_score
            next_node = node

    return next_node

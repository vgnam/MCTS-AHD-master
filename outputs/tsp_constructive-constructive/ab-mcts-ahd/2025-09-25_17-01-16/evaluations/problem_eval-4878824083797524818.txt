def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (num_unvisited - 1 if num_unvisited > 1 else 1)
        exploration_factor = 1.0 / (1.0 + 0.1 * num_unvisited)
        revisit_penalty = 1.0 if node in unvisited_nodes else 0.0
        total_score = (distance_to_current + 0.5 * distance_to_destination + 0.3 * hub_score) * exploration_factor + revisit_penalty

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

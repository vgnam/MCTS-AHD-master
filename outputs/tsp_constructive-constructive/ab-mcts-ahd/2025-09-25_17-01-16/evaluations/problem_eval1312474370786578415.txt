def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)
        destination_distance = distance_matrix[node][destination_node]

        weight = 1.0 / (1.0 + remaining_nodes) if remaining_nodes > 0 else 0.0
        score = -current_distance + 0.5 * destination_distance - (avg_distance_to_unvisited * weight)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_distance = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        connectivity_score = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1) if remaining_nodes > 1 else 0

        if remaining_nodes > 0:
            weight = 1 - (remaining_nodes / (remaining_nodes + 2))
            total_distance = (0.7 * distance_to_current) + (0.3 * connectivity_score) * weight
        else:
            total_distance = distance_to_current

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

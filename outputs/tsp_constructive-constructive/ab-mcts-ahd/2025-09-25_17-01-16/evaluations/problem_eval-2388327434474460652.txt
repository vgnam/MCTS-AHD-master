def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)

        # Dynamic weighting based on remaining nodes
        progress_weight = 0.8 if remaining_nodes > 2 else 0.4
        locality_weight = 0.2 if remaining_nodes > 2 else 0.5
        hub_weight = 0.0 if remaining_nodes > 2 else 0.1

        # Penalty for nodes too far from current
        penalty = 0.0
        if distance_to_current > 1.2 * sum(distance_matrix[current_node][other] for other in unvisited_nodes) / len(unvisited_nodes):
            penalty = 0.2 * distance_to_current

        total_score = (progress_weight * distance_to_destination +
                       locality_weight * distance_to_current +
                       hub_weight * hub_score +
                       penalty)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

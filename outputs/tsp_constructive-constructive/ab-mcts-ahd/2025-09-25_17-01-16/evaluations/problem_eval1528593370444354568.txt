def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_value = float('inf')
    remaining_nodes = len(unvisited_nodes)
    temperature = max(0.1, 1.0 - 0.05 * remaining_nodes)  # Decaying temperature
    exploration_factor = np.random.uniform(0, 1)  # Random exploration term

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        value = (distance_to_current + distance_to_destination) * (1 - exploration_factor * temperature)

        if value < min_value:
            min_value = value
            next_node = node

    return next_node

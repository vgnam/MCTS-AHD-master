def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weighting based on remaining nodes
        weight = 0.5 if remaining_nodes > 1 else 0.8
        penalty = 1.0 if distance_to_current > 2 * distance_to_destination else 1.0

        score = distance_to_current + weight * distance_to_destination * penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

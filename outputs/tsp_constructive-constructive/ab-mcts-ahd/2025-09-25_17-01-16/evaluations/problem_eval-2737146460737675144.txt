def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    dynamic_factor = 0.5 / (1 + 0.1 * remaining_nodes)  # Adjusts based on remaining nodes
    penalty = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        score = -current_distance + dynamic_factor * destination_distance - penalty[node]

        if score > best_score:
            best_score = score
            next_node = node

    # Apply penalty to the next_node to discourage revisiting
    if next_node is not None:
        penalty[next_node] += 0.1

    return next_node

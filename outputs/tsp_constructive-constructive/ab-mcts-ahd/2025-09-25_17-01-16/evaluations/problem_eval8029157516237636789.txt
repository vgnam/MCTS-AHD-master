def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    alpha = 0.6  # Weighting factor for distance to current node (higher = more emphasis on proximity)
    beta = 0.4   # Weighting factor for distance to destination (higher = more emphasis on progress)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 1.0 if node in unvisited_nodes else 1.2  # Penalty for revisiting nodes

        score = (alpha * distance_to_current + beta * distance_to_destination) * penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

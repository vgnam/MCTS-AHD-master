def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    dynamic_factor = 0.5 / (1 + 0.1 * remaining_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        avg_distance_to_unvisited = sum(distance_matrix[node][other] for other in unvisited_nodes) / remaining_nodes
        destination_distance = distance_matrix[node][destination_node]
        score = -current_distance + dynamic_factor * (-avg_distance_to_unvisited + 0.2 * destination_distance)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

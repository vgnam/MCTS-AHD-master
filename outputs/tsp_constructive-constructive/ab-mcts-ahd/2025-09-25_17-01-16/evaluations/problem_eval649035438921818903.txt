def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        average_future_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        destination_proximity = distance_matrix[node][destination_node]
        weighted_score = 0.6 * immediate_distance + 0.2 * average_future_distance + 0.2 * destination_proximity

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

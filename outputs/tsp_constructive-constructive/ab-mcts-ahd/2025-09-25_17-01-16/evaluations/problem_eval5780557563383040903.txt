def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    exploration_weight = max(0.1, 0.5 * (remaining_nodes / len(distance_matrix)))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        penalty = 0.2 if estimated_distance_to_destination < 0.5 * distance_matrix[current_node][destination_node] else 0
        total_score = distance_to_current + exploration_weight * estimated_distance_to_destination + penalty

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

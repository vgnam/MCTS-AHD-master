def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        connectivity_score = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (remaining_nodes - 1) if remaining_nodes > 1 else 0

        if remaining_nodes > 0:
            weight = remaining_nodes / (remaining_nodes + 1)
            score = (1 - weight) * (-current_distance) + weight * (0.6 * destination_distance + 0.4 * connectivity_score)
        else:
            score = -current_distance

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

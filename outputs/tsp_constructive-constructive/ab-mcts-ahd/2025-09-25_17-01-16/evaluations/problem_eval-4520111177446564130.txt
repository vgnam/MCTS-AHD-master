def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    alpha = 0.5 * (1 + (remaining_nodes / (remaining_nodes + 1)))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 1.0 if distance_to_current < 0.1 * distance_matrix[current_node][destination_node] else 0.0
        score = distance_to_current + alpha * distance_to_destination + penalty * distance_to_current

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

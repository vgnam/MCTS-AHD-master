def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_product = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        weighted_product = (distance_to_current * 2 + estimated_distance_to_destination * 0.5) / (num_unvisited ** 0.5)
        weighted_product *= (distance_to_current + estimated_distance_to_destination)

        if weighted_product < min_weighted_product:
            min_weighted_product = weighted_product
            next_node = node

    return next_node

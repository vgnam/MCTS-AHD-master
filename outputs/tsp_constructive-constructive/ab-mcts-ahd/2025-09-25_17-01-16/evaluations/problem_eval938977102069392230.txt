def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight_factor = 0.5 + 0.4 * (remaining_distance / max(distance_matrix.values())) if remaining_distance > 0 else 0.5
        detour_penalty = max(0, distance_to_current - remaining_distance * 0.8)
        total_score = (1 - weight_factor) * distance_to_current + weight_factor * distance_to_destination + detour_penalty

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

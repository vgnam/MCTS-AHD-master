def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    alpha = 0.8  # Weight for current distance (prioritizes proximity)
    beta = 0.4   # Weight for destination distance (moderate emphasis)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        score = -alpha * current_distance + beta * destination_distance

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    weight_current = 1.0 + 0.3 * (num_unvisited / (num_unvisited + 1))
    weight_dest = 0.5 + 0.2 * (num_unvisited / (num_unvisited + 1))
    weight_hub = 0.5 * (1 - (num_unvisited / (num_unvisited + 1)) ** 3)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(1 / (distance_matrix[node][other] + 1e-6) for other in unvisited_nodes if other != node)
        path_memory = sum(distance_matrix[current_node][other] for other in unvisited_nodes if other != node) / (num_unvisited - 1 if num_unvisited > 1 else 1)
        total_score = weight_current * distance_to_current + weight_dest * distance_to_destination + weight_hub * hub_score - 0.1 * path_memory

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

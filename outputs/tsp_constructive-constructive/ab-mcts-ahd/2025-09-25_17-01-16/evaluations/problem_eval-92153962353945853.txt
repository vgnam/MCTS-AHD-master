def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    neighborhood_density = len(unvisited_nodes) / len(distance_matrix)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Hub potential calculation with adaptive normalization
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        hub_score = hub_score / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0

        # Novelty score: reward nodes with unique connections
        novelty_score = sum(1 for other in unvisited_nodes if distance_matrix[node][other] > distance_to_current * 1.5) / len(unvisited_nodes)

        # Path diversity bonus: reward nodes that connect to multiple unvisited regions
        diversity_bonus = sum(1 for other in unvisited_nodes if distance_matrix[node][other] > distance_to_current * 1.2) / len(unvisited_nodes)

        # Adaptive weights based on neighborhood density
        proximity_weight = 0.4 * (0.7 + 0.3 * neighborhood_density)
        destination_weight = 0.3 * (0.6 + 0.4 * (1 - neighborhood_density))
        hub_weight = 0.2 * (0.5 + 0.5 * neighborhood_density)
        novelty_weight = 0.1 * (0.5 + 0.5 * neighborhood_density)

        # Dynamic weight adjustment based on remaining nodes
        remaining_nodes = len(unvisited_nodes)
        weight_adj = 1.0 / (1.0 + 0.1 * remaining_nodes)

        # Penalize outliers in distance distribution
        distances_to_current = [distance_matrix[current_node][other] for other in unvisited_nodes]
        mean_dist = sum(distances_to_current) / len(distances_to_current)
        std_dev = (sum((d - mean_dist) ** 2 for d in distances_to_current) / len(distances_to_current)) ** 0.5
        outlier_penalty = 0.0 if std_dev == 0 else (distance_to_current - mean_dist) / std_dev

        total_score = ((proximity_weight * distance_to_current +
                        destination_weight * distance_to_destination +
                        hub_weight * hub_score +
                        novelty_weight * novelty_score +
                        0.1 * diversity_bonus) * weight_adj +
                       0.1 * outlier_penalty)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

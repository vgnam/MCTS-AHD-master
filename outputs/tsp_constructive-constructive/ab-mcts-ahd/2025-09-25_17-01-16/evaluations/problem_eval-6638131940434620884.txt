def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]

        exploration_factor = 1.0 - (remaining_nodes / len(unvisited_nodes)) ** 2
        proximity_weight = 0.5 - 0.3 * exploration_factor
        destination_weight = 0.5 + 0.3 * exploration_factor

        penalty = 1.0 if current_distance < 0.15 * distance_matrix[current_node][destination_node] else 0.0
        score = -current_distance * (1.0 - penalty) * proximity_weight + destination_distance * destination_weight

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

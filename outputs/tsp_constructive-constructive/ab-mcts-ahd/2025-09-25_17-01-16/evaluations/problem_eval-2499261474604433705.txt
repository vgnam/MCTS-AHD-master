def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        weight_proximity = 0.9 if remaining_nodes > 1 else 0.7
        weight_hub = 0.1 if remaining_nodes > 1 else 0.3
        total_score = weight_proximity * distance_to_current + weight_hub * hub_score

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

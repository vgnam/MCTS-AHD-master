def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    avg_distance_to_destination = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0
    problem_size_factor = 1 / (1 + remaining_nodes)  # Decreases as more nodes are visited
    node_centrality = {}

    for node in unvisited_nodes:
        node_centrality[node] = sum(distance_matrix[node][n] for n in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        distance_ratio = estimated_distance_to_destination / avg_distance_to_destination if avg_distance_to_destination > 0 else 1
        deviation_penalty = (distance_to_current + estimated_distance_to_destination) / (distance_matrix[current_node][destination_node] + 1e-6)
        centrality_boost = 1 / (1 + node_centrality[node])  # Higher for more central nodes
        total_score = (distance_to_current + problem_size_factor * estimated_distance_to_destination) * deviation_penalty * centrality_boost

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

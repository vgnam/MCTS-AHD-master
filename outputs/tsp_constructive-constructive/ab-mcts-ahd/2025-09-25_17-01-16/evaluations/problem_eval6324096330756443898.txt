def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    weight_current = 1.0 + 0.3 * (num_unvisited / (num_unvisited + 1))
    weight_dest = 1.0 - 0.2 * (num_unvisited / (num_unvisited + 1)) ** 3
    weight_hub = 0.5 * (1 - (num_unvisited / (num_unvisited + 1)) ** 2)
    history_penalty = 0.1 * num_unvisited

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (num_unvisited - 1)
        neighborhood_score = sum(distance_matrix[other][node] for other in unvisited_nodes if other != node) / (num_unvisited - 1)
        combined_hub = hub_score + 0.5 * neighborhood_score
        progress_factor = (1 - (num_unvisited / (num_unvisited + 1)) ** 4)
        historical_visits = sum(1 for n in unvisited_nodes if n == node)
        total_score = (weight_current * distance_to_current +
                       progress_factor * weight_dest * distance_to_destination +
                       weight_hub * combined_hub +
                       history_penalty * historical_visits)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

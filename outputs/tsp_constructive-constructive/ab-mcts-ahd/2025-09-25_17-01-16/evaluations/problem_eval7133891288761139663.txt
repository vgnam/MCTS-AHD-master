def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    if num_unvisited == 0:
        return destination_node

    weight_factor = 1 / num_unvisited if num_unvisited > 0 else 0.5
    immediate_weight = 0.7 * (1 - weight_factor)
    future_weight = 0.3 * weight_factor

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        score = immediate_weight * immediate_distance + future_weight * future_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

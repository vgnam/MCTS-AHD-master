def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)

        # Dynamic weighting based on remaining nodes
        locality_weight = 0.5 - 0.1 * (remaining_nodes - 1)
        progress_weight = 0.3 + 0.1 * (remaining_nodes - 1)
        hub_weight = 0.2 + 0.05 * (remaining_nodes - 1)

        # Dynamic penalty based on average remaining distance
        avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes)
        penalty = 0.2 * distance_to_current if distance_to_current > 1.2 * avg_distance else 0.0

        total_score = (locality_weight * distance_to_current +
                       progress_weight * distance_to_destination +
                       hub_weight * hub_score +
                       penalty)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

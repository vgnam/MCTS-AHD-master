def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.7):
    next_node = None
    min_total_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        weighted_destination_distance = alpha * distance_matrix[node][destination_node] + (1 - alpha) * distance_matrix[current_node][destination_node]
        total_score = distance_to_current * weighted_destination_distance

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

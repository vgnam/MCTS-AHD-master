def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_weighted_sum = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight using sigmoid function
        weight = 1 / (1 + np.exp(-(remaining_nodes - 5)))

        # Penalize nodes too far from current when few nodes remain
        if remaining_nodes <= 3:
            distance_penalty = max(0, distance_to_current - 2 * np.mean(distance_matrix[current_node]))
            weighted_sum = (1 - weight) * (distance_to_current + distance_penalty) + weight * estimated_distance_to_destination
        else:
            weighted_sum = (1 - weight) * distance_to_current + weight * estimated_distance_to_destination

        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node

    return next_node

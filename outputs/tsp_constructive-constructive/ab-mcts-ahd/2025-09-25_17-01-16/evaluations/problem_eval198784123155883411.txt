def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_value = float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + 0.1 * remaining_nodes)
    destination_weight = 0.3 * (0.7 ** remaining_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = max(0, distance_to_destination - distance_matrix[current_node][destination_node])
        value = distance_to_current + destination_weight * (distance_to_destination ** weight_factor) + 0.1 * penalty

        if value < min_value:
            min_value = value
            next_node = node

    return next_node

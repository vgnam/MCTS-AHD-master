def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1
    weight_current = 1.0 + 0.1 * remaining_nodes
    weight_dest = 1.0 - 0.05 * remaining_nodes
    weight_hub = 0.5 + 0.05 * remaining_nodes

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        hub_score = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        weight_factor = (total_nodes / remaining_nodes) if remaining_nodes > 0 else 1.0
        score = weight_current * (-current_distance) + weight_factor * weight_dest * destination_distance + weight_hub * (-hub_score)

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

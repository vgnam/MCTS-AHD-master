def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if estimated_distance_to_destination == 0:
            score = float('inf')
        else:
            dynamic_alpha = 1 / (1 + distance_to_current / estimated_distance_to_destination)
            score = dynamic_alpha * distance_to_current + (1 - dynamic_alpha) * estimated_distance_to_destination

            if distance_to_current > estimated_distance_to_destination:
                score += 0.5 * distance_to_current

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Progress factor: scales with fraction of remaining nodes
        progress_factor = (total_unvisited / (total_unvisited + 1)) ** 2

        # Locality bonus: rewards nodes close to both current and destination
        locality_bonus = 1 / (1 + distance_to_current + distance_to_destination)

        # Dynamic weight combining progress and locality
        weight = 0.5 * (1 - progress_factor)

        # Weighted score with locality bonus
        total_score = weight * distance_to_current + (1 - weight) * distance_to_destination + locality_bonus

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

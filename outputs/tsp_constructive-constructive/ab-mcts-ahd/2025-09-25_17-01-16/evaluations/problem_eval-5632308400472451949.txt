import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        progress_factor = 1 / (1 + math.exp(-(total_unvisited - len(unvisited_nodes) + 1)))
        total_score = (1 - progress_factor) * 0.7 * distance_to_current + progress_factor * 0.3 * estimated_distance_to_destination

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

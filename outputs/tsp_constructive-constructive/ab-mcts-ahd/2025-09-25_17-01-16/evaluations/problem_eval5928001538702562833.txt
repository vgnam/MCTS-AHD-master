def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        dynamic_weight = 0.5 * (remaining_nodes / len(unvisited_nodes))
        penalty = 0 if node == destination_node else 1
        score = current_distance + dynamic_weight * destination_distance * penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

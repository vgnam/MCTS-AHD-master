def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        total_distance = distance_matrix[current_node][destination_node]

        normalized_distance = distance_to_current / total_distance
        progress_weight = (1 - normalized_distance) if total_distance > 0 else 0
        penalty_term = (distance_to_destination / total_distance) if total_distance > 0 else 0

        score = (progress_weight * distance_to_current) + (penalty_term * distance_to_destination)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

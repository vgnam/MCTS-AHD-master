def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weight based on remaining unvisited nodes
        weight = 0.5 * (1 - (total_unvisited / (total_unvisited + 1)))

        # Weighted score: closer nodes matter more early, destination proximity matters more later
        total_score = weight * distance_to_current + (1 - weight) * distance_to_destination

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

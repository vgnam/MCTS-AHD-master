def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_distance = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_current == 0 or estimated_distance_to_destination == 0:
            weight = 0.5
        else:
            weight = 1 / (distance_to_current * estimated_distance_to_destination)

        total_distance = (distance_to_current + estimated_distance_to_destination) * weight

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]

        if remaining_nodes > 1:
            weight_immediate = 0.5 + 0.25 * (1 - 1 / remaining_nodes)
            weight_destination = 0.5 - 0.25 * (1 - 1 / remaining_nodes)
        else:
            weight_immediate = 0.7
            weight_destination = 0.3

        penalty = 1.0 if node == destination_node else 0.0

        weighted_score = (weight_immediate * current_distance + weight_destination * destination_distance) * (1 + penalty)

        if weighted_score < best_score:
            best_score = weighted_score
            next_node = node

    return next_node

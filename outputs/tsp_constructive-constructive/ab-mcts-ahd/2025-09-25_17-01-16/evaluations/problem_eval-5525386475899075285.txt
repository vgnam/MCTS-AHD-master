def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    proximity_factor = 1.0 / (1.0 + distance_matrix[current_node][destination_node])

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node_to_dest = distance_matrix[node][destination_node]
        future_weight = 0.5 * (1.0 - proximity_factor) * (remaining_nodes / (remaining_nodes + 1))
        combined_score = distance_to_node + future_weight * distance_from_node_to_dest

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

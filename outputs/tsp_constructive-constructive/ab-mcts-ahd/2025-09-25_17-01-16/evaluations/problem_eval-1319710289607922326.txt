def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_total_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    avg_distance_to_destination = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]
        distance_ratio = estimated_distance_to_destination / (avg_distance_to_destination + 1e-6)
        penalty_factor = 1 + (distance_to_current ** 2) / (distance_matrix[current_node][destination_node] ** 2 + 1e-6)
        total_score = (distance_to_current + estimated_distance_to_destination) * penalty_factor * (1 - distance_ratio)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5, beta=0.5):
    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_score = alpha * distance_to_current + beta * distance_to_destination

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

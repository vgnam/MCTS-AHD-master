def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    alpha = max(0.1, 0.5 * (remaining_nodes / len(distance_matrix)))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        estimated_distance_to_destination = distance_matrix[node][destination_node]

        if estimated_distance_to_destination == 0:
            score = float('inf')
        else:
            score = alpha * distance_to_current + (1 - alpha) * estimated_distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

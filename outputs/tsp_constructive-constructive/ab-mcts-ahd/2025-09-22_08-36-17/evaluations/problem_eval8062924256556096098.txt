def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    if destination_node in unvisited_nodes:
        return destination_node

    min_max_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        max_distance = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                current_distance = distance_matrix[node][other_node]
                if current_distance > max_distance:
                    max_distance = current_distance
        current_to_node_distance = distance_matrix[current_node][node]
        total_penalty = max_distance + current_to_node_distance

        if total_penalty < min_max_distance:
            min_max_distance = total_penalty
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        sum_future_distances = sum(distance_matrix[node][other] for other in unvisited_nodes)
        if sum_future_distances == 0:
            score = float('inf')
        else:
            score = current_distance / sum_future_distances

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

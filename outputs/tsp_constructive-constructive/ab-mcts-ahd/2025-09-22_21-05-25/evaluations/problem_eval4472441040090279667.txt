def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    potential_nodes = {}
    for node in unvisited_nodes:
        min_distance = min(distance_matrix[node][other] for other in unvisited_nodes if other != node)
        potential_nodes[node] = min_distance
    next_node = max(potential_nodes.items(), key=lambda item: item[1])[0]
    return next_node

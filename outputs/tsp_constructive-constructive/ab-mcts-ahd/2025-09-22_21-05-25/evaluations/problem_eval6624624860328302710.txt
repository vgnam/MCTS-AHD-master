defaulting to the nearest node, ensuring adaptability to varying problem scales and densities.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    nearest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    farthest_node = max(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    dynamic_threshold = 1.2 * avg_distance + (10 / len(unvisited_nodes)) if len(unvisited_nodes) > 1 else float('inf')
    next_node = farthest_node if distance_matrix[current_node][farthest_node] > dynamic_threshold else nearest_node
    return next_node

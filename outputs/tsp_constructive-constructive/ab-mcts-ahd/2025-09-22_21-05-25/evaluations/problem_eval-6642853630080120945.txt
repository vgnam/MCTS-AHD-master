def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, lookahead_weight=0.5):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_to_node = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_unvisited = [n for n in unvisited_nodes if n != node]
            future_costs = sorted([distance_matrix[node][n] for n in remaining_unvisited])[:3]
            weighted_future = sum(f * (lookahead_weight ** i) for i, f in enumerate(future_costs))
            return current_to_node + weighted_future
        else:
            return current_to_node

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_to_node = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 1:
            remaining_unvisited = [n for n in unvisited_nodes if n != node]
            avg_remaining_distance = sum(distance_matrix[node][n] for n in remaining_unvisited) / len(remaining_unvisited)
            random_factor = 0.3 * (distance_matrix[node][remaining_unvisited[0]] - distance_matrix[node][remaining_unvisited[-1]])
            return current_to_node + 0.7 * avg_remaining_distance + random_factor
        else:
            return current_to_node

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

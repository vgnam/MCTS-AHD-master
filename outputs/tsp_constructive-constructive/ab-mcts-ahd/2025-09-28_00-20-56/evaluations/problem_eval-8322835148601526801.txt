def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    min_current_distance = float('inf')
    next_node_by_total = None
    next_node_by_current = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = current_distance + remaining_distance

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node_by_total = node

        if current_distance < min_current_distance:
            min_current_distance = current_distance
            next_node_by_current = node

    if destination_node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][destination_node]
        if direct_distance <= min_total_distance * 0.6:
            return destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def calculate_score(node):
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = sum(distance_matrix[node][other_node] for other_node in unvisited_nodes if other_node != node)
        avg_distance = total_distance / len(unvisited_nodes) if unvisited_nodes else 0
        return remaining_distance + avg_distance

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

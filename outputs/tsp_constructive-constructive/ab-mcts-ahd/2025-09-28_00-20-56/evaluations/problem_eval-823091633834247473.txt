def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = current_distance + remaining_distance

        # Dynamic threshold based on remaining nodes and current path
        threshold = 0.8 - (0.2 * (remaining_nodes / (remaining_nodes + 1)))

        if node == destination_node:
            if current_distance <= min_total_distance * threshold:
                min_total_distance = total_distance
                next_node = node
        else:
            if total_distance < min_total_distance:
                min_total_distance = total_distance
                next_node = node

    return next_node

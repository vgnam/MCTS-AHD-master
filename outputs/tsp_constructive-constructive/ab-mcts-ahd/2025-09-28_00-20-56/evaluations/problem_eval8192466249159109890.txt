def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_count = len(unvisited_nodes)
    scaling_factor = 1.0 / remaining_count if remaining_count > 0 else 1.0

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if remaining_count > 2:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            next_best = min([n for n in remaining_nodes if n != best_lookahead], key=lambda n: distance_matrix[best_lookahead][n] + distance_matrix[n][destination_node])
            lookahead_dist = distance_matrix[node][best_lookahead] + distance_matrix[best_lookahead][next_best] + distance_matrix[next_best][destination_node]
            adjusted_lookahead_dist = lookahead_dist * scaling_factor
        elif remaining_count == 2:
            next_node = next(iter([n for n in unvisited_nodes if n != node]))
            lookahead_dist = distance_matrix[node][next_node] + distance_matrix[next_node][destination_node]
            adjusted_lookahead_dist = lookahead_dist * scaling_factor
        else:
            adjusted_lookahead_dist = distance_matrix[node][destination_node]
        return current_dist + 0.3 * adjusted_lookahead_dist

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        detour_penalty = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                detour_penalty += max(0, distance_matrix[current_node][node] + distance_matrix[node][other_node] - distance_matrix[current_node][other_node])
        normalized_penalty = detour_penalty / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
        proximity_bonus = (1.0 - (remaining_distance / max(distance_matrix[node]))) * 0.5
        score = current_distance + normalized_penalty - proximity_bonus
        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance <= min_score * 1.5:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, k=3):
    if not unvisited_nodes:
        return destination_node

    total_remaining_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes)
    avg_remaining_distance = total_remaining_distance / len(unvisited_nodes)

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if len(unvisited_nodes) > k:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: sum(distance_matrix[node][n] + distance_matrix[n][m] for m in remaining_nodes if m != n))
            lookahead_dist = sum(distance_matrix[node][n] + distance_matrix[n][best_lookahead] for n in remaining_nodes if n != best_lookahead)
            adjusted_lookahead_dist = lookahead_dist * (avg_remaining_distance / lookahead_dist) ** 2
        else:
            adjusted_lookahead_dist = distance_matrix[node][destination_node]
        return current_dist + 0.5 * adjusted_lookahead_dist

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

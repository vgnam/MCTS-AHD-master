def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = -1
    total_remaining = sum(distance_matrix[current_node][node] for node in unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        remaining_dist = distance_matrix[node][destination_node]

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            lookahead_dist = distance_matrix[node][best_lookahead] + distance_matrix[best_lookahead][destination_node]
        else:
            lookahead_dist = distance_matrix[node][destination_node]

        proximity_weight = 0.5 + (0.1 * (total_remaining / (distance + 1e-10)))
        lookahead_weight = 0.5 - (0.1 * (total_remaining / (distance + 1e-10)))
        total_score = (remaining_dist / (distance + 1e-10)) * (1 - lookahead_weight) + (lookahead_dist / (distance + 1e-10)) * lookahead_weight

        if total_score > best_score:
            best_score = total_score
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        if destination_distance <= distance_matrix[current_node][next_node] * 1.3:
            next_node = destination_node

    return next_node

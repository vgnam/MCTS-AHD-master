def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = current_distance + remaining_distance

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    # Adjust threshold dynamically based on the number of unvisited nodes
    dynamic_threshold = max(0.5, 0.8 - (len(unvisited_nodes) / len(distance_matrix)))

    if destination_node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][destination_node]
        if direct_distance <= min_total_distance * dynamic_threshold:
            next_node = destination_node

    return next_node

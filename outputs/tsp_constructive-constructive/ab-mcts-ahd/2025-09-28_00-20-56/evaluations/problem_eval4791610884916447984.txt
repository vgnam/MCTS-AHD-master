def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    min_total_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = current_distance + remaining_distance
        adjusted_distance = total_distance + (total_avg_distance * 0.3)

        if adjusted_distance < min_total_distance:
            min_total_distance = adjusted_distance
            next_node = node

    dynamic_threshold = 0.6 + (len(unvisited_nodes) / len(distance_matrix)) * 0.2
    if destination_node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][destination_node]
        if direct_distance <= min_total_distance * dynamic_threshold:
            next_node = destination_node

    return next_node

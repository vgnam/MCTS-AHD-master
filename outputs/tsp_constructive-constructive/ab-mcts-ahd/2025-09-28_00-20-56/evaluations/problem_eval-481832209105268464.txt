def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def calculate_score(node):
        num_unvisited = len(unvisited_nodes)
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        detour_penalty = sum(max(0, distance_matrix[current_node][node] + distance_matrix[node][other_node] - distance_matrix[current_node][other_node]) for other_node in unvisited_nodes if other_node != node)

        base_score = (0.6 * current_distance + 0.3 * remaining_distance) + (0.1 / num_unvisited) * detour_penalty

        if node == destination_node:
            threshold = 0.8 - 0.1 * (num_unvisited / (num_unvisited + 1))
            if current_distance <= base_score * threshold:
                return -float('inf')

        exploration_factor = 1.0 - (0.5 * (num_unvisited / (num_unvisited + 1)))
        score = base_score * exploration_factor

        return score

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_remaining_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes)
    avg_remaining_distance = total_remaining_distance / len(unvisited_nodes) if unvisited_nodes else 0

    def calculate_score(node):
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        detour_penalty = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                detour_penalty += max(0, distance_matrix[current_node][node] + distance_matrix[node][other_node] - distance_matrix[current_node][other_node])
        scaled_penalty = detour_penalty * (current_distance / avg_remaining_distance) if avg_remaining_distance > 0 else detour_penalty
        return current_distance + remaining_distance + scaled_penalty

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def calculate_score(node):
        num_unvisited = len(unvisited_nodes)
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        detour_penalty = sum(max(0, distance_matrix[current_node][node] + distance_matrix[node][other_node] - distance_matrix[current_node][other_node]) for other_node in unvisited_nodes if other_node != node)
        weight_current_distance = 1 / num_unvisited if num_unvisited > 1 else 1
        weight_remaining_distance = 1 / (num_unvisited ** 2) if num_unvisited > 1 else 1
        weight_detour_penalty = num_unvisited
        return (weight_current_distance * current_distance + 
                weight_remaining_distance * remaining_distance + 
                weight_detour_penalty * detour_penalty)

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

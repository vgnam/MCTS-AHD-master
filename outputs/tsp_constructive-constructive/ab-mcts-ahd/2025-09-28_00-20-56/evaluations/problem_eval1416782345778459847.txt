def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        destination_distance = distance_matrix[current_node][destination_node]
        dynamic_threshold = 1.5 + (0.5 * (destination_distance / (avg_distance + 1e-6)))
        prob_threshold = 0.7 * (1 / (1 + (destination_distance / (avg_distance + 1e-6)))) + 0.3 * random.random()

        if destination_distance <= min_distance * dynamic_threshold and prob_threshold > 0.5:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def calculate_score(node):
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        if unvisited_nodes:
            avg_detour = sum(distance_matrix[node][other_node] for other_node in unvisited_nodes) / len(unvisited_nodes)
        else:
            avg_detour = 0
        detour_penalty = max(0, avg_detour - distance_matrix[current_node][node])
        return current_distance + remaining_distance + detour_penalty

    next_node = min(unvisited_nodes, key=calculate_score)
    return next_node

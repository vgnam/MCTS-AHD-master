def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        remaining_count = len(unvisited_nodes)
        weight_immediate = 0.6 / (remaining_count + 1)
        weight_lookahead = 0.3 * (remaining_count / (remaining_count + 1))
        weight_density = 0.1 / (remaining_count + 1)

        current_dist = distance_matrix[current_node][node]

        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            lookahead_dist = distance_matrix[node][best_lookahead] + distance_matrix[best_lookahead][destination_node]
            density = sum(distance_matrix[node][n] for n in remaining_nodes) / len(remaining_nodes)
        else:
            lookahead_dist = distance_matrix[node][destination_node]
            density = 0

        return weight_immediate * current_dist + weight_lookahead * lookahead_dist + weight_density * density

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

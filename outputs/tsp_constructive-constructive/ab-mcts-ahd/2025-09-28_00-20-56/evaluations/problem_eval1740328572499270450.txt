def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]
        total_distance = current_distance + remaining_distance

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    # Calculate dynamic threshold based on remaining flexibility
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / unvisited_count
    threshold_factor = 0.7 + 0.1 * (unvisited_count / len(distance_matrix))  # Adjusts with remaining nodes

    if destination_node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][destination_node]
        if direct_distance <= min_total_distance * threshold_factor and direct_distance <= avg_distance * 1.2:
            next_node = destination_node

    return next_node

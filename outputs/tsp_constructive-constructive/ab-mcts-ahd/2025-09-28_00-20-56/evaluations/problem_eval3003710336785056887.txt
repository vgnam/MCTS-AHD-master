def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 2:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            second_best = min([n for n in remaining_nodes if n != best_lookahead], key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            lookahead_dist = distance_matrix[node][best_lookahead] + distance_matrix[best_lookahead][second_best] + distance_matrix[second_best][destination_node]
        elif len(unvisited_nodes) > 1:
            lookahead_dist = distance_matrix[node][next(iter([n for n in unvisited_nodes if n != node]))] + distance_matrix[next(iter([n for n in unvisited_nodes if n != node]))][destination_node]
        else:
            lookahead_dist = distance_matrix[node][destination_node]
        return 0.7 * current_dist + 0.3 * lookahead_dist

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

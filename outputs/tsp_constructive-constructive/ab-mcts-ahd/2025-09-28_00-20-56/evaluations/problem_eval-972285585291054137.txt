def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_remaining_distance = max(distance_matrix[node][destination_node] for node in unvisited_nodes)

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if len(unvisited_nodes) > 2:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            second_node = best_lookahead
            remaining_nodes = [n for n in remaining_nodes if n != second_node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[second_node][n] + distance_matrix[n][destination_node])
            third_node = best_lookahead
            lookahead_dist = distance_matrix[node][second_node] + distance_matrix[second_node][third_node] + distance_matrix[third_node][destination_node]
            adjusted_lookahead_dist = lookahead_dist * (max_remaining_distance / lookahead_dist)
        elif len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != node]
            best_lookahead = min(remaining_nodes, key=lambda n: distance_matrix[node][n] + distance_matrix[n][destination_node])
            lookahead_dist = distance_matrix[node][best_lookahead] + distance_matrix[best_lookahead][destination_node]
            adjusted_lookahead_dist = lookahead_dist * (max_remaining_distance / lookahead_dist)
        else:
            adjusted_lookahead_dist = distance_matrix[node][destination_node]
        return current_dist + 0.3 * adjusted_lookahead_dist

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

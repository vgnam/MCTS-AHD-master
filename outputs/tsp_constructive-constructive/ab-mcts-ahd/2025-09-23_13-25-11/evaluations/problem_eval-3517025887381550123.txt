def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        future_dists = [distance_matrix[node][other] for other in remaining_nodes]
        max_future_dist = max(future_dists)
        min_future_dist = min(future_dists)
        variance = (max_future_dist - min_future_dist) / (max_future_dist + 1e-10)

        dynamic_weight = len(remaining_nodes) / (len(unvisited_nodes) + 1e-10)
        balanced_future = (max_future_dist + min_future_dist) / 2

        score = (current_dist * (1 - dynamic_weight) + balanced_future * dynamic_weight) * (1 + variance)
        normalized_score = score / (1 + np.log1p(len(remaining_nodes)))
        return normalized_score

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

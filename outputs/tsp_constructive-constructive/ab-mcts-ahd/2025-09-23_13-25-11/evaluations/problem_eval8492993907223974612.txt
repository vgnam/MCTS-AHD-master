def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    alpha = 0.7  # Tunable parameter to balance immediate and average distances
    next_node = min(unvisited_nodes, key=lambda node: alpha * distance_matrix[current_node][node] + (1 - alpha) * (sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes)))
    return next_node

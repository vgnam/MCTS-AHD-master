def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic(node):
        immediate_distance = distance_matrix[current_node][node]
        future_avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
        return immediate_distance / (future_avg_distance + 1)

    next_node = min(unvisited_nodes, key=heuristic)
    return next_node

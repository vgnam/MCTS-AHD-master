def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)
    weight_current = math.log(remaining_nodes + 1) / math.log(total_nodes + 1)
    weight_future = 1 - weight_current
    avg_dist = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes if remaining_nodes else 0
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        avg_remaining_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes) if remaining_nodes else 0
        return weight_current * current_dist + weight_future * (min_future_dist + avg_remaining_dist) / 2
    next_node = min(unvisited_nodes, key=lambda node: evaluate_node(node) if node != destination_node else float('inf'))
    return next_node

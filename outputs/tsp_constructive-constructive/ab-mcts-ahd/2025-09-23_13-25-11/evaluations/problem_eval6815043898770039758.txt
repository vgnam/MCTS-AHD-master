def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)
    centrality = sum(1 for node in unvisited_nodes if distance_matrix[current_node][node] < sum(distance_matrix[node]) / total_nodes)
    weight_current = (centrality / total_nodes) * (remaining_nodes / total_nodes)
    weight_future = 1 - weight_current
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            return current_dist
        min_future_dist = min(distance_matrix[node][other] for other in unvisited_nodes - {node})
        avg_future_dist = sum(distance_matrix[node][other] for other in unvisited_nodes - {node}) / len(unvisited_nodes - {node})
        return weight_current * current_dist + weight_future * (0.7 * min_future_dist + 0.3 * avg_future_dist)
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

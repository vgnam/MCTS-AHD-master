def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_count = len(unvisited_nodes)
    decay_factor = 0.9 ** remaining_count
    weight_immediate = 0.5 + 0.4 * (1 - decay_factor)
    weight_adapt = 1 - weight_immediate
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        min_remaining_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        future_dist = max(distance_matrix[node][other] for other in remaining_nodes)
        return weight_immediate * current_dist + weight_adapt * (future_dist - min_remaining_dist)
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

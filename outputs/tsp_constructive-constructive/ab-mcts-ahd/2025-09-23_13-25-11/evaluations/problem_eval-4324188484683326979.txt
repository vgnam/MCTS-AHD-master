def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        centrality = sum(distance_matrix[node][other] for other in unvisited_nodes) / len(unvisited_nodes)
        future_avg = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        weight = len(remaining_nodes) / len(unvisited_nodes)

        proximity_weight = 0.6
        centrality_weight = 0.3 * weight
        future_weight = 0.2

        return (proximity_weight * current_dist +
                centrality_weight * centrality +
                future_weight * future_avg)

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

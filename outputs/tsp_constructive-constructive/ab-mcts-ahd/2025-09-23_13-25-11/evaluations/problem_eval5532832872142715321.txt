def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    decay_factor = 0.5
    next_node = min(unvisited_nodes, key=lambda node: (distance_matrix[current_node][node] * (1 - decay_factor)) + (sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) * decay_factor))
    return next_node

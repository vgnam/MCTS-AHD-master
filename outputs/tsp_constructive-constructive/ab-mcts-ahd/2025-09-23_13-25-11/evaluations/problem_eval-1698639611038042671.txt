def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def heuristic_score(node):
        distance = distance_matrix[current_node][node]
        future_cost = min(distance_matrix[node][n] for n in unvisited_nodes - {node}) if unvisited_nodes - {node} else 0
        return distance + 0.5 * future_cost

    next_node = max(unvisited_nodes, key=heuristic_score)
    return next_node

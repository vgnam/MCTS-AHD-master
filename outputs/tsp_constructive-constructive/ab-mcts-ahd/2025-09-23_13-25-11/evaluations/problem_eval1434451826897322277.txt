def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    base_weight = 0.5  # Base weight for balancing current and future distances
    penalty_factor = 1.0 / (1.0 + total_unvisited)  # Penalty decreases as nodes are visited
    weight_current = base_weight + (1.0 - base_weight) * penalty_factor
    weight_future = 1.0 - weight_current

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        penalty = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        return weight_current * current_dist + weight_future * min_future_dist - penalty_factor * penalty

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist

        max_future_dist = max(distance_matrix[node][other] for other in remaining_nodes)
        ratio = len(remaining_nodes) / len(unvisited_nodes)
        weight = 1 / (1 + math.exp(-10 * (ratio - 0.5)))
        penalty = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes) if remaining_nodes else 0
        return (1 - weight) * current_dist + weight * max_future_dist + 0.1 * penalty

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_count = len(unvisited_nodes)
    weight_immediate = 0.5 * (remaining_count / (remaining_count + 1))
    weight_future = 0.5 * (1 / (remaining_count + 1))

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist

        distances_to_remaining = [distance_matrix[node][other] for other in remaining_nodes]
        mean_future_dist = sum(distances_to_remaining) / len(distances_to_remaining)
        variance_future_dist = sum((d - mean_future_dist) ** 2 for d in distances_to_remaining) / len(distances_to_remaining)

        return weight_immediate * current_dist + weight_future * variance_future_dist

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

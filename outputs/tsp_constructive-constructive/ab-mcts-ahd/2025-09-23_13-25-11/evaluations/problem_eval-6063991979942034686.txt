def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_nodes = len(unvisited_nodes) + 1
    current_weight = 0.5 + (0.5 * (remaining_nodes / (remaining_nodes + 1)))
    dest_weight = 1.0 - current_weight
    next_node = min(unvisited_nodes, key=lambda node: (current_weight * distance_matrix[current_node][node] + dest_weight * distance_matrix[node][destination_node]))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    penalty_factor = 0.5
    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] + penalty_factor * distance_matrix[node][destination_node])
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    remaining_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        avg_future_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        future_impact = (current_dist - min_future_dist) if remaining_nodes else 0
        dynamic_weight = 0.5 + 0.4 * remaining_ratio
        smoothing_factor = 0.1 + 0.8 * (1 - remaining_ratio)
        penalized_impact = future_impact * (1 + 0.2 * (avg_future_dist - min_future_dist) / (avg_future_dist + 1e-6))
        return dynamic_weight * current_dist + (1 - dynamic_weight) * penalized_impact * smoothing_factor
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    closest_half = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])[:len(unvisited_nodes)//2]
    next_node = max(closest_half, key=lambda node: distance_matrix[current_node][node])
    return next_node

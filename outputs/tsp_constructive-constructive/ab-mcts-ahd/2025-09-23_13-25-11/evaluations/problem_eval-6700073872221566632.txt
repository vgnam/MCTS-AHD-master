def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist

        # Average future distance
        avg_future_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)

        # Node centrality: average distance to remaining nodes
        centrality = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)

        # Dynamic exploration factor using sigmoid function
        remaining_ratio = len(remaining_nodes) / len(unvisited_nodes)
        exploration_factor = 1 / (1 + math.exp(-5 * (remaining_ratio - 0.5)))

        # Weighted score combining current distance, future distance, and centrality
        score = (1 - exploration_factor) * current_dist + \
                exploration_factor * (0.7 * avg_future_dist + 0.3 * centrality)

        return score

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

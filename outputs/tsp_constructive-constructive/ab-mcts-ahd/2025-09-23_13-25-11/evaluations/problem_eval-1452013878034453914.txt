def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        # Calculate average and variance of future distances
        future_dists = [distance_matrix[node][other] for other in remaining_nodes]
        avg_future_dist = sum(future_dists) / len(future_dists)
        variance = sum((d - avg_future_dist) ** 2 for d in future_dists) / len(future_dists)

        # Dynamic weighting: prioritize immediate distance when few nodes remain
        weight_immediate = 0.7 if len(remaining_nodes) <= 3 else 0.5
        weight_avg = 1 - weight_immediate

        # Penalize high variance to avoid unstable paths
        penalty = 0.1 * variance

        return weight_immediate * current_dist + weight_avg * avg_future_dist + penalty

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

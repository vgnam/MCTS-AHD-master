def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)
    weight_current = (remaining_nodes / total_nodes) ** 2
    weight_future = 1 - weight_current
    heuristic_scale = 1 + (1 / (remaining_nodes + 1))
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist * heuristic_scale
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        avg_remaining_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        normalized_score = (current_dist + min_future_dist + avg_remaining_dist) / 3
        return weight_current * current_dist + weight_future * normalized_score * heuristic_scale
    next_node = min(unvisited_nodes, key=lambda node: evaluate_node(node) if node != destination_node else float('inf'))
    return next_node

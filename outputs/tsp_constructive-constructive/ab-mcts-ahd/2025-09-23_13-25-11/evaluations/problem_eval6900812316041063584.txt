def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1
    remaining_nodes = len(unvisited_nodes)
    weight_current = remaining_nodes / total_nodes
    weight_future = 1 - weight_current
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            return current_dist
        avg_future_dist = sum(distance_matrix[node][other] for other in unvisited_nodes - {node}) / len(unvisited_nodes - {node})
        return weight_current * current_dist + weight_future * (current_dist / (avg_future_dist + 1))
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

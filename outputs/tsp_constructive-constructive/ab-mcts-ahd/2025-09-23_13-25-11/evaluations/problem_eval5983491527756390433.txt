def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight1=0.4, weight2=0.6):
    if not unvisited_nodes:
        return destination_node
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        if not unvisited_nodes - {node}:
            return current_dist
        sum_future_dist = sum(distance_matrix[node][other] for other in unvisited_nodes - {node})
        return weight1 * current_dist + weight2 * sum_future_dist
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

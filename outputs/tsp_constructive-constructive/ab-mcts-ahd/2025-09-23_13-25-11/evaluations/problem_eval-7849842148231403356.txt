def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_count = len(unvisited_nodes)
    penalty_weight = 0.3 * (remaining_count / (len(distance_matrix) - 1))
    next_node = max(unvisited_nodes, key=lambda node: distance_matrix[current_node][node] - penalty_weight * distance_matrix[node][destination_node])
    return next_node

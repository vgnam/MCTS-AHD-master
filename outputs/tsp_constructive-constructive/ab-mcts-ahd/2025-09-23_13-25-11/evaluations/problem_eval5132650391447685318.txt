def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    remaining_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0

    def heuristic(node):
        current_dist = distance_matrix[current_node][node]
        future_dist = distance_matrix[node][destination_node]
        penalty = 2.0 if node == current_node else 0.0  # Stronger penalty for revisiting
        dynamic_weight = 1.0 - 0.75 * (remaining_ratio ** 2)  # Non-linear weight adjustment
        return current_dist + dynamic_weight * future_dist + penalty

    next_node = min(unvisited_nodes, key=heuristic)
    return next_node

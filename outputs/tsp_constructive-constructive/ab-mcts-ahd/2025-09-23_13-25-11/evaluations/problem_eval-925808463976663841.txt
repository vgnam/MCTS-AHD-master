def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    remaining_count = len(unvisited_nodes)
    weight_immediate = 0.3 + 0.5 * (1 - min(remaining_count / (len(distance_matrix) - 1), 1))
    weight_future = 1 - weight_immediate
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        avg_future_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)
        max_future_dist = max(distance_matrix[node][other] for other in remaining_nodes)
        centrality_penalty = 0.3 * avg_future_dist if remaining_count <= 3 else 0
        future_score = (min_future_dist + max_future_dist) / 2
        return weight_immediate * current_dist + weight_future * (future_score - current_dist) + centrality_penalty
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

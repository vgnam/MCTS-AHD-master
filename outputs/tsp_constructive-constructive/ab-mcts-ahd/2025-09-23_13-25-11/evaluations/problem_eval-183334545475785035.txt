def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = min(unvisited_nodes, key=lambda node: (0.5 * distance_matrix[current_node][node] + 0.3 * distance_matrix[node][destination_node] + 0.2 * min(distance_matrix[node][n] for n in unvisited_nodes if n != node)))
    return next_node

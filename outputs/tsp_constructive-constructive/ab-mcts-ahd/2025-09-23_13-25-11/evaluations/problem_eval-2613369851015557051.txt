def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}
        if not remaining_nodes:
            return current_dist
        future_dists = [distance_matrix[node][other] for other in remaining_nodes]
        max_future_dist = max(future_dists)
        min_future_dist = min(future_dists)
        density_factor = 1 / len(remaining_nodes)
        variance_factor = (max_future_dist - min_future_dist) / (max_future_dist + 1e-6)
        weight = 0.5 + 0.5 * (density_factor * variance_factor)
        return weight * current_dist + (1 - weight) * (current_dist - max_future_dist)
    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

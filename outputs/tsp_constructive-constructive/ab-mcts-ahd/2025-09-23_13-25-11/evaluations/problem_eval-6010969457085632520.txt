def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    remaining_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0
    dynamic_weight = 0.5 + 0.4 * remaining_ratio
    randomness_factor = 0.1 + 0.3 * remaining_ratio

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        avg_future_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)
        centrality = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)

        weighted_score = (dynamic_weight * current_dist +
                          (1 - dynamic_weight) * avg_future_dist +
                          randomness_factor * centrality)

        return weighted_score

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

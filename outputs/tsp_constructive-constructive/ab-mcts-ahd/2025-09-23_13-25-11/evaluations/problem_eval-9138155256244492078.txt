def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    remaining_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0

    def heuristic(node):
        current_dist = distance_matrix[current_node][node]
        future_dist = distance_matrix[node][destination_node]
        penalty = 1.0 if node == current_node else 0.0
        dynamic_weight = 1.0 - 0.5 * remaining_ratio
        lookahead = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        return current_dist + dynamic_weight * future_dist + penalty + 0.3 * lookahead

    next_node = min(unvisited_nodes, key=heuristic)
    return next_node

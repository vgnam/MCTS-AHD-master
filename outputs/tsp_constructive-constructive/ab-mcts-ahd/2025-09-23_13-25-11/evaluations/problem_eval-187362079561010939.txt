def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        # Dynamic weight adjustment based on remaining nodes
        weight_factor = 1 / len(remaining_nodes) if len(remaining_nodes) > 1 else 1.0
        immediate_weight = 0.7 - 0.3 * weight_factor
        future_weight = 1.0 - immediate_weight

        # Lookahead factor: average distance to remaining nodes
        avg_future_dist = sum(distance_matrix[node][other] for other in remaining_nodes) / len(remaining_nodes)

        # Additional heuristic: minimum distance to any remaining node
        min_future_dist = min(distance_matrix[node][other] for other in remaining_nodes)

        # Combined score with dynamic weights and lookahead
        score = immediate_weight * current_dist + future_weight * (0.6 * avg_future_dist + 0.4 * min_future_dist)
        return score

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

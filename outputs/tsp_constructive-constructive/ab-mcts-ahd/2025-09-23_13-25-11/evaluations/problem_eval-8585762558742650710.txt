def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def evaluate_node(node):
        current_dist = distance_matrix[current_node][node]
        remaining_nodes = unvisited_nodes - {node}

        if not remaining_nodes:
            return current_dist

        future_dists = [distance_matrix[node][other] for other in remaining_nodes]
        max_future_dist = max(future_dists)
        min_future_dist = min(future_dists)
        mean_future_dist = sum(future_dists) / len(future_dists)

        variance = (max_future_dist - min_future_dist) / (max_future_dist + 1e-10)
        entropy = -sum((d - min_future_dist) / (max_future_dist - min_future_dist + 1e-10) * np.log2((d - min_future_dist) / (max_future_dist - min_future_dist + 1e-10) + 1e-10) for d in future_dists) / len(future_dists)

        dynamic_weight = 1 / (1 + np.exp(-(current_dist - mean_future_dist)))
        normalized_score = (current_dist + max_future_dist) * (1 - dynamic_weight) + entropy * dynamic_weight
        return normalized_score

    next_node = min(unvisited_nodes, key=evaluate_node)
    return next_node

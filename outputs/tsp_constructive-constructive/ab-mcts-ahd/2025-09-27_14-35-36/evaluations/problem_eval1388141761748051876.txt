import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    unvisited_list = list(unvisited_nodes)
    distances = [distance_matrix[current_node][node] for node in unvisited_list]
    total_distance = sum(distances)
    probabilities = [(total_distance - d) / total_distance for d in distances]  # Higher probability for shorter distances
    next_node = random.choices(unvisited_list, weights=probabilities, k=1)[0]
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    avg_distance = total_distance / len(unvisited_nodes) if unvisited_nodes else float('inf')

    next_node = None
    min_distance = float('inf')
    threshold_factor = 0.7

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance and distance <= avg_distance * threshold_factor:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        if dest_distance < min_distance and dest_distance <= avg_distance * 0.5:
            next_node = destination_node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_weighted_score = float('-inf')
    heuristic_factor = 0.7

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[current_node][destination_node] if destination_node in unvisited_nodes else float('inf')
        weighted_score = (1 - heuristic_factor) * distance + heuristic_factor * (1 / (dest_distance + 1e-10))

        if weighted_score > max_weighted_score:
            max_weighted_score = weighted_score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    heuristic_factor = 0.9 if remaining_nodes < 3 else 0.5  # Adjust based on remaining nodes
    penalty_factor = 0.2 if remaining_nodes == 1 else 0.0

    next_node = min(unvisited_nodes, key=lambda node: (
        heuristic_factor * distance_matrix[current_node][node] +
        (1 - heuristic_factor) * (node % 10) +
        penalty_factor * (1 if node == destination_node and remaining_nodes == 1 else 0)
    ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_distance = float('inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node

    if destination_node in unvisited_nodes:
        unvisited_count = len(unvisited_nodes)
        dest_distance = distance_matrix[current_node][destination_node]

        if unvisited_count > 0:
            avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / unvisited_count
            threshold_factor = 0.5 + (0.5 * (1 - (unvisited_count / len(distance_matrix))))
            threshold = avg_distance * threshold_factor

            if dest_distance < threshold:
                next_node = destination_node

    return next_node

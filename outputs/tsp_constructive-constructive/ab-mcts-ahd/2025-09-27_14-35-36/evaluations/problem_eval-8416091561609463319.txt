import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    scores = {}
    
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        weight = (remaining_nodes + 5) / (remaining_nodes + 15)
        score = weight * (1 / (1 + distance_to_node)) + (1 - weight) * (distance_to_dest / (1 + distance_to_node))
        scores[node] = score

    total_score = sum(scores.values())
    probabilities = {node: score / total_score for node, score in scores.items()}
    
    next_node = random.choices(list(probabilities.keys()), list(probabilities.values()), k=1)[0]
    return next_node

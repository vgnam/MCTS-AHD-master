def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('-inf')

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        avg_future_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(len(unvisited_nodes) - 1, 1)
        score = (1 / distance_to_node) * (1 + (1 / (1 + avg_future_distance)))

        if score > best_score:
            best_score = score
            next_node = node

    return next_node

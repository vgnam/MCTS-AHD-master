def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = float('-inf')

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node] if node != destination_node else 0
        exploration_factor = 1 - (distance / (sum(distance_matrix[current_node]) / len(distance_matrix[current_node])))
        score = (distance * exploration_factor) - (dest_distance * 0.3)

        if score > max_score:
            max_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        if dest_distance <= distance_matrix[current_node][next_node]:
            next_node = destination_node

    return next_node

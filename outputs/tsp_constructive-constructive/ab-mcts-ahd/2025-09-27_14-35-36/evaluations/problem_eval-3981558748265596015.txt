def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, revisit_penalty=0.1):
    if not unvisited_nodes:
        return destination_node

    def weighted_distance(node):
        base_distance = distance_matrix[current_node][node]
        destination_influence = distance_matrix[node][destination_node]
        proximity_weight = 1 / (1 + base_distance)
        revisit_factor = 1.0 - revisit_penalty if node in unvisited_nodes else 1.0 + revisit_penalty
        return (proximity_weight + destination_influence) * revisit_factor

    next_node = min(unvisited_nodes, key=lambda node: weighted_distance(node))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        if node == destination_node:
            adjusted_distance = distance_to_node * (unvisited_count / (unvisited_count + 1))
        else:
            distance_to_dest = distance_matrix[node][destination_node]
            adjusted_distance = (distance_to_node + distance_to_dest) / 2  # Harmonic mean approximation

        if adjusted_distance < min_score:
            min_score = adjusted_distance
            next_node = node

    return next_node

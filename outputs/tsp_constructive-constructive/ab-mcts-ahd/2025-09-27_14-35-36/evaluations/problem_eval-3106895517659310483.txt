def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weighted_nodes = []
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        cumulative_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        proximity_weight = 1 / (1 + distance_to_node)
        cumulative_weight = 1 / (1 + cumulative_distance)
        score = 0.6 * proximity_weight + 0.4 * cumulative_weight
        weighted_nodes.append((node, score))

    weighted_nodes.sort(key=lambda x: -x[1])
    next_node = weighted_nodes[0][0]
    return next_node

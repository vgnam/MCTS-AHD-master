import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    max_distances = max(distances)
    temperature = max(0.1, 10 * (len(unvisited_nodes) / len(distance_matrix)))  # Dynamic temperature adjustment
    probabilities = [math.exp(-d / (temperature * max_distances)) for d in distances]
    total_probability = sum(probabilities)
    normalized_probabilities = [p / total_probability for p in probabilities]

    next_node = random.choices(list(unvisited_nodes), weights=normalized_probabilities, k=1)[0]
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    dest_distance = distance_matrix[current_node][destination_node]
    avg_unvisited_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes) if unvisited_nodes else 0
    threshold = dest_distance / avg_unvisited_distance if avg_unvisited_distance > 0 else 0.6

    scores = {node: distance_matrix[current_node][node] * (1 + (1 if node == destination_node else 0) * threshold) for node in unvisited_nodes}
    next_node = min(scores, key=scores.get)
    
    return next_node

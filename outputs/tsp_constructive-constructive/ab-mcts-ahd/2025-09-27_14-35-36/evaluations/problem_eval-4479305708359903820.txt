def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_distance = float('inf')
    remaining_nodes = len(unvisited_nodes)
    decay_factor = 1.0 / (remaining_nodes + 1)  # Decays as unvisited nodes decrease

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        bottleneck_penalty = sum(distance_matrix[node][u] for u in unvisited_nodes) / remaining_nodes
        if node == destination_node:
            weighted_distance = distance * (1 - decay_factor) + bottleneck_penalty * decay_factor
        else:
            weighted_distance = distance * (1 + decay_factor) + bottleneck_penalty * (1 - decay_factor)
        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node

    return next_node

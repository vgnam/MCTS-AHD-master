import random
import numpy as np

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    scaled_distances = [1 / (d + 1) for d in distances]  # Adding 1 to avoid division by zero
    probabilities = np.exp(np.array(scaled_distances) / temperature)
    total_probability = np.sum(probabilities)
    probabilities /= total_probability

    next_node = random.choices(list(unvisited_nodes), probabilities, k=1)[0]
    return next_node

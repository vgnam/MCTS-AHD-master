import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    dest_distance = distance_matrix[current_node][destination_node]
    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    threshold = dest_distance / total_distance if total_distance > 0 else 0.7

    next_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    
    if destination_node in unvisited_nodes:
        if dest_distance < distance_matrix[current_node][next_node] * threshold:
            next_node = destination_node

    return next_node

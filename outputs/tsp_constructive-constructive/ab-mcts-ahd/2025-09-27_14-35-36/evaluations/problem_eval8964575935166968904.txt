import random
import numpy as np

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = np.array([distance_matrix[current_node][node] for node in unvisited_nodes])
    temperature = len(unvisited_nodes) / len(distance_matrix)
    probabilities = (distances ** -1) * (1 + temperature * random.random())
    probabilities /= probabilities.sum()

    next_node = random.choices(list(unvisited_nodes), weights=probabilities, k=1)[0]
    return next_node

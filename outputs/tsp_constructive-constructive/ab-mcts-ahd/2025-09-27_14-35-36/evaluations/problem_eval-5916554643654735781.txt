def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')
    weight_factor = 1 + (len(unvisited_nodes) / len(distance_matrix)) ** 0.5

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        average_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1)
        score = distance_to_node + weight_factor * average_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    def adaptive_weighted_distance(node):
        base_distance = distance_matrix[current_node][node]
        revisit_factor = 1.0 - (0.5 * len(unvisited_nodes) / len(distance_matrix)) if node in unvisited_nodes else 1.0 + (1.5 * (1 - len(unvisited_nodes) / len(distance_matrix)))
        return base_distance * revisit_factor

    next_node = min(unvisited_nodes, key=lambda node: adaptive_weighted_distance(node))
    return next_node

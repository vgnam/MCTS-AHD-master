import random
import numpy as np

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    mean_distance = np.mean(distances)
    dest_distance = distance_matrix[current_node][destination_node]

    if dest_distance > mean_distance:
        threshold = mean_distance * 0.8
    else:
        threshold = mean_distance * 0.6

    candidates = [node for node, dist in zip(unvisited_nodes, distances) if dist <= threshold]

    if not candidates:
        candidates = list(unvisited_nodes)

    next_node = min(candidates, key=lambda node: distance_matrix[current_node][node])
    return next_node

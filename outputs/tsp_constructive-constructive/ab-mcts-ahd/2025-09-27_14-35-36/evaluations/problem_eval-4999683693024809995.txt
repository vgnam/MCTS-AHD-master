def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, threshold=1.5):
    if not unvisited_nodes:
        return destination_node

    closest_nodes = [(float('inf'), None), (float('inf'), None)]

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < closest_nodes[0][0]:
            closest_nodes[1] = closest_nodes[0]
            closest_nodes[0] = (distance, node)
        elif distance < closest_nodes[1][0]:
            closest_nodes[1] = (distance, node)

    next_node = closest_nodes[0][1]

    if closest_nodes[1][1] is not None and closest_nodes[1][0] < closest_nodes[0][0] * threshold:
        next_node = closest_nodes[1][1]

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    dest_distance = distance_matrix[current_node][destination_node]
    weight = dest_distance / len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        heuristic_factor = distance_matrix[node][destination_node]
        score = distance + weight * heuristic_factor
        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes and distance_matrix[current_node][destination_node] < min_score:
        next_node = destination_node

    return next_node

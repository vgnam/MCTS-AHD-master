def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    min_distance = min(distances)
    avg_distance = sum(distances) / len(distances)

    next_node = unvisited_nodes[distances.index(min_distance)]

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        unvisited_ratio = len(unvisited_nodes) / (len(unvisited_nodes) + len(distance_matrix))
        threshold = min_distance + (avg_distance - min_distance) * (1 - unvisited_ratio)
        if dest_distance < threshold:
            next_node = destination_node

    return next_node

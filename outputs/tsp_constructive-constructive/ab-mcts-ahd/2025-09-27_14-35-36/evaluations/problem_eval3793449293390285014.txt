import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Calculate probabilities based on inverse distance
    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    probabilities = [1/d if d != 0 else float('inf') for d in distances]
    total = sum(probabilities)
    probabilities = [p / total for p in probabilities]

    # Select next node based on probabilities
    next_node = random.choices(list(unvisited_nodes), weights=probabilities, k=1)[0]
    return next_node

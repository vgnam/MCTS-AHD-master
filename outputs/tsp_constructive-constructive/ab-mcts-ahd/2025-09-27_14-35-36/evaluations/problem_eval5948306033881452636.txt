import random
import numpy as np

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    distances = np.array([distance_matrix[current_node][node] for node in unvisited_nodes])
    probabilities = (distances ** -1) ** (1 / temperature)
    probabilities /= probabilities.sum()

    if len(unvisited_nodes) <= 2:
        probabilities = np.array([1.0 if node == destination_node else 0.0 for node in unvisited_nodes])

    next_node = random.choices(list(unvisited_nodes), weights=probabilities, k=1)[0]
    return next_node

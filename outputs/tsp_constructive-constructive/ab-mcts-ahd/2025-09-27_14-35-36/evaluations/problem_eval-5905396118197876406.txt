def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('-inf')
    visits = {node: 0 for node in unvisited_nodes}

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        penalty = 1 / (1 + 2 ** visits[node])
        score = penalty * (1 / (1 + distance_to_node)) * (1 / (1 + distance_to_dest))

        if score > best_score:
            best_score = score
            next_node = node

    # Heuristic to prioritize intermediate nodes with shorter paths to the destination
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        total_distance = distance_to_node + distance_matrix[node][destination_node]
        if node != destination_node and total_distance < best_score * 4:
            score = (1 / (1 + distance_to_node)) * (1 / total_distance)
            if score > best_score:
                best_score = score
                next_node = node

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        if dest_distance < distance_matrix[current_node][next_node] * 0.75:
            next_node = destination_node

    visits[next_node] = visits.get(next_node, 0) + 1
    return next_node

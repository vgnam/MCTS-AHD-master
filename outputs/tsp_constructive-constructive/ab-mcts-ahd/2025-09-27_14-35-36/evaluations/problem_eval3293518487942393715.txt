def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('-inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        # Calculate centrality (inverse of average distance to other nodes)
        centrality = 1 / (1 + sum(distance_matrix[node]) / len(distance_matrix[node]))

        # Proximity weight scaled by centrality
        proximity_weight = (1 / (1 + distance_to_node)) * centrality

        # Destination influence adjusted by unvisited count
        destination_influence = (unvisited_count / (unvisited_count + 2)) * (1 / (1 + distance_to_dest))

        score = proximity_weight + destination_influence

        if score > best_score:
            best_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        if dest_distance < distance_matrix[current_node][next_node] * 0.5:
            next_node = destination_node

    return next_node

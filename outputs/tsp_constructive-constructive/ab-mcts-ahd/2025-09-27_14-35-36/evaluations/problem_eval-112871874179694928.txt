def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        weight = 0.5 + 0.5 * (1 - remaining_nodes / len(unvisited_nodes))
        score = distance + weight * dest_distance + 0.1 * random.random()

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        dest_score = distance_matrix[current_node][destination_node]
        if dest_score < min_score:
            next_node = destination_node

    return next_node

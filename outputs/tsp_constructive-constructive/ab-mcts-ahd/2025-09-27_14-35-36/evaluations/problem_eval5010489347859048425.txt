def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    unvisited_count = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        dest_factor = (dest_distance / distance) * (unvisited_count / (unvisited_count + 1))
        score = (distance * 0.7) + (dest_factor * 0.3)

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        dest_score = distance_matrix[current_node][destination_node]
        if dest_score < min_score * 0.5:
            next_node = destination_node

    return next_node

import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    distances = [distance_matrix[current_node][node] for node in unvisited_nodes]
    global_distances = [distance_matrix[node][destination_node] for node in unvisited_nodes]
    combined_distances = [d + g for d, g in zip(distances, global_distances)]
    probabilities = [math.exp(-distance / temperature) * (1 / (combined + 1e-10)) for distance, combined in zip(distances, combined_distances)]
    normalized_probabilities = [p / sum(probabilities) for p in probabilities]

    next_node = random.choices(list(unvisited_nodes), weights=normalized_probabilities, k=1)[0]
    return next_node

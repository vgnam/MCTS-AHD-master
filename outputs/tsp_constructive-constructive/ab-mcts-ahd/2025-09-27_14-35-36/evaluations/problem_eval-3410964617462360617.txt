def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]

        if current_node != destination_node:
            deviation = abs((distance_matrix[current_node][destination_node] - distance_to_current - distance_to_dest))
            direct_distance = distance_matrix[current_node][destination_node]
            remaining_weight = remaining_nodes / total_nodes
            deviation_weight = 0.5 * (1 - remaining_weight)
            centrality = sum(distance_matrix[node]) / (total_nodes - 1)
            score = distance_to_current + distance_to_dest + deviation_weight * deviation + 0.2 * centrality
        else:
            score = distance_to_current

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

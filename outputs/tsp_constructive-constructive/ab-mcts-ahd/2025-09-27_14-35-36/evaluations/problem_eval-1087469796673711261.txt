import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    candidates = [(node, distance_matrix[current_node][node], distance_matrix[node][destination_node]) for node in unvisited_nodes]
    candidates.sort(key=lambda x: (x[2] / x[1]) * (1 / (1 + x[1])))
    
    if random.random() < 0.1:  # 10% chance to choose a random node for exploration
        next_node = random.choice(list(unvisited_nodes))
    else:
        next_node = candidates[0][0]

    if destination_node in unvisited_nodes:
        dest_distance = distance_matrix[current_node][destination_node]
        if dest_distance < distance_matrix[current_node][next_node] * 0.8:
            next_node = destination_node

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    nearest_node = min(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
    dest_distance = distance_matrix[current_node][destination_node]

    if random.random() < 0.1:  # 10% chance to explore a random node
        next_node = random.choice(list(unvisited_nodes))
    else:
        next_node = nearest_node

    if destination_node in unvisited_nodes and dest_distance < distance_matrix[current_node][next_node] * 0.8:
        next_node = destination_node

    return next_node

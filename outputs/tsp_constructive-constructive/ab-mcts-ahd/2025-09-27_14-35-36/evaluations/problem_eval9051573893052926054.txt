def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    heuristic_factor = 0.6
    next_node = min(unvisited_nodes, key=lambda node: (
        heuristic_factor * distance_matrix[current_node][node] +
        (1 - heuristic_factor) * (
            3 / ((1 / distance_matrix[node][destination_node]) + (1 / (distance_matrix[current_node][node] + 1)) + (1 / (distance_matrix[current_node][destination_node] + 1)))
        )
    ))
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        local_distance = distance_matrix[current_node][node]
        global_potential = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        score = 0.7 * local_distance + 0.3 * global_potential
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

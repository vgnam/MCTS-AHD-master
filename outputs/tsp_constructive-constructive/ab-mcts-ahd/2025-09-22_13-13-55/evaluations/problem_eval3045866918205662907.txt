def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    alpha = 0.7  # Weight for current node distance
    beta = 0.3   # Weight for destination distance
    for node in unvisited_nodes:
        score = (alpha * distance_matrix[current_node][node] +
                 beta * distance_matrix[node][destination_node])
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

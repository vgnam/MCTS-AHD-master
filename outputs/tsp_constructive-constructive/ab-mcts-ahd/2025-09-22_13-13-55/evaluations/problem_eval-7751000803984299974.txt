def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, heuristic_factor=None):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if heuristic_factor is not None:
            score = distance * (1 + heuristic_factor.get(node, 0))
        else:
            score = distance
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

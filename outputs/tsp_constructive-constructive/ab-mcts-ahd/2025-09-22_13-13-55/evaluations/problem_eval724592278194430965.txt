def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        # Heuristic factor: angular deviation from current path direction (simplified)
        heuristic_factor = abs(distance_matrix[current_node][node] - distance_matrix[node][destination_node]) / distance_matrix[current_node][destination_node] if distance_matrix[current_node][destination_node] != 0 else 0
        # Weighted score: 70% distance, 30% heuristic
        score = 0.7 * distance + 0.3 * heuristic_factor
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

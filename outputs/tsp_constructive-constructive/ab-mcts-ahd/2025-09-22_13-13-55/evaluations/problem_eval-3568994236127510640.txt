def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        direction = abs(node - current_node)  # Simplified direction heuristic
        score = distance + direction  # Balances distance and direction
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

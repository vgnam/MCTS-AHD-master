def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, w1=0.5, w2=0.5):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_sum = w1 * distance_to_current + w2 * distance_to_destination

        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_nodes = len(unvisited_nodes) + 1
    visited_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        proximity_weight = 1 - visited_ratio
        destination_weight = visited_ratio
        penalty = 0.5 if distance_to_current < 0.1 * sum(distance_matrix[current_node]) else 0

        score = (proximity_weight * distance_to_current +
                 destination_weight * distance_to_destination +
                 penalty * distance_to_current)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

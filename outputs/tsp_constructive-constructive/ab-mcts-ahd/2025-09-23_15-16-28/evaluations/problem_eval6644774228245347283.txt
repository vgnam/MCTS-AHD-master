def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_distance = float('inf')

    total_nodes = len(unvisited_nodes) + 1
    position_factor = (total_nodes - len(unvisited_nodes)) / total_nodes

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        sum_distance = distance_to_current + distance_to_destination
        max_distance = max(distance_to_current, distance_to_destination)

        if position_factor < 0.3:
            weight_sum = 0.2
            weight_max = 0.8
        elif position_factor < 0.7:
            weight_sum = 0.5
            weight_max = 0.5
        else:
            weight_sum = 0.8
            weight_max = 0.2

        weighted_distance = weight_sum * sum_distance + weight_max * max_distance

        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        weight_factor = 1.0 / (1.0 + 0.1 * remaining_nodes)
        score = (1.0 + weight_factor) * current_distance + 0.5 * (1.0 - weight_factor) * destination_distance
        score += 0.2 * (current_distance * destination_distance) / (current_distance + destination_distance)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]

        if len(unvisited_nodes) == 1:
            distance_to_destination = distance_matrix[node][destination_node]
            score = distance_to_current + distance_to_destination
        else:
            avg_distance_to_unvisited = sum(distance_matrix[node][u] for u in unvisited_nodes) / len(unvisited_nodes)
            score = 0.6 * distance_to_current + 0.4 * avg_distance_to_unvisited

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

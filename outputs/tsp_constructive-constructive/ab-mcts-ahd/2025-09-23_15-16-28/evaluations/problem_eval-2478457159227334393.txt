def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        sum_distance = distance_to_current + distance_to_destination
        product_distance = distance_to_current * distance_to_destination

        weight = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
        score = weight * sum_distance + (1 - weight) * product_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

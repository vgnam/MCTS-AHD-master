def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')
    threshold = float('inf')
    remaining_nodes = len(unvisited_nodes)
    priority_factor = min(1.0, (remaining_nodes / (remaining_nodes + 1)) ** 0.5)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance = (distance_to_current + distance_to_destination) / 2
        weighted_sum = (0.7 * distance_to_current) + (0.3 * (distance_to_destination / avg_distance)) * priority_factor
        max_distance = max(distance_to_current, distance_to_destination)

        if max_distance < threshold and weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
            threshold = max_distance

    return next_node

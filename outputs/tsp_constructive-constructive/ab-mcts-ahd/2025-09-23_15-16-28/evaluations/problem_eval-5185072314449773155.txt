def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_unvisited = len(unvisited_nodes)
    progress_factor = 1.0 - (total_unvisited / (total_unvisited + 1))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            weight = 1.0
        else:
            weight = 0.6 * (1.0 / distance_to_destination)

        score = (1 - progress_factor) * distance_to_current + progress_factor * (distance_to_current + weight * distance_to_destination)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

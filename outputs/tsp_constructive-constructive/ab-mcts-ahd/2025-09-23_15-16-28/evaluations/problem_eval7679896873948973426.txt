def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_combined_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        sum_distance = distance_to_current + distance_to_destination
        product_distance = distance_to_current * distance_to_destination

        weight_current = 1 / (1 + num_unvisited)
        weight_destination = 1 - weight_current
        weighted_sum = weight_current * distance_to_current + weight_destination * distance_to_destination

        combined_score = (weighted_sum + sum_distance + product_distance) / 3

        if combined_score < min_combined_score:
            min_combined_score = combined_score
            next_node = node

    return next_node

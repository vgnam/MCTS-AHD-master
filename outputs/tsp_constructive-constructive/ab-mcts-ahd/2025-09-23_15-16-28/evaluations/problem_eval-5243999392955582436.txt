def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')
    destination_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_sum = 0.7 * distance_to_current + 0.3 * distance_to_destination
        penalty = 0.1 * abs(distance_to_destination - destination_distance)
        total_score = weighted_sum + penalty

        if total_score < min_weighted_sum:
            min_weighted_sum = total_score
            next_node = node

    return next_node

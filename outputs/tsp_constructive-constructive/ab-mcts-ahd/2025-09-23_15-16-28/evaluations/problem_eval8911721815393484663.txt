def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        avg_distance = sum(distance_matrix[node][n] for n in unvisited_nodes) / num_unvisited
        weight_factor = 0.5 + (0.5 * (distance_to_current / (avg_distance + 1e-6)))
        combined_score = distance_to_current + weight_factor * distance_to_destination + 0.5 * avg_distance

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

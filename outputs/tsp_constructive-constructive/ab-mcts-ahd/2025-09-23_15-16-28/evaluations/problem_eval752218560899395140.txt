def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.5):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_combined_distance = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        combined_distance = distance_to_current + weight * distance_to_destination

        if combined_distance < min_combined_distance:
            min_combined_distance = combined_distance
            next_node = node

    return next_node

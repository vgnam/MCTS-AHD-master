def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    max_distance = max(max(row) for row in distance_matrix)
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            return node

        # Dynamic weight based on remaining nodes
        weight = 0.7 * (1 - (remaining_nodes / (remaining_nodes + 1)))

        # Heuristic: combine direct and detour distances
        detour_factor = (distance_to_current + distance_to_destination) / (distance_matrix[current_node][destination_node] + 1e-6)
        normalized_score = (weight * (distance_to_current / max_distance)) + ((1 - weight) * detour_factor)

        if normalized_score < min_score:
            min_score = normalized_score
            next_node = node

    return next_node

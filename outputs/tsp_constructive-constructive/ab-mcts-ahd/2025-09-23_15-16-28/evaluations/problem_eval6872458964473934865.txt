def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    progress_factor = 1.0 - (remaining_nodes / (remaining_nodes + 1.5))  # Adjusted normalization

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        proximity_weight = (1 - progress_factor) * distance_to_current
        potential_weight = progress_factor * (distance_to_current + distance_to_destination) / 2
        penalty = (distance_to_current * distance_to_destination) ** 0.3  # Reduced exponent for balance

        score = proximity_weight + potential_weight - penalty

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

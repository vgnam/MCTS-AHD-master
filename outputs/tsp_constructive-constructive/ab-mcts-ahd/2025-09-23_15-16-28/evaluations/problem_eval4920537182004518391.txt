def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, selection_history=None):
    if not unvisited_nodes:
        return destination_node

    if selection_history is None:
        selection_history = {}

    next_node = None
    min_heuristic = float('inf')
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Calculate centrality
        centrality = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, len(unvisited_nodes) - 1)

        # Dynamic weight adjustment based on remaining nodes
        weight_current = max(0.3, 0.5 - 0.1 * (remaining_nodes / len(distance_matrix)))
        weight_destination = max(0.3, 0.4 + 0.1 * (remaining_nodes / len(distance_matrix)))
        weight_centrality = 0.3

        # Novelty factor based on selection history
        novelty = 1.0 / (1.0 + selection_history.get(node, 0))

        # Combine with dynamic weights and novelty
        heuristic = (weight_current * distance_to_current +
                     weight_destination * distance_to_destination +
                     weight_centrality * centrality) * novelty

        if heuristic < min_heuristic:
            min_heuristic = heuristic
            next_node = node

    # Update selection history
    selection_history[next_node] = selection_history.get(next_node, 0) + 1

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_distance = float('inf')

    # Calculate non-linear weight and average distance to unvisited nodes
    remaining_nodes = len(unvisited_nodes)
    weight = 1.0 / (1.0 + remaining_nodes ** 2) if remaining_nodes > 0 else 1.0
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / remaining_nodes if remaining_nodes > 0 else 0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Weighted sum with normalized distance to current and average distance
        normalized_distance = distance_to_current / avg_distance if avg_distance > 0 else 1.0
        total_distance = weight * normalized_distance + (1 - weight) * distance_to_destination

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

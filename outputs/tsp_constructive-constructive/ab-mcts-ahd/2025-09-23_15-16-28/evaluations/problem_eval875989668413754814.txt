def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            return node

        weighted_sum = 0.4 * distance_to_current + 0.3 * distance_to_destination
        product_distance = distance_to_current * distance_to_destination
        ratio = distance_to_current / (distance_to_destination + 1e-10)

        score = 0.4 * weighted_sum + 0.3 * product_distance + 0.3 * ratio

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    avg_distance = sum(sum(row) for row in distance_matrix) / (len(distance_matrix) * len(distance_matrix))
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            return node

        ratio = distance_to_current / distance_to_destination
        normalized_ratio = ratio / avg_distance
        potential_future_cost = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / remaining_nodes
        weighted_score = (0.5 * normalized_ratio) + (0.5 * potential_future_cost / avg_distance)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

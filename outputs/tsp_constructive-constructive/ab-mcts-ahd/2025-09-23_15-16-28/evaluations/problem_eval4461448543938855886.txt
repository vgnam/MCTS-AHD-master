def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node_distance = distance_matrix[current_node][node]
        node_to_destination_distance = distance_matrix[node][destination_node]
        total_distance = 0.3 * current_to_node_distance + 0.7 * node_to_destination_distance

        if total_distance < min_total_distance:
            min_total_distance = total_total_distance
            next_node = node

    return next_node

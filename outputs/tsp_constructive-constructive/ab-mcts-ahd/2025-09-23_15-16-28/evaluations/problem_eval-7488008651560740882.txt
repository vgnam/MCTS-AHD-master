def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_unvisited = len(unvisited_nodes)
    progress_factor = 1.0 - (total_unvisited / (total_unvisited + 1))  # Normalized progress

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            ratio = 0.0
        else:
            ratio = distance_to_current / distance_to_destination

        weighted_score = (1 - progress_factor) * ratio + progress_factor * (distance_to_current + distance_to_destination)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

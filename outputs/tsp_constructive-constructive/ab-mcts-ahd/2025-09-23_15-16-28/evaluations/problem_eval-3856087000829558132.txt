def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]

        # Stronger dynamic weight adjustment based on remaining nodes
        current_weight = 0.6 + 0.3 * (remaining_nodes / (remaining_nodes + 1))
        destination_weight = 1 - current_weight

        # More aggressive penalty for nodes far from destination
        score = (current_weight * current_distance) + (destination_weight * destination_distance)
        if destination_distance > 2 * distance_matrix[current_node][destination_node]:
            score *= 1.5  # Stronger penalty

        # Additional term to minimize total path length
        total_path_length = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        score += 0.1 * total_path_length / (remaining_nodes + 1)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

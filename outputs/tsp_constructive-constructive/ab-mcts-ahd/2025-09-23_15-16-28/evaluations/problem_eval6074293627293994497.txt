def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    best_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            return node

        normalized_ratio = (distance_to_current / distance_to_destination) / (1 + 0.1 * (distance_to_current / distance_to_destination))
        detour_penalty = (distance_to_current + distance_to_destination) - distance_matrix[current_node][destination_node]
        score = 0.6 * normalized_ratio + 0.4 * detour_penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Dynamic weighting factor based on remaining nodes
        weighting_factor = 0.5 + 0.5 * (1 - (total_unvisited / (total_unvisited + 1)))

        # Weighted sum of distances
        combined_score = (distance_to_current * weighting_factor) + (distance_to_destination * (1 - weighting_factor))

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

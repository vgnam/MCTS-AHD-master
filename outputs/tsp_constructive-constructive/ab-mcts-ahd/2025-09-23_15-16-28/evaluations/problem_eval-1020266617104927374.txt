def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        detour_risk = distance_to_current + distance_to_destination - distance_matrix[current_node][destination_node]

        if detour_risk <= 0:
            weight = 1.0
        else:
            weight = 1.0 / (1.0 + detour_risk)

        score = (distance_to_current * distance_to_destination) * (1 - weight)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_priority = -float('inf')
    remaining_nodes = len(unvisited_nodes)

    if remaining_nodes == 0:
        return destination_node

    avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            lookahead_factor = 1.0
        else:
            lookahead_factor = (avg_distance_to_dest / distance_to_destination) ** (1.0 / remaining_nodes)

        priority = (1.0 / distance_to_current) * lookahead_factor

        if priority > max_priority:
            max_priority = priority
            next_node = node

    return next_node

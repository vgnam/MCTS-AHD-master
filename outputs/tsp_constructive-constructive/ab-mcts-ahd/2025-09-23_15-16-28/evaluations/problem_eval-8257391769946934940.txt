def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_combined_value = float('inf')
    next_node = None
    weight = 0.5  # Adjustable weight to balance sum and product

    for node in unvisited_nodes:
        sum_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        product_distance = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        combined_value = weight * sum_distance + (1 - weight) * product_distance
        if combined_value < min_combined_value:
            min_combined_value = combined_value
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_ratio = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node_distance = distance_matrix[current_node][node]
        node_to_destination_distance = distance_matrix[node][destination_node]
        if node_to_destination_distance == 0:
            ratio = current_to_node_distance
        else:
            ratio = current_to_node_distance / node_to_destination_distance

        if ratio < min_ratio:
            min_ratio = ratio
            next_node = node

    return next_node

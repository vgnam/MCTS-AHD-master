def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_to_node_distance = distance_matrix[current_node][node]
        average_unvisited_distance = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes)
        total_distance = current_to_node_distance + average_unvisited_distance

        if total_distance < min_total_distance:
            min_total_distance = total_total_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_total_distance = float('inf')
    next_node = None
    progress = (len(unvisited_nodes) + 1) / (len(unvisited_nodes) + 1)  # Normalized progress (0 to 1)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        combined_distance = immediate_distance + future_distance
        weighted_distance = (1 - progress) * immediate_distance + progress * combined_distance

        if weighted_distance < min_total_distance:
            min_total_distance = weighted_distance
            next_node = node

    return next_node

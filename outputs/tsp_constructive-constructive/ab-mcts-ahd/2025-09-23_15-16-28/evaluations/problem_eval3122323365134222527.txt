def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        closest_unvisited_distance = min([distance_matrix[node][n] for n in unvisited_nodes if n != node], default=0)
        score = 0.6 * current_distance + 0.4 * closest_unvisited_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

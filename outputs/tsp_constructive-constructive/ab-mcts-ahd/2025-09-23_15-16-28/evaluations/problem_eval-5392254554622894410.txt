def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_distance = float('inf')

    remaining_nodes = len(unvisited_nodes)
    distance_to_destination = distance_matrix[current_node][destination_node]

    # Dynamic weight based on remaining nodes and distance to destination
    weight = (1.0 / (1.0 + 2 ** remaining_nodes)) * (1.0 + distance_to_destination / max(distance_matrix[current_node]))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]

        # Weighted sum with dynamic adjustment
        total_distance = weight * distance_to_current + (1 - weight) * future_distance

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + remaining_nodes) if remaining_nodes > 0 else 1.0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0:
            return node

        ratio = distance_to_current / distance_to_destination
        weighted_ratio = weight_factor * ratio
        weighted_distance = (1.0 - weight_factor) * distance_to_current
        score = weighted_ratio + weighted_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

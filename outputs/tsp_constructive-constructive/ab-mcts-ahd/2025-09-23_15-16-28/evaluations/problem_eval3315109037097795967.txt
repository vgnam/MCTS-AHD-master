def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_score = float('inf')
    current_to_destination = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if current_to_destination == 0:
            weight = 0.5
        else:
            ratio = (distance_to_current + 0.1) / (current_to_destination + 0.1)
            weight = 1.0 if ratio > 1.0 else 0.3

        combined_score = (distance_to_current * distance_to_destination) + weight * (distance_to_current + distance_to_destination)

        if combined_score < min_score:
            min_score = combined_score
            next_node = node

    return next_node

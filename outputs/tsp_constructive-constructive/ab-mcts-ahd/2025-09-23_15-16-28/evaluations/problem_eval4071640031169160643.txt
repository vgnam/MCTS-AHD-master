def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')
    threshold = float('inf')
    current_path_length = sum(distance_matrix[current_node][destination_node] for _ in range(len(unvisited_nodes)))

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        dynamic_weight = 0.5 + 0.5 * (distance_to_current / (current_path_length + 1e-10))
        weighted_sum = (dynamic_weight * distance_to_current) + ((1 - dynamic_weight) * distance_to_destination)
        max_distance = max(distance_to_current, distance_to_destination)

        if max_distance < threshold and weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
            threshold = max_distance

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_ratio = float('inf')
    total_unvisited = len(unvisited_nodes)
    progress_factor = 1.0 - (total_unvisited / (total_unvisited + 1))  # Normalized progress

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_destination == 0 or distance_to_current == 0:
            ratio = 1.0
        else:
            ratio = distance_to_current / distance_to_destination

        weighted_ratio = (1 - progress_factor) * distance_to_current + progress_factor * ratio

        if weighted_ratio < min_weighted_ratio:
            min_weighted_ratio = weighted_ratio
            next_node = node

    return next_node

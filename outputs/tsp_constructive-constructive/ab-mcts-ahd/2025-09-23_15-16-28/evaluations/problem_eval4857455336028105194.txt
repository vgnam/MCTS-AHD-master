def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')
    remaining_nodes = len(unvisited_nodes)
    alpha = 0.7 - 0.1 * (remaining_nodes / len(distance_matrix))  # Dynamic weight for distance to current node
    beta = 0.3 + 0.1 * (remaining_nodes / len(distance_matrix))   # Dynamic weight for distance to destination

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_sum = alpha * distance_to_current + beta * distance_to_destination

        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node

    return next_node

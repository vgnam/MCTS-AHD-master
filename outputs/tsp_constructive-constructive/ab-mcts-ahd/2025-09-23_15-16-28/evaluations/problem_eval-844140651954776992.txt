def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_distance = float('inf')

    total_nodes = len(unvisited_nodes) + 1  # Including current node
    position_factor = (total_nodes - len(unvisited_nodes)) / total_nodes  # Normalized position (0 to 1)

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Adjust weights based on tour position
        if position_factor < 0.3:  # Early in tour
            weight_current = 0.4
            weight_future = 0.6
        elif position_factor < 0.7:  # Middle of tour
            weight_current = 0.5
            weight_future = 0.5
        else:  # Late in tour
            weight_current = 0.7
            weight_future = 0.3

        weighted_distance = (weight_current * distance_to_current) + (weight_future * distance_to_destination)

        if weighted_distance < min_weighted_distance:
            min_weighted_distance = weighted_distance
            next_node = node

    return next_node

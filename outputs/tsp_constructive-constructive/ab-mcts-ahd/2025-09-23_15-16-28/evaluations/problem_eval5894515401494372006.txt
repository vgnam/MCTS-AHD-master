def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_weighted_sum = float('inf')
    threshold = float('inf')
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 0.6 if remaining_nodes > 3 else 0.3

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weighted_sum = (weight_factor * distance_to_current) + ((1 - weight_factor) * distance_to_destination)
        max_distance = max(distance_to_current, distance_to_destination)

        if max_distance <= threshold and weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
            threshold = max_distance

    return next_node

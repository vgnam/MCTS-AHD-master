def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_combined_score = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        sum_distance = distance_to_current + distance_to_destination
        product_distance = distance_to_current * distance_to_destination
        ratio = distance_to_destination / distance_to_current if distance_to_current != 0 else float('inf')

        combined_score = sum_distance + 0.5 * product_distance - 0.2 * ratio

        if combined_score < min_combined_score:
            min_combined_score = combined_score
            next_node = node

    return next_node

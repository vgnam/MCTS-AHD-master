def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_total_distance = float('inf')

    remaining_nodes = len(unvisited_nodes)
    base_weight = remaining_nodes / (1.0 + remaining_nodes)
    # Non-linear scaling to emphasize either proximity or efficiency more
    weight = base_weight ** 2 if remaining_nodes > 1 else 0.0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Bonus for nodes closer to destination than current node
        bonus = 1.0 if distance_to_destination < distance_matrix[current_node][destination_node] else 0.2

        # Weighted sum with bonus
        total_distance = (weight * distance_to_current + (1 - weight) * distance_to_destination) * bonus

        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node

    return next_node

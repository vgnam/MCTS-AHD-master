def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.5, penalty_threshold=2.0):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_current == 0:
            ratio = float('inf')
        else:
            ratio = distance_to_destination / distance_to_current

        penalty = 1.0
        if distance_to_destination > penalty_threshold * distance_matrix[current_node][destination_node]:
            penalty = 0.1

        weighted_score = weight * ratio + (1 - weight) * penalty

        if weighted_score > max_score:
            max_score = weighted_score
            next_node = node

    return next_node

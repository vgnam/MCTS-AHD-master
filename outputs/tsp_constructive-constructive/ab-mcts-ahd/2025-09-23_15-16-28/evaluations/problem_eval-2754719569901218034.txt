def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    min_combined_value = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        sum_distance = distance_to_current + distance_to_destination
        max_distance = max(distance_to_current, distance_to_destination)

        if total_unvisited > 1:
            weight = 1.0 / total_unvisited
        else:
            weight = 1.0

        combined_value = (sum_distance + max_distance) * weight

        if combined_value < min_combined_value:
            min_combined_value = combined_value
            next_node = node

    return next_node

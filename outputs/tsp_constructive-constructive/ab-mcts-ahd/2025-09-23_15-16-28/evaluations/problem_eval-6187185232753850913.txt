def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 - (1.0 / (remaining_nodes + 1)) if remaining_nodes > 0 else 0.0

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if remaining_nodes <= 2:
            score = distance_to_current * (1 - weight_factor) + distance_to_destination * weight_factor
        else:
            score = distance_to_current * 0.5 + distance_to_destination * 0.5

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

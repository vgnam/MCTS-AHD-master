def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None

    for node in unvisited_nodes:
        if distance_matrix[current_node][node] == 0:
            score = float('inf')
        else:
            dest_to_current = distance_matrix[destination_node][current_node]
            current_to_node = distance_matrix[current_node][node]
            avg_unvisited_dist = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
            score = (dest_to_current * 0.6 + avg_unvisited_dist * 0.4) / current_to_node

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

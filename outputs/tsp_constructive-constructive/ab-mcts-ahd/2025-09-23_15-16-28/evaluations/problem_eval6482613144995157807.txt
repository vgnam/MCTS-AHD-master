def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    min_combined_metric = float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        sum_distance = distance_to_current + distance_to_destination
        product_distance = distance_to_current * distance_to_destination

        # Normalize the product distance by the sum to balance the metrics
        normalized_product = product_distance / (sum_distance + 1e-10)  # Avoid division by zero
        combined_metric = sum_distance + normalized_product

        if combined_metric < min_combined_metric:
            min_combined_metric = combined_metric
            next_node = node

    return next_node

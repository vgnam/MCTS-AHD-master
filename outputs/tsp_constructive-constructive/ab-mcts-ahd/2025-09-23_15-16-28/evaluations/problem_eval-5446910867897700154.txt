def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')
    current_to_dest = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        if distance_to_current == 0:
            score = float('inf')
        else:
            proximity_weight = 1 - (distance_to_destination / current_to_dest) if current_to_dest != 0 else 1
            score = (distance_to_destination / distance_to_current) * (1 + proximity_weight)

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_score = -float('inf')
    next_node = None

    for node in unvisited_nodes:
        if distance_matrix[current_node][node] == 0:
            score = float('inf')
        else:
            ratio = distance_matrix[destination_node][current_node] / distance_matrix[current_node][node]
            inverse_distance = 1 / distance_matrix[destination_node][node]
            score = 0.7 * ratio + 0.3 * inverse_distance
        if score > max_score:
            max_score = score
            next_node = node

    return next_node

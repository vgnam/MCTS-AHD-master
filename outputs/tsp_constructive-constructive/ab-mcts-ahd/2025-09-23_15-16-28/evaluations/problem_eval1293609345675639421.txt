def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    next_node = None
    max_score = -float('inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        normalized_current = 1 / (1 + distance_to_current)
        normalized_destination = 1 / (1 + distance_to_destination)
        score = 0.6 * normalized_current + 0.4 * normalized_destination - 0.2 * (distance_to_current * distance_to_destination)

        if score > max_score:
            max_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_combined_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        avg_future_cost = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                avg_future_cost += distance_matrix[node][other_node]
        avg_future_cost /= (remaining_nodes - 1) if remaining_nodes > 1 else 1
        dynamic_weight = 1.0 / remaining_nodes
        combined_cost = immediate_cost + dynamic_weight * avg_future_cost
        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = node
    return next_node

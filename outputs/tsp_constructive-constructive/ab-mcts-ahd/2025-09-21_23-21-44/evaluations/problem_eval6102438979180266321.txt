def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    visited_count = len(distance_matrix) - len(unvisited_nodes) - 1
    weight_factor = visited_count / (len(distance_matrix) - 1)
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        score = ((1 - weight_factor) * immediate_distance) + (weight_factor * potential_distance)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

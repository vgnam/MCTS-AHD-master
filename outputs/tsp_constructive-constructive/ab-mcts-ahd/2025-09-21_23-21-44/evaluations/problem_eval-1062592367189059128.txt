def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        sum_distance = distance_current + distance_destination
        product_distance = distance_current * distance_destination
        score = (sum_distance + product_distance) + (distance_current * 0.5) * (distance_destination * 0.5)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_weighted_product = float('inf')
    next_node = None
    avg_distance_to_dest = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        weighted_product = current_to_node * (node_to_dest / avg_distance_to_dest)
        if weighted_product < min_weighted_product:
            min_weighted_product = weighted_product
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    alpha = 0.7  # Tunable parameter to balance multiplicative and additive components
    for node in unvisited_nodes:
        product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        additive = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        score = alpha * product + (1 - alpha) * additive
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

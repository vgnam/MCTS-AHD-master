defaulting to the destination node if no unvisited nodes remain, creating a constructive approach to TSP with a focus on maximizing immediate distance.}

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    max_distance = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] > max_distance:
            max_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    dest_dist_total = distance_matrix[current_node][destination_node]
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            ratio = immediate_distance
        else:
            ratio = immediate_distance / dest_dist
        dynamic_weight = 1.0 + (dest_dist_total - dest_dist) / (dest_dist_total + 1e-6)
        combined_score = dynamic_weight * immediate_distance + (1 - dynamic_weight) * ratio
        if combined_score < min_score:
            min_score = combined_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_weighted_score = float('inf')
    next_node = None
    n = len(distance_matrix)
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            weighted_score = current_dist
        else:
            # Centrality measure: average distance to all other nodes
            centrality = sum(distance_matrix[node]) / (n - 1)
            # Greedy component (60%) + centrality component (30%) + penalty (10%)
            projected_path_length = current_dist + dest_dist
            penalty = 1 + (current_dist / (projected_path_length + 1e-10)) ** 2
            weighted_score = 0.6 * current_dist + 0.3 * centrality + 0.1 * dest_dist * penalty
        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_total_sum = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        return_distance = distance_matrix[node][destination_node]
        dynamic_weight = 1 + (1 / remaining_nodes)
        total_sum = immediate_distance + (dynamic_weight * return_distance)
        if total_sum < min_total_sum:
            min_total_sum = total_sum
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    num_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node = distance_matrix[node][destination_node]
        centrality_heuristic = sum(distance_matrix[node]) / (num_nodes - 1)
        dynamic_weight = 0.8 if remaining_nodes > num_nodes / 2 else 0.6
        score = (dynamic_weight * distance_to_node + (1 - dynamic_weight) * distance_from_node) + (0.1 * centrality_heuristic)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

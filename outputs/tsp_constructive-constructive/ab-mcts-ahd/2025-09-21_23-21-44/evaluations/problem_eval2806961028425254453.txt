def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        dynamic_weight = 0.5 * (1 - (remaining_nodes / len(unvisited_nodes)))
        score = distance_to_current + (0.5 + dynamic_weight) * distance_to_destination
        if distance_to_current > 1.5 * distance_matrix[current_node][destination_node]:
            score += 2 * distance_to_current
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

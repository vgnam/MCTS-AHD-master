def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_ratio = float('inf')
    next_node = None
    for node in unvisited_nodes:
        product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        sum_distances = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        ratio = product / sum_distances if sum_distances != 0 else float('inf')
        if ratio < min_ratio:
            min_ratio = ratio
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        dest_distance = distance_matrix[node][destination_node]
        ratio = immediate_distance / dest_distance if dest_distance != 0 else immediate_distance
        avg_distance = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                avg_distance += distance_matrix[node][other_node]
        avg_distance /= len(unvisited_nodes) if len(unvisited_nodes) > 1 else 1
        combined_score = 0.6 * immediate_distance + 0.4 * ratio + 0.2 * avg_distance
        if combined_score < min_score:
            min_score = combined_score
            next_node = node
    return next_node

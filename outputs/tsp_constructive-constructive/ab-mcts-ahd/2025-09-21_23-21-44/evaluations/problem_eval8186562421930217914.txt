def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = min(0.7, len(unvisited_nodes) / (len(unvisited_nodes) + 2))
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        lookahead_score = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        score = (weight_factor * immediate_distance) + ((1 - weight_factor) * potential_distance) + (0.2 * lookahead_score)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

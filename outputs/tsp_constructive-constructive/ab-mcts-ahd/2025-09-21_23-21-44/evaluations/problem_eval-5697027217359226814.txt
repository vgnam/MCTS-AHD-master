def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        if distance_destination == 0:
            score = -distance_current
        else:
            weight = 1.0 + remaining_nodes
            penalty = 0.5 if distance_current > 2 * distance_matrix[current_node][destination_node] else 1.0
            score = -distance_current + weight * (1 / (distance_destination + 1)) * penalty
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    total_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) if unvisited_nodes else 0
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        degree_weight = 0.3 if total_distance > 0 else 0.5  # Adjust weight based on remaining distance
        score = -distance + degree_weight * (1 / (degree + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

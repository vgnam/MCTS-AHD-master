def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    current_path_length = sum(distance_matrix[current_node][node] for node in unvisited_nodes if node in unvisited_nodes)
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        centrality = sum(1 for n in unvisited_nodes if distance_matrix[node][n] > 0) / len(unvisited_nodes)
        normalized_distance = distance / (current_path_length + 1e-6)
        dynamic_weight = 0.4 + 0.4 * (1 - normalized_distance) + 0.2 * centrality
        score = -normalized_distance + dynamic_weight * (1 / (degree + 1)) + 0.1 * centrality
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        avg_distance_to_unvisited = sum(distance_matrix[node][n] for n in unvisited_nodes) / remaining_nodes
        dynamic_weight = 1 / (1 + remaining_nodes)  # Weight decreases as more nodes are visited
        score = -distance + dynamic_weight * (1 / (degree + 1)) - 0.2 * avg_distance_to_unvisited
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

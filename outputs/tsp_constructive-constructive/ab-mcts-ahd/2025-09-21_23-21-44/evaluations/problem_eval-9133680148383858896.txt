def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_normalized_value = float('inf')
    next_node = None
    for node in unvisited_nodes:
        sum_term = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        product_term = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        normalized_value = (sum_term + product_term) / 2
        if normalized_value < min_normalized_value:
            min_normalized_value = normalized_value
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, weight=0.5):
    if not unvisited_nodes:
        return destination_node
    min_total_weight = float('inf')
    next_node = None
    for node in unvisited_nodes:
        total_weight = weight * distance_matrix[current_node][node] + (1 - weight) * distance_matrix[node][destination_node]
        if total_weight < min_total_weight:
            min_total_weight = total_weight
            next_node = node
    return next_node

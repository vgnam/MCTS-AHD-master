def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    min_score = float('inf')
    next_node = None
    min_total_distance = float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        total_distance = immediate_distance + potential_distance
        score = (weight_factor * immediate_distance) + ((1 - weight_factor) * potential_distance)

        if score < min_score or (score == min_score and total_distance < min_total_distance):
            min_score = score
            next_node = node
            min_total_distance = total_distance

    return next_node

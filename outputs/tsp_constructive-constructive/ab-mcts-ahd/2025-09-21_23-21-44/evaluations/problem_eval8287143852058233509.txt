def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            score = current_dist
        else:
            ratio_weight = 0.6 if dest_dist > 1.5 else 0.8
            sum_weight = 1 - ratio_weight
            score = ratio_weight * (current_dist / dest_dist) + sum_weight * (current_dist + dest_dist)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

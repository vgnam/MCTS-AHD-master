def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_ratio = 0.7 * (remaining_nodes / (remaining_nodes + 1))
    weight_total = 0.3 * (1 - (remaining_nodes / (remaining_nodes + 1)))
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            score = current_dist
        else:
            score = weight_ratio * (current_dist / dest_dist) + weight_total * (current_dist + dest_dist)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

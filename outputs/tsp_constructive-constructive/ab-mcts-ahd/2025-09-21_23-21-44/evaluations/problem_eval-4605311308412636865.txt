def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes) + 1
    weight_factor = len(unvisited_nodes) / total_unvisited
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        potential_distance = distance_matrix[node][destination_node]
        distance_ratio = immediate_distance / (potential_distance + 1e-10)  # Avoid division by zero
        adjusted_weight = weight_factor * (1 + distance_ratio)
        score = adjusted_weight * immediate_distance + (1 - adjusted_weight) * potential_distance

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

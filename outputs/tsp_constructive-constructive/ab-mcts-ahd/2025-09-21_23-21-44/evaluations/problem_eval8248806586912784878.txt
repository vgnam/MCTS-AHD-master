def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        product_term = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        avg_term = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if unvisited_nodes else 0
        total_score = 0.7 * product_term + 0.3 * avg_term
        if total_score < min_score:
            min_score = total_score
            next_node = node
    return next_node

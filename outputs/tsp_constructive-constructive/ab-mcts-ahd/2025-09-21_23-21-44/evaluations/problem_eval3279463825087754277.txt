def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_weighted_sum = float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)
    for node in unvisited_nodes:
        weight = (num_unvisited / (num_unvisited + 1)) ** 0.5
        weighted_sum = weight * distance_matrix[current_node][node] + (1 - weight) * distance_matrix[node][destination_node]
        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
    return next_node

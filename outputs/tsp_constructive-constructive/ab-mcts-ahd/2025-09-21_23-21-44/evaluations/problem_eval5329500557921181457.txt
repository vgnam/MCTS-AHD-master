def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_total_product = float('inf')
    next_node = None
    closest_node = None
    min_distance = float('inf')
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            closest_node = node
    for node in unvisited_nodes:
        if node == closest_node:
            total_product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
            if total_product < min_total_product:
                min_total_product = total_product
                next_node = node
    return next_node

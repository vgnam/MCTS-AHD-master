def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    remaining = len(unvisited_nodes)
    max_remaining = len(distance_matrix)
    weight_factor = 1 / (1 + math.exp(-(remaining - max_remaining/2) / (max_remaining/5)))
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        distance_weight = 1 - weight_factor
        degree_weight = weight_factor
        score = -distance_weight * distance + degree_weight * (1 / (degree + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        score = (weight_factor * distance_current) + ((1 - weight_factor) * distance_destination)
        weighted_score = score * (distance_current * distance_destination)

        if weighted_score < min_score:
            min_score = weighted_score
            next_node = node

    return next_node

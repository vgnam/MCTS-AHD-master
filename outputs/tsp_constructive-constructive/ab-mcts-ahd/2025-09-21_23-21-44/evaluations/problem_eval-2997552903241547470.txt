def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][node]
        indirect_distance = distance_matrix[node][destination_node]
        normalized_direct = direct_distance / (remaining_nodes + 1)
        normalized_indirect = indirect_distance / (remaining_nodes + 1)
        score = (direct_distance * indirect_distance) / (normalized_direct * normalized_indirect)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    avg_unvisited_distance = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / len(unvisited_nodes)
    visited_penalty = 1.0 + (len(unvisited_nodes) / (len(distance_matrix) - 1))  # Dynamic penalty based on tour progress

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node]
        utility_factor = 1.0 - (future_distance / (2 * avg_unvisited_distance))  # Reinforcement for nodes closer to destination
        balance_factor = 0.5 * (immediate_distance / avg_unvisited_distance) + 0.5 * (future_distance / avg_unvisited_distance) * utility_factor
        total_score = immediate_distance + future_distance * balance_factor * visited_penalty

        if total_score < min_score:
            min_score = total_score
            next_node = node

    return next_node

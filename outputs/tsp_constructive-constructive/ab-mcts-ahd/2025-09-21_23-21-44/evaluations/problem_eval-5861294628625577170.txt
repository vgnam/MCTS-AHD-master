def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    num_nodes = len(distance_matrix)
    remaining_nodes = num_nodes - len(unvisited_nodes)
    weight_proximity = 0.5 + 0.1 * (remaining_nodes / num_nodes)
    weight_long_term = 0.4 + 0.1 * (remaining_nodes / num_nodes)
    weight_centrality = 0.1 - 0.05 * (remaining_nodes / num_nodes)
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node = distance_matrix[node][destination_node]
        centrality_heuristic = sum(distance_matrix[node]) / (num_nodes - 1)
        score = (weight_proximity * distance_to_node + weight_long_term * distance_from_node) + (weight_centrality * centrality_heuristic)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

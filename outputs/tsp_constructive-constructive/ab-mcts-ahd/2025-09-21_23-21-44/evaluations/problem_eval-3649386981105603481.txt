def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_total_product = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_node_to_dest = distance_matrix[node][destination_node]
        heuristic_factor = 1 + (remaining_nodes / (remaining_nodes + 1))  # Adjusts based on remaining nodes
        total_product = distance_to_node * (distance_node_to_dest ** heuristic_factor)
        if total_product < min_total_product:
            min_total_product = total_product
            next_node = node
    return next_node

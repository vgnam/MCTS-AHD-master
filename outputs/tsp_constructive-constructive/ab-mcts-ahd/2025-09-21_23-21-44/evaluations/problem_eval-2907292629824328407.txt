def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_total_distance = float('inf')
    next_node = None
    current_to_destination = distance_matrix[current_node][destination_node]
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_destination = distance_matrix[node][destination_node]
        weight = 1.0 - (current_to_node / (current_to_node + current_to_destination + 1e-10))
        total_distance = current_to_node + (weight * node_to_destination)
        if total_distance < min_total_distance:
            min_total_distance = total_distance
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    alpha = 1.0 + (len(unvisited_nodes) / (len(unvisited_nodes) + 1))
    beta = 1.0 / alpha
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        score = (distance_current ** alpha) * (distance_destination ** beta)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    min_score = float('inf')
    next_node = None
    num_nodes = len(distance_matrix)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        centrality_heuristic = sum(distance_matrix[node]) / (num_nodes - 1)
        score = (weight_factor * immediate_distance) + (0.5 * weight_factor * destination_distance) + (0.2 * (1 - weight_factor) * centrality_heuristic)

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

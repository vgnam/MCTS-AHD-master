def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    current_path_length = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            ratio = current_dist
        else:
            ratio = current_dist / dest_dist
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        normalized_distance = current_dist / (current_path_length + 1e-6)
        dynamic_weight = 0.5 + 0.3 * (1 - normalized_distance)
        score = -0.7 * normalized_distance + 0.3 * ratio + dynamic_weight * (1 / (degree + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

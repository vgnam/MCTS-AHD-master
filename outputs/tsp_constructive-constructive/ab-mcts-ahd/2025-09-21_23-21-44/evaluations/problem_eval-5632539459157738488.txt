def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        if distance_destination == 0:
            score = distance_current
        else:
            weight = 1.0 / (1.0 + distance_destination) * (1.0 + remaining_nodes)
            penalty = 1.0 if distance_current > 2 * distance_matrix[current_node][destination_node] else 1.0
            score = distance_current * weight * penalty
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_combined_cost = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_immediate = 1.0 / (1.0 + remaining_nodes)  # Dynamic weight for immediate cost
    weight_global = 1.0 - weight_immediate  # Dynamic weight for global cost
    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        max_cost = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                max_cost = max(max_cost, distance_matrix[node][other_node])
        combined_cost = (weight_immediate * immediate_cost) + (weight_global * max_cost)
        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_weighted_sum = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1 / (1 + remaining_nodes)  # Adjusts based on remaining nodes
    for node in unvisited_nodes:
        current_to_node = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        weighted_sum = (1 - weight_factor) * current_to_node + weight_factor * node_to_dest
        if weighted_sum < min_weighted_sum:
            min_weighted_sum = weighted_sum
            next_node = node
    return next_node

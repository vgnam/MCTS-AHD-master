def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    weight_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    min_combined_cost = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        destination_cost = distance_matrix[node][destination_node]
        combined_cost = (weight_factor * immediate_cost) + (0.5 * destination_cost)

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = node

    return next_node

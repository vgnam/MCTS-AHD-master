def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = -float('inf')

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_distance = distance_matrix[node][destination_node] if unvisited_nodes == {node} else min(distance_matrix[node][n] for n in unvisited_nodes - {node})
        score = -immediate_distance + 0.5 * future_distance

        if score > best_score:
            best_score = score
            best_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = 0.3 * (remaining_nodes / (remaining_nodes + 1))
        score = distance_to_current + weight * distance_to_destination
        if distance_to_current < min_score:
            min_score = distance_to_current
            next_node = node
        elif distance_to_current == min_score and score < min_score + weight * distance_to_destination:
            next_node = node
    return next_node

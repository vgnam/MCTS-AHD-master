def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    total_nodes = len(unvisited_nodes) + 1  # +1 for current_node
    unvisited_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        penalty = 1 + 0.5 * (1 - unvisited_ratio)
        score = (distance_to_current * penalty) + (0.4 * distance_to_destination)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

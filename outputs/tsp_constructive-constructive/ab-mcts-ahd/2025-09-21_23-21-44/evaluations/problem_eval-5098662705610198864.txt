def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    remaining_nodes = len(unvisited_nodes)
    total_nodes = len(distance_matrix)
    progress = 1 - (remaining_nodes / total_nodes)  # Normalized progress (0 to 1)
    weight = 1 / (1 + math.exp(-10 * (progress - 0.5)))  # Sigmoid for dynamic weighting

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        distance_to_dest = distance_matrix[node][destination_node]

        # Penalize nodes too close to destination to avoid early termination
        penalty = 0.3 if distance_to_dest < (max(distance_matrix[node]) * 0.3) else 0

        # Dynamic weighted score: prioritize distance early, degree later
        score = -distance * (1 - weight) + 0.5 * (1 / (degree + 1)) * weight - penalty

        if score > best_score:
            best_score = score
            next_node = node
    return next_node

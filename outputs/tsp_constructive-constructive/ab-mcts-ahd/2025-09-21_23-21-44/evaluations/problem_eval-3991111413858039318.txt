def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_combined_cost = float('inf')
    next_node = None
    for node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][node]
        max_cost = 0
        avg_cost = 0
        for other_node in unvisited_nodes:
            if other_node != node:
                max_cost = max(max_cost, distance_matrix[node][other_node])
                avg_cost += distance_matrix[node][other_node]
        avg_cost /= len(unvisited_nodes) if len(unvisited_nodes) > 1 else 1
        combined_cost = immediate_cost + 0.5 * max_cost + 0.3 * avg_cost
        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = node
    return next_node

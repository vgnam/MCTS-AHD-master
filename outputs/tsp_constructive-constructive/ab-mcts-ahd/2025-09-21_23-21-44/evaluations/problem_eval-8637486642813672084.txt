def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    total_path_length = sum(distance_matrix[current_node][node] for node in unvisited_nodes)
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        weight = 0.5 + 0.5 * (distance_current / total_path_length) if total_path_length > 0 else 0.5
        score = (distance_current * distance_destination) * (1 - weight) + distance_destination * weight
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

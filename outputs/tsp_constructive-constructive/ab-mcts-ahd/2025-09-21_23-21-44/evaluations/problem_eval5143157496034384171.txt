def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_normalized_product = float('inf')
    next_node = None
    for node in unvisited_nodes:
        additive_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        multiplicative_product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        normalized_product = multiplicative_product / additive_distance if additive_distance != 0 else float('inf')
        if normalized_product < min_normalized_product:
            min_normalized_product = normalized_product
            next_node = node
    return next_node

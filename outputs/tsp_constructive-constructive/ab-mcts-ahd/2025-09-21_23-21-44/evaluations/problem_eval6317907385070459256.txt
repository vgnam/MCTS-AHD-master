def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        add_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        mult_distance = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        avg_distance = (add_distance + mult_distance) / 2
        norm_factor = (distance_matrix[current_node][node] + distance_matrix[node][destination_node]) / 2
        score = (0.6 * add_distance + 0.4 * mult_distance) / norm_factor
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

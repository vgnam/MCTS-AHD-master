def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_distance = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]

        if remaining_distance > 0:
            weight = 0.5 + (distance_destination / remaining_distance) * 0.5
        else:
            weight = 1.0

        score = (1 - weight) * distance_current + weight * distance_destination

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

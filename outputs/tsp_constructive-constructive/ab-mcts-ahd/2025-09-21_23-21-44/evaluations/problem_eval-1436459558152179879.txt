def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        # Sigmoid-based weighting for degree
        degree_weight = 1 / (1 + math.exp(-(degree - 5)))  # Adjust threshold (5) as needed
        # Distance to destination as secondary factor
        dest_distance = distance_matrix[node][destination_node]
        # Combined score with non-linear terms
        score = -distance + 0.3 * degree_weight + 0.2 * (1 / (dest_distance + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

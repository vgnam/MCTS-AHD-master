def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    num_nodes = len(distance_matrix)
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node = distance_matrix[node][destination_node]
        centrality_heuristic = sum(distance_matrix[node]) / (num_nodes - 1)
        hub_potential = sum(distance_matrix[node][other] for other in unvisited_nodes) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
        dynamic_weight = 0.5 + 0.3 * (len(unvisited_nodes) / num_nodes)
        score = (dynamic_weight * distance_to_node + (1 - dynamic_weight) * distance_from_node) + (0.2 * centrality_heuristic) + (0.1 * hub_potential)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

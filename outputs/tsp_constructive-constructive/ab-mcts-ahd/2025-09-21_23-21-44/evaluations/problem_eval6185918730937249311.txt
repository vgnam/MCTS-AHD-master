def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining = len(unvisited_nodes)
    for node in unvisited_nodes:
        weight = 1.0 / (1.0 + remaining) if remaining > 0 else 1.0
        score = (distance_matrix[current_node][node] +
                 weight * distance_matrix[node][destination_node])
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

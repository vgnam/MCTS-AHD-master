def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        weight = 0.5 if remaining_nodes > len(unvisited_nodes) // 2 else 0.2
        penalty = 0.1 * (distance_to_destination / distance_matrix[current_node][destination_node] - 1) if distance_to_destination > 1.5 * distance_matrix[current_node][destination_node] else 0
        score = distance_to_current + weight * distance_to_destination + penalty
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

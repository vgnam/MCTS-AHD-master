def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        sum_score = distance_current + distance_destination
        product_score = distance_current * distance_destination
        weight = 1.0 / remaining_nodes if remaining_nodes > 0 else 1.0
        score = (1 - weight) * sum_score + weight * product_score
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

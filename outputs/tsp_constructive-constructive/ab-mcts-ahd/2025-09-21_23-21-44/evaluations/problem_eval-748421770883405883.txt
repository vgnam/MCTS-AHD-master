def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        sum_distance = distance_current + distance_destination
        product_distance = distance_current * distance_destination
        normalized_score = (sum_distance + product_distance) / (distance_current + distance_destination + 1e-6)
        if normalized_score < min_score:
            min_score = normalized_score
            next_node = node
    return next_node

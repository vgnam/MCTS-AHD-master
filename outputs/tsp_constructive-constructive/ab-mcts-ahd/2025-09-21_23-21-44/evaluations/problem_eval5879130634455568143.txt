def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    total_nodes = len(distance_matrix)
    remaining_nodes_ratio = len(unvisited_nodes) / total_nodes if total_nodes > 0 else 0
    alpha = 0.3  # Weight for remaining nodes ratio
    beta = 0.5    # Weight for degree term
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        score = (-distance * (1 + alpha * (1 - remaining_nodes_ratio))) + beta * (1 / (degree + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

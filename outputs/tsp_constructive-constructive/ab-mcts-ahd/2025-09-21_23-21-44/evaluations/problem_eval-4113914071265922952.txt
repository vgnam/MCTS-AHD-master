def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    weight = 0.7  # Tunable parameter balancing current and destination distances
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            score = current_dist
        else:
            score = (1 - weight) * current_dist + weight * dest_dist
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

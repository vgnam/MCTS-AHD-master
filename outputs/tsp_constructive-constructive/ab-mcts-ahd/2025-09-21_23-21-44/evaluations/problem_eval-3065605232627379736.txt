def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    best_score = float('-inf')
    path_length = sum(distance_matrix[current_node][destination_node] for _ in range(len(unvisited_nodes)))
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        degree = sum(1 for d in distance_matrix[node] if d > 0)
        dynamic_weight = 1 / (1 + path_length) if path_length > 0 else 1
        score = -distance + dynamic_weight * (1 / (degree + 1))
        if score > best_score:
            best_score = score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    weight_factor = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    min_total_product = float('inf')
    next_node = None
    for node in unvisited_nodes:
        total_product = (distance_matrix[current_node][node] * distance_matrix[node][destination_node]) * weight_factor
        if total_product < min_total_product:
            min_total_product = total_product
            next_node = node
    return next_node

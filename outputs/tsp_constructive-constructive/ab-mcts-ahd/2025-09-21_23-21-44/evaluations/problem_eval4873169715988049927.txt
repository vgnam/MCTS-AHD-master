def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    num_nodes = len(distance_matrix)
    remaining_nodes = len(unvisited_nodes)
    total_nodes = num_nodes
    proximity_weight = 0.7 - 0.1 * (remaining_nodes / total_nodes)
    return_weight = 0.3 + 0.1 * (remaining_nodes / total_nodes)
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node = distance_matrix[node][destination_node]
        avg_distance_to_unvisited = sum(distance_matrix[current_node][n] for n in unvisited_nodes) / remaining_nodes
        outlier_penalty = abs(distance_to_node - avg_distance_to_unvisited) * 0.2
        centrality_heuristic = sum(distance_matrix[node]) / (num_nodes - 1)
        normalized_proximity = 1 / (1 + math.exp(-distance_to_node + avg_distance_to_unvisited))
        score = (proximity_weight * distance_to_node * normalized_proximity +
                 return_weight * distance_from_node +
                 0.1 * centrality_heuristic +
                 outlier_penalty)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

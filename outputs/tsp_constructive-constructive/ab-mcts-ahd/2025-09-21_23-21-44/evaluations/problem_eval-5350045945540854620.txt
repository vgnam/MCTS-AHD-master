def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_weighted_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    current_weight = 0.7 - 0.1 * (remaining_nodes / len(distance_matrix))  # Dynamic weight adjustment
    future_weight = 1.0 - current_weight
    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        future_distance = min(distance_matrix[node][n] for n in unvisited_nodes if n != node) if len(unvisited_nodes) > 1 else 0
        detour_penalty = sum(distance_matrix[current_node][n] + distance_matrix[n][node] - distance_matrix[current_node][node] for n in unvisited_nodes if n != node) / len(unvisited_nodes) if len(unvisited_nodes) > 1 else 0
        weighted_score = current_weight * current_distance + future_weight * future_distance + 0.1 * detour_penalty
        if weighted_score < min_weighted_score:
            min_weighted_score = weighted_score
            next_node = node
    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    min_score = float('inf')
    avg_distance = sum(distance_matrix[current_node][node] for node in unvisited_nodes) / len(unvisited_nodes)
    for node in unvisited_nodes:
        distance_current = distance_matrix[current_node][node]
        distance_destination = distance_matrix[node][destination_node]
        weight = 0.7 if distance_current < distance_destination else 0.3
        score = (weight * distance_current + (1 - weight) * distance_destination) / (avg_distance + 1e-6)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

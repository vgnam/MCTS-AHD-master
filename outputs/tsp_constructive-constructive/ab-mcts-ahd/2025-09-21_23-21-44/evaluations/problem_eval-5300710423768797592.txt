def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        add_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        mul_distance = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        normalized_score = (add_distance * 0.6) + (mul_distance * 0.4) / (add_distance + mul_distance + 1e-10)
        if normalized_score < min_score:
            min_score = normalized_score
            next_node = node
    return next_node

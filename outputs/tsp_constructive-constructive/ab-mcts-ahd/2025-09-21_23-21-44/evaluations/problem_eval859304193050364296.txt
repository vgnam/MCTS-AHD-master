def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    next_node = None
    min_score = float('inf')
    num_unvisited = len(unvisited_nodes)
    num_nodes = len(distance_matrix)
    exploration_decay = (num_unvisited / num_nodes) ** 2
    for node in unvisited_nodes:
        distance_to_node = distance_matrix[current_node][node]
        distance_from_node = distance_matrix[node][destination_node]
        avg_distance_to_unvisited = sum(distance_matrix[node][unvisited] for unvisited in unvisited_nodes) / num_unvisited
        neighborhood_factor = avg_distance_to_unvisited * (1 - exploration_decay)
        centrality_heuristic = (sum(distance_matrix[node]) / (num_nodes - 1)) ** 0.3
        exploration_term = (num_unvisited / num_nodes) * 0.2
        score = (0.4 * distance_to_node + 0.3 * distance_from_node + 0.2 * neighborhood_factor) * (1 + 0.1 * centrality_heuristic + exploration_term)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

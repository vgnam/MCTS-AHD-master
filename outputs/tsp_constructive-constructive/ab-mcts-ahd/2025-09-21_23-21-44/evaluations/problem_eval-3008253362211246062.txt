def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    weight_ratio = 0.7 if len(unvisited_nodes) > 1 else 0.5
    weight_sum = 0.3 if len(unvisited_nodes) > 1 else 0.5
    for node in unvisited_nodes:
        current_dist = distance_matrix[current_node][node]
        dest_dist = distance_matrix[node][destination_node]
        if dest_dist == 0:
            score = current_dist
        else:
            score = weight_ratio * (current_dist / dest_dist) + weight_sum * (current_dist + dest_dist)
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

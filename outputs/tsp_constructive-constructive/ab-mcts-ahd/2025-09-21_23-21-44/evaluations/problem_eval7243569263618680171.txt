def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_total_score = float('inf')
    next_node = None
    for node in unvisited_nodes:
        distance_sum = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        distance_product = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
        total_score = 0.6 * distance_sum + 0.4 * distance_product
        if total_score < min_total_score:
            min_total_score = total_score
            next_node = node
    return next_node

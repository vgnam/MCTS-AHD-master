def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node
    min_score = float('inf')
    next_node = None
    remaining_nodes_ratio = len(unvisited_nodes) / (len(unvisited_nodes) + 1)
    exploration_factor = 0.5 * (1 - remaining_nodes_ratio) ** 2
    penalty_factor = 1.2
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]
        score = distance_to_current + exploration_factor * distance_to_destination
        if node in unvisited_nodes and node != destination_node:
            score *= penalty_factor
        if score < min_score:
            min_score = score
            next_node = node
    return next_node

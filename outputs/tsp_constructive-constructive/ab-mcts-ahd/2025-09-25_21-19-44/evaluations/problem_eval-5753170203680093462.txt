def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        if node == destination_node:
            return node

        distance_to_current = distance_matrix[current_node][node]
        if distance_to_current > 2 * min(distance_matrix[current_node][n] for n in unvisited_nodes if n != node):
            penalty = 1.5
        else:
            penalty = 1.0

        avg_distance_to_remaining = sum(distance_matrix[node][n] for n in unvisited_nodes if n != node) / max(1, len(unvisited_nodes) - 1)
        dynamic_weight = 0.5 * (1 + 0.5 * (remaining_nodes / len(distance_matrix)))
        score = penalty * distance_to_current + dynamic_weight * avg_distance_to_remaining

        if score < min_score:
            min_score = score
            next_node = node

    if destination_node in unvisited_nodes:
        distance_to_destination = distance_matrix[current_node][destination_node]
        avg_distance_to_remaining = sum(distance_matrix[destination_node][n] for n in unvisited_nodes if n != destination_node) / max(1, len(unvisited_nodes) - 1)
        dynamic_weight = 0.5 * (1 + 0.5 * (remaining_nodes / len(distance_matrix)))
        destination_score = distance_to_destination + dynamic_weight * avg_distance_to_remaining

        if destination_score < min_score:
            next_node = destination_node

    return next_node

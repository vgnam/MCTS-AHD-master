def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, temperature=1.0):
    if not unvisited_nodes:
        return destination_node

    best_node = None
    best_score = float('-inf')

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Novel scoring function combining distance to current and distance to destination
        score = (1 - temperature) * (1 / (1 + distance_to_current)) + temperature * (1 / (1 + distance_to_destination))

        if score > best_score:
            best_score = score
            best_node = node

    return next_node

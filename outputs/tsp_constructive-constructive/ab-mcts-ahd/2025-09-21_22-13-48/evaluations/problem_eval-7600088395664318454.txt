def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    # Dynamic adjustment of Î» based on the number of remaining unvisited nodes
    lambda_param = 0.5 if len(unvisited_nodes) > 2 else 0.8

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Weighted score combining distance to current and distance to destination
        score = lambda_param * distance_to_current + (1 - lambda_param) * distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

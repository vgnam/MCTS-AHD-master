def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    dynamic_weight = 1.0 if remaining_nodes > 2 else 0.5  # Adjust weight based on remaining nodes
    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        proximity_penalty = 0.1 * immediate_distance if immediate_distance < 0.5 * distance_matrix[current_node][destination_node] else 0  # Penalize nodes too close to current
        score = immediate_distance - dynamic_weight * future_savings + proximity_penalty  # Updated weighted combination

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

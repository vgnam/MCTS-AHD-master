def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, λ=0.5, randomness=0.1):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    candidates = []

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        score = λ * distance_to_current + (1 - λ) * distance_to_dest

        if score < min_score or (score == min_score and random.random() < randomness):
            min_score = score
            next_node = node
            candidates = [node]
        elif score == min_score:
            candidates.append(node)

    if len(candidates) > 1:
        next_node = random.choice(candidates)

    return next_node

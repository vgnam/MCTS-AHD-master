def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # Prefer nodes with better immediate-to-future distance ratios
            current_ratio = distance / (distance_matrix[node][destination_node] + 1e-10)
            next_ratio = distance / (distance_matrix[next_node][destination_node] + 1e-10)
            if current_ratio < next_ratio:
                next_node = node

    return next_node

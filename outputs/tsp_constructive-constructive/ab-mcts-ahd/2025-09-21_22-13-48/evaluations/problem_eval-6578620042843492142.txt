def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, balance=0.5):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        destination_distance = distance_matrix[node][destination_node]
        combined_score = balance * current_distance + (1 - balance) * destination_distance

        if combined_score < best_score:
            best_score = combined_score
            next_node = node
        elif combined_score == best_score:
            if current_distance < distance_matrix[next_node][node]:
                next_node = node

    return next_node

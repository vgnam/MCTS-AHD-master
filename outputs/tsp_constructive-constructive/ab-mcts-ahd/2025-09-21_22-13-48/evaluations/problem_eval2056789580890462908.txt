def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_dest = distance_matrix[node][destination_node]
        if distance_to_dest < min_distance:
            min_distance = distance_to_dest
            next_node = node
        elif distance_to_dest == min_distance:
            # If multiple nodes have the same minimal distance to destination, choose the one closer to current node
            if distance_matrix[current_node][node] < distance_matrix[current_node][next_node]:
                next_node = node

    return next_node

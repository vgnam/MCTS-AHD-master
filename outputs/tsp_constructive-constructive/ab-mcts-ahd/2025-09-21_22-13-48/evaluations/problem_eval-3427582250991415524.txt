def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if unvisited_nodes:
            min_distance = min(distance_matrix[node][other] for other in unvisited_nodes if other != node)
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes)
        else:
            min_distance = 0
            avg_distance = 0
        score = 0.5 * immediate_distance + 0.3 * min_distance + 0.2 * avg_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

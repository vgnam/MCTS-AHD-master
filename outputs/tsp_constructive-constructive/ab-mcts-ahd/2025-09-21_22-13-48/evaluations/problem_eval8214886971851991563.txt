def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    weight_current = 0.6
    weight_destination = 0.4

    for node in unvisited_nodes:
        score = (weight_current * distance_matrix[current_node][node]) + (weight_destination * distance_matrix[node][destination_node])
        if score < min_score:
            min_score = score
            next_node = node

    return next_node

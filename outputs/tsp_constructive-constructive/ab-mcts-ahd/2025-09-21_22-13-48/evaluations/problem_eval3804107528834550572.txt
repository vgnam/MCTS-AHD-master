def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    remaining_nodes = len(unvisited_nodes)
    total_nodes = remaining_nodes + 1  # Including current node

    # Dynamic weight adjustment based on remaining nodes
    weight_immediate = 2.0 * (remaining_nodes / total_nodes)  # Higher when more nodes left
    weight_future = 0.3 * (1 - remaining_nodes / total_nodes)  # Higher when fewer nodes left

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        score = weight_immediate * immediate_distance - weight_future * future_savings

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix, alpha=0.5, beta=0.5):
    if not unvisited_nodes:
        return destination_node

    candidates = []
    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_dest = distance_matrix[node][destination_node]
        combined_score = alpha * distance_to_current + beta * distance_to_dest
        candidates.append((node, combined_score))

    # Sort candidates by combined score
    candidates.sort(key=lambda x: x[1])

    # Select top 3 candidates and choose randomly among them
    top_candidates = candidates[:min(3, len(candidates))]
    next_node = random.choice(top_candidates)[0]

    return next_node

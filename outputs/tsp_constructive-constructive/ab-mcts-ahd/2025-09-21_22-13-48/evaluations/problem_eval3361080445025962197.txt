def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        score = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if score < min_score:
            min_score = score
            next_node = node
        elif score == min_score:
            # If multiple nodes have the same score, choose the one with the higher sum
            if distance_matrix[current_node][node] + distance_matrix[node][destination_node] > \
               distance_matrix[current_node][next_node] + distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

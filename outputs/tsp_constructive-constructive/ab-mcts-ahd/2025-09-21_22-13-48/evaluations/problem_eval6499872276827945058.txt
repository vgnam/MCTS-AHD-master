def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    # Dynamic weight based on remaining unvisited nodes
    remaining_nodes = len(unvisited_nodes)
    weight_factor = 1.0 / (1.0 + math.exp(-(remaining_nodes - 5)))  # Sigmoid adjustment

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]

        # Penalty for large immediate distances
        penalty = max(0, immediate_distance - 2 * distance_matrix[current_node][destination_node])
        score = immediate_distance - (0.5 + 0.5 * weight_factor) * future_savings + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

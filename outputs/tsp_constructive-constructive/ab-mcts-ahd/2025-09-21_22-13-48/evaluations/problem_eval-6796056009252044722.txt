def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        # Weighted score: balance between distance to current node and distance to destination
        weight_current = remaining_nodes / (remaining_nodes + 1)
        weight_destination = 1 - weight_current
        score = (weight_current * distance_matrix[current_node][node]) + (weight_destination * distance_matrix[node][destination_node])

        if score < min_score:
            min_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        destination_penalty = distance_matrix[node][destination_node]
        score = immediate_distance - 0.7 * future_savings + 0.5 * destination_penalty

        if score < best_score:
            best_score = score
            next_node = node
        elif score == best_score:
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    remaining_nodes = len(unvisited_nodes)
    exploration_weight = 1 / (1 + math.exp(-0.1 * remaining_nodes))  # Sigmoid-based dynamic weighting

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if remaining_nodes > 1:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1)
        else:
            avg_distance = 0

        # Penalize high immediate distances more as nodes remain
        penalty_factor = 1 + (0.5 * immediate_distance / sum(sum(row) for row in distance_matrix) * (remaining_nodes / len(distance_matrix)))
        score = immediate_distance + exploration_weight * avg_distance * penalty_factor

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

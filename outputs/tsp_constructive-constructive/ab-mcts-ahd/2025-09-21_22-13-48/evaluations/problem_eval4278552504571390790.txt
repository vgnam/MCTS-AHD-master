def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Weighted score: balance proximity and potential path cost
        score = 0.6 * distance_to_current + 0.4 * distance_to_destination

        if score < min_score:
            min_score = score
            next_node = node
        elif score == min_score:
            # Tie-breaker: prefer the node closer to the destination
            if distance_to_destination < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

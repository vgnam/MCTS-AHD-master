def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    alpha = 0.7  # Weight for distance to current node
    beta = 0.3   # Weight for distance to destination

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance_to_current = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Combined score equation
        score = (alpha * distance_to_current) + (beta * distance_to_destination)

        if score < best_score:
            best_score = score
            next_node = node
        elif score == best_score:
            if distance_to_destination < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

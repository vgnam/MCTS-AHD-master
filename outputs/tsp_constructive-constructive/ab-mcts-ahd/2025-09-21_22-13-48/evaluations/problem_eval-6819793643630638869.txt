def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if unvisited_nodes:
            max_distance = max(distance_matrix[node][other] for other in unvisited_nodes if other != node) if unvisited_nodes else 0
        else:
            max_distance = 0
        score = immediate_distance + 0.5 * max_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

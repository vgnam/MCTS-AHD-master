def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        heuristic_cost = distance_matrix[node][destination_node]
        combined_score = immediate_distance + 0.5 * heuristic_cost  # Weighted combination

        if combined_score < min_score:
            min_score = combined_score
            next_node = node
        elif combined_score == min_score:
            if heuristic_cost < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    current_to_dest = distance_matrix[current_node][destination_node]

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        node_to_dest = distance_matrix[node][destination_node]
        weight_factor = (current_to_dest + 1e-6) / (node_to_dest + 1e-6)  # Avoid division by zero

        weighted_distance = distance * weight_factor

        if weighted_distance < min_distance:
            min_distance = weighted_distance
            next_node = node
        elif weighted_distance == min_distance:
            if node_to_dest < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None
    penalty_factor = 1.5  # Penalty factor to discourage revisiting nodes

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        adjusted_distance = distance * penalty_factor if node in unvisited_nodes else distance

        if adjusted_distance < min_distance:
            min_distance = adjusted_distance
            next_node = node
        elif adjusted_distance == min_distance:
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

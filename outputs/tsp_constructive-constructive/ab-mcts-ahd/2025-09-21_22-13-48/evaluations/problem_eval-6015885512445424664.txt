def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    remaining_nodes = len(unvisited_nodes)
    avg_future_savings = sum(distance_matrix[node][destination_node] for node in unvisited_nodes) / remaining_nodes

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        dynamic_weight = 0.5 * (1 + (future_savings / (avg_future_savings + 1e-6)))  # Normalized and bounded weight

        score = immediate_distance - dynamic_weight * future_savings

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_unvisited = len(unvisited_nodes)
    progress = 1 - (total_unvisited / (total_unvisited + 1))  # Normalized progress (0 to 1)
    weight_factor = 1 / (1 + math.exp(-10 * (progress - 0.5)))  # Sigmoid function for dynamic weighting

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        future_savings = distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]
        dynamic_weight = 1.0 - weight_factor * 0.5  # Adjust weight based on progress
        score = immediate_distance - dynamic_weight * future_savings

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    total_unvisited = len(unvisited_nodes)
    dynamic_weight = 0.3 if total_unvisited > 3 else 0.1

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if total_unvisited > 1:
            normalized_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (total_unvisited - 1)
        else:
            normalized_distance = 0
        score = immediate_distance + dynamic_weight * normalized_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

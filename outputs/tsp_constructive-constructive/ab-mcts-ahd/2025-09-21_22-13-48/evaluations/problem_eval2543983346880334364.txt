def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        proximity_factor = 1.0 / (1.0 + distance_matrix[node][destination_node])
        adjusted_future_savings = (distance_matrix[node][destination_node] - distance_matrix[current_node][destination_node]) * proximity_factor
        score = immediate_distance + 0.7 * adjusted_future_savings  # Adjusted weights

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)
    weight_factor = 0.1 + 0.2 * (1 - num_unvisited / len(distance_matrix))

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / max(1, num_unvisited - 1)
        max_distance_to_dest = max(distance_matrix[node][dest] for dest in unvisited_nodes if dest != node)
        score = immediate_distance + weight_factor * avg_distance + 0.1 * max_distance_to_dest

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

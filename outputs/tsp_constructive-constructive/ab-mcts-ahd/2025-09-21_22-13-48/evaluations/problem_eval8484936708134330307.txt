def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        long_term_distance = distance_matrix[node][destination_node]
        dynamic_weight = 1 + (0.5 * (len(unvisited_nodes) / len(distance_matrix)))

        score = immediate_distance + (dynamic_weight * long_term_distance)

        if score < min_score:
            min_score = score
            next_node = node
        elif score == min_score:
            if immediate_distance < distance_matrix[next_node][current_node]:
                next_node = node

    return next_node

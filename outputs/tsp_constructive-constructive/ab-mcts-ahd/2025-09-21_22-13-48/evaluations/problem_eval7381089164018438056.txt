def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    decay_factor = 0.1 * (1 - remaining_nodes / (remaining_nodes + 1))  # Adjusts weight dynamically

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if unvisited_nodes:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / len(unvisited_nodes)
        else:
            avg_distance = 0
        penalty = 1 if node in unvisited_nodes and node != current_node else 0  # Penalize revisits
        score = immediate_distance + (0.2 + decay_factor) * avg_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

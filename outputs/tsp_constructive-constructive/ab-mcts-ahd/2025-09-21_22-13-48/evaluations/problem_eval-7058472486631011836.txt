def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        if distance < min_distance:
            min_distance = distance
            next_node = node
        elif distance == min_distance:
            # Break ties by choosing the node closer to the destination but also not too close to avoid premature clustering
            dest_distance_current = distance_matrix[node][destination_node]
            dest_distance_next = distance_matrix[next_node][destination_node]
            if dest_distance_current < dest_distance_next or (dest_distance_current == dest_distance_next and distance_matrix[node][current_node] > distance_matrix[next_node][current_node]):
                next_node = node

    return next_node

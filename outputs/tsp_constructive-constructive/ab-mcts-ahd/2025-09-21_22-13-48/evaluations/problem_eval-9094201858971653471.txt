def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    remaining_nodes = len(unvisited_nodes)
    weight_factor = 0.3 if remaining_nodes > 2 else 0.1

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if remaining_nodes > 1:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1)
        else:
            avg_distance = 0

        penalty = 0.5 if immediate_distance < 0.1 * max(distance_matrix[current_node]) else 0
        score = immediate_distance + weight_factor * avg_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

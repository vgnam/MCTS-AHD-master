def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    num_unvisited = len(unvisited_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if num_unvisited > 1:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (num_unvisited - 1)
        else:
            avg_distance = 0

        # Dynamic weight based on unvisited nodes
        weight = 0.3 / (1 + 0.1 * num_unvisited) if num_unvisited > 0 else 0

        # Penalize nodes too close to avoid local minima
        penalty = 0.1 * immediate_distance if immediate_distance < 0.2 * sum(distance_matrix[current_node]) / len(distance_matrix) else 0

        score = immediate_distance + weight * avg_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

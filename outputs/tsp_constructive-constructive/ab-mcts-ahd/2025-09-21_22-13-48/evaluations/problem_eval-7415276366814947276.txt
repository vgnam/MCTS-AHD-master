def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if unvisited_nodes:
            distances_to_unvisited = [distance_matrix[node][other] for other in unvisited_nodes if other != node]
            avg_distance = sum(distances_to_unvisited) / len(distances_to_unvisited)
            variance_distance = sum((d - avg_distance) ** 2 for d in distances_to_unvisited) / len(distances_to_unvisited)
            weight_factor = 0.2 * (len(unvisited_nodes) / len(distance_matrix))
            score = immediate_distance + weight_factor * avg_distance + 0.1 * variance_distance
        else:
            score = immediate_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    min_sum_distance = float('inf')
    next_node = None

    for node in unvisited_nodes:
        sum_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if sum_distance < min_sum_distance:
            min_sum_distance = sum_distance
            next_node = node
        elif sum_distance == min_sum_distance:
            if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                next_node = node

    return next_node

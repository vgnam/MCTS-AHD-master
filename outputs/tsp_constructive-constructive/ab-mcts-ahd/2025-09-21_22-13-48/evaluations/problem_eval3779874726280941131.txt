def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)
    decay_factor = 1.0 / (1.0 + remaining_nodes)  # Decreases as nodes are visited

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if remaining_nodes > 1:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1)
        else:
            avg_distance = 0

        # Dynamic weighting: prioritize immediate distance early, then balance
        weighted_distance = (1 - decay_factor) * immediate_distance + decay_factor * avg_distance

        # Penalty for nodes that were previously considered but not chosen
        penalty = 0.1 * (remaining_nodes - 1)  # Penalize more as fewer nodes remain
        score = weighted_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

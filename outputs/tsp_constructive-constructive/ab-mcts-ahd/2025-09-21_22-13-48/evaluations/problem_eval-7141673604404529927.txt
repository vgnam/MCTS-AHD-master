def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    max_remaining_distance = -1
    next_node = None

    for node in unvisited_nodes:
        current_distance = distance_matrix[current_node][node]
        remaining_distance = distance_matrix[node][destination_node]

        if current_distance < distance_matrix[current_node][next_node] if next_node is not None else True:
            if remaining_distance > max_remaining_distance:
                max_remaining_distance = remaining_distance
                next_node = node
            elif remaining_distance == max_remaining_distance:
                if distance_matrix[node][destination_node] < distance_matrix[next_node][destination_node]:
                    next_node = node

    return next_node

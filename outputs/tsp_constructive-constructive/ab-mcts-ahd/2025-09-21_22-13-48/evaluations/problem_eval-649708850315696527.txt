def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    best_score = float('inf')
    next_node = None
    remaining_nodes = len(unvisited_nodes)

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if remaining_nodes > 1:
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (remaining_nodes - 1)
        else:
            avg_distance = 0

        # Dynamic weighting using sigmoid function
        weight_factor = 1 / (1 + math.exp(-(remaining_nodes - 5)))
        weighted_distance = immediate_distance + weight_factor * avg_distance

        # Penalty for high average distances to encourage diversification
        if remaining_nodes > 1:
            max_avg_distance = max(distance_matrix[node][other] for other in unvisited_nodes if other != node)
            penalty = 0.1 * max_avg_distance
        else:
            penalty = 0

        score = weighted_distance + penalty

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

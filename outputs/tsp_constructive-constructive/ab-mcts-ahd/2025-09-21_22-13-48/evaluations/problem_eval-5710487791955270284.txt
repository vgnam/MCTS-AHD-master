def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return destination_node

    total_nodes = len(unvisited_nodes) + 1
    visited_ratio = (total_nodes - len(unvisited_nodes)) / total_nodes
    dynamic_weight = 0.1 + 0.2 * visited_ratio

    best_score = float('inf')
    next_node = None

    for node in unvisited_nodes:
        immediate_distance = distance_matrix[current_node][node]
        if unvisited_nodes:
            distances_to_unvisited = [distance_matrix[node][other] for other in unvisited_nodes if other != node]
            avg_distance = sum(distances_to_unvisited) / len(distances_to_unvisited)
            variance = sum((d - avg_distance) ** 2 for d in distances_to_unvisited) / len(distances_to_unvisited)
            score = immediate_distance + dynamic_weight * avg_distance - 0.1 * variance
        else:
            score = immediate_distance

        if score < best_score:
            best_score = score
            next_node = node

    return next_node

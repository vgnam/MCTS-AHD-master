import random
import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    if destination_node in unvisited_nodes:
        return destination_node

    # Calculate exploration factor based on remaining unvisited nodes
    exploration_factor = 1.0 - (len(unvisited_nodes) / len(distance_matrix))

    # Compute weighted probabilities for each unvisited node
    weights = []
    for node in unvisited_nodes:
        distance = distance_matrix[current_node][node]
        # Weight is inverse of distance raised to a power that increases with exploration factor
        weight = (1.0 / (distance + 1e-6)) ** (1.0 + 2.0 * exploration_factor)
        weights.append(weight)

    # Normalize weights to form a probability distribution
    total_weight = sum(weights)
    probabilities = [w / total_weight for w in weights]

    # Select next node based on weighted probabilities
    next_node = random.choices(unvisited_nodes, weights=probabilities, k=1)[0]

    return next_node

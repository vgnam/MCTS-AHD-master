import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node
    if not unvisited_nodes:
        return None

    nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    # Higher stochasticity: 70% chance to pick from top 2 nearest, 20% from next 3, 10% random
    if random.random() < 0.7:
        candidates = nearest_neighbors[:min(2, len(nearest_neighbors))]
    elif random.random() < 0.9:
        candidates = nearest_neighbors[:min(5, len(nearest_neighbors))]
    else:
        candidates = nearest_neighbors

    next_node = random.choice(candidates)
    return next_node

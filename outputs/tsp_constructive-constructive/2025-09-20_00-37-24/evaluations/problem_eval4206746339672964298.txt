def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node
    if not unvisited_nodes:
        return None

    if random.random() < 0.8:  # 80% chance to use weighted score
        min_score = float('inf')
        next_node = None
        for node in unvisited_nodes:
            current_distance = distance_matrix[current_node][node]
            avg_distance = sum(distance_matrix[node][other] for other in unvisited_nodes if other != node) / (len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 0
            score = 0.7 * current_distance + 0.3 * avg_distance  # Weighted sum
            if score < min_score:
                min_score = score
                next_node = node
    else:  # 20% chance to pick from top 3 nearest
        nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
        candidates = nearest_neighbors[:min(3, len(nearest_neighbors))]
        next_node = random.choice(candidates)

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node
    if not unvisited_nodes:
        return None

    farthest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node], reverse=True)

    if random.random() < 0.9:
        candidates = farthest_neighbors[:min(3, len(farthest_neighbors))]
        next_node = random.choice(candidates)
    else:
        next_node = random.choice(farthest_neighbors)

    return next_node

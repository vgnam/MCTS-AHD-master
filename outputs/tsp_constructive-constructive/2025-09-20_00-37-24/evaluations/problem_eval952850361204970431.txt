import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node
    if not unvisited_nodes:
        return None

    nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    # 80% chance to pick from top 2 nearest, 15% chance to pick next 2, 5% chance to pick random
    if random.random() < 0.8:
        candidates = nearest_neighbors[:min(2, len(nearest_neighbors))]
    elif random.random() < 0.95:
        candidates = nearest_neighbors[min(2, len(nearest_neighbors)):min(4, len(nearest_neighbors))]
    else:
        candidates = nearest_neighbors

    next_node = random.choice(candidates)
    return next_node

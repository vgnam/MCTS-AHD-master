import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        return destination_node
    if not unvisited_nodes:
        return None

    nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    # 70% chance to pick from top 2 nearest
    if random.random() < 0.7:
        candidates = nearest_neighbors[:min(2, len(nearest_neighbors))]
        next_node = random.choice(candidates)
    # 20% chance to pick from top 5 nearest
    elif random.random() < 0.9:
        candidates = nearest_neighbors[:min(5, len(nearest_neighbors))]
        next_node = random.choice(candidates)
    # 10% chance to pick a random node
    else:
        next_node = random.choice(nearest_neighbors)

    return next_node

import random

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        # If destination is among unvisited, prioritize it if it's among the top 5 nearest
        nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])
        if destination_node in nearest_neighbors[:min(5, len(nearest_neighbors))]:
            return destination_node

    if not unvisited_nodes:
        return None

    nearest_neighbors = sorted(unvisited_nodes, key=lambda node: distance_matrix[current_node][node])

    # Adaptive stochasticity: lower probability of picking nearest neighbors as fewer nodes remain
    exploration_prob = 0.1 + (0.8 * len(unvisited_nodes) / len(distance_matrix))

    if random.random() < exploration_prob:
        candidates = nearest_neighbors[:min(3, len(nearest_neighbors))]
        next_node = random.choice(candidates)
    else:
        next_node = random.choice(nearest_neighbors)

    return next_node

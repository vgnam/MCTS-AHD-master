def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if destination_node in unvisited_nodes:
        if len(unvisited_nodes) == 1:
            return destination_node
        min_value = float('inf')
        next_node = None
        for node in unvisited_nodes:
            if node == destination_node:
                continue
            product_distance = distance_matrix[current_node][node] * distance_matrix[node][destination_node]
            if product_distance < min_value:
                min_value = product_distance
                next_node = node
        return next_node if next_node is not None else destination_node
    min_distance = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if distance_matrix[current_node][node] < min_distance:
            min_distance = distance_matrix[current_node][node]
            next_node = node
    return next_node

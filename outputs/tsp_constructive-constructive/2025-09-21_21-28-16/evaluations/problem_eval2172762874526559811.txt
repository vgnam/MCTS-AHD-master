def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    min_value = float('inf')
    next_node = None
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        combined_distance = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if combined_distance < min_value:
            min_value = combined_distance
            next_node = node
    return next_node

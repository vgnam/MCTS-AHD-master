def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    max_ratio = -float('inf')
    next_node = None
    for node in unvisited_nodes:
        if node == destination_node:
            continue
        current_to_dest = distance_matrix[current_node][destination_node]
        sum_distances = distance_matrix[current_node][node] + distance_matrix[node][destination_node]
        if current_to_dest == 0:
            ratio = float('inf')
        else:
            ratio = sum_distances / current_to_dest
        if ratio > max_ratio:
            max_ratio = ratio
            next_node = node
    if next_node is None:
        return destination_node
    return next_node
